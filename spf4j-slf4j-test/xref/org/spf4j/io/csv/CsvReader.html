32  package org.spf4j.io.csv;
33  
34  import java.io.IOException;
35  import java.util.Iterator;
36  import java.util.function.Consumer;
37  import javax.annotation.Nonnull;
38  import javax.annotation.Nullable;
39  
40  /**
41   * Char separated value file Reader.
42   * @author zoly
43   */
44  public interface CsvReader {
45  
46    enum TokenType {
47       ELEMENT, END_ROW, END_DOCUMENT
48    }
49  
50    /**
51     * read next CSV element, and return its type.
52     *
53     * @return return CSV element type.
54     * @throws IOException exception is something goes wrong.
55     */
56    @Nonnull
57    TokenType next() throws IOException, CsvParseException;
58  
59    /**
60     * @return the currently parsed token type,  null if no current token is available (next has never been called)
61     * @throws IOException
62     * @throws CsvParseException
63     */
64    @Nullable
65    TokenType current();
66  
67    /**
68     * the CSV element string. the underlying instance is reused, so you will need to make a copy of this if planning to
69     * use it.
70     *
71     * @return CharSequence representing a csv cell.
72     */
73    CharSequence getElement();
74  
75    default int skipRow() throws IOException, CsvParseException {
76      int skipped = 0;
77      TokenType current = current();
78      if (current == null) { // beginning of file.
79        next();
80      }
81      while ((current = current()) != CsvReader.TokenType.END_ROW
82              && current != CsvReader.TokenType.END_DOCUMENT) {
83        next();
84        skipped++;
85      }
86      if (current != CsvReader.TokenType.END_DOCUMENT) {
87        next();
88      }
89      return skipped;
90    }
91  
92    default void readRow(final Consumer<CharSequence> consumer) throws IOException, CsvParseException {
93      TokenType current = current();
94      if (current == null) { // beginning of file.
95        next();
96      }
97      while ((current = current()) != CsvReader.TokenType.END_ROW
98              && current != CsvReader.TokenType.END_DOCUMENT) {
99        consumer.accept(getElement());
100       next();
101     }
102     if (current != CsvReader.TokenType.END_DOCUMENT) {
103       next();
104     }
105   }
106 
107 
108   default CsvReader toReader(final Iterator<? extends CharSequence> it) {
109     return new IterableCsvReader(it);
110   }
111 }
