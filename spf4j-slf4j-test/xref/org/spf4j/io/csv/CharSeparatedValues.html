55  /**
56   * Supports Character Separated values format as described at: https://en.wikipedia.org/wiki/Comma-separated_values.
57   * either of \n \r or \r\n generalized to custom separator character.
58   * are valid end of line delimiters
59   *
60   * why another implementation? because I need one that is as fast as possible, and as flexible as possible.
61   *
62   * @author zoly

76    public CharSeparatedValues(final char separator) {
77      if (separator == '\n' || separator == '\r' || separator == '"') {
78        throw new IllegalArgumentException("Illegal separator character " + separator);
79      }
80      this.separator = separator;
81      this.toEscape = new char[]{separator, '\n', '\r', '"'};
82    }
83  
84    public CharSeparatedValues(final char separator, final char... extraCharsToEscape) {
85      if (separator == '\n' || separator == '\r' || separator == '"') {
86        throw new IllegalArgumentException("Illegal separator character " + separator);
87      }
88      this.separator = separator;
89      this.toEscape = new char[4 + extraCharsToEscape.length];
90      this.toEscape[0] = separator;
91      this.toEscape[1] = '\n';
92      this.toEscape[2] = '\r';
93      this.toEscape[3] = '"';
94      System.arraycopy(extraCharsToEscape, 0, this.toEscape, 4, extraCharsToEscape.length);
95    }

102   public void writeCsvRowNoEOL(final Appendable writer, final Object... elems) throws IOException {
103     if (elems.length > 0) {
104       int i = 0;
105       Object elem = elems[i++];
106       if (elem != null) {
107         writeCsvElement(elem.toString(), writer);
108       }
109       while (i < elems.length) {
110         writer.append(separator);
111         elem = elems[i++];
112         if (elem != null) {
113           writeCsvElement(elem.toString(), writer);
114         }
115       }
116     }
117   }

119   public void writeCsvRow2(final Appendable writer, final Object obj, final Object... elems)
120           throws IOException {
121     if (obj != null) {
122       writeCsvElement(obj.toString(), writer);
123     }
124     for (Object elem : elems) {
125       writer.append(separator);
126       if (elem != null) {
127         writeCsvElement(elem.toString(), writer);
128       }
129     }
130     writer.append('\n');
131   }

154   public void writeCsvRowNoEOL(final Iterable<?> elems, final Appendable writer) throws IOException {
155     Iterator<?> it = elems.iterator();
156     if (it.hasNext()) {
157       Object next = it.next();
158       if (next != null) {
159         writeCsvElement(next.toString(), writer);
160       }
161       while (it.hasNext()) {
162         writer.append(separator);
163         next = it.next();
164         if (next != null) {
165           writeCsvElement(next.toString(), writer);
166         }
167       }
168     }
169   }

202   public <T> T read(final Reader preader,
203           final CsvHandler<T> handler) throws IOException, CsvParseException {
204     PushbackReader reader = new PushbackReader(preader);
205     int firstChar = reader.read();
206     if (firstChar != UTF_BOM && firstChar >= 0) {
207       reader.unread(firstChar);
208     }
209     return readNoBom(reader, handler);
210   }

212   /**
213    * reads CSV format until EOF of reader.
214    *
215    * @param <T>
216    * @param preader
217    * @param handler
218    * @return
219    * @throws IOException
220    */
221   public <T> T readNoBom(final PushbackReader reader, final CsvHandler<T> handler)
222           throws IOException, CsvParseException {
223     CsvReader r = reader(reader);
224     handler.startRow(0);
225     CsvReader.TokenType token = r.next();
226     while (token != CsvReader.TokenType.END_DOCUMENT) {
227       if (token == CsvReader.TokenType.ELEMENT) {
228         handler.element(r.getElement());
229         token = r.next();
230       } else if (token == CsvReader.TokenType.END_ROW) {
231         handler.endRow();
232         token = r.next();
233         if (token == CsvReader.TokenType.ELEMENT) {
234           handler.startRow(r.currentLineNumber());
235         }
236       }
237     }
238     return handler.eof();
239   }

305   public void writeCsvElement(final CharSequence elem, final Appendable writer) throws IOException {
306     if (CharSequences.containsAnyChar(elem, toEscape)) {
307       writeQuotedCsvElement(elem, writer);
308     } else {
309       writer.append(elem);
310     }
311   }

313   public static void writeQuotedCsvElement(final CharSequence elem, final Appendable writer) throws IOException {
314     writer.append('"');
315     writeQuotedElementContent(elem, 0, elem.length(), writer);
316     writer.append('"');
317   }
318 
319   public static void writeQuotedElementContent(final CharSequence elem,
320           final int start, final int end, final Appendable writer) throws IOException {
321     for (int i = start; i < end; i++) {
322       char c = elem.charAt(i);
323       writeQuotedChar(c, writer);
324     }
325   }
326 
327   public static void writeQuotedChar(final char c, final Appendable writer) throws IOException {
328     if (c == '"') {
329       writer.append("\"\"");
330     } else {
331       writer.append(c);
332     }
333   }

335    public CharSequence toCsvElement(final CharSequence elem) {
336     if (CharSequences.containsAnyChar(elem, toEscape)) {
337       StringBuilder sw = new StringBuilder(elem.length() + 4);
338       try {
339         writeQuotedCsvElement(elem, sw);
340       } catch (IOException ex) {
341         throw new UncheckedIOException(ex);
342       }
343       return sw;
344     } else {
345       return elem;
346     }
347   }

349    public String toCsvElement(final String elem) {
350     if (CharSequences.containsAnyChar(elem, toEscape)) {
351       StringBuilder sw = new StringBuilder(elem.length() + 4);
352       try {
353         writeQuotedCsvElement(elem, sw);
354       } catch (IOException ex) {
355         throw new UncheckedIOException(ex);
356       }
357       return sw.toString();
358     } else {
359       return elem;
360     }

363   /**
364    * returns next character.
365    *
366    * @param reader
367    * @param addElemTo
368    * @return - next character or -1 if eof has been reached.
369    * @throws IOException
370    */
371   @CheckReturnValue
372   public int readCsvElement(final Reader reader, final StringBuilder addElemTo, final long lineNr)
373           throws IOException, CsvParseException {
374     int c = reader.read();
375     if (c < 0) {
376       return c;
377     }
378     if (c == '"') {
379       c = reader.read();
380       while (c >= 0) {
381         if (c == '"') {
382           int c2 = reader.read();
383           if (c2 >= 0) {
384             if (c2 == '"') {
385               addElemTo.append((char) c);
386             } else {
387               return c2;
388             }
389           } else {
390             return c2;
391           }
392         } else {
393           addElemTo.append((char) c);
394         }
395         c = reader.read();
396       }
397       throw new CsvParseException("Escaped CSV element " + addElemTo + " not terminated correctly at " + lineNr);
398     } else {
399       while (c != separator && c != '\n' && c != '\r' && c >= 0) {
400         addElemTo.append((char) c);
401         c = reader.read();
402       }
403     }
404     return c;
405   }

427   private static class CsvMapHandler2CsvHandler<T> implements CsvHandler<T> {
428 
429     private final CsvMapHandler<T> handler;
430     private boolean first = true;
431     private final List<String> header = new ArrayList<>();
432     private int elemIdx;
433     private Map<String, String> row = null;
434     private long lineNr;
435 
436 
437     CsvMapHandler2CsvHandler(final CsvMapHandler<T> handler) {
438       this.handler = handler;
439     }
440 
441     @Override
442     public void startRow(final long ln) {
443       lineNr = ln;
444       elemIdx = 0;
445       if (!first) {
446         row = new THashMap<>(header.size());
447       }
448     }
449 
450     @Override
451     public void element(final CharSequence elem) throws CsvParseException {
452       if (first) {
453         header.add(elem.toString());
454       } else {
455         if (header.size() <= elemIdx) {
456           throw new CsvParseException("Too many elements in row " + row + " at line " + lineNr);
457         }
458         row.put(header.get(elemIdx), elem.toString());
459       }
460       elemIdx++;
461     }
462 
463     @Override
464     public void endRow() {
465       if (first) {
466         first = false;
467       } else {
468         handler.row(row);
469       }
470     }
471 
472     @Override
473     public T eof() {
474       return handler.eof();
475     }
476   }

478  private class CsvReaderImpl implements CsvReader {
479 
480     private final PushbackReader reader;
481     private final StringBuilder currentElement = new StringBuilder();
482     private CsvReader.TokenType currentToken;
483     private CsvReader.TokenType nextToken;
484     private long lineNr = 0;
485 
486     CsvReaderImpl(final PushbackReader reader) {
487       this.reader = reader;
488       this.currentToken = CsvReader.TokenType.START_DOCUMENT;
489       this.nextToken = null;
490     }
491 
492    @SuppressFBWarnings("SF_SWITCH_FALLTHROUGH")
493    private void readNext() throws IOException, CsvParseException {
494      // nextToken will always be null;
495      switch (currentToken) {
496        case END_DOCUMENT:
497          nextToken = TokenType.END_DOCUMENT;
498          return;
499        case END_ROW:
500          // handle special case of EOF followed by EOL.
501          int peek = reader.read();
502          if (peek < 0) {
503            currentToken = TokenType.END_DOCUMENT;
504            nextToken = TokenType.END_DOCUMENT;
505            return;
506          }
507          reader.unread(peek);
508        case START_DOCUMENT:
509        case ELEMENT:
510          currentElement.setLength(0);
511          int next = readCsvElement(reader, currentElement, lineNr);
512          currentToken = CsvReader.TokenType.ELEMENT;
513          switch (next) {
514            case '\r':
515              lineNr++;
516              nextToken = CsvReader.TokenType.END_ROW;
517              int c2 = reader.read();
518              if (c2 < 0) {
519                return;
520              }
521              if (c2 != '\n') {
522                reader.unread(c2);
523              }
524              return;
525            case '\n':
526              lineNr++;
527              nextToken = CsvReader.TokenType.END_ROW;
528              c2 = reader.read();
529              if (c2 < 0) {
530                return;
531              }
532              if (c2 != '\r') {
533                reader.unread(c2);
534                break;
535              }
536              break;
537            default:
538              if (next != separator) {
539                if (next < 0) {
540                  nextToken = CsvReader.TokenType.END_ROW;
541                } else {
542                  throw new CsvParseException("Unexpected character " + next + " at line" + lineNr);
543                }
544              }
545          }
546          return;
547        default:
548          throw new IllegalStateException("Invalid current token " + currentToken);
549 
550      }
551 
552    }
553 
554     @Override
555     public CsvReader.TokenType next() throws IOException, CsvParseException {
556       if (nextToken == null) {
557         readNext();
558         return currentToken;
559       } else {
560         CsvReader.TokenType result = nextToken;
561         if (result != CsvReader.TokenType.END_DOCUMENT) {
562           nextToken = null;
563         }
564         currentToken = result;
565         return result;
566       }
567     }
568 
569     @Override
570     public CsvReader.TokenType current() {
571       return currentToken;
572     }
573 
574     @Override
575     public CharSequence getElement() {
576       if (currentToken != TokenType.ELEMENT) {
577         throw new IllegalStateException("No current element, current token is " + currentToken);
578       }
579       return currentElement;
580     }
581 
582     @Override
583     public long currentLineNumber() {
584       return lineNr;
585     }

632   private class CsvWriterImpl implements CsvWriter {
633 
634     private final Writer writer;
635 
636     CsvWriterImpl(final Writer writer) {
637       this.writer = writer;
638     }
639     private boolean isStartLine = true;
640 
641     @Override
642     public void writeElement(final CharSequence cs) throws IOException {
643       addComma();
644       writeCsvElement(cs, writer);
645     }
646 
647     private void addComma() throws IOException {
648       if (isStartLine) {
649         isStartLine = false;
650       } else {
651         writer.append(separator);
652       }
653     }
654 
655     @Override
656     public void writeEol() throws IOException {
657       writer.append('\n');
658       isStartLine = true;
659     }
660 
661     @Override
662     public void flush() throws IOException {
663       writer.flush();
664     }
665 
666 
667     @Override
668     public ElementAppendable startQuotedElement() throws IOException {
669       addComma();
670       writer.write('"');
671       return new ElementAppendable() {
672         @Override
673         public Appendable append(final CharSequence csq) throws IOException {
674           writeQuotedElementContent(csq, 0, csq.length(), writer);
675           return this;
676         }
677 
678         @Override
679         public Appendable append(final CharSequence csq, final int start, final int end) throws IOException {
680           writeQuotedElementContent(csq, start, end, writer);
681           return this;
682         }
683 
684         @Override
685         public Appendable append(final char c) throws IOException {
686           writeQuotedChar(c, writer);
687           return this;
688         }
689 
690         @Override
691         public void close() throws IOException {
692           writer.write('"');
693         }
694       };
695     }
696 
697     @Override
698     public Appendable startRawElement() throws IOException {
699       addComma();
700       return new Appendable() {
701         @Override
702         public Appendable append(final CharSequence csq) throws IOException {
703           if (CharSequences.containsAnyChar(csq, toEscape)) {
704             throw new IllegalStateException("Attempting to write str containing escapeable seq " + csq);
705           }
706           writer.append(csq);
707           return this;
708         }
709 
710         @Override
711         public Appendable append(final CharSequence csq, final int start, final int end) throws IOException {
712           if (CharSequences.containsAnyChar(csq, start, end, toEscape)) {
713             throw new IllegalStateException("Attempting to write str containing escapeable seq " + csq);
714           }
715           writer.append(csq, start, end);
716           return this;
717         }
718 
719         @Override
720         public Appendable append(final char c) throws IOException {
721           if (Arrays.search(toEscape, c) >= 0) {
722             throw new IllegalStateException("Attempting to write str containing escapeable seq " + c);
723           }
724           writer.append(c);
725           return this;
726         }
727       };
728     }
