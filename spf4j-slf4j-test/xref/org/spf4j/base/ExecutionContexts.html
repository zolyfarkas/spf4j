64    private static ExecutionContextFactory<ExecutionContext> initFactory() {
65      String factoryClass = System.getProperty("spf4j.execContentFactoryClass");
66      ExecutionContextFactory<ExecutionContext> factory;
67      if (factoryClass == null) {
68        factory = new BasicExecutionContextFactory();
69      } else {
70        try {
71          factory = ((Class<ExecutionContextFactory<ExecutionContext>>) Class.forName(factoryClass)).newInstance();
72        } catch (ClassNotFoundException | InstantiationException | IllegalAccessException ex) {
73          throw new ExceptionInInitializerError(ex);
74        }
75      }
76      String factoryWrapperClass = System.getProperty("spf4j.execContentFactoryWrapperClass");
77      if (factoryWrapperClass != null) {
78        try {
79          factory = (ExecutionContextFactory<ExecutionContext>) Class.forName(factoryWrapperClass)
80                  .getConstructor(ExecutionContextFactory.class).newInstance(factory);
81        } catch (ClassNotFoundException | InstantiationException | IllegalAccessException
82                | NoSuchMethodException | InvocationTargetException ex) {
83          throw new ExceptionInInitializerError(ex);
84        }
85      }
86      return factory;
87    }

90      return CTX_FACTORY;
91    }

160   public static ExecutionContext start(final String name,
161           @Nullable final ExecutionContext parent, final long timeout, final TimeUnit tu) {
162     ExecutionContext localCtx = EXEC_CTX.get();
163     long nanoTime = TimeSource.nanoTime();
164     ExecutionContext nCtx;
165     if (localCtx == null) {
166       nCtx = CTX_FACTORY.startThreadRoot(name, parent,
167               nanoTime, computeDeadline(nanoTime, parent, tu, timeout), () -> ExecutionContexts.setCurrent(null));
168     } else if (parent == null) {
169       nCtx = CTX_FACTORY.start(name, localCtx, nanoTime, computeDeadline(nanoTime, localCtx, tu, timeout),
170               () -> ExecutionContexts.setCurrent(localCtx));
171     } else {
172       nCtx = CTX_FACTORY.start(name, parent, nanoTime, computeDeadline(nanoTime, parent, tu, timeout),
173               () -> ExecutionContexts.setCurrent(localCtx));
174     }
175     EXEC_CTX.set(nCtx);
176     return nCtx;
177   }

184   public static ExecutionContext start(final String name,
185           @Nullable final ExecutionContext parent, final long startTimeNanos, final long deadlineNanos) {
186     ExecutionContext localCtx = EXEC_CTX.get();
187     ExecutionContext nCtx;
188     if (localCtx == null) {
189       nCtx = CTX_FACTORY.startThreadRoot(name, parent,
190               startTimeNanos, deadlineNanos, () -> ExecutionContexts.setCurrent(null));
191     } else {
192       nCtx = CTX_FACTORY.start(name, parent == null ? localCtx : parent,
193               startTimeNanos, deadlineNanos, () -> ExecutionContexts.setCurrent(localCtx));
194     }
195     EXEC_CTX.set(nCtx);
196     return nCtx;
197   }

199   public static long getContextDeadlineNanos() {
200     ExecutionContext ec = ExecutionContexts.current();
201     if (ec == null) {
202       return TimeSource.nanoTime() + DEFAULT_TIMEOUT_NANOS;
203     } else {
204       return ec.getDeadlineNanos();
205     }
206   }
207 
208   public static long getContextDeadlineNanos(final long currentTime) {
209     ExecutionContext ec = ExecutionContexts.current();
210     if (ec == null) {
211       return currentTime + DEFAULT_TIMEOUT_NANOS;
212     } else {
213       return ec.getDeadlineNanos();
214     }
215   }

217   @Signed
218   public static long getTimeRelativeToDeadline(final TimeUnit unit) {
219     return unit.convert(getContextDeadlineNanos() - TimeSource.nanoTime(), TimeUnit.NANOSECONDS);
220   }

222   @Nonnegative
223   public static long getTimeToDeadline(final TimeUnit unit) throws TimeoutException {
224     long timeRelativeToDeadline = getTimeRelativeToDeadline(unit);
225     if (timeRelativeToDeadline <= 0) {
226       throw new TimeoutException("Deadline exceeded by " + (-timeRelativeToDeadline) + ' ' + unit);
227     }
228     return timeRelativeToDeadline;
229   }

231   @Nonnegative
232   public static long getMillisToDeadline() throws TimeoutException {
233     return getTimeToDeadline(TimeUnit.MILLISECONDS);
234   }

236   @Nonnegative
237   public static int getSecondsToDeadline() throws TimeoutException {
238     long secondsToDeadline = getTimeToDeadline(TimeUnit.SECONDS);
239     if (secondsToDeadline > Integer.MAX_VALUE) {
240       return Integer.MAX_VALUE;
241     } else {
242       return (int) secondsToDeadline;
243     }

246   public static long computeDeadline(@Nullable final ExecutionContext current,
247           final TimeUnit unit, final long timeout) {
248     if (current == null) {
249       return TimeSource.getDeadlineNanos(timeout, unit);
250     }
251     long nanoTime = TimeSource.nanoTime();
252     long ctxDeadlinenanos = current.getDeadlineNanos();
253     long timeoutNanos = unit.toNanos(timeout);
254     return (ctxDeadlinenanos - nanoTime < timeoutNanos) ? ctxDeadlinenanos : nanoTime + timeoutNanos;
255   }
256 
257   public static long computeDeadline(final long startTimeNanos, @Nullable final ExecutionContext current,
258           final TimeUnit unit, final long timeout) {
259     if (current == null) {
260       return TimeSource.getDeadlineNanos(startTimeNanos, timeout, unit);
261     }
262     long ctxDeadlinenanos = current.getDeadlineNanos();
263     long timeoutNanos = unit.toNanos(timeout);
264     return (ctxDeadlinenanos - startTimeNanos < timeoutNanos) ? ctxDeadlinenanos : startTimeNanos + timeoutNanos;
265   }
266 
267   public static TimeoutDeadline computeTimeoutDeadline(@Nullable final ExecutionContext current,
268           final TimeUnit unit, final long timeout) throws TimeoutException {
269     if (current == null) {
270       return TimeoutDeadline.of(unit.toNanos(timeout), TimeSource.getDeadlineNanos(timeout, unit));
271     }
272     long nanoTime = TimeSource.nanoTime();
273     long ctxDeadlinenanos = current.getDeadlineNanos();
274     long timeoutNanos = unit.toNanos(timeout);
275     long contextTimeoutNanos = ctxDeadlinenanos - nanoTime;
276     return (contextTimeoutNanos < timeoutNanos)
277             ? TimeoutDeadline.of(contextTimeoutNanos, ctxDeadlinenanos)
278             : TimeoutDeadline.of(timeoutNanos, nanoTime + timeoutNanos);
279   }

291   public static <T> Callable<T> propagatingCallable(final Callable<T> callable) {
292     ExecutionContext current = current();
293     return current == null ? callable : propagatingCallable(callable, current);
294   }

296   public static <T> Callable<T> propagatingCallable(final Callable<T> callable, final ExecutionContext ctx) {
297     return new PropagatingCallable<T>(callable, ctx);
298   }

300   public static <T> Collection<? extends Callable<T>> propagatingCallables(
301           final Collection<? extends Callable<T>> tasks) {
302     ExecutionContext current = current();
303     return current == null ? tasks : propagatingCallables(tasks, current);
304   }
305 
306   public static <T> Collection<? extends Callable<T>> propagatingCallables(
307           final Collection<? extends Callable<T>> tasks,
308           final ExecutionContext ctx) {
309     return tasks.stream().map(
310             (c) -> new PropagatingCallable<>(c, ctx))
311             .collect(Collectors.toCollection(() -> new ArrayList<>(tasks.size())));
312   }
313 
314   public static <T> Collection<? extends Callable<T>> deadlinedPropagatingCallables(
315           final Collection<? extends Callable<T>> tasks,
316           final ExecutionContext ctx, final long deadlineNanos) {
317     return tasks.stream().map(
318             (c) -> new DeadlinedPropagatingCallable<>(c, ctx, deadlineNanos))
319             .collect(Collectors.toCollection(() -> new ArrayList<>(tasks.size())));
320   }

336   private static final class PropagatingCallable<T> implements Callable<T> {
337 
338     private final Callable<T> task;
339     private final ExecutionContext current;
340 
341     PropagatingCallable(final Callable<T> task, final ExecutionContext current) {
342       this.task = task;
343       this.current = current;
344     }
345 
346     @Override
347     public T call() throws Exception {
348       try (ExecutionContext ctx = ExecutionContexts.start(task.toString(), current)) {
349         return task.call();
350       }
351     }
352   }
353 
354   private static final class DeadlinedPropagatingCallable<T> implements Callable<T> {
355 
356     private final Callable<T> task;
357     private final ExecutionContext current;
358     private final long deadlineNanos;
359 
360     DeadlinedPropagatingCallable(final Callable<T> task, final ExecutionContext current,
361             final long deadlineNanos) {
362       this.task = task;
363       this.current = current;
364       this.deadlineNanos = deadlineNanos;
365     }
366 
367     @Override
368     public T call() throws Exception {
369       try (ExecutionContext ctx = ExecutionContexts.start(task.toString(), current,
370               TimeSource.nanoTime(), deadlineNanos)) {
371         return task.call();
372       }
373     }
374   }
