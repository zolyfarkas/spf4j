50    @Test
51    public void testLogging() {
52      TestLoggers tLog = TestLoggers.sys();
53      try (HandlerRegistration printer = tLog.print("org.spf4j.test", Level.TRACE)) {
54        logTests();
55        logMarkerTests();
56        LogAssert expect = tLog.expect("org.spf4j.test", Level.ERROR,
57                LogMatchers.hasMatchingFormat(Matchers.equalTo("Booo")));
58        LOG.error("Booo", new RuntimeException());
59        expect.assertObservation();
60      }
61    }

63    public static void logTests() {
64      LOG.trace("Hello logger", new RuntimeException());
65      LOG.trace("Hello logger");
66      LOG.trace("Hello logger {}", 1);
67      LOG.trace("Hello logger {} {} {}", 1, 2, 3);
68      LOG.trace("Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
69      LOG.debug("Hello logger", new RuntimeException());
70      LOG.debug("Hello logger");
71      LOG.debug("Hello logger {}", 1);
72      LOG.debug("Hello logger {} {} {}", 1, 2, 3);
73      LOG.debug("Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
74      LOG.info("Hello logger", new RuntimeException());
75      LOG.info("Hello logger");
76      LOG.info("Hello logger {}", 1);
77      LOG.info("Hello logger {} {} {}", 1, 2, 3);
78      LOG.info("Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
79      LOG.warn("Hello logger", new RuntimeException());
80      LOG.warn("Hello logger");
81      LOG.warn("Hello logger {}", 1);
82      LOG.warn("Hello logger {} {} {}", 1, 2, 3);
83      LOG.warn("Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
84      LogAssert expect = TestLoggers.sys().expect("org.spf4j.test", Level.ERROR, 5,
85              LogMatchers.hasMatchingFormat(Matchers.containsString("Hello logger")));
86      LOG.error("Hello logger", new RuntimeException());
87      LOG.error("Hello logger");
88      LOG.error("Hello logger {}", 1);
89      LOG.error("Hello logger {} {} {}", 1, 2, 3);
90      LOG.error("Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
91      expect.assertObservation();
92    }

94    public static void logMarkerTests() {
95      Marker marker = MarkerFactory.getMarker("TEST");
96      Marker marker2 = MarkerFactory.getMarker("TEST2");
97      marker2.add(marker);
98      marker2.add(MarkerFactory.getMarker("TEST3"));
99  
100     LOG.trace(marker, "Hello logger", new RuntimeException());
101     LOG.trace(marker, "Hello logger");
102     LOG.trace(marker, "Hello logger {}", 1);
103     LOG.trace(marker, "Hello logger {} {} {}", 1, 2, 3);
104     LOG.trace(marker, "Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
105     LOG.debug(marker, "Hello logger", new RuntimeException());
106     LOG.debug(marker, "Hello logger");
107     LOG.debug(marker, "Hello logger {}", 1);
108     LOG.debug(marker, "Hello logger {} {} {}", 1, 2, 3);
109     LOG.debug(marker, "Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
110     LOG.info(marker, "Hello logger", new RuntimeException());
111     LOG.info(marker2, "Hello logger");
112     LOG.info(marker2, "Hello logger {}", 1);
113     LOG.info(marker2, "Hello logger {} {} {}", 1, 2, 3);
114     LOG.info(marker2, "Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
115     LOG.warn(marker, "Hello logger", new RuntimeException());
116     LOG.warn(marker, "Hello logger");
117     LOG.warn(marker, "Hello logger {}", 1);
118     LOG.warn(marker, "Hello logger {} {} {}", 1, 2, 3);
119     LOG.warn(marker, "Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
120     LogAssert expect = TestLoggers.sys().expect("org.spf4j.test", Level.ERROR, 5,
121             Matchers.allOf(
122                     LogMatchers.hasMatchingFormat(Matchers.containsString("Hello logger")),
123                     LogMatchers.hasMarker(marker)));
124     LOG.error(marker, "Hello logger", new RuntimeException());
125     LOG.error(marker, "Hello logger");
126     LOG.error(marker, "Hello logger {}", 1);
127     LOG.error(marker, "Hello logger {} {} {}", 1, 2, 3);
128     LOG.error(marker, "Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
129     expect.assertObservation();
130   }

133   @SuppressFBWarnings("LO_TOSTRING_PARAMETER") // this is on purpose configs change over time.
134   public void testIsEnabled() {
135     LOG.debug("Log Config: {}", TestLoggers.sys().toString());
136     Marker marker = MarkerFactory.getMarker("TEST");
137     Assert.assertFalse(LOG.isTraceEnabled());
138     Assert.assertFalse(LOG.isTraceEnabled(marker));
139 
140     //DEBUG is enabled since debug logs are collected in the background...
141     Assert.assertTrue(LOG.isDebugEnabled());
142     Assert.assertTrue(LOG.isDebugEnabled(marker));
143 
144     Assert.assertTrue(LOG.isInfoEnabled());
145     Assert.assertTrue(LOG.isInfoEnabled(marker));
146     Assert.assertTrue(LOG.isWarnEnabled());
147     Assert.assertTrue(LOG.isWarnEnabled(marker));
148     Assert.assertTrue(LOG.isErrorEnabled());
149     Assert.assertTrue(LOG.isErrorEnabled(marker));
150   }

239   @Test
240   public void testUncaught() throws InterruptedException {
241     IllegalStateException ex = new IllegalStateException();
242     Thread thread = new Thread(() -> {
243       throw ex;
244     });
245     TestLoggers config = TestLoggers.sys();
246     AsyncObservationAssert obs = config
247             .expectUncaughtException(
248                     UncaughtExceptionDetail.hasThrowable(Matchers.equalTo(ex)));
249     thread.start();
250     obs.assertObservation(5, TimeUnit.SECONDS);
251   }
