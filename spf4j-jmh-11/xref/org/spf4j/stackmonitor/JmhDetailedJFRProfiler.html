46  /**
47   * A profiler based on Java Flight Recorder.
48   * Inspired a bit by a JMH contrib from Jason Zaugg.
49   * This implementation has been tested to work on a JDK 11 u8.
50   * Relies on FlightRecorderMXBean being available, with the recorder option "destination" implemented.
51   *
52   * @author Zoltan Farkas
53   */
54  @SuppressFBWarnings("PATH_TRAVERSAL_IN")
55  public final class JmhDetailedJFRProfiler implements ExternalProfiler, InternalProfiler {
56  
57  
58    private final File outDir;
59    private final boolean debugNonSafePoints;
60    private final String configName;
61    private final Collection<String> flightRecorderOptions = new ArrayList<>();
62    private PostProcessor postProcessor = null;
63  
64    private volatile boolean measurementStarted = false;
65    private final List<File> generated = new ArrayList<>();
66    private volatile long currentRecording;
67    private File currentRecordingFile;
68    private RecordingGranularity granularity;
69  
70    private final AtomicInteger measurementIterationCounter = new AtomicInteger(0);
71  
72    private final AtomicInteger warmupIterationCounter = new AtomicInteger(0);

75     * required for serviceloader.
76     */
77    public JmhDetailedJFRProfiler() {
78      this.outDir = new File(org.spf4j.base.Runtime.USER_DIR);
79      this.debugNonSafePoints = true;
80      this.configName = "profile";
81    }

84    @SuppressFBWarnings("PATH_TRAVERSAL_IN")
85    public JmhDetailedJFRProfiler(final String initLine) throws
86            ProfilerException {
87      OptionParser parser = new OptionParser();
88  
89      parser.formatHelpWith(new CmdLineUtils.ProfilerOptionFormatter("jfr_detail"));
90  
91      OptionSpec<String> optDir = parser.accepts("dir",
92              "Output directory to write jfr profile files.")
93              .withRequiredArg().ofType(String.class).describedAs("dir");
94  
95      OptionSpec<String> optConfig = parser.accepts("configName",
96              "Name of a predefined Flight Recorder configuration, e.g. profile or default.")
97              .withRequiredArg().ofType(String.class).describedAs("name").defaultsTo("profile");
98  
99      OptionSpec<Boolean> optDebugNonSafePoints
100             = parser.accepts("debugNonSafePoints",
101                     "Gather cpu samples asynchronously, rather than with safepoint bias.")
102                     .withRequiredArg().ofType(Boolean.class).describedAs("bool").defaultsTo(Boolean.TRUE);
103 
104     OptionSpec<Integer> optStackDepth = parser.accepts("stackDepth",
105             "Maximum number of stack frames collected for each event.")
106             .withRequiredArg().ofType(Integer.class).describedAs("frames");
107 
108     OptionSpec<RecordingGranularity> optGranularity = parser.accepts("detailLevel",
109             "The detail granularity of the collected profiles. Can be iteration level,"
110                     + " or warmup|measurement level aggregates")
111             .withRequiredArg().ofType(RecordingGranularity.class)
112             .withValuesConvertedBy(new GranularityValueConverter()).defaultsTo(RecordingGranularity.PER_ITERATION_TYPE)
113             .describedAs("granularity");
114 
115     OptionSpec<String> optPostProcessor = parser.accepts("postProcessor",
116             "The fully qualified name of a class that implements "
117             + PostProcessor.class
118             + ". This must have a public, no-argument constructor.")
119             .withRequiredArg().ofType(String.class).describedAs("fqcn");
120 
121     OptionSet set = CmdLineUtils.parseInitLine(initLine, parser);
122 
123     try {
124       this.granularity = optGranularity.value(set);
125       this.debugNonSafePoints = optDebugNonSafePoints.value(set);
126       this.configName = optConfig.value(set);
127       if (set.has(optStackDepth)) {
128         flightRecorderOptions.add("stackdepth="
129                 + optStackDepth.value(set));
130       }
131       if (!set.has(optDir)) {
132         outDir = new File(org.spf4j.base.Runtime.USER_DIR);
133       } else {
134         outDir = new File(set.valueOf(optDir));
135       }
136       if (set.has(optPostProcessor)) {
137         try {
138           ClassLoader loader
139                   = Thread.currentThread().getContextClassLoader();
140           Class<?> postProcessorClass
141                   = loader.loadClass(optPostProcessor.value(set));
142           postProcessor = (PostProcessor) postProcessorClass.getDeclaredConstructor().newInstance();
143         } catch (ReflectiveOperationException e) {
144           ProfilerException pex =  new ProfilerException(e);
145           pex.addSuppressed(e);
146           throw pex;
147         }
148       }
149     } catch (OptionException e) {
150       ProfilerException profilerException = new ProfilerException(e.getMessage());
151       profilerException.addSuppressed(e);
152       throw profilerException;
153     }
154   }

156   @Override
157   public void beforeIteration(final BenchmarkParams benchmarkParams,
158           final IterationParams iterationParams) {
159            IterationType type = iterationParams.getType();
160         switch (type) {
161           case WARMUP:
162             warmupIterationCounter.incrementAndGet();
163             break;
164           case MEASUREMENT:
165             measurementIterationCounter.incrementAndGet();
166             break;
167           default:
168             throw new UnsupportedOperationException("Unsupported Iteration type: " + type);
169         }
170       if (!measurementStarted) {
171         measurementStarted = true;
172         switch (type) {
173           case WARMUP:
174             currentRecordingFile = new File(outDir, "w_" + benchmarkParams.id()
175                 + '_' + warmupIterationCounter.get() + ".jfr");
176             break;
177           case MEASUREMENT:
178             currentRecordingFile = new File(outDir, "m_" + benchmarkParams.id()
179                 + '_' + measurementIterationCounter.get() + ".jfr");
180             break;
181           default:
182             throw new UnsupportedOperationException("Unsupported Iteration type: " + type);
183         }
184         currentRecording = JFRControler.startRecording(currentRecordingFile, configName);
185       }
186   }
187 
188   private boolean isEndOfWarmupOrMeasuremenets(final IterationParams iterationParams) {
189           IterationType type = iterationParams.getType();
190       switch (type) {
191         case WARMUP:
192           return warmupIterationCounter.get() == iterationParams.getCount();
193         case MEASUREMENT:
194           return measurementIterationCounter.get() == iterationParams.getCount();
195         default:
196           throw new UnsupportedOperationException("Unsupported Iteration type: " + type);
197       }
198   }
199 
200   @Override
201   public Collection<? extends Result>
202           afterIteration(final BenchmarkParams benchmarkParams, final IterationParams iterationParams,
203                   final IterationResult iterationResult) {
204       if (measurementStarted
205               && (granularity == RecordingGranularity.PER_ITERATION
206               || isEndOfWarmupOrMeasuremenets(iterationParams))) {
207         JFRControler.closeRecording(currentRecording);
208         currentRecording = -1L;
209         measurementStarted = false;
210         generated.add(currentRecordingFile);
211         if (postProcessor != null) {
212           generated.addAll(postProcessor.postProcess(benchmarkParams, currentRecordingFile));
213         }
214         return Collections.singletonList(result());
215       }
216 
217     return Collections.emptyList();
218   }
219 
220   private TextResult result() {
221     StringWriter output = new StringWriter();
222     try (PrintWriter pw = new PrintWriter(output)) {
223       pw.println("JFR profiler results:");
224       for (File file : generated) {
225         pw.print("  ");
226         pw.println(file.getPath());
227       }
228       pw.flush();
229     }
230     return new TextResult(output.toString(), "jfr");
231   }

238   @Override
239   public Collection<String> addJVMOptions(final BenchmarkParams params) {
240     List<String> args = new ArrayList<>();
241     if (debugNonSafePoints) {
242       args.add("-XX:+UnlockDiagnosticVMOptions");
243       args.add("-XX:+DebugNonSafepoints");
244     }
245 
246     if (!flightRecorderOptions.isEmpty()) {
247       args.add("-XX:FlightRecorderOptions="
248               + Utils.join(flightRecorderOptions, ","));
249     }
250     // Unnecessary / deprecated for removal on JDKs 13.
251     // TODO Could make this conditional on   majorVersion(params.getJdkVersion()) < 13
252     args.add("-XX:+IgnoreUnrecognizedVMOptions");
253     args.add("-XX:+FlightRecorder");
254 
255     return args;
256   }

301   private static class GranularityValueConverter implements ValueConverter<RecordingGranularity> {
302 
303     @Override
304     public RecordingGranularity convert(final String arg) {
305       return RecordingGranularity.valueOf(arg);
306     }
307 
308     @Override
309     public Class<RecordingGranularity> valueType() {
310       return RecordingGranularity.class;
311     }

314     public String valuePattern() {
315       return "PER_ITERATION_TYPE|PER_ITERATION";
