7   package org.spf4j.zel.vm;
8   
9   import com.google.common.base.Function;
10  import java.util.Map;
11  import org.spf4j.zel.instr.Instruction;
12  import org.spf4j.zel.instr.LODAX;
13  import org.spf4j.zel.instr.LODAXF;
14  import org.spf4j.zel.instr.LODX;
15  import org.spf4j.zel.instr.LODXF;
16  
17  /**
18   *
19   * @author zoly
20   */
21  public final class RefOptimizer implements Function<Program, Program> {
22  
23      private RefOptimizer() { }
24  
25      public static final Function<Program, Program> INSTANCE = new RefOptimizer();
26  
27      @Override
28      public Program apply(final Program input) {
29          if (input == null) {
30              throw new IllegalArgumentException("Input cannot be null for " + this);
31          }
32          Instruction [] instructions = input.getInstructions().clone();
33          Map<String, Integer> lsym = input.getLocalSymbolTable();
34          Map<String, Integer> gsym = input.getGlobalSymbolTable();
35  
36          for (int i = 0; i < instructions.length; i++) {
37              Instruction instr = instructions[i];
38              if (LODX.class.equals(instr.getClass())) {
39                  String symbol = ((LODX) instr).getSymbol();
40                  Address addr = getAddress(lsym, symbol, gsym);
41                  instructions[i] = new LODXF(addr);
42              } else if (LODAX.class.equals(instr.getClass())) {
43                  String symbol = ((LODAX) instr).getSymbol();
44                  Address addr = getAddress(lsym, symbol, gsym);
45                  instructions[i] = new LODAXF(addr);
46              }
47          }
48  
49          try {
50              return new Program(input.getName(), input.getGlobalSymbolTable(),
51                      input.getGlobalMem(), input.getLocalSymbolTable(),
52                      instructions, input.getDebug(), input.getSource(),
53                      input.getType(), input.getExecType(),
54                      input.hasDeterministicFunctions());
55          } catch (CompileException ex) {
56              throw new RuntimeException(ex);
57          }
58      }
59  
60      private static Address getAddress(final Map<String, Integer> lsym,
61              final String symbol, final Map<String, Integer> gsym) {
62          Address addr;
63          Integer idx = lsym.get(symbol);
64          if (idx == null) {
65              idx = gsym.get(symbol);
66              addr = new Address(idx, Address.Scope.GLOBAL);
67          } else {
68              addr = new Address(idx, Address.Scope.LOCAL);
69          }
70          return addr;
71      }

73  }
