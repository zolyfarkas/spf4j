117         this.debug = debug;
118         this.source = source;

156     private static Map<String, Integer> buildLocalSymTable(final Instruction [] instructions,
157             final String[] parameterNames1,
158             final int length, final Map<String, Integer> globalTable,
159             final Map<String, Integer> addTo) throws CompileException {
160         final int addToSize = addTo.size();
161         Map<String, Integer> symbolTable = new HashMap<>(addToSize + parameterNames1.length);
162         symbolTable.putAll(addTo);
163         // allocate program params
164         int i = addToSize;
165         for (String param : parameterNames1) {
166             Integer existing = symbolTable.put(param, i++);
167             if (existing != null) {
168                 throw new CompileException("Duplicate parameter defined: " + param);
169             }
170         }
171         // allocate variables used in Program
172         for (int j = 0; j < length; j++) {
173             Instruction code = instructions[j];
174             if (code instanceof LValRef) {
175                 String ref = ((LValRef) code).getSymbol();
176                 Integer idxr = symbolTable.get(ref);
177                 if (idxr == null) {
178                     idxr = globalTable.get(ref);
179                     if (idxr == null) {
180                         idxr = i++;
181                         symbolTable.put(ref, idxr);
182                     }
183                 }
184             }
185         }
186         return symbolTable;
187     }

253     @Nonnull
254     public static Program compile(@Nonnull final String zExpr, @Nonnull final String... varNames)
255             throws CompileException {
256 
257         ParsingContext cc = new CompileContext(ZEL_GLOBAL_FUNC.copy());
258         final String srcId = ZelFrame.newSource(zExpr);
259         try {
260             ZCompiler.compile(srcId, zExpr, cc);
261         } catch (TokenMgrError | ParseException err) {
262             throw new CompileException(err);
263         }
264         return RefOptimizer.INSTANCE.apply(cc.getProgramBuilder().toProgram("anon@root", srcId, varNames));
265     }

268     static Program compile(@Nonnull final String zExpr,
269             final Map<String, Integer> localTable,
270             final Object [] globalMem,
271             final Map<String, Integer> globalTable,
272             @Nonnull final String... varNames)
273             throws CompileException {
274 
275         ParsingContext cc = new CompileContext(new MemoryBuilder(
276                 new ArrayList<>(Arrays.asList(globalMem)), globalTable));
277         final String srcId = ZelFrame.newSource(zExpr);
278         try {
279             ZCompiler.compile(srcId, zExpr, cc);
280         } catch (TokenMgrError | ParseException err) {
281             throw new CompileException(err);
282         }
283         return cc.getProgramBuilder().toProgram("anon@root", srcId, varNames, localTable);
284     }

347     public static Object executeSyncOrAsync(@Nonnull final ExecutionContext ectx)
348             throws ExecutionException, InterruptedException {
349         final VMExecutor.Suspendable<Object> execution = ectx.getCallable();
350         if (ectx.execService != null && ectx.code.getExecType() == ExecutionType.ASYNC) {
351             if (ectx.isChildContext()) {
352                 return ectx.execService.submitInternal(VMExecutor.synchronize(execution));
353             } else {
354                 return ectx.execService.submit(VMExecutor.synchronize(execution));
355             }
356         } else {
357             try {
358                 return execution.call();
359             } catch (SuspendedException ex) {
360                 throw new RuntimeException(ex);
361             }
362         }
363     }

394     public static Object execute(@Nonnull final ExecutionContext ectx)
395             throws ExecutionException, InterruptedException {
396         Object result = executeSyncOrAsync(ectx);
397         if (result instanceof Future) {
398             return ((Future<Object>) result).get();
399         } else {
400             return result;

458     /**
459      * Output Core, in hierarchical tab indented mode
460      *
461      * @param name
462      * @param mem
463      * @param indent
464      * @param maxIndent
465      * @return
466      */
467     @SuppressWarnings("unchecked")
468     @CheckReturnValue
469     public static String dumpCore(final String name, final Object mem, final int indent, final int maxIndent) {
470         if (mem == null) {
471             return "";
472         }
473         if (maxIndent > 0 && indent > maxIndent) {
474             return "";
475         }
476         StringBuilder result = new StringBuilder();
477         if (mem instanceof java.util.Map) {
478             result.append(strIndent(indent)).append(name).append('\n');
479             for (Map.Entry<Object, Object> elem : ((Map<Object, Object>) mem).entrySet()) {
480                 result.append(dumpCore(elem.getKey().toString(), elem.getValue(), indent + 1, maxIndent));
481             }
482         } else {
483             result.append(strIndent(indent)).append(name).append('=').append(mem).append('\n');
484         }
485         return result.toString();
486     }

488     /**
489      * *
490      * This allows to run ZEL in an interactive mode
491      *
492      * @param args
493      */
494     public static void main(final String[] args)
495             throws IOException, ZExecutionException, InterruptedException {
496         System.out.println("ZEL Shell");
497         boolean terminated = false;
498         Map<String, Integer> localSymTable = Collections.EMPTY_MAP;
499         Pair<Object[], Map<String, Integer>> gmemPair = ZEL_GLOBAL_FUNC.build();
500         Map<String, Integer> globalSymTable = gmemPair.getSecond();
501         Object [] mem = new Object [] {};
502         Object [] gmem = gmemPair.getFirst();
503         ResultCache resCache = new SimpleResultCache();
504         InputStreamReader inp = new InputStreamReader(System.in, Charsets.UTF_8);
505         BufferedReader br = new BufferedReader(inp);
506         while (!terminated) {
507             System.out.print("zel>");
508             String line = br.readLine();
509             if (line != null) {
510                 if ("QUIT".equalsIgnoreCase(line)) {
511                     terminated = true;
512                 } else {
513                     try {
514                         final Program prog = Program.compile(line, localSymTable, gmem, globalSymTable);
515                         localSymTable = prog.getLocalSymbolTable();
516                         globalSymTable = prog.getGlobalSymbolTable();
517                         gmem = prog.getGlobalMem();
518                         Pair<Object, ExecutionContext> res = prog.executeX(
519                                 VMExecutor.Lazy.DEFAULT, System.in, System.out, System.err, resCache, mem);
520                         System.out.println("result>" + res.getFirst());
521                         final ExecutionContext execCtx = res.getSecond();
522                         mem = execCtx.mem;
523                         resCache = execCtx.resultCache;
524                     } catch (CompileException ex) {
525                         System.out.println("Syntax Error: " + Throwables.getStackTraceAsString(ex));
526                     } catch (ExecutionException ex) {
527                         System.out.println("Execution Error: " + Throwables.getStackTraceAsString(ex));
528                     }
529                 }
530             }
531         }
532     }

534     public String toAssemblyString() {
535         StringBuilder result = new StringBuilder();
536         result.append("Program: \n");
537         for (int i = 0; i < instructions.length; i++) {
538             Object obj = instructions[i];
539             result.append(Strings.padEnd(Integer.toString(i), 8, ' '));
540             result.append(':');
541             result.append(obj);
542             result.append('\n');
543         }
544         result.append("execType = ").append(this.execType).append('\n');
545         result.append("type = ").append(this.type).append('\n');
546         return result.toString();
547     }

562     public boolean contains(final Class<? extends Instruction> instr) {
563         Boolean res = itterate(new HasClass(instr));
564         if (res == null) {
565             return false;
566         }
567         return res;
568     }

570     @Nullable
571     public <T> T itterate(final Function<Object, T> func) {
572         for (Instruction code : instructions) {
573             T res = func.apply(code);
574             if (res != null) {
575                 return res;
576             }
577             for (Object param : code.getParameters()) {
578                 res = func.apply(param);
579                 if (res != null) {
580                    return res;
581                 }
582                 if (param instanceof Program) {
583                     res = ((Program) param).itterate(func);
584                 }
585                 if (res != null) {
586                     return res;
587                 }
588             }
589         }
590         return null;
591     }

597     public static final class HasClass implements Function<Object, Boolean> {
598 
599         private final Class<? extends Instruction> instr;
600 
601         public HasClass(final Class<? extends Instruction> instr) {
602             this.instr = instr;
603         }
604 
605         @Override
606         @SuppressFBWarnings("TBP_TRISTATE_BOOLEAN_PATTERN")
607         public Boolean apply(@Nonnull final Object input) {
608             if (input.getClass() == instr) {
609                 return Boolean.TRUE;
610             }
611             return null;
612         }
