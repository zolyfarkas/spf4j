102     public BigInteger fibB(final int i) {
103         if (i <= 1) {
104             return BigInteger.valueOf(i);
105         } else {
106             return fibB(i - 1).add(fibB(i - 2));
107         }
108     }
109 
110     public long fib(final long i) {
111         if (i <= 1) {
112             return i;
113         } else {
114             return fib(i - 1) + fib(i - 2);
115         }
116     }
117 
118     public BigInteger fibBNr(final int i) {
119         if (i <= 1) {
120             return BigInteger.valueOf(i);
121         } else {
122             BigInteger v1 = BigInteger.ZERO;
123             BigInteger v2 = BigInteger.ONE;
124             BigInteger tmp;
125             for (int j = 2; j <= i; j++) {
126                 tmp = v2;
127                 v2 = v1.add(v2);
128                 v1 = tmp;
129             }
130             return v2;
131         }
132     }

134     @Test
135     public void testFibPerformance() throws CompileException, ExecutionException, InterruptedException {
136         String fib = "func det fib (x) {fib(x-1) + fib(x-2)}; fib(0) = 0; fib(1) = 1; fib(x)";
137         Program fibZel = Program.compile(fib, "x");
138         long startTime = System.currentTimeMillis();
139         Number zelResult = (Number) fibZel.execute(40);
140         long intTime = System.currentTimeMillis();
141         long javaResult = fib(40);
142         long stopTime = System.currentTimeMillis();
143         Assert.assertEquals(zelResult.longValue(), javaResult);
144         System.out.println("zel exec " + (intTime - startTime));
145         System.out.println("java exec " + (stopTime - intTime));
146 
147         startTime = System.currentTimeMillis();
148         BigInteger zelResult2 = (BigInteger) fibZel.execute(10000);
149         intTime = System.currentTimeMillis();
150         BigInteger javaResult2 = fibBNr(10000);
151         stopTime = System.currentTimeMillis();
152         System.out.println("fib(10000) = " + zelResult2);
153         System.out.println("fib(10000) = " + javaResult2);
154         Assert.assertEquals(zelResult2, javaResult2);
155         System.out.println("zel exec " + (intTime - startTime));
156         System.out.println("java exec " + (stopTime - intTime));
157     }

190     @Test
191     @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
192     public void testCond() throws CompileException, ExecutionException, InterruptedException {
193         Boolean result = (Boolean) Program.compile("x == 1", "x").execute(1);
194         Assert.assertTrue(result);
195         result = (Boolean) Program.compile("x != 1", "x").execute(1);
196         Assert.assertFalse(result);
197         result = (Boolean) Program.compile("x < 1", "x").execute(1);
198         Assert.assertFalse(result);
199         result = (Boolean) Program.compile("x >= 1", "x").execute(1);
200         Assert.assertTrue(result);
201         result = (Boolean) Program.compile("x <= 1", "x").execute(1);
202         Assert.assertTrue(result);
203         result = (Boolean) Program.compile("x > 1", "x").execute(1);
204         Assert.assertFalse(result);
205         Number n = (Number) Program.compile("min(3, 1, 8)").execute();
206         Assert.assertEquals(1, n.intValue());
207         n = (Number) Program.compile("max(3, 1, 8, 10)").execute();
208         Assert.assertEquals(10, n.intValue());
209         n = (Number) Program.compile("sqrt(4)").execute();
210         Assert.assertEquals(2, n.intValue());
211         n = (Number) Program.compile("log(E)").execute();
212         Assert.assertEquals(1, n.intValue());
213
