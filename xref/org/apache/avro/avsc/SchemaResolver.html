16  package org.apache.avro.avsc;
17  
18  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
19  import java.util.ArrayList;
20  import java.util.Collection;
21  import java.util.Collections;
22  import java.util.IdentityHashMap;
23  import java.util.List;
24  import java.util.Map;
25  import java.util.concurrent.atomic.AtomicInteger;
26  
27  import org.apache.avro.Schema;
28  
29  /**
30   * Utility class to resolve schemas that are unavailable at the time they are referenced in the IDL.
31   */
32  public final class SchemaResolver {
33  
34    private SchemaResolver() {
35    }
36  
37    private static final String UR_SCHEMA_ATTR = "org.apache.avro.compiler.idl.unresolved.name";
38  
39    private static final String UR_SCHEMA_NAME = "UnresolvedSchema";
40  
41    private static final String UR_SCHEMA_NS = "org.apache.avro.compiler";
42  
43    private static final AtomicInteger COUNTER = new AtomicInteger();
44  
45    /**
46     * Create a schema to represent a "unresolved" schema.
47     * (used to represent a schema where the definition is not known at the time)
48     * This concept might be generalizable...
49     *
50     * @param name
51     * @return
52     */
53    public static Schema unresolvedSchema(final String name) {
54      Schema schema = Schema.createRecord(UR_SCHEMA_NAME + '_' + COUNTER.getAndIncrement(), "unresolved schema",
55          UR_SCHEMA_NS, false, Collections.EMPTY_LIST);
56      schema.addProp(UR_SCHEMA_ATTR, name);
57      return schema;
58    }
59  
60    /**
61     * Is this a unresolved schema.
62     *
63     * @param schema
64     * @return
65     */
66    static boolean isUnresolvedSchema(final Schema schema) {
67      return (schema.getType() == Schema.Type.RECORD && schema.getProp(UR_SCHEMA_ATTR) != null
68          && schema.getName().startsWith(UR_SCHEMA_NAME)
69          && UR_SCHEMA_NS.equals(schema.getNamespace()));
70    }
71  
72    /**
73     * get the unresolved schema name.
74     *
75     * @param schema
76     * @return
77     */
78    @SuppressFBWarnings("AI_ANNOTATION_ISSUES_NEEDS_NULLABLE")
79    static String getUnresolvedSchemaName(final Schema schema) {
80      if (!isUnresolvedSchema(schema)) {
81        throw new IllegalArgumentException("Not a unresolved schema: " + schema);
82      }
83      return schema.getProp(UR_SCHEMA_ATTR);
84    }
85  
86    /**
87     * Will clone the provided protocol while resolving all unreferenced schemas
88     *
89     * @param protocol
90     * @return
91     */
92    public static List<Schema> resolve(final Map<String, Schema> schemas, final boolean allowUndefinedLogicalTypes) {
93      final Collection<Schema> types = schemas.values();
94      // replace unresolved schemas.
95      List<Schema> newSchemas = new ArrayList(types.size());
96      IdentityHashMap<Schema, Schema> replacements = new IdentityHashMap<Schema, Schema>();
97      for (Schema schema : types) {
98        newSchemas.add(Schemas.visit(schema, new ResolvingVisitor(schema, replacements,
99                schemas::get, allowUndefinedLogicalTypes)));
100     }
101     return newSchemas;
102   }
103 
104 }
