81     * Removes the first System property.
82     * @param pname the name of the system property to remove.
83     * @return the value of the removed system property. or null if there is no such property.
84     */
85    @Nullable
86    public String removeSystemProperty(final String pname) {
87      Iterator<String> itr = arguments.iterator();
88      itr.next(); // skip command.
89      int nl = pname.length();
90      while (itr.hasNext()) {
91        String s = itr.next();
92        if (s.startsWith("-D") && s.regionMatches(2, pname, 0, nl)) {
93          int l = nl + 2;
94          if (s.length() == l) {
95            itr.remove();
96            return "";
97          } else if (s.charAt(l) == '=') {
98            itr.remove();
99            return s.substring(l + 1);
100         }
101       }
102     }
103     return null;
104   }

125   @Nullable
126   public void createOrUpdateSystemProperty(final String pname, final Function<String, String> replacer) {
127     ListIterator<String> itr = arguments.listIterator();
128     itr.next(); // skip command.
129     int nl = pname.length();
130     while (itr.hasNext()) {
131       String s = itr.next();
132       if (s.startsWith("-D") && s.regionMatches(2, pname, 0, nl)) {
133         int l = nl + 2;
134         if (s.length() == l) {
135           itr.set("-D" + pname + '=' + replacer.apply(""));
136           return;
137         } else if (s.charAt(l) == '=') {
138           itr.set("-D" + pname + '=' + replacer.apply(s.substring(l + 1)));
139           return;
140         }
141       }
142     }
143     arguments.add(1, "-D" + pname + '=' + replacer.apply(null));
144   }

275   public static JVMArguments of(final int pid) throws IOException {
276     String os = Runtime.OS_NAME;
277     switch (os) {
278       case "Linux":
279         return ofLinux(pid);
280       case "SunOS":
281         return ofSolaris(pid);
282       case "Mac OS X":
283         return ofMac(pid);
284       case "FreeBSD":
285         return ofFreeBSD(pid);
286       default:
287         throw new UnsupportedOperationException("Unsupported Operating System " + os);
288     }
289   }

305   private static JVMArguments ofSolaris(final int ppid) throws IOException {
306     // /proc shows different contents based on the caller's memory model, so we need to know if we are 32 or 64.
307     // 32 JVMs are the norm, so err on the 32bit side.
308     boolean areWe64 = "64".equals(System.getProperty("sun.arch.data.model"));
309     int pid = resolvePID(ppid);
310     try (RandomAccessFile psinfo = new RandomAccessFile(new File("/proc/" + pid + "/psinfo"), "r")) {
311       // see http://cvs.opensolaris.org/source/xref/onnv/onnv-gate/usr/src/uts/common/sys/procfs.h
312       //typedef struct psinfo {
313       // int pr_flag; /* process flags */
314       // int pr_nlwp; /* number of lwps in the process */
315       // pid_t pr_pid; /* process id */
316       // pid_t pr_ppid; /* process id of parent */
317       // pid_t pr_pgid; /* process id of process group leader */
318       // pid_t pr_sid; /* session id */
319       // uid_t pr_uid; /* real user id */
320       // uid_t pr_euid; /* effective user id */
321       // gid_t pr_gid; /* real group id */
322       // gid_t pr_egid; /* effective group id */
323       // uintptr_t pr_addr; /* address of process */
324       // size_t pr_size; /* size of process image in Kbytes */
325       // size_t pr_rssize; /* resident set size in Kbytes */
326       // dev_t pr_ttydev; /* controlling tty device (or PRNODEV) */
327       // ushort_t pr_pctcpu; /* % of recent cpu time used by all lwps */
328       // ushort_t pr_pctmem; /* % of system memory used by process */
329       // timestruc_t pr_start; /* process start time, from the epoch */
330       // timestruc_t pr_time; /* cpu time for this process */
331       // timestruc_t pr_ctime; /* cpu time for reaped children */
332       // char pr_fname[PRFNSZ]; /* name of exec'ed file */
333       // char pr_psargs[PRARGSZ]; /* initial characters of arg list */
334       // int pr_wstat; /* if zombie, the wait() status */
335       // int pr_argc; /* initial argument count */
336       // uintptr_t pr_argv; /* address of initial argument vector */
337       // uintptr_t pr_envp; /* address of initial environment vector */
338       // char pr_dmodel; /* data model of the process */
339       // lwpsinfo_t pr_lwp; /* information for representative lwp */
340       //} psinfo_t;
341 
342       // see http://cvs.opensolaris.org/source/xref/onnv/onnv-gate/usr/src/uts/common/sys/types.h
343       // for the size of the various datatype.
344       // see http://cvs.opensolaris.org/source/xref/onnv/onnv-gate/usr/src/cmd/ptools/pargs/pargs.c
345       // for how to read this information
346       psinfo.seek(8);
347       if (adjust(psinfo.readInt()) != pid) {
348         throw new IOException("psinfo PID mismatch: " + pid + ", " + psinfo);   // sanity check
349       }
350       /* The following program computes the offset:
351                     #include <stdio.h>
352                     #include <sys/procfs.h>
353                     int main() {
354                       printf("psinfo_t = %d\n", sizeof(psinfo_t));
355                       psinfo_t *x;
356                       x = 0;
357                       printf("%x\n", &(x->pr_argc));
358                     }
359        */
360 
361       psinfo.seek(areWe64 ? 0xEC : 0xBC);  // now jump to pr_argc
362       int argc = adjust(psinfo.readInt());
363       long argp = areWe64 ? adjust(psinfo.readLong()) : to64(adjust(psinfo.readInt()));
364       File asFile = new File("/proc/" + pid + "/as");
365       if (areWe64) {
366         // 32bit and 64bit basically does the same thing, but because the stream position
367         // is computed with signed long, doing 64bit seek to a position bigger than Long.MAX_VALUE
368         // requres some real hacking. Hence two different code path.
369         // (RandomAccessFile uses Java long for offset, so it just can't get to anywhere beyond Long.MAX_VALUE)
370         FILE fp = CLibrary.INSTANCE.fopen(asFile.getPath(), "r");
371         try {
372           JVMArguments args = new JVMArguments(16);
373           Memory m = new Memory(8);
374           for (int n = 0; n < argc; n++) {
375             // read a pointer to one entry
376             seek64(fp, argp + ((long) n) * 8);
377             m.setLong(0, 0); // just to make sure failed read won't result in bogus value
378             CLibrary.INSTANCE.fread(m, 1, 8, fp);
379             long p = m.getLong(0);
380             args.add(readLine(fp, p));
381           }
382           return args;
383         } finally {
384           CLibrary.INSTANCE.fclose(fp);
385         }
386       } else {
387         try (RandomAccessFile as = new RandomAccessFile(asFile, "r")) {
388           JVMArguments args = new JVMArguments(16);
389           for (int n = 0; n < argc; n++) {
390             // read a pointer to one entry
391             as.seek(argp + n * 4);
392             int p = adjust(as.readInt());
393 
394             args.add(readLine(as, p));
395           }
396           return args;
397         }
398       }
399     }
400   }

402   /**
403    * Seek to the specified position. This method handles offset bigger than {@link Long#MAX_VALUE} correctly.
404    *
405    * @param upos This value is interpreted as unsigned 64bit integer (even though it's typed 'long')
406    */
407   private static void seek64(final FILE fp, final long pupos) {
408     long upos = pupos;
409     CLibrary.INSTANCE.fseek(fp, 0, 0); // start at the beginning
410     while (upos < 0) {
411       long chunk = Long.MAX_VALUE;
412       upos -= chunk;
413       CLibrary.INSTANCE.fseek(fp, chunk, 1);
414     }
415     CLibrary.INSTANCE.fseek(fp, upos, 1);
416   }

418   /**
419    * {@link DataInputStream} reads a value in big-endian, so convert it to the correct value on little-endian systems.
420    */
421   private static int adjust(final int i) {
422     if (Runtime.IS_LITTLE_ENDIAN) {
423       return (i << 24) | ((i << 8) & 0x00FF0000) | ((i >> 8) & 0x0000FF00) | (i >>> 24);
424     } else {
425       return i;
426     }
427   }
428 
429   private static long adjust(final long i) {
430     if (Runtime.IS_LITTLE_ENDIAN) {
431       return (i << 56)
432               | ((i << 40) & 0x00FF000000000000L)
433               | ((i << 24) & 0x0000FF0000000000L)
434               | ((i << 8) & 0x000000FF00000000L)
435               | ((i >> 8) & 0x00000000FF000000L)
436               | ((i >> 24) & 0x0000000000FF0000L)
437               | ((i >> 40) & 0x000000000000FF00L)
438               | (i >> 56);
439     } else {
440       return i;
441     }
442   }

445    * int to long conversion with zero-padding.
446    */
447   private static long to64(final int i) {
448     return i & 0xFFFFFFFFL;

451   private static String readLine(final RandomAccessFile as, final int p) throws IOException {
452     as.seek(to64(p));
453     ByteArrayOutputStream buf = new ByteArrayOutputStream();
454     int ch;
455     while ((ch = as.read()) > 0) {
456       buf.write(ch);
457     }
458     return buf.toString();
459   }

461   private static String readLine(final FILE as, final long p) {
462     seek64(as, p);
463     Memory m = new Memory(1);
464     ByteArrayOutputStream buf = new ByteArrayOutputStream();
465     while (true) {
466       if (CLibrary.INSTANCE.fread(m, 1, 1, as) == 0) {
467         break;
468       }
469       byte b = m.getByte(0);
470       if (b == 0) {
471         break;
472       }
473       buf.write(b);
474     }
475     return buf.toString();
476   }

478   /**
479    * Mac support
480    *
481    * See http://developer.apple.com/qa/qa2001/qa1123.html http://www.osxfaq.com/man/3/kvm_getprocs.ws
482    * http://matburt.net/?p=16 (libkvm is removed from OSX) where is kinfo_proc?
483    * http://lists.apple.com/archives/xcode-users/2008/Mar/msg00781.html
484    *
485    * This code uses sysctl to get the arg/env list:
486    * http://www.psychofx.com/psi/trac/browser/psi/trunk/src/arch/macosx/macosx_process.c which came from
487    * http://www.opensource.apple.com/darwinsource/10.4.2/top-15/libtop.c
488    *
489    * sysctl is defined in libc.
490    *
491    * PS source code for Mac: http://www.opensource.apple.com/darwinsource/10.4.1/adv_cmds-79.1/ps.tproj/
492    */
493   @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
494   private static JVMArguments ofMac(final int pid) {
495     // local constants
496     final int ctlKern = 1;
497     final int kernArgMax = 8;
498     final int kernProcArgs2 = 49;
499     final int sizeOfInt = Native.getNativeSize(int.class);
500     IntByReference ibf = new IntByReference();
501 
502     IntByReference argmaxRef = new IntByReference(0);
503     IntByReference size = new IntByReference(sizeOfInt);
504 
505     // for some reason, I was never able to get sysctlbyname work.
506 //        if(LIBC.sysctlbyname("kern.argmax", argmaxRef.getPointer(), size, NULL, ibf)!=0)
507     if (CLibrary.INSTANCE.sysctl(new int[]{ctlKern, kernArgMax}, 2, argmaxRef.getPointer(), size, NULL, ibf) != 0) {
508       throw new UnsupportedOperationException("Failed to get kernl.argmax: "
509               + CLibrary.INSTANCE.strerror(Native.getLastError()));
510     }
511 
512     int argmax = argmaxRef.getValue();
513     StringArrayMemory m = new StringArrayMemory(argmax, sizeOfInt);
514     size.setValue(argmax);
515     if (CLibrary.INSTANCE.sysctl(new int[]{ctlKern, kernProcArgs2, resolvePID(pid)}, 3, m, size, NULL, ibf) != 0) {
516       throw new UnsupportedOperationException("Failed to obtain ken.procargs2: "
517               + CLibrary.INSTANCE.strerror(Native.getLastError()));
518     }
519     /*
520          * Make a sysctl() call to get the raw argument space of the
521          * process.  The layout is documented in start.s, which is part
522          * of the Csu project.  In summary, it looks like:
523          *
524          * /---------------\ 0x00000000
525          * :               :
526          * :               :
527          * |---------------|
528          * | argc          |
529          * |---------------|
530          * | arg[0]        |
531          * |---------------|
532          * :               :
533          * :               :
534          * |---------------|
535          * | arg[argc - 1] |
536          * |---------------|
537          * | 0             |
538          * |---------------|
539          * | env[0]        |
540          * |---------------|
541          * :               :
542          * :               :
543          * |---------------|
544          * | env[n]        |
545          * |---------------|
546          * | 0             |
547          * |---------------| <-- Beginning of data returned by sysctl()
548          * | exec_path     |     is here.
549          * |:::::::::::::::|
550          * |               |
551          * | String area.  |
552          * |               |
553          * |---------------| <-- Top of stack.
554          * :               :
555          * :               :
556          * \---------------/ 0xffffffff
557      */
558     int nargs = m.readInt();
559     JVMArguments args = new JVMArguments(nargs);
560     m.readString(); // exec path
561     for (int i = 0; i < nargs; i++) {
562       m.skip0();
563       args.add(m.readString());
564     }
565     return args;
566   }

568   @SuppressFBWarnings("EQ_DOESNT_OVERRIDE_EQUALS")
569   private static final class StringArrayMemory extends Memory {
570 
571     private long offset = 0;
572     private final int sizeOfInt;
573 
574     StringArrayMemory(final long l, final int sizeOfInt) {
575       super(l);
576       this.sizeOfInt = sizeOfInt;
577     }
578 
579     private int readInt() {
580       int r = getInt(offset);
581       offset += sizeOfInt;
582       return r;
583     }
584 
585     private String readString() {
586       ByteArrayOutputStream baos = new ByteArrayOutputStream();
587       byte ch;
588       while ((ch = getByte(offset++)) != '\0') {
589         baos.write(ch);
590       }
591       return baos.toString();
592     }
593 
594     void skip0() {
595       // skip trailing '\0's
596       while (getByte(offset) == '\0') {
597         offset++;
598       }
599     }
600   }

602   private static JVMArguments ofFreeBSD(final int pid) {
603     // taken from sys/sysctl.h
604     final int ctlKern = 1;
605     final int kernArgMax = 8;
606     final int kernProc = 14;
607     final int kernProcArgs = 7;
608 
609     IntByReference ibr = new IntByReference();
610     IntByReference sysctlArgMax = new IntByReference();
611     IntByReference size = new IntByReference();
612 
613     size.setValue(4);
614     if (CLibrary.INSTANCE.sysctl(new int[]{ctlKern, kernArgMax},
615             2, sysctlArgMax.getPointer(), size, NULL, ibr) != 0) {
616       throw new UnsupportedOperationException("Failed to sysctl kern.argmax");
617     }
618 
619     int argmax = sysctlArgMax.getValue();
620     Memory m = new Memory(argmax);
621     size.setValue(argmax);
622 
623     if (CLibrary.INSTANCE.sysctl(new int[]{ctlKern, kernProc, kernProcArgs, resolvePID(pid)},
624             4, m, size, NULL, ibr) != 0) {
625       throw new UnsupportedOperationException();
626     }
627 
628     ByteArrayOutputStream baos = new ByteArrayOutputStream();
629     ArrayList<String> lArgs = new ArrayList<String>();
630     byte ch;
631     int offset = 0;
632     while (offset < size.getValue()) {
633       while ((ch = m.getByte(offset++)) != '\0') {
634         baos.write(ch);
635       }
636       lArgs.add(baos.toString());
637       baos.reset();
638     }
639 
640     return new JVMArguments(lArgs);
641   }

653   @Override
654   public boolean equals(final Object obj) {
655     if (this == obj) {
656       return true;
657     }
658     if (obj == null) {
659       return false;
660     }
661     if (getClass() != obj.getClass()) {
662       return false;
663     }
664     final JVMArguments other = (JVMArguments) obj;
665     return Objects.equals(this.arguments, other.arguments);
666   }

668   public String[] toArray() {
669     return arguments.toArray(new String[arguments.size()]);
