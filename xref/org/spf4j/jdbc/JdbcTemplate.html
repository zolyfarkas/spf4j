1   package org.spf4j.jdbc;
2   
3   import com.google.common.annotations.Beta;
4   import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
5   import java.sql.Connection;
6   import java.sql.SQLException;
7   import java.util.concurrent.TimeUnit;
8   import javax.sql.DataSource;
9   import org.spf4j.base.CallablesNano;
10  import org.spf4j.base.HandlerNano;
11  
12  /**
13   * A very simple JdbTemplate.
14   *
15   * @author zoly
16   */
17  @Beta
18  public final class JdbcTemplate {
19  
20    private final DataSource dataSource;
21  
22    public JdbcTemplate(final DataSource dataSource) {
23      this.dataSource = dataSource;
24    }
25  
26    @SuppressFBWarnings("BED_BOGUS_EXCEPTION_DECLARATION")
27    public <R> R transactOnConnection(final HandlerNano<Connection, R, SQLException> handler,
28            final long timeout, final TimeUnit tu)
29            throws SQLException, InterruptedException {
30      return CallablesNano.executeWithRetry(new CallablesNano.TimeoutCallable<R, SQLException>(tu.toNanos(timeout)) {
31  
32        @Override
33        // CHECKSTYLE IGNORE RedundantThrows FOR NEXT 100 LINES
34        @SuppressFBWarnings("NP_LOAD_OF_KNOWN_NULL_VALUE")
35        public R call(final long deadlineNanos)
36                throws SQLException {
37          try (Connection conn = dataSource.getConnection()) {
38            boolean autocomit = conn.getAutoCommit();
39            if (autocomit) {
40              conn.setAutoCommit(false);
41            }
42            try {
43              R result = handler.handle(conn, deadlineNanos);
44              conn.commit();
45              return result;
46            } catch (SQLException | RuntimeException ex) {
47              conn.rollback();
48              throw ex;
49            } finally {
50              if (autocomit) {
51                conn.setAutoCommit(true);
52              }
53            }
54          }
55        }
56      }, 2, 1000);
57  
58    }
59  
60    @Override
61    public String toString() {
62      return "JdbcTemplate{" + "dataSource=" + dataSource + '}';
63    }
64    
65    
66  }
