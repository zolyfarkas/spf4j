2   package org.spf4j.ds;

4   import gnu.trove.set.hash.THashSet;
5   import java.util.Collection;
6   
7   public final class IdentityHashSet<T> extends THashSet<T> {
8   
9     public IdentityHashSet() {
10    }
11  
12    public IdentityHashSet(final int initialCapacity) {
13      super(initialCapacity);
14    }
15  
16    public IdentityHashSet(final int initialCapacity, final float loadFactor) {
17      super(initialCapacity, loadFactor);
18    }
19  
20    public IdentityHashSet(final Collection<? extends T> collection) {
21      super(collection);
22    }
23  
24    @Override
25    protected int hash(final Object notnull) {
26      return System.identityHashCode(notnull);
27    }
28  
29    @Override
30    protected boolean equals(final Object notnull, final Object two) {
31      return notnull == two;
32    }
33  }
