39      public interface TraversalCallback<V, E> {
40  
41          void handle(V vertex, Map<E, V> edges);
42      }
43  
44      public static <V, E> void traverse(final Graph<V, E> graph, final V startNode,
45              final TraversalCallback<V, E> handler, final boolean isBreadth) {
46          Set<V> traversedNodes = new HashSet<V>();
47          Deque<V> traversalQueue = new ArrayDeque<V>();
48          traversalQueue.add(startNode);
49          boolean done = false;
50          do {
51              boolean first = true;
52              while (!traversalQueue.isEmpty()) {
53                  V node;
54                  if (isBreadth) {
55                      node = traversalQueue.removeFirst();
56                  } else {
57                      node = traversalQueue.removeLast();
58                  }
59                  VertexEdges<V, E> edges = graph.getEdges(node);
60                  if (traversedNodes.contains(node)) {
61                      continue;
62                  }
63                  Map<E, V> incomming = edges.getIncomming();
64                  boolean hasIncomingBeenTraversed = true;
65                  for (V val : incomming.values()) {
66                      if (!traversedNodes.contains(val)) {
67                          hasIncomingBeenTraversed = false;
68                          break;
69                      }
70                  }
71                  if (!first && !hasIncomingBeenTraversed) {
72                      continue;
73                  }
74                  handler.handle(node, incomming);
75                  traversedNodes.add(node);
76                  first = false;
77                  Map<E, V> outgoing = edges.getOutgoing();
78                  for (V next : outgoing.values()) {
79                      traversalQueue.add(next);
80                  }
81              }
82  
83              Set<V> leftNodes = Sets.difference(graph.getVertices(), traversedNodes);
84              if (leftNodes.isEmpty()) {
85                  done = true;
86              } else {
87                  boolean added = false;
88                  for (V node : leftNodes) {
89                      Collection<V> incomingNodes = graph.getEdges(node).getIncomming().values();
90                      for (V incoming : incomingNodes) {
91                          if (traversedNodes.contains(incoming)) {
92                              traversalQueue.add(node);
93                              added = true;
94                              break;
95                          }
96                      }
97                      if (added) {
98                          break;
99                      }
100                 }
101             }
102         } while (!done);
103     }

106     public static final class VertexHolder<V> implements Comparable<VertexHolder<V>> {
107 
108         private final V vertex;
109         
110         private final int order;
111         
112         private final int nrImcoming;
113 
114         public VertexHolder(final V vertex, final int order, final int nrImcoming) {
115             this.vertex = vertex;
116             this.order = order;
117             this.nrImcoming = nrImcoming;
118         }
119            
120         @Override
121         public int compareTo(final VertexHolder<V> o) {
122             int result = this.nrImcoming - o.nrImcoming;
123             if (result == 0) {
124                 return  this.order - o.order;
125             } else {
126                 return result;
127             }
128         }
129 
130         public V getVertex() {
131             return vertex;
132         }
133 
134         @Override
135         public int hashCode() {
136             int hash = 7;
137             hash = 29 * hash + (this.vertex != null ? this.vertex.hashCode() : 0);
138             hash = 29 * hash + this.order;
139             return 29 * hash + this.nrImcoming;
140         }
141 
142         @Override
143         public boolean equals(final Object obj) {
144             if (obj == null) {
145                 return false;
146             }
147             if (getClass() != obj.getClass()) {
148                 return false;
149             }
150             final VertexHolder<V> other = (VertexHolder<V>) obj;
151             return (this.compareTo(other) == 0);
152         }
153         
154         
155         
156         
157     }

160     public static <V, E> void customTraverse(final Graph<V, E> graph, final V startNode,
161             final TraversalCallback<V, E> handler) {
162         Set<V> traversedNodes = new HashSet<V>();
163         Queue<VertexHolder<V>> traversalQueue = new PriorityQueue<VertexHolder<V>>(16);
164         int counter = 0;
165         traversalQueue.add(new VertexHolder<V>(startNode, counter++, 0));
166         boolean done = false;
167         do {
168             boolean first = true;
169             while (!traversalQueue.isEmpty()) {
170                 V node = traversalQueue.remove().getVertex();
171                 VertexEdges<V, E> edges = graph.getEdges(node);
172                 if (traversedNodes.contains(node)) {
173                     continue;
174                 }
175                 Map<E, V> incomming = edges.getIncomming();
176                 boolean hasIncomingBeenTraversed = true;
177                 for (V val : incomming.values()) {
178                     if (!traversedNodes.contains(val)) {
179                         hasIncomingBeenTraversed = false;
180                         break;
181                     }
182                 }
183                 if (!first && !hasIncomingBeenTraversed) {
184                     continue;
185                 }
186                 handler.handle(node, incomming);
187                 traversedNodes.add(node);
188                 first = false;
189                 Map<E, V> outgoing = edges.getOutgoing();
190                 for (V next : outgoing.values()) {
191                     traversalQueue.add(
192                             new VertexHolder<V>(next, counter++, graph.getEdges(next).getIncomming().size()));
193                 }
194             }
195 
196             Set<V> leftNodes = Sets.difference(graph.getVertices(), traversedNodes);
197             if (leftNodes.isEmpty()) {
198                 done = true;
199             } else {
200                 boolean added = false;
201                 for (V node : leftNodes) {
202                     Collection<V> incomingNodes = graph.getEdges(node).getIncomming().values();
203                     for (V incoming : incomingNodes) {
204                         if (traversedNodes.contains(incoming)) {
205                             traversalQueue.add(new VertexHolder<V>(node, counter++, 0));
206                             added = true;
207                             break;
208                         }
209                     }
210                     if (added) {
211                         break;
212                     }
213                 }
214             }
215         } while (!done);
216     }
