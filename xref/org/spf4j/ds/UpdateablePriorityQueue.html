17  /**
18   * A priority queue which allows efficient element update.
19   * Element update can be either a replacement or element mutation.
20   * Implementation is based on jdk priority queue.
21   * @author zoly
22   * @param <E> - the type of the elements in the queue.
23   */
24  
25  public final class UpdateablePriorityQueue<E> implements Iterable<E>, Serializable {
26  
27      private static final int DEFAULT_INITIAL_CAPACITY = 16;
28      private static final long serialVersionUID = 1L;
29  
30      public final class ElementRef implements Comparable<ElementRef> {
31  
32          private E elem;
33          private int index;
34  
35          ElementRef(@Nonnull final E elem, final int idx) {
36              this.elem = elem;
37              this.index = idx;
38          }
39  
40          @Override
41          public int compareTo(final ElementRef e) {
42              return comparator.compare(elem, e.elem);
43          }
44  
45          public E getElem() {
46              return elem;
47          }
48  
49          public int getIndex() {
50              return index;
51          }
52  
53          public void setElem(final E elem) {
54  
55              final int compare = comparator.compare(this.elem, elem);
56              this.elem = elem;
57              if (compare > 0) {
58                  siftUp(index, this);
59              } else if (compare < 0) {
60                  siftDown(index, this);
61              }
62          }
63  
64          public void elementMutated() {
65              int idx = this.index;
66              siftUp(this.index, this);
67              if (idx == this.index) {
68                  siftDown(idx, this);
69              }
70          }
71  
72          public boolean remove() {
73              if (this.index < 0) {
74                  return false;
75              }
76              removeAt(this.index);
77              return true;
78          }
79  
80          @Override
81          public int hashCode() {
82              int hash = 7;
83              hash = 97 * hash + Objects.hashCode(this.elem);
84              return 97 * hash + this.index;
85          }
86  
87  
88          @Override
89          public boolean equals(final Object obj) {
90              if (obj == null) {
91                  return false;
92              }
93              if (getClass() != obj.getClass()) {
94                  return false;
95              }
96              final ElementRef other = (ElementRef) obj;
97              if (!Objects.equals(this.elem, other.elem)) {
98                  return false;
99              }
100             return this.index == other.index;
101         }
102 
103 
104 
105         @Override
106         public String toString() {
107             return "QueueElement{" + "elem=" + elem + ", index=" + index + '}';
108         }
109

134             new Comparator<Comparable>() {

139                 }

142     public UpdateablePriorityQueue(final int initialCapacity,
143             final Comparator<? super E> comparator) {
144 
145         if (initialCapacity < 0) {
146             throw new IllegalArgumentException("Invalid initial capacity " + initialCapacity);
147         }
148         this.queue = (ElementRef[]) Array.newInstance(ElementRef.class, initialCapacity);
149         if (comparator == null) {
150             this.comparator = (Comparator<? super E>) DEFAULT_COMPARATOR;
151         } else {
152             this.comparator = comparator;
153         }
154     }
155 
156     private void grow(final int minCapacity) {
157         if (minCapacity < 0) { // overflow
158             throw new OutOfMemoryError();
159         }
160         int oldCapacity = queue.length;
161         // Double size if small; else grow by 50%
162         int newCapacity = ((oldCapacity < 64) ? ((oldCapacity + 1) * 2) : ((oldCapacity / 2) * 3));
163         if (newCapacity < 0) { // overflow
164             newCapacity = Integer.MAX_VALUE;
165         }
166         if (newCapacity < minCapacity) {
167             newCapacity = minCapacity;
168         }
169         queue = Arrays.copyOf(queue, newCapacity);
170     }
171 
172     public ElementRef add(@Nonnull final E e) {
173         if (e == null) {
174             throw new NullPointerException();
175         }
176         modCount++;
177         int i = size;
178         if (i >= queue.length) {
179             grow(i + 1);
180         }
181         size = i + 1;
182         ElementRef qe = new ElementRef(e, 0);
183         if (i == 0) {
184             queue[0] = qe;
185         } else {
186             siftUp(i, qe);
187         }
188         return qe;
189     }

208     private int indexOf(final E o) {
209         if (o != null) {
210             for (int i = 0; i < size; i++) {
211                 if (o.equals(queue[i].elem)) {
212                     return i;
213                 }
214             }
215         }
216         return -1;
217     }

219     public boolean remove(final ElementRef qe) {
220         int i = qe.index;

222             return false;
223         } else {
224             removeAt(i);
225             return true;
226         }
227     }

229     public boolean remove(final E o) {
230         int i = indexOf(o);
231         if (i == -1) {
232             return false;

234             removeAt(i);

239     public boolean isEmpty() {
240         return size == 0;

243     boolean removeEq(final Object o) {
244         for (int i = 0; i < size; i++) {
245             if (o == queue[i]) {
246                 removeAt(i);
247                 return true;
248             }
249         }
250         return false;
251     }

253     public boolean contains(final E o) {

262     @SuppressWarnings("unchecked")
263     public <T> T[] toArray(final T[] a) {
264         if (a.length < size) { // Make a new array of a's runtime type, but my contents:
265             return (T[]) Arrays.copyOf(queue, size, a.getClass());
266         }
267         System.arraycopy(queue, 0, a, 0, size);
268         if (a.length > size) {
269             a[size] = null;
270         }
271         return a;

279     private final class Itr implements Iterator<E> {
280 
281         /**
282          * Index (into queue array) of element to be returned by subsequent call to next.
283          */
284         private int cursor = 0;
285 
286         /**
287          * Index of element returned by most recent call to next, unless that element came from the forgetMeNot list.
288          * Set to -1 if element is deleted by a call to remove.
289          */
290         private int lastRet = -1;
291 
292         /**
293          * A queue of elements that were moved from the unvisited portion of the heap into the visited portion as a
294          * result of "unlucky" element removals during the iteration. (Unlucky element removals are those that require a
295          * siftup instead of a siftdown.) We must visit all of the elements in this list to complete the iteration. We
296          * do this after we've completed the "normal" iteration.
297          *
298          * We expect that most iterations, even those involving removals, will not need to store elements in this field.
299          */
300         private ArrayDeque<E> forgetMeNot = null;
301 
302         /**
303          * Element returned by the most recent call to next iff that element was drawn from the forgetMeNot list.
304          */
305         private E lastRetElt = null;
306 
307 
308         private int expectedModCount = modCount;
309 
310         @Override
311         public boolean hasNext() {
312             return cursor < size
313                     || (forgetMeNot != null && !forgetMeNot.isEmpty());
314         }
315 
316         @Override
317         public E next() {
318             if (expectedModCount != modCount) {
319                 throw new ConcurrentModificationException();
320             }
321             if (cursor < size) {
322                 //CHECKSTYLE:OFF
323                 return queue[lastRet = cursor++].elem;
324                 //CHECKSTYLE:ON
325             }
326             if (forgetMeNot != null) {
327                 lastRet = -1;
328                 lastRetElt = forgetMeNot.poll();
329                 if (lastRetElt != null) {
330                     return lastRetElt;
331                 }
332             }
333             throw new NoSuchElementException();
334         }
335 
336         @Override
337         public void remove() {
338             if (expectedModCount != modCount) {
339                 throw new ConcurrentModificationException();
340             }
341             if (lastRet != -1) {
342                 E moved = UpdateablePriorityQueue.this.removeAt(lastRet);
343                 lastRet = -1;
344                 if (moved == null) {
345                     cursor--;
346                 } else {
347                     if (forgetMeNot == null) {
348                         forgetMeNot = new ArrayDeque<>();
349                     }
350                     forgetMeNot.add(moved);
351                 }
352             } else if (lastRetElt != null) {
353                 UpdateablePriorityQueue.this.removeEq(lastRetElt);
354                 lastRetElt = null;
355             } else {
356                 throw new IllegalStateException();
357             }
358             expectedModCount = modCount;
359         }
360     }

362     public int size() {
363         return size;
364     }

366     public void clear() {
367         modCount++;
368         for (int i = 0; i < size; i++) {
369             final ElementRef ref = queue[i];
370             ref.index = -1;
371             queue[i] = null;
372         }
373         size = 0;
374     }
375 
376     @Nullable
377     @SuppressFBWarnings("BAS_BLOATED_ASSIGNMENT_SCOPE")
378     public E poll() {
379         if (size == 0) {
380             return null;
381         }
382         int s = --size;
383         modCount++;
384         final ElementRef removedRef = queue[0];
385         removedRef.index = -1;
386         E result = removedRef.elem;
387         ElementRef x = queue[s];
388         queue[s] = null;
389         if (s != 0) {
390             siftDown(0, x);
391         }
392         return result;
393     }
394 
395 
396     private E removeAt(final int i) {
397         assert i >= 0 && i < size;
398         modCount++;
399         queue[i].index = -1;
400         int s = --size;
401         if (s == i) { // removed last element
402             queue[i] = null;
403         } else {
404             ElementRef moved = queue[s];
405             queue[s] = null;
406             siftDown(i, moved);
407             if (queue[i] == moved) {
408                 siftUp(i, moved);
409                 if (queue[i] != moved) {
410                     return moved.elem;
411                 }
412             }
413         }
414         return null;
415     }
416 
417 
418     private void siftUp(final int pk, final ElementRef x) {
419         int k = pk;
420         while (k > 0) {
421             int parent = (k - 1) >>> 1;
422             ElementRef e = queue[parent];
423             if (x.compareTo(e) >= 0) {
424                 break;
425             }
426             queue[k] = e;
427             e.index = k;
428             k = parent;
429         }
430         queue[k] = x;
431         x.index = k;
432     }
433 
434     private void siftDown(final int pk, final ElementRef x) {
435         int k = pk;
436         int half = size >>> 1;        // loop while a non-leaf
437         while (k < half) {
438             int child = (k << 1) + 1; // assume left child is least
439             ElementRef c = queue[child];
440             int right = child + 1;
441             if (right < size && c.compareTo(queue[right]) > 0) {
442                 //CHECKSTYLE:OFF
443                 c = queue[child = right];
444                 //CHECKSTYLE:ON
445             }
446             if (x.compareTo(c) <= 0) {
447                 break;
448             }
449             queue[k] = c;
450             c.index = k;
451             k = child;
452         }
453         queue[k] = x;
454         x.index = k;
455     }

462     private void writeObject(final java.io.ObjectOutputStream s)
463         throws java.io.IOException {
464         // Write out element count, and any hidden stuff
465         s.defaultWriteObject();
466 
467         // Write out array length, for compatibility with 1.5 version
468         s.writeInt(Math.max(2, size + 1));
469 
470         // Write out all elements in the "proper order".
471         for (int i = 0; i < size; i++) {
472             s.writeObject(queue[i].elem);
473         }
474     }
475 
476 
477     private void readObject(final java.io.ObjectInputStream s)
478         throws java.io.IOException, ClassNotFoundException {
479         // Read in size, and any hidden stuff
480         s.defaultReadObject();
481 
482         // Read in (and discard) array length
483         s.readInt();
484 
485         queue = (ElementRef[]) Array.newInstance(ElementRef.class, size);
486 
487         // Read in all elements.
488         for (int i = 0; i < size; i++) {
489             queue[i] = new ElementRef((E) s.readObject(), i);
490         }
491 
492         // Elements are guaranteed to be in "proper order", but the
493         // spec has never explained what that might be.
494         heapify();
495     }
496 
497     public void heapify() {
498         for (int i = (size >>> 1) - 1; i >= 0; i--) {
499             siftDown(i, queue[i]);
500         }
501     }
