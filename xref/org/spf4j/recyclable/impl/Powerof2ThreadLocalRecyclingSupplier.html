18  package org.spf4j.recyclable.impl;
19  
20  import java.lang.ref.Reference;
21  import java.util.ArrayDeque;
22  import java.util.Deque;
23  import org.spf4j.base.IntMath;
24  import org.spf4j.base.ReferenceType;
25  import org.spf4j.recyclable.SizedRecyclingSupplier;
26  
27  /**
28   * recycling supplier that allows you to recycle objects.
29   * Recycling objects is dangerous business, care should be used.
30   * @author zoly
31   */
32  public final class Powerof2ThreadLocalRecyclingSupplier<T> implements SizedRecyclingSupplier<T> {
33  
34      private final SizedRecyclingSupplier.Factory<T> factory;
35  
36      private final ReferenceType refType;
37  
38      private final ThreadLocal<Deque<Reference<T>> []> localObjects;
39  
40      public Powerof2ThreadLocalRecyclingSupplier(final Factory<T> factory, final ReferenceType refType) {
41          this.factory = factory;
42          this.refType = refType;
43          localObjects = new ThreadLocal<Deque<Reference<T>> []>() {
44  
45              @Override
46              protected  Deque<Reference<T>>[] initialValue() {
47                  Deque<Reference<T>>[] result =  new Deque[28];
48                  for (int i = 0; i < result.length; i++) {
49                      result[i] = new ArrayDeque<>();
50                  }
51                  return result;
52              }
53          };
54      }
55  
56      @Override
57      public T get(final int size) {
58          Deque<Reference<T>>[] available = localObjects.get();
59          int idx = IntMath.closestPowerOf2(size);
60          Deque<Reference<T>> refs = available[idx];
61          if (refs.isEmpty()) {
62              int actualSize = 1 << idx;
63              return factory.create(actualSize);
64          } else {
65              T result;
66              do {
67                  Reference<T> removeLast = refs.removeLast();
68                  result = removeLast.get();
69              } while (result == null && !refs.isEmpty());
70              if (result == null) {
71                  int actualSize = 1 << idx;
72                  return factory.create(actualSize);
73              } else {
74                  return result;
75              }
76          }
77      }
78  
79      @Override
80      public void recycle(final T object) {
81          int size = factory.size(object);
82          int idx = IntMath.closestPowerOf2(size);
83          Deque<Reference<T>>[] available = localObjects.get();
84          Deque<Reference<T>> refs = available[idx];
85          refs.addLast(refType.create(object));
86      }
87  
88      @Override
89      public String toString() {
90          return "Powerof2ThreadLocalRecyclingSupplier{" + "factory=" + factory + ", refType=" + refType + '}';
91      }
92  
93  }
