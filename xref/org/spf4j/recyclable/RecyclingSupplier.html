18  package org.spf4j.recyclable;
19  
20  import java.util.concurrent.TimeoutException;
21  import javax.annotation.CheckReturnValue;
22  import javax.annotation.Nonnull;
23  import javax.annotation.Nullable;
24  import javax.annotation.ParametersAreNonnullByDefault;
25  
26  /**
27   * Object pool interface.
28   *
29   * My goal is to create a simpler and better object pool interface and implementation.
30   *
31   *
32   * @author zoly
33   */
34  @ParametersAreNonnullByDefault
35  public interface RecyclingSupplier<T> extends Disposable {
36  
37      /**
38       * block until a object is available and return it.
39       *
40       * @return
41       * @throws ObjectCreationException
42       * @throws ObjectBorrowException
43       * @throws InterruptedException
44       * @throws TimeoutException
45       */
46      @Nonnull
47      T get() throws ObjectCreationException, ObjectBorrowException,
48              InterruptedException, TimeoutException;
49  
50      /**
51       * return a object previously borrowed from the pool,
52       * together with a optional exception in case one was encountered
53       * while using the object. passing an exception will cause the object
54       * to be validated and potentially retired from the pool.
55       *
56       * @param object
57       * @param e
58       */
59      void recycle(T object, @Nullable Exception e);
60      
61      
62  
63      @ParametersAreNonnullByDefault
64      public interface Factory<T> {
65  
66          /**
67           * create the object.
68           *
69           * @return
70           * @throws ObjectCreationException
71           */
72          T create() throws ObjectCreationException;
73  
74          /**
75           * Dispose the object.
76           *
77           * @param object
78           * @throws ObjectDisposeException
79           */
80          void dispose(T object) throws ObjectDisposeException;
81  
82          /**
83           * Validate the object, return true if valid,
84           * false of throw an Exception with validation detail otherwise.
85           */
86          @Nullable
87          @CheckReturnValue
88          boolean validate(T object, @Nullable Exception e) throws Exception;
89          
90          
91      }
92  }
