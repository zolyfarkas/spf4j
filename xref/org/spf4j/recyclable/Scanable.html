18  package org.spf4j.recyclable;
19  
20  /**
21   *
22   * @author zoly
23   * @param <T> - type of the objects to scan.
24   */
25  public interface Scanable<T> {
26      
27      /**
28       * Scan through  objects and call handler...
29       * if handler throws exception scan is not aborted.
30       * scan is aborted only if handler returns false.
31       * @param handler - the handler to handle the scanned objects.
32       * @return - false if scanning is stopped by the handler, true otherwise.
33       * @throws Exception - whatever exception is thrown during scanning.
34       */
35      boolean scan(ScanHandler<T> handler) throws Exception;
36              
37      interface ScanHandler<O> {
38          
39          /**
40           * method to handle object
41           * @param object - the scanned object.
42           * @return true if scan operation is to continue, false otherwise.
43           * @throws java.lang.Exception - whatever exception this handler needs to throw.
44           */
45          boolean handle(O object) throws Exception;
46      }
47      
48  }
