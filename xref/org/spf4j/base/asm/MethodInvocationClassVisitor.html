2   package org.spf4j.base.asm;
3   
4   import java.lang.reflect.Method;
5   import java.util.ArrayDeque;
6   import java.util.Arrays;
7   import java.util.Collection;
8   import java.util.Deque;
9   import java.util.HashSet;
10  import java.util.Set;
11  import org.objectweb.asm.Label;
12  import org.objectweb.asm.MethodVisitor;
13  import org.objectweb.asm.Type;
14  import org.objectweb.asm.commons.EmptyVisitor;
15  
16  /**
17   * The purpose of this visitor is to find all incovation of certain method.
18   * Also to try to get the invocation parameters in case they are know at compile time.
19   * Since not all stack operations are simulated yet, the invocation parameters might not be accurate.
20   *
21   * @author zoly
22   */
23  class MethodInvocationClassVisitor extends EmptyVisitor {
24  
25      private final Collection<Invocation> addCaleesTo;
26      private final Set<String> methodStrings;
27      private String className;
28      private String source;
29  
30      public MethodInvocationClassVisitor(final Collection addCaleesTo,
31              final Set<Method> methods) {
32          this.addCaleesTo = addCaleesTo;
33          this.methodStrings = new HashSet<>();
34          for (Method m : methods) {
35              this.methodStrings.add(TypeUtils.toString(m));
36          }
37      }
38  
39      @Override
40      public void visit(final int version, final int access, final String name,
41              final String signature, final String superName, final String[] interfaces) {
42          className = name;
43      }
44  
45      @Override
46      public void visitSource(final String psource, final String debug) {
47          this.source = psource;
48      }
49  
50      @Override
51      public MethodVisitor visitMethod(final int access, final String methodName,
52              final String methodDesc, final String signature,
53              final String[] exceptions) {
54          return new EmptyVisitor() {
55  
56              private int lineNumber;
57  
58              private final Deque<Object> stack = new ArrayDeque<>();
59  
60              {
61                  Type[] argumentTypes = Type.getArgumentTypes(methodDesc);
62                  stack.addAll(Arrays.asList(argumentTypes));
63              }
64  
65              @Override
66              public void visitMethodInsn(final int opcode, final String owner, final String name, final String desc) {
67                  boolean isCallingtarget = methodStrings.contains(owner + '/' + name + desc);
68                  Type[] parameterTypes = Type.getArgumentTypes(desc);
69                  Object[] parameters = new Object[parameterTypes.length];
70                  for (int i = parameterTypes.length - 1; i >= 0; i--) {
71                      if (stack.isEmpty()) {
72                          parameters[i] = "UNKNOWN";
73                      } else {
74                          parameters[i] = stack.pop();
75                      }
76                  }
77                  if (isCallingtarget) {
78                      addCaleesTo.add(new Invocation(className, methodName, methodDesc, parameters,
79                              source, lineNumber));
80                  }
81              }
82  
83              @Override
84              public void visitLdcInsn(final Object o) {
85                  stack.push(o);
86              }
87  
88              @Override
89              public void visitLineNumber(final int line, final Label start) {
90                  this.lineNumber = line;
91              }
92  
93          };
94      }
95  
96  }
