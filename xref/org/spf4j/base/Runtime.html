91      @SuppressFBWarnings("MDM_RUNTIME_EXIT_OR_HALT")
92      public static void goDownWithError(@Nullable final Throwable t, final int exitCode) {
93          try {
94            if (t != null) {
95              Throwables.writeTo(t, System.err, Throwables.Detail.NONE); //High probability attempt to log first
96              Throwables.writeTo(t, System.err, Throwables.Detail.STANDARD); //getting more curageous :-)
97              Lazy.LOGGER.error("Error, going down with exit code {}", exitCode, t); //Now we are pushing it...
98            } else {
99              Lazy.LOGGER.error("Error, going down with exit code {}", exitCode);
100           }
101         } finally {
102           java.lang.Runtime.getRuntime().halt(exitCode);
103         }
104     }

124     private static final SortedMap<Integer, Set<Runnable>> SHUTDOWN_HOOKS = new TreeMap<>();
125 
126     static {
127         final java.lang.Runtime runtime = java.lang.Runtime.getRuntime();
128         RuntimeMXBean runtimeMxBean = ManagementFactory.getRuntimeMXBean();
129         final int availableProcessors = runtime.availableProcessors();
130         if (availableProcessors <= 0) {
131             System.err.println("Invalid number of processors " + availableProcessors
132                 + " defaulting to 1");
133             NR_PROCESSORS = 1;
134         } else {
135             NR_PROCESSORS = availableProcessors;
136         }
137         String mxBeanName = runtimeMxBean.getName();
138         PROCESS_NAME = mxBeanName;
139         PROCESS_ID = mxBeanName + ':' + Long.toHexString(System.currentTimeMillis());
140         int atIdx = mxBeanName.indexOf('@');
141         if (atIdx < 0) {
142             PID = -1;
143         } else {
144             PID = Integer.parseInt(mxBeanName.substring(0, atIdx));
145         }
146         final String osName = System.getProperty("os.name");
147         OS_NAME = osName;
148         IS_MAC_OSX =  "Mac OS X".equals(osName);
149         IS_WINDOWS = osName.startsWith("Windows");
150         runtime.addShutdownHook(new Thread(new AbstractRunnable(false) {
151             @Override
152             public void doRun() throws Exception {
153                 Exception rex = null;
154                 SortedMap<Integer, Set<Runnable>> hooks;
155                 synchronized (SHUTDOWN_HOOKS) {
156                     hooks = new TreeMap<>(SHUTDOWN_HOOKS);
157                     for (Map.Entry<Integer, Set<Runnable>> entry: hooks.entrySet()) {
158                         entry.setValue(new HashSet<>(entry.getValue()));
159                     }
160                 }
161                 for (Map.Entry<Integer, Set<Runnable>> runnables : hooks.entrySet()) {
162                     final Set<Runnable> values = runnables.getValue();
163                     if (values.size() <= 1) {
164                         for (Runnable runnable : values) {
165                             try {
166                                 runnable.run();
167                             } catch (RuntimeException ex) {
168                               if (rex == null) {
169                                 rex = ex;
170                               } else {
171                                 rex.addSuppressed(ex);
172                               }
173                             }
174                         }
175                     } else if (((int) runnables.getKey()) >= Integer.MAX_VALUE) {
176                         Thread[] threads = new Thread[values.size()];
177                         int i = 0;
178                         for (Runnable runnable : values) {
179                             Thread thread = new Thread(runnable);
180                             thread.start();
181                             threads[i++] = thread;
182                         }
183                         long deadline = System.currentTimeMillis() + WAIT_FOR_SHUTDOWN_MILLIS;
184                         for (Thread thread : threads) {
185                             try {
186                                 thread.join(deadline - System.currentTimeMillis());
187                             } catch (InterruptedException ex) {
188                               if (rex == null) {
189                                 rex = ex;
190                               } else {
191                                 rex.addSuppressed(ex);
192                               }
193                             }
194                         }
195                     } else {
196                         List<Future<?>> futures = new ArrayList<>(values.size());
197                         for (Runnable runnable : values) {
198                             futures.add(DefaultExecutor.INSTANCE.submit(runnable));
199                         }
200                         for (Future<?> future : futures) {
201                             try {
202                                 future.get();
203                             } catch (InterruptedException | ExecutionException | RuntimeException ex) {
204                               if (rex == null) {
205                                 rex = ex;
206                               } else {
207                                 rex.addSuppressed(ex);
208                               }
209                             }
210                         }
211                     }
212                 }
213                 // print out info on all remaining non daemon threads.
214                 Thread[] threads = FastStackCollector.getThreads();
215                 Thread current = Thread.currentThread();
216                 boolean first = true;
217                 for (Thread thread : threads) {
218                     if (thread.isAlive() && !thread.isDaemon() && !thread.equals(current)) {
219                         if (first) {
220                             System.err.println("Non daemon threads still running:");
221                             first = false;
222                         }
223                         System.err.println("Non daemon thread " + thread + ", stackTrace = "
224                                 + java.util.Arrays.toString(thread.getStackTrace()));
225                     }
226                 }
227                 if (rex != null) {
228                   throw rex;
229                 }
230             }
231         }, "spf4j queued shutdown"));
232         JAVA_PLATFORM = Version.fromSpecVersion(JAVA_VERSION);
233         Registry.export(Jmx.class);
234     }

293         public static final int MAX_NR_OPENFILES;
294 
295         static {
296             int mfiles;
297             if (ULIMIT.exists() && ULIMIT.canExecute()) {
298                 ULIMIT_CMD = new String[]{ULIMIT.getPath()};
299             } else if (BASH.exists() && BASH.canExecute()) {
300                 ULIMIT_CMD = new String[]{BASH.getPath(), "-c", "ulimit"};
301             }  else if (SH.exists() && SH.canExecute()) {
302                 ULIMIT_CMD = new String[]{SH.getPath(), "-c", "ulimit"};
303             } else {
304                 ULIMIT_CMD = null;
305             }
306             if (ULIMIT_CMD == null) {
307                 Lazy.LOGGER.warn("No ulimit available on {}, assuming no limit for nr open files",
308                         Runtime.OS_NAME);
309                 mfiles = Integer.MAX_VALUE;
310             } else {
311                 mfiles = runUlimit("-Sn");
312             }
313             MAX_NR_OPENFILES = mfiles;
314         }
315 
316         public static int runUlimit(final String ... options) {
317             if (ULIMIT_CMD == null) {
318                 throw new RuntimeException("Ulimit not available on " + Runtime.OS_NAME);
319             }
320             int mfiles;
321             try {
322                 String result = Runtime.run(Arrays.concat(ULIMIT_CMD, options), 10000);
323                 if (result.contains("unlimited")) {
324                     mfiles = Integer.MAX_VALUE;
325                 } else {
326                     mfiles = Integer.parseInt(result.trim());
327                 }
328             } catch (TimeoutException | IOException | InterruptedException | ExecutionException ex) {
329                 Lazy.LOGGER.error("Error while running ulimit, assuming no limit", ex);
330                 mfiles = Integer.MAX_VALUE;
331             }
332             return mfiles;
333         }

339      * get the number of open files by current java process.
340      * @return -1 if cannot get nr of open files
341      * @throws IOException - IO exception while accessing procfs.
342      * @throws InterruptedException - interrupted.
343      * @throws ExecutionException - execution exception while invoking lsof.
344      * @throws java.util.concurrent.TimeoutException - timeout while obtaining info.

347     public static int getNrOpenFiles() throws IOException, InterruptedException, ExecutionException, TimeoutException {
348         if (isMacOsx()) {
349             if (Lsof.LSOF == null) {
350                 return -1;
351             }
352             LineCountCharHandler handler = new LineCountCharHandler();
353             run(LSOF_CMD, handler, 60000);
354             return handler.getLineCount() - 1;
355         } else {
356             if (Files.isDirectory(FD_FOLDER)) {
357                 int result = 0;
358                 try (DirectoryStream<Path> stream = Files.newDirectoryStream(FD_FOLDER)) {
359                     Iterator<Path> iterator = stream.iterator();
360                     while (iterator.hasNext()) {
361                         iterator.next();
362                         result++;
363                     }
364                 }
365                 return result;
366             } else {
367                 return -1;
368             }
369         }

405     public static int killProcess(final Process proc, final long terminateTimeoutMillis,
406             final long forceTerminateTimeoutMillis)
407             throws InterruptedException {
408       proc.destroy();
409       if (proc.waitFor(terminateTimeoutMillis, TimeUnit.MINUTES)) {
410           return proc.exitValue();
411       } else {
412         proc.destroyForcibly();
413         if (!proc.waitFor(forceTerminateTimeoutMillis, TimeUnit.SECONDS)) {
414           throw new RuntimeException("Cannot terminate " + proc);
415         } else {
416           return proc.exitValue();
417         }
418       }
419     }

428   @SuppressFBWarnings("LEST_LOST_EXCEPTION_STACK_TRACE") // not really lost, suppressed exceptions are used.
429   public static int run(final String[] command, final ProcOutputHandler handler,
430           final long timeoutMillis, final long terminationTimeoutMillis)
431           throws IOException, InterruptedException, ExecutionException, TimeoutException {
432     final Process proc = java.lang.Runtime.getRuntime().exec(command);
433     try (InputStream pos = proc.getInputStream();
434             InputStream pes = proc.getErrorStream();
435             OutputStream pis = proc.getOutputStream()) {
436       Future<?> esh;
437       Future<?> osh;
438       try {
439         esh = DefaultExecutor.INSTANCE.submit(new StdErrHandlerRunnable(handler, pes));
440       } catch (RuntimeException ex) {
441         int result = killProcess(proc, terminationTimeoutMillis, 5000);
442         throw new ExecutionException("Error, process terminated and returned " + result, ex);
443       }
444       try {
445         osh = DefaultExecutor.INSTANCE.submit(new StdOutHandlerRunnable(handler, pos));
446       } catch (RuntimeException ex) {
447         RuntimeException cex = Futures.cancelAll(true, esh);
448         if (cex != null) {
449           ex.addSuppressed(cex);
450         }
451         int result = killProcess(proc, terminationTimeoutMillis, 5000);
452         throw new ExecutionException("Error, process terminated and returned " + result, ex);
453       }
454       
455       long deadlineNanos = System.nanoTime() + TimeUnit.NANOSECONDS.convert(timeoutMillis, TimeUnit.MILLISECONDS);
456       boolean isProcessFinished;
457       try {
458         isProcessFinished = proc.waitFor(timeoutMillis, TimeUnit.MILLISECONDS);
459       } catch (InterruptedException ex) {
460         RuntimeException cex = Futures.cancelAll(true, osh, esh);
461         if (cex != null) {
462           ex.addSuppressed(cex);
463         }
464         killProcess(proc, terminationTimeoutMillis, 5000);
465         throw ex;
466       }
467       if (isProcessFinished) {
468         Exception hex = Futures.getAllWithDeadline(deadlineNanos, osh, esh).getSecond();
469         if (hex == null) {
470           return proc.exitValue();
471         } else {
472           throwException(hex);
473           throw new IllegalStateException();
474         }
475       } else {
476         RuntimeException cex = Futures.cancelAll(true, osh, esh);
477         int result = killProcess(proc, terminationTimeoutMillis, 5000);
478         TimeoutException tex = new TimeoutException("Timed out while executing: " + java.util.Arrays.toString(command)
479                 + ";\n process returned " + result + ";\n output handler: " + handler);
480         if (cex != null) {
481           tex.addSuppressed(cex);
482         }
483         throw tex;
484       }
485     }
486   }

488   @SuppressFBWarnings("ITC_INHERITANCE_TYPE_CHECKING")
489   public static void throwException(final Exception ex) throws IOException, InterruptedException,
490           ExecutionException, TimeoutException {
491     if (ex instanceof IOException) {
492       throw (IOException) ex;
493     } else if (ex instanceof InterruptedException) {
494       throw (InterruptedException) ex;
495     } else if (ex instanceof ExecutionException) {
496       throw (ExecutionException) ex;
497     } else if (ex instanceof TimeoutException) {
498       throw (TimeoutException) ex;
499     } else {
500       throw new ExecutionException(ex);
501     }

607     public static void queueHook(final int priority, final Runnable runnable) {
608         synchronized (SHUTDOWN_HOOKS) {
609             Integer pr = priority;
610             Set<Runnable> runnables = SHUTDOWN_HOOKS.get(pr);
611             if (runnables == null) {
612                 runnables = new HashSet<>();
613                 SHUTDOWN_HOOKS.put(pr, runnables);
614             }
615             runnables.add(runnable);
616         }
617     }

619     public static boolean removeQueuedShutdownHook(final Runnable runnable) {
620         if ("spf4j queued shutdown".equals(Thread.currentThread().getName())) {
621             return false;
622         }
623         synchronized (SHUTDOWN_HOOKS) {
624             for (Set<Runnable> entry : SHUTDOWN_HOOKS.values()) {
625                 if (entry.remove(runnable)) {
626                     return true;
627                 }
628             }
629         }
630         return false;
631     }

660     /**
661      * Attempts to run the GC in a verifiable way.
662      * @param timeoutMillis - timeout for GC attempt
663      * @return true if GC executed for sure, false otherwise, gc might have been executed though,
664      * but we cannot be sure.
665      */
666     @SuppressFBWarnings
667     public static boolean gc(final long timeoutMillis) {
668         WeakReference<Object> ref = new WeakReference<>(new Object());
669         long deadline = System.currentTimeMillis() + timeoutMillis;
670         do {
671             System.gc();
672         } while (ref.get() != null && System.currentTimeMillis() < deadline);
673         return ref.get() == null;
674     }

683     public static String jrun(final Class<?> classWithMain, final String classPath, final long timeoutMillis,
684             final String... arguments) throws InterruptedException, ExecutionException, TimeoutException, IOException {
685         final String jvmPath = JAVA_HOME + File.separatorChar + "bin" + File.separatorChar + "java";
