69      // Calling Halt is the only sensible thing to do when the JVM is hosed.
70      @edu.umd.cs.findbugs.annotations.SuppressWarnings
71      public static void goDownWithError(final Throwable t, final int exitCode) {
72          try {
73              Lazy.LOGGER.error("Unrecoverable Error, going down", t);
74          } finally {
75              try {
76                  t.printStackTrace();
77              } finally {
78                  java.lang.Runtime.getRuntime().halt(exitCode);
79              }
80          }
81      }

91      static {
92          final java.lang.Runtime runtime = java.lang.Runtime.getRuntime();
93          RuntimeMXBean runtimeMxBean = ManagementFactory.getRuntimeMXBean();
94          final int availableProcessors = runtime.availableProcessors();
95          if (availableProcessors <= 0) {
96              System.err.println("Invalid number of processors " + availableProcessors
97                  + " defaulting to 1");
98              NR_PROCESSORS = 1;
99          } else {
100             NR_PROCESSORS = availableProcessors;
101         }
102         PROCESS_NAME = runtimeMxBean.getName();
103         int atIdx = PROCESS_NAME.indexOf('@');
104         if (atIdx < 0) {
105             PID = -1;
106         } else {
107             PID = Integer.parseInt(PROCESS_NAME.substring(0, atIdx));
108         }
109         OS_NAME = System.getProperty("os.name");
110 
111         runtime.addShutdownHook(new Thread(new AbstractRunnable(false) {
112             @Override
113             public void doRun() throws Exception {
114                 synchronized (SHUTDOWN_HOOKS) {
115                     for (Runnable runnable : SHUTDOWN_HOOKS) {
116                         try {
117                             runnable.run();
118                         } catch (Exception e) {
119                             e.printStackTrace();
120                         }
121                     }
122                 }
123             }
124         }, "tsdb shutdown"));
125         //JAVA_PLATFORM = Version.fromSpecVersion(runtimeMxBean.getSpecVersion());
126         JAVA_PLATFORM = Version.fromSpecVersion(JAVA_VERSION);

171     public static int getNrOpenFiles() throws IOException, InterruptedException, ExecutionException {
172         if (isMacOsx()) {
173             if (Lsof.LSOF == null) {
174                 return -1;
175             }
176             LineCountCharHandler handler = new LineCountCharHandler();
177             run(LSOF_CMD, handler, 60000);
178             return handler.getLineCount() - 1;
179         } else {
180             if (FD_FOLDER.exists()) {
181                 final String[] list = FD_FOLDER.list();
182                 if (list == null) {
183                     return -1;
184                 }
185                 return list.length;
186             } else {
187                 return -1;
188             }
189         }
190     }

202     public interface ProcOutputHandler {
203 
204         void handleStdOut(byte[] bytes, int length);
205 
206         void stdOutDone();
207 
208         void handleStdErr(byte[] bytes, int length);
209 
210         void stdErrDone();
211     }
212 
213     public static int run(final String [] command, final ProcOutputHandler handler,
214             final long timeoutMillis)
215             throws IOException, InterruptedException, ExecutionException {
216         final Process proc = java.lang.Runtime.getRuntime().exec(command);
217         ScheduledFuture<?> schedule = DefaultScheduler.INSTANCE.schedule(new AbstractRunnable(false) {
218 
219             @Override
220             public void doRun() throws Exception {
221                 if (JAVA_PLATFORM.ordinal() >= Version.V1_8.ordinal()) {
222                     final Class<? extends Process> aClass = proc.getClass();
223                     if ((Boolean) aClass.getMethod("isAlive").invoke(proc)) {
224                         aClass.getMethod("destroyForcibly").invoke(proc);
225                     }
226                 } else {
227                     proc.destroy();
228                 }
229             }
230         }, timeoutMillis, TimeUnit.MILLISECONDS);
231         try (InputStream pos = proc.getInputStream();
232                 InputStream pes = proc.getErrorStream();
233                 OutputStream pis = proc.getOutputStream()) {
234             Future<?> esh = DefaultExecutor.INSTANCE.submit(new AbstractRunnable() {
235                 @Override
236                 public void doRun() throws Exception {
237                     int eos;
238                     byte [] buffer = ArraySuppliers.Bytes.TL_SUPPLIER.get(8192);
239                     try {
240                         while ((eos = pes.read(buffer)) >= 0) {
241                             handler.handleStdErr(buffer, eos);
242                         }
243                     } finally {
244                         ArraySuppliers.Bytes.TL_SUPPLIER.recycle(buffer);
245                         handler.stdErrDone();
246                     }
247                 }
248             });
249             int cos;
250             byte [] buffer = ArraySuppliers.Bytes.TL_SUPPLIER.get(8192);
251             try {
252                 while ((cos = pos.read(buffer)) >= 0) {
253                     handler.handleStdOut(buffer, cos);
254                 }
255             } finally {
256                 ArraySuppliers.Bytes.TL_SUPPLIER.recycle(buffer);
257                 handler.stdOutDone();
258             }
259             esh.get();
260             return proc.waitFor();
261         } finally {
262             schedule.cancel(false);
263         }
264     }

269     public static final class LineCountCharHandler implements ProcOutputHandler {
270 
271         public LineCountCharHandler() {
272             lineCount = 0;
273         }
274         private int lineCount;
275 
276         @Override
277         public void handleStdOut(final byte [] buffer, final int length) {
278             for (int i = 0; i < length; i++) {
279                 byte c = buffer[i];
280                 if (c ==  (byte) '\n') {
281                     lineCount++;
282                 }
283             }
284         }
285 
286         public int getLineCount() {
287             return lineCount;
288         }
289 
290         @Override
291         public void handleStdErr(final byte [] buffer, final int length) {
292             handleStdOut(buffer, length);
293         }
294 
295         @Override
296         public void stdOutDone() {
297         }
298 
299         @Override
300         public void stdErrDone() {
301         }
302     }

304     public static final class StringBuilderCharHandler implements ProcOutputHandler {
305 
306         private final Charset charset;
307 
308         public StringBuilderCharHandler(final Charset charset) {
309             builder = new ByteArrayBuilder(PID, ArraySuppliers.Bytes.JAVA_NEW);
310             this.charset = charset;
311         }
312 
313         public StringBuilderCharHandler() {
314             this(Charset.defaultCharset());
315         }
316         private final ByteArrayBuilder builder;
317 
318         @Override
319         public void handleStdOut(final byte [] buffer, final int length) {
320             builder.write(buffer, 0, length);
321         }
322 
323         @Override
324         public String toString() {
325             return new String(builder.getBuffer(), 0, builder.size(), charset);
326         }
327 
328         @Override
329         public void handleStdErr(final byte [] buffer, final int length) {
330             handleStdOut(buffer, length);
331         }
332 
333         @Override
334         public void stdOutDone() {
335         }
336 
337         @Override
338         public void stdErrDone() {
339         }

341     private static final LinkedList<Runnable> SHUTDOWN_HOOKS = new LinkedList<>();
342 
343 
344     public static void addHookAtBeginning(final Runnable runnable) {
345         synchronized (SHUTDOWN_HOOKS) {
346             SHUTDOWN_HOOKS.addFirst(runnable);
347         }
348     }
349 
350     public static void addHookAtEnd(final Runnable runnable) {
351         synchronized (SHUTDOWN_HOOKS) {
352             SHUTDOWN_HOOKS.addLast(runnable);
353         }
