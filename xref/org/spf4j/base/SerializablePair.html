19  package org.spf4j.base;
20  
21  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
22  import java.io.Serializable;
23  
24  /**
25   *
26   * @author zoly
27   */
28  @SuppressFBWarnings({ "PIS_POSSIBLE_INCOMPLETE_SERIALIZATION", "SE_NO_SUITABLE_CONSTRUCTOR" })
29  public class SerializablePair<A extends Serializable, B extends Serializable>
30          extends Pair<A, B>
31          implements Serializable {
32  
33      private static final long serialVersionUID = 1L;
34  
35      public SerializablePair(final A first, final B second) {
36          super(first, second);
37      }
38  
39      @edu.umd.cs.findbugs.annotations.SuppressWarnings("UMTP_UNBOUND_METHOD_TEMPLATE_PARAMETER")
40      public static <A extends Serializable, B extends Serializable>
41              SerializablePair<A, B> of(final A first, final B second) {
42          return new SerializablePair<>(first, second);
43      }
44  
45      private Object writeReplace()
46              throws java.io.ObjectStreamException {
47          PairProxy result = new PairProxy();
48          result.first = getFirst();
49          result.second = getSecond();
50          return result;
51      }
52  
53      private static final class PairProxy implements Serializable {
54  
55          private static final long serialVersionUID = 1L;
56          private Serializable first;
57          private Serializable second;
58  
59          private Object readResolve() {
60              return SerializablePair.of(first, second);
61          }
62  
63      }
64  
65  }
