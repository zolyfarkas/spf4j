26  /**
27   * some "standard" process exit codes from:
28   * http://tldp.org/LDP/abs/html/index.html
29   * https://www.freebsd.org/cgi/man.cgi?query=sysexits&apropos=0&sektion=0&manpath=FreeBSD+4.3-RELEASE&format=html
30   * http://journal.thobe.org/2013/02/jvms-and-kill-signals.html
31   * @author zoly
32   */
33  public enum SysExits {
34  
35    /**
36     * Everything is OK.
37     */
38    OK(0),
39  
40    /**
41     * Catch all for general errors.
42     */
43    EX_GENERAL(1),
44  
45    /**
46     * Shell build in miss-use.
47     */
48    EX_SHELL_BUILTIN_MISSUSE(2),
49  
50    /**
51     * The command was used incorrectly, e.g., with the wrong number of arguments, a bad flag, a bad syntax in a
52     * parameter, or whatever.
53     */
54    EX_USAGE(64),
55    /**
56     * The input data was incorrect in some way. This should only be used for user's data and not system files.
57     */
58    EX_DATAERR(65),
59    /**
60     * An input file (not a system file) did not exist or was not readable. This could also include errors like ``No
61     * message'' to a mailer (if it cared to catch it).
62     */
63    EX_NOINPUT(66),
64    /**
65     * The user specified did not exist. This might be used for mail addresses or remote logins.
66     */
67    EX_NOUSER(67),
68    /**
69     * The host specified did not exist. This is used in mail addresses or network requests.
70     */
71    EX_NOHOST(68),
72    /**
73     * A service is unavailable. This can occur if a support program or file does not exist. This can also be used as a
74     * catchall message when something you wanted to do doesn't work, but you don't know why.
75     */
76    EX_UNAVAILABLE(69),
77    /**
78     * An internal software error has been detected. This should be limited to non-operating system related errors as
79     * possible.
80     */
81    EX_SOFTWARE(70),
82    /**
83     * An operating system error has been detected. This is intended to be used for such things as ``cannot fork'',
84     * ``cannot create pipe'', or the like. It includes things like getuid returning a user that does not exist in the
85     * passwd file.
86     */
87    EX_OSERR(71),
88    /**
89     * Some system file (e.g., /etc/passwd, /var/run/utmp,etc.) does not exist, cannot be opened, or has some sort of
90     * error (e.g., syntax error).
91     */
92    EX_OSFILE(72),
93    /**
94     * A (user specified) output file cannot be created.
95     */
96    EX_CANTCREAT(73),
97    /**
98     * An error occurred while doing I/O on some file.
99     */
100   EX_IOERR(74),
101   /**
102    * Temporary failure, indicating something that is not really an error. In sendmail, this means that a mailer (e.g.)
103    * could not create a connection, and the request should be reattempted later.
104    */
105   EX_TEMPFAIL(75),
106   /**
107    * The remote system returned something that was ``not possible'' during a protocol exchange.
108    */
109   EX_PROTOCOL(76),
110   /**
111    * You did not have sufficient permission to perform the operation. This is not intended for file system problems,
112    * which should use EX_NOINPUT or EX_CANTCREAT, but rather for higher level permissions.
113    */
114   EX_NOPERM(77),
115   /**
116    * Something was found in an unconfigured or misconfigured state.
117    */
118   EX_CONFIG(78),
119 
120 
121   /**
122    * cannot execute invoked command.
123    */
124   EX_CANNOT_EXEC_CMD(126),
125 
126   /**
127    * Command not found.
128    */
129   EX_CMD_NOT_FOUND(127),
130 
131   /**
132    * Invalid argument to exit.
133    */
134   EX_INVALID_ARG_TO_EXIT(128),
135 
136   /**
137    * Section caused by exit due to signal.
138    * where signal name is same on Linux, Solaris and MacOS enum has the appropriate name.
139    * for linux see: http://man7.org/linux/man-pages/man7/signal.7.html
140    * or run man signal on you OS of choice.
141    */
142   EX_SIG_HUP(129),
143   EX_SIG_INT(130),
144   EX_SIG_QUIT(131),
145   EX_SIG_ILL(132),
146   EX_SIG_TRAP(133),
147   EX_SIG_ABRT(134),
148   EX_SIG_7(135),
149   EX_SIG_FPE(136),
150   EX_SIG_KILL(137),
151   EX_SIG_10(138),
152   EX_SIG_11(139),
153   EX_SIG_12(140),
154   EX_SIG_PIPE(141),
155   EX_SIG_ALRM(142),
156   EX_SIG_TERM(143),
157   EX_SIG_16(144),
158   EX_SIG_17(145),
159   EX_SIG_18(146),
160   EX_SIG_19(147),
161   EX_SIG_20(148),
162   EX_SIG_21(149),
163   EX_SIG_22(150),
164   EX_SIG_23(151),
165   EX_SIG_24(152),
166   EX_SIG_25(153),
167   EX_SIG_26(154),
168   EX_SIG_27(155),
169   EX_SIG_28(156),
170   EX_SIG_29(157),
171   EX_SIG_30(158),
172   EX_SIG_31(159),
173 
174   EX_STATUS_OUT_OF_RANGE(255),
175   /**
176    * Any return codes not explicitly defined will be associated with "EX_UNKNOWN"
177    */
178   EX_UNKNOWN(-1);
179 
180   SysExits(final int code) {
181     this.exitCode = code;
182   }
183 
184   final int exitCode;
185 
186   public int exitCode() {
187     return exitCode;
188   }
189 
190   public boolean isOk() {
191     return exitCode == 0;
192   }
193 
194   public boolean isError() {
195     return exitCode != 0;
196   }
197 
198   private static final TIntObjectMap<SysExits> CODE2ENUM;
199 
200   static {
201     SysExits[] values = SysExits.values();
202     TIntObjectMap<SysExits> c2e = new TIntObjectHashMap<>(values.length);
203     for (SysExits e : values) {
204       if (c2e.put(e.exitCode(), e) != null) {
205         throw new ExceptionInInitializerError("Duplicate exit code " + e);
206       }
207     }
208     CODE2ENUM = c2e;
209   }
210 
211 
212   /**
213    * @param exitCode
214    * @return corresponding enum.
215    */
216   @Nonnull
217   public static SysExits fromCode(final int exitCode) {
218     SysExits result =  CODE2ENUM.get(exitCode);
219     if (result == null) {
220       return SysExits.EX_UNKNOWN;
221     } else {
222       return result;
223     }
224   }
