14    @Nullable
15    public static RuntimeException runAll(final Runnable... runnables) {
16      RuntimeException ex = null;
17      for (Runnable closeable : runnables) {
18        try {
19          closeable.run();
20        } catch (RuntimeException ex1) {
21          if (ex == null) {
22            ex = ex1;
23          } else {
24            ex = Throwables.suppress(ex1, ex);
25          }
26        }
27      }
28      return ex;
29    }
