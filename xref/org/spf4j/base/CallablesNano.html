61      public static final AdvancedRetryPredicate<Exception> DEFAULT_EXCEPTION_RETRY =
62              new AdvancedRetryPredicate<Exception>() {
63          @Override
64          public AdvancedAction apply(@Nonnull final Exception input) {
65              Throwable rootCause = com.google.common.base.Throwables.getRootCause(input);
66              if (rootCause instanceof RuntimeException) {
67                  return AdvancedAction.ABORT;
68              }
69              if (rootCause instanceof SQLTransientException
70                      || rootCause instanceof SQLRecoverableException
71                      || rootCause instanceof IOException
72                      || rootCause instanceof TimeoutException) {
73                  LOG.debug("Exception encountered, retrying...", input);
74                  return AdvancedAction.RETRY;
75              }
76              return AdvancedAction.ABORT;
77          }
78      };

111      * After the immediate retries are done,
112      * delayed retry with randomized Fibonacci values up to the specified max is executed.
113      * @param <T> - the type returned by the Callable that is retried.
114      * @param <EX> - the Exception thrown by the retried callable.
115      * @param what - the callable to retry.
116      * @param nrImmediateRetries - the number of immediate retries.
117      * @param maxWaitNanos - maximum wait time in between retries.
118      * @param retryOnReturnVal - predicate to control retry on return value;
119      * @param retryOnException - predicate to retry on thrown exception.
120      * @return the result of the callable.
121      * @throws java.lang.InterruptedException - thrown if interrupted.
122      * @throws EX - the exception declared to be thrown by the callable.

125             final int nrImmediateRetries, final long maxWaitNanos,

127             final AdvancedRetryPredicate<Exception> retryOnException)

135     private static final class RetryData {
136 
137         private long immediateLeft;
138 
139         private long p1;
140 
141         private long p2;
142 
143         private final long maxDelayNanos;
144 
145         RetryData(final long immediateLeft, final long p1, final long maxDelayNanos) {
146             this.immediateLeft = immediateLeft;
147             if (p1 < 1) {
148                 this.p1 = 0;
149                 this.p2 = 1;
150             } else {
151                 this.p1 = p1;
152                 this.p2 = p1;
153             }
154             this.maxDelayNanos = maxDelayNanos;
155         }
156 
157         long nextDelayNanos() {
158             if (immediateLeft > 0) {
159                 immediateLeft--;
160                 return 0;
161             } else if (p2 > maxDelayNanos) {
162                 return maxDelayNanos;
163             } else {
164                 long result = p2;
165                 p2 = p1 + p2;
166                 p1 = result;
167                 return result;
168             }
169         }
170 
171     }

184     public static final class FibonacciBackoffRetryPredicate<T> implements TimeoutRetryPredicate<T> {
185 
186         private final IntMath.XorShift32 random;
187 
188         private final AdvancedRetryPredicate<T> arp;
189 
190         private final int nrImmediateRetries;
191 
192         private final long maxWaitNanos;
193 
194         private final long minWaitNanos;
195 
196         private Map<Object, RetryData> retryRegistry;
197 
198         private final Function<T, Object> mapper;
199 
200         public FibonacciBackoffRetryPredicate(final AdvancedRetryPredicate<T> arp,
201                 final int nrImmediateRetries, final long minWaitNanos, final long maxWaitNanos,
202                 final Function<T, Object> mapper) {
203             this.arp = arp;
204             this.nrImmediateRetries = nrImmediateRetries;
205             this.maxWaitNanos = maxWaitNanos;
206             this.minWaitNanos = minWaitNanos;
207             retryRegistry = null;
208             this.mapper = mapper;
209             this.random = new IntMath.XorShift32();
210         }
211 
212 
213         @Override
214         @SuppressFBWarnings("MDM_THREAD_YIELD")
215         public Action apply(final T value, final long deadlineNanos) throws InterruptedException, TimeoutException {
216             long currentTimeNanos = System.nanoTime();
217             if (currentTimeNanos > deadlineNanos) {
218                 return Action.ABORT;
219             }
220             if (retryRegistry == null) {
221                 retryRegistry = new HashMap<>();
222             }
223             AdvancedAction action = arp.apply(value, deadlineNanos);
224             switch (action) {
225                 case ABORT:
226                     return Action.ABORT;
227                 case RETRY_IMMEDIATE:
228                     return Action.RETRY;
229                 case RETRY_DELAYED:
230                 case RETRY:
231                     RetryData retryData = getRetryData(value, action);
232                     final long nextDelay = retryData.nextDelayNanos();
233                     long delayNanos = Math.min(nextDelay, deadlineNanos - currentTimeNanos);
234                     if (delayNanos > 0) {
235                         delayNanos = Math.abs(random.nextInt()) % delayNanos;
236                         Thread.sleep(TimeUnit.NANOSECONDS.toMillis(delayNanos));
237                     }
238                     return Action.RETRY;
239                 default:
240                     throw new RuntimeException("Unsupperted Retry Action " + action);
241 
242             }
243         }
244 
245         RetryData getRetryData(final T value, final AdvancedAction action) {
246             Object rootCauseClass = mapper.apply(value);
247             RetryData data = retryRegistry.get(rootCauseClass);
248             if (data == null) {
249                 data  = createRetryData(action);
250                 retryRegistry.put(rootCauseClass, data);
251             }
252             return data;
253         }
254 
255         private RetryData createRetryData(final AdvancedAction action) {
256             if (action == AdvancedAction.RETRY_DELAYED) {
257                 return new RetryData(0, minWaitNanos, maxWaitNanos);
258             } else {
259                 return new RetryData(nrImmediateRetries, minWaitNanos, maxWaitNanos);
260             }
261         }
262 
263 
264     }

374     public static final class SmartRetryPredicate2TimeoutRetryPredicate<T>
375     implements TimeoutRetryPredicate<T> {
376 
377         private final TimeoutDelayPredicate predicate;
378 
379         public SmartRetryPredicate2TimeoutRetryPredicate(final TimeoutDelayPredicate<T> predicate) {
380             this.predicate = predicate;
381         }
382 
383 
384 
385         @Override
386         @SuppressFBWarnings("MDM_THREAD_YIELD")
387         public Action apply(final T value, final long deadlineNanos) throws InterruptedException {
388             long apply = predicate.apply(value, deadlineNanos);
389             if (apply < 0) {
390                 return Action.ABORT;
391             } else if (apply == 0) {
392                 return Action.RETRY;
393             } else {
394                 Thread.sleep(TimeUnit.NANOSECONDS.toMillis(apply));
395                 return Action.RETRY;
396             }
397         }

538     /**
539      * Naive implementation of execution with retry logic. a callable will be executed and retry attempted in current
540      * thread if the result and exception predicates. before retry, a callable can be executed that can abort the retry
541      * and finish the function with the previous result.
542      *
543      * @param <T> - The type of callable to retry result;
544      * @param <EX> - the exception thrown by the callable to retry.
545      * @param what - the callable to retry.
546      * @param retryOnReturnVal - the predicate to control retry on return value.
547      * @param retryOnException - the predicate to return on retry value.
548      * @return the result of the retried callable if successful.
549      * @throws java.lang.InterruptedException - thrown if retry interrupted.
550      * @throws EX - the exception thrown by callable.
551      */
552     @SuppressFBWarnings("BC_UNCONFIRMED_CAST_OF_RETURN_VALUE")
553     public static <T, EX extends Exception> T executeWithRetry(
554             final RetryCallable<T, EX> what,
555             final RetryPredicate<? super T> retryOnReturnVal, final RetryPredicate<Exception> retryOnException)
556             throws InterruptedException, EX {
557         T result = null;
558         Exception lastEx = null; // last exception
559         try {
560             result = what.call();
561         } catch (InterruptedException ex1) {
562             throw what.lastException(ex1);
563         } catch (Exception e) { // only EX and RuntimeException
564             lastEx = e;
565         }
566         Exception lastExChain = lastEx; // last exception chained with all previous exceptions
567         while ((lastEx != null && retryOnException.apply(lastEx) == Action.RETRY)
568                 || retryOnReturnVal.apply(result) == Action.RETRY) {
569             if (Thread.interrupted()) {
570                 Thread.currentThread().interrupt();
571                 throw what.lastException(new InterruptedException());
572             }
573             result = null;
574             lastEx = null;
575             try {
576                 result = what.call();
577             } catch (InterruptedException ex1) {
578                 throw what.lastException(ex1);
579             } catch (Exception e) { // only EX and RuntimeException
580                 lastEx = e;
581                 if (lastExChain != null) {
582                     lastExChain = Throwables.suppress(e, lastExChain);
583                 } else {
584                     lastExChain = e;
585                 }
586             }
587         }
588         if (lastEx != null) {
589             if (lastExChain instanceof RuntimeException) {
590                 throw what.lastException((RuntimeException) lastExChain);
591             } else {
592                 throw what.lastException((EX) lastExChain);
593             }
594         }
595         return what.lastReturn(result);
596     }

608     public static <T> Callable<T> withName(final Callable<T> callable, final String name) {
609         return new Callable<T>() {
610 
611             @Override
612             public T call() throws Exception {
613                 Thread currentThread = Thread.currentThread();
614                 String origName = currentThread.getName();
615                 try {
616                     currentThread.setName(origName + '[' + name + ']');
617                     return callable.call();
618                 } finally {
619                     currentThread.setName(origName);
620                 }
621             }
622 
623             @Override
624             public String toString() {
625                 return name;
626             }
