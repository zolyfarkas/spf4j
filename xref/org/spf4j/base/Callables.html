65      public static final AdvancedRetryPredicate<Exception> DEFAULT_EXCEPTION_RETRY =
66              new AdvancedRetryPredicate<Exception>() {
67          @Override
68          public AdvancedAction apply(@Nonnull final Exception input) {
69              Throwable rootCause = com.google.common.base.Throwables.getRootCause(input);
70              if (rootCause instanceof RuntimeException) {
71                  return AdvancedAction.ABORT;
72              }
73              if (rootCause instanceof SQLTransientException
74                      || rootCause instanceof SQLRecoverableException
75                      || rootCause instanceof IOException
76                      || rootCause instanceof TimeoutException) {
77                  LOG.debug("Exception encountered, retrying...", input);
78                  return AdvancedAction.RETRY;
79              }
80              return AdvancedAction.ABORT;
81          }
82      };
83  
84      public static final Predicate<Exception> DEFAULT_EXCEPTION_RETRY_PREDICATE =
85              new Predicate<Exception>() {
86  
87          @Override
88          @SuppressFBWarnings("NP_PARAMETER_MUST_BE_NONNULL_BUT_MARKED_AS_NULLABLE")
89          public boolean apply(final Exception t) {
90              return DEFAULT_EXCEPTION_RETRY.apply(t) != AdvancedAction.ABORT;
91          }

138     private static final class RetryData {
139 
140         private int immediateLeft;
141 
142         private int p1;
143 
144         private int p2;
145 
146         private final int maxDelay;
147 
148         public RetryData(final int immediateLeft, final int p1, final int maxDelay) {
149             this.immediateLeft = immediateLeft;
150             if (p1 < 1) {
151                 this.p1 = 0;
152                 this.p2 = 1;
153             } else {
154                 this.p1 = p1;
155                 this.p2 = p1;
156             }
157             this.maxDelay = maxDelay;
158         }
159 
160         int nextDelay() {
161             if (immediateLeft > 0) {
162                 immediateLeft--;
163                 return 0;
164             } else if (p2 > maxDelay) {
165                 return maxDelay;
166             } else {
167                 int result = p2;
168                 p2 = p1 + p2;
169                 p1 = result;
170                 return result;
171             }
172         }
173 
174     }

187     public static final class FibonacciBackoffRetryPredicate<T> implements TimeoutRetryPredicate<T> {
188 
189         private final IntMath.XorShift32 random;
190 
191         private final AdvancedRetryPredicate<T> arp;
192 
193         private final int nrImmediateRetries;
194 
195         private final int maxWaitMillis;
196 
197         private final int minWaitMillis;
198 
199         private Map<Object, RetryData> retryRegistry;
200 
201         private final Function<T, Object> mapper;
202 
203         public FibonacciBackoffRetryPredicate(final AdvancedRetryPredicate<T> arp,
204                 final int nrImmediateRetries, final int minWaitMillis, final int maxWaitMillis,
205                 final Function<T, Object> mapper) {
206             this.arp = arp;
207             this.nrImmediateRetries = nrImmediateRetries;
208             this.maxWaitMillis = maxWaitMillis;
209             this.minWaitMillis = minWaitMillis;
210             retryRegistry = null;
211             this.mapper = mapper;
212             this.random = new IntMath.XorShift32();
213         }
214 
215 
216         @Override
217         @SuppressFBWarnings("MDM_THREAD_YIELD")
218         public Action apply(final T value, final long deadline) throws InterruptedException, TimeoutException {
219             long currentTime = System.currentTimeMillis();
220             if (currentTime > deadline) {
221                 return Action.ABORT;
222             }
223             if (retryRegistry == null) {
224                 retryRegistry = new HashMap<>();
225             }
226             AdvancedAction action = arp.apply(value);
227             switch (action) {
228                 case ABORT:
229                     return Action.ABORT;
230                 case RETRY_IMMEDIATE:
231                     return Action.RETRY;
232                 case RETRY_DELAYED:
233                 case RETRY:
234                     RetryData retryData = getRetryData(value, action);
235                     final int nextDelay = retryData.nextDelay();
236                     long delay = Math.min(nextDelay, deadline - currentTime);
237                     if (delay > 0) {
238                         delay = Math.abs(random.nextInt()) % delay;
239                         Thread.sleep(delay);
240                     }
241                     return Action.RETRY;
242                 default:
243                     throw new RuntimeException("Unsupperted Retry Action " + action);
244 
245             }
246         }
247 
248         RetryData getRetryData(final T value, final AdvancedAction action) {
249             Object rootCauseClass = mapper.apply(value);
250             RetryData data = retryRegistry.get(rootCauseClass);
251             if (data == null) {
252                 data  = createRetryData(action);
253                 retryRegistry.put(rootCauseClass, data);
254             }
255             return data;
256         }
257 
258         private RetryData createRetryData(final AdvancedAction action) {
259             if (action == AdvancedAction.RETRY_DELAYED) {
260                 return new RetryData(0, minWaitMillis, maxWaitMillis);
261             } else {
262                 return new RetryData(nrImmediateRetries, minWaitMillis, maxWaitMillis);
263             }
264         }
265 
266 
267     }

282     public abstract static class TimeoutCallable<T, EX extends Exception> implements CheckedCallable<T, EX> {
283 
284         private final long mdeadline;
285 
286         public TimeoutCallable(final int timeoutMillis) {
287             mdeadline = System.currentTimeMillis() + timeoutMillis;
288         }
289 
290         @Override
291         public final T call() throws EX, InterruptedException {
292             return call(mdeadline);
293         }

298             return mdeadline;

320     public static final class TimeoutRetryPredicate2RetryPredicate<T> implements RetryPredicate<T> {
321 
322         private final long deadline;
323 
324         private final TimeoutRetryPredicate<T> predicate;
325 
326         public TimeoutRetryPredicate2RetryPredicate(final long deadline, final TimeoutRetryPredicate<T> predicate) {
327             this.deadline = deadline;
328             this.predicate = predicate;
329         }
330 
331 
332 
333         @Override
334         public Action apply(final T value) throws InterruptedException {
335             try {
336                 return predicate.apply(value, deadline);
337             } catch (TimeoutException ex) {
338                 throw new RuntimeException(ex);
339             }
340         }

343     }

361     /**
362      * Naive implementation of execution with retry logic. a callable will be executed and retry attempted in current
363      * thread if the result and exception predicates. before retry, a callable can be executed that can abort the retry
364      * and finish the function with the previous result.
365      *
366      * @param what
367      * @param doBeforeRetry
368      * @param retryOnReturnVal
369      * @param retryOnException
370      * @return
371      * @throws InterruptedException
372      */
373     public static <T, EX extends Exception> T executeWithRetry(
374             final CheckedCallable<T, EX> what,
375             final RetryPredicate<? super T> retryOnReturnVal, final RetryPredicate<Exception> retryOnException)
376             throws InterruptedException, EX {
377         T result = null;
378         Exception lastEx = null; // last exception
379         try {
380             result = what.call();
381         } catch (InterruptedException ex1) {
382             throw ex1;
383         } catch (Exception e) { // only EX and RuntimeException
384             lastEx = e;
385         }
386         Exception lastExChain = lastEx; // last exception chained with all previous exceptions
387         while ((lastEx != null && retryOnException.apply(lastEx) == Action.RETRY)
388                 || retryOnReturnVal.apply(result) == Action.RETRY) {
389             if (Thread.interrupted()) {
390                 Thread.currentThread().interrupt();
391                 throw new InterruptedException();
392             }
393             result = null;
394             lastEx = null;
395             try {
396                 result = what.call();
397             } catch (InterruptedException ex1) {
398                 throw ex1;
399             } catch (Exception e) { // only EX and RuntimeException
400                 lastEx = e;
401                 if (lastExChain != null) {
402                     lastExChain = Throwables.suppress(e, lastExChain);
403                 } else {
404                     lastExChain = e;
405                 }
406             }
407         }
408         if (lastEx != null) {
409             if (lastExChain instanceof RuntimeException) {
410                 throw (RuntimeException) lastExChain;
411             } else {
412                 throw (EX) lastExChain;
413             }
414         }
415         return result;
416     }
