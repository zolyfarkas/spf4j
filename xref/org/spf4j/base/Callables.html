159   /**
160    * After the immediate retries are done, delayed retry with randomized Fibonacci values up to the specified max is
161    * executed.
162    *
163    * @param <T> - the type returned by the Callable that is retried.
164    * @param <EX> - the Exception thrown by the retried callable.
165    * @param what - the callable to retry.
166    * @param nrImmediateRetries - the number of immediate retries.
167    * @param maxWaitMillis - maximum wait time in between retries.
168    * @param retryOnReturnVal - predicate to control retry on return value;
169    * @param retryOnException - predicate to retry on thrown exception.
170    * @return the result of the callable.
171    * @throws java.lang.InterruptedException - thrown if interrupted.
172    * @throws EX - the exception declared to be thrown by the callable.
173    * @deprecated use RetryPolicy
174    */
175   @Deprecated
176   @Nullable
177   public static <T, EX extends Exception> T executeWithRetry(final TimeoutCallable<T, EX> what,
178           final int nrImmediateRetries, final int maxWaitMillis,
179           final TimeoutRetryPredicate<? super T, T> retryOnReturnVal,
180           final AdvancedRetryPredicate<Exception> retryOnException,
181           final Class<EX> exceptionClass)
182           throws InterruptedException, EX, TimeoutException {
183     long deadline = what.getDeadline();
184     return executeWithRetry(what,  new TimeoutRetryPredicate2RetryPredicate<>(deadline, retryOnReturnVal),
185             new FibonacciBackoffRetryPredicate<>(retryOnException, nrImmediateRetries,
186                     maxWaitMillis / 100, maxWaitMillis, Callables::rootClass, deadline,
187                     () -> System.currentTimeMillis(), TimeUnit.MILLISECONDS),
188     exceptionClass);
189   }

191   /**
192    * @deprecated use RetryPolicy
193    */
194   @Deprecated
195   private static final class RetryData {
196 
197     private int immediateLeft;
198 
199     private long p1;
200 
201     private long p2;
202 
203     private final long maxDelay;
204 
205     RetryData(final int immediateLeft, final long p1, final long maxDelay) {
206       this.immediateLeft = immediateLeft;
207       if (p1 < 1) {
208         this.p1 = 0;
209         this.p2 = 1;
210       } else {
211         this.p1 = p1;
212         this.p2 = p1;
213       }
214       this.maxDelay = maxDelay;
215     }
216 
217     private long nextDelay() {
218       if (immediateLeft > 0) {
219         immediateLeft--;
220         return 0;
221       } else if (p2 > maxDelay) {
222         return maxDelay;
223       } else {
224         long result = p2;
225         p2 = p1 + p2;
226         p1 = result;
227         return result;
228       }
229     }
230 
231   }

241   /**
242    * @deprecated use RetryPolicy
243    */
244   @Deprecated
245   public static final class FibonacciBackoffRetryPredicate<T, R> implements RetryPredicate<T, R> {
246 
247     private final IntMath.XorShift32 random;
248 
249     private final AdvancedRetryPredicate<T> arp;
250 
251     private final int nrImmediateRetries;
252 
253     private final long maxWaitUnits;
254 
255     private final long minWaitUnits;
256 
257     private Map<Object, RetryData> retryRegistry;
258 
259     private final Function<T, ?> mapper;
260 
261     private final long deadline;
262 
263     private final LongSupplier currTimeSuplier;
264 
265     private final TimeUnit tu;
266 
267     public FibonacciBackoffRetryPredicate(final AdvancedRetryPredicate<T> arp,
268             final int nrImmediateRetries, final long minWaitUnits, final long maxWaitUnits,
269             final Function<T, ?> mapper, final long deadline, final LongSupplier currTimeSuplier,
270             final TimeUnit tu) {
271       this.arp = arp;
272       this.nrImmediateRetries = nrImmediateRetries;
273       this.maxWaitUnits = maxWaitUnits;
274       this.minWaitUnits = minWaitUnits;
275       retryRegistry = null;
276       this.mapper = mapper;
277       this.random = new IntMath.XorShift32();
278       this.deadline = deadline;
279       this.currTimeSuplier = currTimeSuplier;
280       this.tu = tu;
281     }
282 
283     private RetryData getRetryData(final T value, final AdvancedAction action) {
284       Object rootCauseClass = mapper.apply(value);
285       RetryData data = retryRegistry.get(rootCauseClass);
286       if (data == null) {
287         data = createRetryData(action);
288         retryRegistry.put(rootCauseClass, data);
289       }
290       return data;
291     }
292 
293     private RetryData createRetryData(final AdvancedAction action) {
294       if (action == AdvancedAction.RETRY_DELAYED) {
295         return new RetryData(0, minWaitUnits, maxWaitUnits);
296       } else {
297         return new RetryData(nrImmediateRetries, minWaitUnits, maxWaitUnits);
298       }
299     }
300 
301     @Override
302     public RetryDecision<R> getDecision(final T value, final Callable<R> callable) {
303       long currentTime = currTimeSuplier.getAsLong();
304       if (currentTime > deadline) {
305         return RetryDecision.abort(new TimeoutException("Deadline " + Instant.ofEpochMilli(deadline)
306                 + " passed, current time is " + Instant.ofEpochMilli(currentTime)));
307       }
308       if (retryRegistry == null) {
309         retryRegistry = new HashMap<>();
310       }
311       AdvancedAction action = arp.apply(value, deadline);
312       switch (action) {
313         case ABORT:
314           return RetryDecision.abort();
315         case RETRY_IMMEDIATE:
316           return RetryDecision.retry(0, callable);
317         case RETRY_DELAYED:
318         case RETRY:
319           RetryData retryData = getRetryData(value, action);
320           final long nextDelay = retryData.nextDelay();
321           long delay = Math.min(nextDelay, deadline - currentTime);
322           if (delay > 0) {
323             delay = Math.abs(random.nextInt()) % delay;
324           }
325           if (currentTime + delay > deadline) {
326             return RetryDecision.abort(new TimeoutException("No time left for retry "
327                     + Instant.ofEpochMilli(deadline) + ' ' + tu
328                 + " passed, current time is " + Instant.ofEpochMilli(currentTime) + ' ' + tu));
329           }
330           return RetryDecision.retry(tu.toMillis(delay), callable);
331         default:
332           throw new UnsupportedOperationException("Unsupperted Retry Action " + action);
333 
334       }
335     }
336 
337   }

597   /**
598    * Naive implementation of execution with retry logic. a callable will be executed and retry attempted in current
599    * thread if the result and exception predicates. before retry, a callable can be executed that can abort the retry
600    * and finish the function with the previous result.
601    *
602    * @param <T> - The type of callable to retry result;
603    * @param <EX> - the exception thrown by the callable to retry.
604    * @param pwhat - the callable to retry.
605    * @param retryOnReturnVal - the predicate to control retry on return value.
606    * @param retryOnException - the predicate to return on retry value.
607    * @return the result of the retried callable if successful.
608    * @throws java.lang.InterruptedException - thrown if retry interrupted.
609    * @throws EX - the exception thrown by callable.
610    * @deprecated use RetryPolicy
611    */
612   @Deprecated
613   @SuppressFBWarnings({ "RCN_REDUNDANT_NULLCHECK_OF_NONNULL_VALUE", "MDM_THREAD_YIELD" })
614   @Nullable
615   public static <T, EX extends Exception> T executeWithRetry(
616           final CheckedCallable<T, EX> pwhat,
617           final RetryPredicate<? super T, T> retryOnReturnVal,
618           final RetryPredicate<Exception, T> retryOnException,
619           final Class<EX> exceptionClass)
620           throws InterruptedException, TimeoutException, EX {
621     Callable<T> what = pwhat;
622     T result = null;
623     Exception lastEx = null; // last exception
624     try {
625       result = what.call();
626     } catch (InterruptedException ex1) {
627       throw ex1;
628     } catch (Exception e) { // only EX and RuntimeException
629       lastEx = e;
630     }
631     Exception lastExChain = lastEx; // last exception chained with all previous exceptions
632     RetryDecision decision = null;
633     //CHECKSTYLE IGNORE InnerAssignment FOR NEXT 5 LINES
634     while ((lastEx != null
635             && (decision = retryOnException.getDecision(lastEx, what)).getDecisionType()
636                 == RetryDecision.Type.Retry)
637             || (lastEx == null && (decision = retryOnReturnVal.getDecision(result, what)).getDecisionType()
638                 == RetryDecision.Type.Retry)) {
639       if (Thread.interrupted()) {
640         throw new InterruptedException();
641       }
642       long delayMillis = decision.getDelayMillis();
643       if (delayMillis > 0) {
644         Thread.sleep(delayMillis);
645       }
646       what = decision.getNewCallable();
647       result = null;
648       lastEx = null;
649       try {
650         result = what.call();
651       } catch (InterruptedException ex1) {
652         throw ex1;
653       } catch (Exception e) { // only EX and RuntimeException
654         lastEx = e;
655         if (lastExChain != null) {
656           lastExChain = Throwables.suppress(lastEx, lastExChain);
657         } else {
658           lastExChain = lastEx;
659         }
660       }
661     }
662     if (decision == null) {
663       throw new IllegalStateException("Decission should have ben initialized " + lastEx + ", " + result);
664     }
665     if (decision.getDecisionType() == RetryDecision.Type.Abort) {
666         Exception ex = decision.getException();
667         if (ex != null) {
668           lastEx = ex;
669           if (lastExChain != null) {
670             lastExChain = Throwables.suppress(lastEx, lastExChain);
671           } else {
672             lastExChain = lastEx;
673           }
674         }
675     }
676     if (lastEx != null) {
677       if (lastExChain instanceof RuntimeException) {
678         throw (RuntimeException) lastExChain;
679       } else if (lastExChain instanceof TimeoutException) {
680         throw (TimeoutException) lastExChain;
681       } else if (lastExChain == null) {
682         return null;
683       } else if (exceptionClass.isAssignableFrom(lastExChain.getClass())) {
684         throw (EX) lastExChain;
685       } else {
686         throw new UncheckedExecutionException(lastExChain);
687       }
688     }
689     return result;
690   }

711         String origName = currentThread.getName();

728   static long overflowSafeAdd(final long currentTime, final long timeout) {
729     if (currentTime < 0) {
730       throw new IllegalArgumentException("Time must be positive, not " + currentTime);
731     }
732     if (timeout < 0) {
733       return currentTime;
734     }
735     long result = currentTime + timeout;
736     if ((currentTime ^ timeout) < 0 || (currentTime ^ result) >= 0) {
737       return result;
738     } else {
739       return Long.MAX_VALUE;
740     }
741   }
