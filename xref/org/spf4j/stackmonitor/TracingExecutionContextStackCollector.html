32  package org.spf4j.stackmonitor;
33  
34  import gnu.trove.map.TMap;
35  import gnu.trove.map.hash.THashMap;
36  import java.util.Arrays;
37  import java.util.Map;
38  import java.util.function.Supplier;
39  import javax.annotation.concurrent.NotThreadSafe;
40  import org.spf4j.base.ExecutionContext;
41  import org.spf4j.base.Threads;
42  
43  /**
44   * A stack sample collector that collects samples only for code executed within a execution context.
45   * Plus this collector will attach the samples to the contexts.
46   *
47   * This context requires ProfiledExecutionContextFactory wrapper.
48   *
49   * @author Zoltan Farkas
50   */
51  @NotThreadSafe
52  public final class TracingExecutionContextStackCollector implements ISampler {
53  
54    private final Supplier<Iterable<Map.Entry<Thread, ExecutionContext>>> execCtxSupplier;
55  
56    private Thread[] requestFor;
57  
58    private ExecutionContext[] contexts;
59  
60    private final TMap<String, StackCollector> collections;
61  
62    public TracingExecutionContextStackCollector(
63            final Supplier<Iterable<Map.Entry<Thread, ExecutionContext>>> execCtxSupplier) {
64      this(100, execCtxSupplier);
65    }
66  
67    public TracingExecutionContextStackCollector(final int maxNrThreads,
68            final Supplier<Iterable<Map.Entry<Thread, ExecutionContext>>> execCtxSupplier) {
69      requestFor = new Thread[maxNrThreads];
70      contexts = new ExecutionContext[maxNrThreads];
71      this.execCtxSupplier = execCtxSupplier;
72      collections = new THashMap<>();
73    }
74  
75    @Override
76    public void sample() {
77      Iterable<Map.Entry<Thread, ExecutionContext>> currentThreads = execCtxSupplier.get();
78      int i = 0;
79      for (Map.Entry<Thread, ExecutionContext> entry : currentThreads) {
80        requestFor[i] = entry.getKey();
81        contexts[i++] = entry.getValue();
82        if (i >= requestFor.length) {
83          break;
84        }
85      }
86      Arrays.fill(requestFor, i, requestFor.length, null);
87      StackTraceElement[][] stackTraces = Threads.getStackTraces(requestFor);
88      for (int j = 0; j < i; j++) {
89        StackTraceElement[] stackTrace = stackTraces[j];
90        ExecutionContext context = contexts[j];
91        String name = context.getName();
92        StackCollector c = collections.computeIfAbsent(name, (k) -> new StackCollectorImpl());
93        if (stackTrace != null && stackTrace.length > 0) {
94          c.collect(stackTrace);
95          context.compute("TSS", (String k, SampleNode v) -> {
96            if (v == null) {
97              return SampleNode.createSampleNode(stackTrace);
98            } else {
99              SampleNode.addToSampleNode(v, stackTrace);
100             return v;
101           }
102         });
103       } else {
104         c.collect(new StackTraceElement[]{
105           new StackTraceElement("Thread", requestFor[j].getName(), "", 0)
106         });
107       }
108     }
109   }
110 
111   @Override
112   public Map<String, SampleNode> getCollectionsAndReset() {
113     TMap<String, SampleNode> result = new THashMap<>(collections.size());
114     collections.forEachEntry((k, v) -> {
115       result.put(k, v.getAndReset());
116       return true;
117     });
118     return result;
119   }
120 
121   @Override
122   public Map<String, SampleNode> getCollections() {
123     TMap<String, SampleNode> result = new THashMap<>(collections.size());
124     collections.forEachEntry((k, v) -> {
125       result.put(k, v.get());
126       return true;
127     });
128     return result;
129   }
130 
131   @Override
132   public String toString() {
133     return "TracingExecutionContextStackCollector{" + "execCtxSupplier=" + execCtxSupplier
134             + ", collections=" + collections + '}';
135   }
