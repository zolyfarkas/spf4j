30  @edu.umd.cs.findbugs.annotations.SuppressWarnings("CBX_CUSTOM_BUILT_XML")

33      private StackVisualizer() {
34      }

36      private static final String[] COLORS = {"#CCE01B",
37          "#DDE01B", "#EEE01B", "#FFE01B", "#FFD01B",
38          "#FFC01B", "#FFA01B", "#FF901B", "#FF801B",
39          "#FF701B", "#FF601B", "#FF501B", "#FF401B"};

41      public static void generateHtmlTable(final Writer writer, final Method m,
42              final SampleNode node, final int tableWidth, final int maxDepth) throws IOException {
43          Map<Method, SampleNode> subNodes = node.getSubNodes();
44          writer.append("<table border=\"0\" cellpadding=\"0\" cellspacing=\"0\" "
45                  + "style=\"overflow:hidden;table-layout:fixed;width:").
46                  append(Integer.toString(tableWidth)).append("px\">\n");
47          int totalSamples = node.getSampleCount();
48  
49          if (subNodes != null && maxDepth > 0) {
50              writer.append("<tr style=\"height:1em\">");
51              for (Map.Entry<Method, SampleNode> entry : subNodes.entrySet()) {
52                  int width = entry.getValue().getSampleCount() * tableWidth / totalSamples;
53                  writer.append("<td style=\"vertical-align:bottom; width:").
54                          append(Integer.toString(width)).append("px\">");
55                  generateHtmlTable(writer, entry.getKey(), entry.getValue(), width, maxDepth - 1);
56                  writer.append("</td>");
57              }
58              writer.append("<td></td>");
59              writer.append("</tr>\n");
60          }
61          writer.append("<tr style=\"height:1em\" ><td ");
62          if (subNodes != null) {
63              writer.append("colspan=\"").append(Integer.toString(subNodes.size() + 1)).append("\" ");
64          }
65          writer.append(" title=\"");
66          m.toHtmlWriter(writer);
67          writer.append(":");
68          writer.append(Integer.toString(node.getSampleCount()));
69          writer.append("\" style=\"overflow:hidden;width:100%;vertical-align:bottom ;background:").
70                  append(COLORS[(int) (Math.random() * COLORS.length)]).append("\">");
71          m.toHtmlWriter(writer);
72          writer.append(":");
73          writer.append(Integer.toString(totalSamples));
74          writer.append("</td></tr>\n");
75  
76          writer.append("</table>\n");
77      }

79      public static void generateSvg(final Writer writer, final Method m,
80              final SampleNode node, final int x, final int y,
81              final int width, final int maxDepth, final String idPfx) throws IOException {
82          if (width < 200) {
83              throw new IllegalArgumentException("width must be greater than 200, it is " + width);
84          }
85          writer.append("<svg width=\"" + width + "\" height= \"" + (15 * node.height() + 15) + "\" onload=\""
86                  + idPfx + "init(evt)\" >\n");
87          writer.append("<script type=\"text/ecmascript\">\n"
88                  + "<![CDATA[\n"
89                  + "var " + idPfx + "tooltip;\n"
90                  + "var " + idPfx + "tooltip_bg;\n"
91                  + "function " + idPfx + "init(evt)\n"
92                  + "  {\n"
93                  + "    if ( window.svgDocument == null )\n"
94                  + "    {\n"
95                  + "      svgDocument = evt.target.ownerDocument;\n"
96                  + "    }\n"
97                  + "    " + idPfx + "tooltip = svgDocument.getElementById('" + idPfx + "tooltip'); "
98                  + idPfx + "tooltip_bg = svgDocument.getElementById('" + idPfx + "tooltip_bg');"
99                  + "  }\n"
100                 + "function " + idPfx + "ss(evt, mouseovertext, xx, yy)\n"
101                 + "{\n"
102                 + "  " + idPfx + "tooltip.setAttributeNS(null,\"x\",xx );\n"
103                 + "  " + idPfx + "tooltip.setAttributeNS(null,\"y\",yy+13 );\n"
104                 + "  " + idPfx + "tooltip.firstChild.data = mouseovertext;\n"
105                 + "  " + idPfx + "tooltip.setAttributeNS(null,\"visibility\",\"visible\");\n"
106                 + "length = " + idPfx + "tooltip.getComputedTextLength();\n"
107                 + idPfx + "tooltip_bg.setAttributeNS(null,\"width\",length+8);\n"
108                 + idPfx + "tooltip_bg.setAttributeNS(null,\"x\",xx);\n"
109                 +  idPfx + "tooltip_bg.setAttributeNS(null,\"y\",yy);\n"
110                 +  idPfx + "tooltip_bg.setAttributeNS(null,\"visibility\",\"visibile\");"
111                 + "}\n"
112                 + "function " + idPfx + "hh()\n"
113                 + "{\n"
114                 + "  " + idPfx + "tooltip.setAttributeNS(null,\"visibility\",\"hidden\");\n"
115                 + idPfx + "tooltip_bg.setAttributeNS(null,\"visibility\",\"hidden\");\n"
116                 + "}]]>"
117                 + "</script>");
118 
119         generateSubSvg(writer, m, node, x, y + 15, width - 100, maxDepth, idPfx);
120 
121         writer.append("<rect fill=\"rgb(255,255,255)\" id=\"" + idPfx + "tooltip_bg\"\n"
122                 + "      x=\"0\" y=\"0\" rx=\"4\" ry=\"4\"\n"
123                 + "      width=\"55\" height=\"17\" visibility=\"hidden\"/>");
124         writer.append("<text font-size=\"12\" font-family=\"Verdana\" fill=\"rgb(0,0,0)\"  id=\""
125                 + idPfx + "tooltip\" x=\"0\" y=\"0\" visibility=\"hidden\">Tooltip</text>");
126         writer.append("</svg>\n");
127     }

129     @SuppressFBWarnings("ISB_TOSTRING_APPENDING")
130     public static void generateSubSvg(final Writer writer, final Method m, final SampleNode node,
131             final int x, final int y, final int width, final int maxDepth, final String idPfx) throws IOException {
132 
133 
134         Map<Method, SampleNode> subNodes = node.getSubNodes();
135 
136         int totalSamples = node.getSampleCount();
137         String id = idPfx + "ix" + x + 'y' + y;
138         String content = HtmlEscapers.htmlEscaper().escape(m.toString() + ':' + totalSamples);
139         writer.append("<g onmouseover=\"" + idPfx + "ss(evt,'" + content + "'," + x + ", "
140                 + y + " )\" onmouseout=\"" + idPfx + "hh()\">");
141         writer.append("<rect id=\"" + id + "\" x=\"" + x + "\" y=\"" + y + "\" width=\"" + width
142                 + "\" height=\"15\" fill=\""
143                 + COLORS[(int) (Math.random() * COLORS.length)] + "\"  />");
144 
145         writer.append("<text x=\"" + x + "\" y=\"" + (y + 13)
146                 + "\" font-size=\"12\" font-family=\"Verdana\" fill=\"rgb(0,0,0)\" "
147                 + " >");
148         writer.append(content.substring(0, Math.min(width / 9, content.length())));
149         writer.append("</text>\n");
150         writer.append("</g>");
151 
152         if (subNodes != null && maxDepth > 0) {
153             int rx = 0;
154             for (Map.Entry<Method, SampleNode> entry : subNodes.entrySet()) {
155                 int cwidth = (int) (((long) entry.getValue().getSampleCount()) * width / totalSamples);
156                 generateSubSvg(writer, entry.getKey(), entry.getValue(), rx + x, y + 15, cwidth, maxDepth - 1, idPfx);
157                 rx += cwidth;
158             }
159 
160         }
161     }
