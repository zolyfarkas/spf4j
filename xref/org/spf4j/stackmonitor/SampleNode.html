33  @ParametersAreNonnullByDefault
34  public final class SampleNode implements Serializable {

38      private int sampleCount;
39      private HashMap<Method, SampleNode> subNodes;

41      public SampleNode(final StackTraceElement[] stackTrace, final int from) {
42          sampleCount = 1;
43          if (from >= 0) {
44              subNodes = new HashMap();
45              subNodes.put(Method.getMethod(stackTrace[from]), new SampleNode(stackTrace, from - 1));
46          }
47      }
48      
49      public static SampleNode createSampleNode(final StackTraceElement... stackTrace) {
50          SampleNode result = new SampleNode(1, null);
51          SampleNode prevResult = result;
52          for (int i = stackTrace.length - 1; i >= 0; i--) {
53              StackTraceElement elem = stackTrace[i];
54              if (prevResult.subNodes == null) {
55                  prevResult.subNodes = new HashMap<>();
56              }
57              SampleNode node = new SampleNode(1, null);
58              prevResult.subNodes.put(Method.getMethod(elem), node);
59              prevResult = node;
60          }
61          return result;
62      }
63      
64      
65      public static void addToSampleNode(final SampleNode node, final StackTraceElement... stackTrace) {
66          SampleNode prevResult = node;
67          prevResult.sampleCount++;
68          for (int i = stackTrace.length - 1; i >= 0; i--) {
69              StackTraceElement elem = stackTrace[i];
70              final Method method = Method.getMethod(elem);
71              SampleNode nNode;
72              if (prevResult.subNodes == null) {
73                  prevResult.subNodes = new HashMap<>();
74                  nNode = new SampleNode(1, null);
75                  prevResult.subNodes.put(method, nNode);
76              } else {
77                  nNode = prevResult.subNodes.get(method);
78                  if (nNode != null) {
79                      nNode.sampleCount++;
80                  } else {
81                      nNode = new SampleNode(1, null);
82                      prevResult.subNodes.put(method, nNode);
83                  }
84              }
85              prevResult = nNode;
86          }
87      }
88      
89      public static SampleNode clone(final SampleNode node) {
90          if (node.subNodes == null) {
91              return new SampleNode(node.sampleCount, null);
92          }
93          HashMap<Method, SampleNode> newSubNodes = new HashMap<>(node.subNodes.size());
94          for (Map.Entry<Method, SampleNode> entry: node.subNodes.entrySet()) {
95              newSubNodes.put(entry.getKey(), clone(entry.getValue()));
96          }
97          return new SampleNode(node.sampleCount, newSubNodes);
98      }
99      
100     public static SampleNode aggregate(final SampleNode node1, final SampleNode node2) {
101         int newSampleCount = node1.sampleCount + node2.sampleCount;
102         HashMap<Method, SampleNode> newSubNodes;
103         if (node1.subNodes == null && node2.subNodes == null) {
104             newSubNodes = null;
105         } else if (node1.subNodes == null) {
106             newSubNodes = new HashMap<>(node2.subNodes.size());
107             for (Map.Entry<Method, SampleNode> entry: node2.subNodes.entrySet()) {
108                 newSubNodes.put(entry.getKey(), clone(entry.getValue()));
109             }
110         } else if (node2.subNodes == null) {
111             newSubNodes = new HashMap<>(node1.subNodes.size());
112             for (Map.Entry<Method, SampleNode> entry: node1.subNodes.entrySet()) {
113                 newSubNodes.put(entry.getKey(), clone(entry.getValue()));
114             }
115         } else {
116             newSubNodes = new HashMap<>(node1.subNodes.size());
117             for (Map.Entry<Method, SampleNode> entry: node1.subNodes.entrySet()) {
118                 Method m = entry.getKey();
119                 SampleNode other = node2.subNodes.get(m);
120                 if (other == null) {
121                     newSubNodes.put(m, clone(entry.getValue()));
122                 } else {
123                     newSubNodes.put(m, aggregate(entry.getValue(), other));
124                 }
125             }
126             for (Map.Entry<Method, SampleNode> entry: node2.subNodes.entrySet()) { // on left not it right
127                 Method m = entry.getKey();
128                 if (!node1.subNodes.containsKey(m)) {
129                     newSubNodes.put(m, clone(entry.getValue()));
130                 }
131             }
132         }
133         return new SampleNode(newSampleCount, newSubNodes);
134     }

136     public SampleNode(final int count, @Nullable final HashMap<Method, SampleNode> subNodes) {
137         this.sampleCount = count;
138         this.subNodes = subNodes;
139     }

141     public void addSample(final StackTraceElement[] stackTrace, final int from) {
142         sampleCount++;
143         if (from >= 0) {
144             Method method = Method.getMethod(stackTrace[from]);
145             SampleNode subNode = null;
146             if (subNodes == null) {
147                 subNodes = new HashMap();
148             } else {
149                 subNode = subNodes.get(method);
150             }
151             if (subNode == null) {
152                 subNodes.put(method, new SampleNode(stackTrace, from - 1));
153             } else {
154                 subNode.addSample(stackTrace, from - 1);
155             }
156         }
157     }

173     public int height() {
174         if (subNodes == null) {
175             return 1;
176         } else {
177             int subHeight = 0;
178             for (SampleNode node : subNodes.values()) {
179                 int nHeight = node.height();
180                 if (nHeight > subHeight) {
181                     subHeight = nHeight;
182                 }
183             }
184             return subHeight + 1;
185         }
186 
187     }

190     public int getNrNodes() {
191         if (subNodes == null) {
192             return 1;
193         } else {
194             int nrNodes = 0;
195             for (SampleNode node : subNodes.values()) {
196                 nrNodes += node.getNrNodes();
197             }
198             return nrNodes + 1;
199         }
200     }

202     @Nullable
203     public SampleNode filteredBy(final Predicate<Method> predicate) {
204 
205         int newCount = this.sampleCount;
206 
207         HashMap<Method, SampleNode> sns = null;
208         if (this.subNodes != null) {
209             for (Map.Entry<Method, SampleNode> entry : this.subNodes.entrySet()) {
210                 Method method = entry.getKey();
211                 SampleNode sn = entry.getValue();
212                 if (predicate.apply(method)) {
213                     newCount -= sn.getSampleCount();
214                 } else {
215                     if (sns == null) {
216                         sns = new HashMap<>();
217                     }
218                     SampleNode sn2 = sn.filteredBy(predicate);
219                     if (sn2 == null) {
220                         newCount -= sn.getSampleCount();
221                     } else {
222                         newCount -= sn.getSampleCount() - sn2.getSampleCount();
223                         sns.put(method, sn2);
224                     }
225 
226                 }
227             }
228         }
229         if (newCount == 0) {
230             return null;
231         } else if (newCount < 0) {
232             throw new IllegalStateException("child sample counts must be <= parent sample count, detail: " + this);
233         } else {
234             return new SampleNode(newCount, sns);
235         }
236     }

238     public interface InvocationHandler {
239 
240         void handle(Method from, Method to, int count, final Map<Method, Integer> ancestors);
241     }
242 
243     public void forEach(final InvocationHandler handler, final Method from,
244             final Method to, final Map<Method, Integer> ancestors) {
245 
246         handler.handle(from, to, sampleCount, ancestors);
247 
248         if (subNodes != null) {
249             Integer val = ancestors.get(to);
250             if (val != null) {
251                 val = val + 1;
252             } else {
253                 val = 1;
254             }
255             ancestors.put(to, val);
256             for (Map.Entry<Method, SampleNode> subs : subNodes.entrySet()) {
257                 Method toKey = subs.getKey();
258                 subs.getValue().forEach(handler, to, toKey, ancestors);
259             }
260             val = ancestors.get(to);
261             if (val == 1) {
262                 ancestors.remove(to);
263             } else {
264                 val = val - 1;
265                 ancestors.put(to, val);
266             }
267         }
268     }

270     public static final class InvocationCount {
271 
272         private int value;
273 
274         public InvocationCount(final int value) {
275             this.value = value;
276         }
277 
278         public int getValue() {
279             return value;
280         }
281 
282         public void setValue(final int value) {
283             this.value = value;
284         }
285 
286         @Override
287         public String toString() {
288             return "InvocationCount{" + "value=" + value + '}';
289         }
290     }
291 
292     @Nonnull
293     public static Graph<Method, InvocationCount> toGraph(final SampleNode rootNode) {
294         final HashMapGraph<Method, InvocationCount> result = new HashMapGraph<>();
295 
296         rootNode.forEach(new InvocationHandler() {
297             @Override
298             public void handle(final Method pfrom, final Method pto, final int count,
299                     final Map<Method, Integer> ancestors) {
300 
301                 Method from = pfrom;
302                 Method to = pto;
303                 Integer val = ancestors.get(from);
304                 if (val != null) {
305                     from = from.withId(val - 1);
306                 }
307                 val = ancestors.get(to);
308                 if (val != null) {
309                     to = to.withId(val);
310                 }
311                 
312                 InvocationCount ic = result.getEdge(from, to);
313 
314                 if (ic == null) {
315                     result.add(new InvocationCount(count), from, to);
316                 } else {
317                     ic.setValue(count + ic.getValue());
318                 }
319 
320 
321             }
322         }, Method.ROOT, Method.ROOT, new HashMap<Method, Integer>());
323 
324         return result;
325 
326     }
