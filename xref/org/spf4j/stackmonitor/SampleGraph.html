85    public static final class SampleKey {
86  
87      private final Method method;
88      private int idxInHierarchy;
89  
90      public SampleKey(final Method method, final int idxInHierarchy) {
91        this.method = method;
92        this.idxInHierarchy = idxInHierarchy;
93      }
94  
95      @Override
96      public int hashCode() {
97        int hash = 59 + Objects.hashCode(this.method);
98        return 59 * hash + this.idxInHierarchy;
99      }
100 
101     @Override
102     public boolean equals(final Object obj) {
103       if (this == obj) {
104         return true;
105       }
106       if (obj == null) {
107         return false;
108       }
109       if (getClass() != obj.getClass()) {
110         return false;
111       }
112       final SampleKey other = (SampleKey) obj;
113       if (this.idxInHierarchy != other.idxInHierarchy) {
114         return false;
115       }
116       return Objects.equals(this.method, other.method);
117     }
118 
119     public Method getMethod() {
120       return method;
121     }
122 
123     @Override
124     public String toString() {
125       StringBuilder sb = new StringBuilder(32);
126       Methods.writeTo(method, sb);
127       sb.append('_');
128       sb.append(idxInHierarchy);
129       return sb.toString();
130     }

210   public SampleGraph(final Method m, final SampleNode node) {
211     int nrNodes = node.getNrNodes();
212     vertexMap = MultimapBuilder.hashKeys(nrNodes).hashSetValues(1).build();
213     aggregates = new THashMap<>(nrNodes);
214     sampleTree = GraphBuilder.directed()
215             .nodeOrder(ElementOrder.unordered())
216             .allowsSelfLoops(false)
217             .expectedNodeCount(nrNodes)
218             .build();
219     aggGraph = GraphBuilder.directed()
220             .nodeOrder(ElementOrder.unordered())
221             .allowsSelfLoops(false)
222             .expectedNodeCount(nrNodes)
223             .build();
224     rootVertex = tree2Graph(m, node);
225   }

227   /**
228    * Compute a duplication occurrence from root index for a method. (number of occurrences of this method on a stack
229    * path.
230    *
231    * @param from
232    * @param m
233    * @return
234    */
235   private int computeMethodIdx(final Sample from, final Method m) {
236     if (from.key.method.equals(m)) {
237       return from.key.idxInHierarchy + 1;
238     } else {
239       Set<Sample> predecessors = sampleTree.predecessors(from);
240       int size = predecessors.size();
241       if (size == 1) {
242         return computeMethodIdx(predecessors.iterator().next(), m);
243       } else if (size > 1) {
244         throw new IllegalStateException("Cannot have multiple predecesors for " + from + ", pred = " + predecessors);
245       } else {
246         return 0;
247       }
248     }
249   }

251   private Sample tree2Graph(final Method m, final SampleNode node) {
252     Sample parentVertex = new Sample(new SampleKey(m, 0), node.getSampleCount(), 0, node);
253     if (!sampleTree.addNode(parentVertex)) {
254       throw new IllegalStateException();
255     }
256     if (!vertexMap.put(parentVertex.key, parentVertex)) {
257       throw new IllegalStateException();
258     }
259     AggSample aggSampleVertex = new AggSample(parentVertex);
260     aggGraph.addNode(aggSampleVertex);
261     aggregates.put(parentVertex.key, aggSampleVertex);
262     Deque<TraversalData> dq = new ArrayDeque<>();
263     TMap<Method, SampleNode> subNodes = node.getSubNodes();
264     if (subNodes != null) {
265       subNodes.forEachEntry((k, v) -> {
266         dq.add(new TraversalData(parentVertex, k, v));
267         return true;
268       });
269     }
270     TraversalData t;
271     while ((t = dq.pollFirst()) != null) {
272       SampleKey vtxk = new SampleKey(t.method, computeMethodIdx(t.parent, t.method));
273       Sample vtx = new Sample(vtxk, t.node.getSampleCount(), t.parent.level + 1, t.node);
274       AggSample aggParent = aggregates.get(t.parent.key);
275       AggSample current;
276       do {
277         current = aggregates.get(vtxk);
278         if (current == null) {
279           current = new AggSample(vtx);
280           aggGraph.addNode(current);
281           aggregates.put(vtxk, current);
282         } else {
283 //          if (isParentDescendant(current, aggParent)) {
284 //            vtxk = new SampleKey(vtxk.method, vtxk.idxInHierarchy + 1);
285 //            vtx = new Sample(vtxk, t.node.getSampleCount(), t.parent.level + 1, t.node);
286 //            continue;
287 //          }
288           current.add(vtx);
289         }
290         break;
291       } while (true);
292       if (!sampleTree.addNode(vtx)) {
293         throw new IllegalStateException();
294       }
295       if (!vertexMap.put(vtx.key, vtx)) {
296         throw new IllegalStateException();
297       }
298       aggGraph.putEdge(aggParent, current);
299       if (!sampleTree.putEdge(t.parent, vtx)) {
300         throw new IllegalStateException();
301       }
302       TMap<Method, SampleNode> subNodes2 = t.node.getSubNodes();
303       final Sample vvv = vtx;
304       if (subNodes2 != null) {
305         subNodes2.forEachEntry((k, v) -> {
306           dq.add(new TraversalData(vvv, k, v));
307           return true;
308         });
309       }
310     }
311     updateLevels(aggSampleVertex);
312     return parentVertex;
313   }

315   /**
316    * Update level of an aggregate sample...
317    * @param sample
318    * @param l
319    */
320   private void updateLevels(final AggSample sample) {
321     Set<AggSample> seen = new THashSet<>();
322     ArrayDeque<Pair<AggSample, Integer>> trq = new ArrayDeque<>();
323     aggGraph.successors(sample).forEach((n) -> trq.add(Pair.of(n, sample.level + 1)));
324     Pair<AggSample, Integer> curr;
325     while ((curr = trq.pollFirst()) != null) {
326       AggSample first = curr.getFirst();
327       if (seen.add(first)) {
328         Integer second = curr.getSecond();
329         first.addLevel(second);
330         Set<AggSample> successors = aggGraph.successors(first);
331         successors.forEach((n) -> trq.add(Pair.of(n, second + 1)));
332       }
333     }
334   }

348   public boolean haveCommonChild(final AggSample a, final AggSample b) {
349     if (a.getKey().equals(b.getKey())) {
350       return true;
351     }
352     Set<SampleKey> traversed = new THashSet<>();
353     ArrayDeque<AggSample> trq = new ArrayDeque<>();
354     trq.add(a);
355     trq.add(b);
356     AggSample curr;
357     while ((curr = trq.pollFirst()) != null) {
358       if (!traversed.add(curr.getKey())) {
359         return true;
360       }
361       aggGraph.successors(curr).forEach((n) -> trq.add(n));
362     }
363     return false;
364   }

366   public boolean isParentDescendant(final AggSample a, final AggSample b) {
367     ArrayDeque<AggSample> trq = new ArrayDeque<>();
368     trq.add(a);
369     AggSample curr;
370     while ((curr = trq.pollFirst()) != null) {
371       if (curr.equals(b)) {
372         return true;
373       }
374       aggGraph.successors(curr).forEach((n) -> trq.add(n));
375     }
376     return false;
377   }
