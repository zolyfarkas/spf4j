78    private static Map<String,
79           Either<PartialExceptionRetryPredicateSupplier, PartialResultRetryPredicateSupplier>> load() {
80      Map<String, Either<PartialExceptionRetryPredicateSupplier, PartialResultRetryPredicateSupplier>> result
81              = new HashMap<>();
82      for (PartialExceptionRetryPredicateSupplier s
83              : ServiceLoader.load(PartialExceptionRetryPredicateSupplier.class)) {
84        result.put(s.getName(), Either.left(s));
85      }
86      for (PartialResultRetryPredicateSupplier s
87              : ServiceLoader.load(PartialResultRetryPredicateSupplier.class)) {
88        result.put(s.getName(), Either.right(s));
89      }
90      return result;
91    }

93    public static Either<PartialExceptionRetryPredicateSupplier, PartialResultRetryPredicateSupplier>
94            getRetryPredicateSupplier(final String name) {
95      return REGISTRY.get(name);

98    @Nullable
99    public static <T, C extends Callable<? extends T>>
100           Either<TimedSupplier<PartialExceptionRetryPredicate<T, C>>,
101          TimedSupplier<PartialResultRetryPredicate<T, C>>> getResultRetryPredicateSupplier(
102                   final String resultName, final RetryParams rp,
103                  final ResultMatchers.Supplier resultMatcherSupplier) {
104     Either<Predicate<Throwable>, Predicate<Object>> resPred = resultMatcherSupplier.apply(resultName);
105     if (resPred == null) {
106       return null;
107     }
108 
109     if (resPred.isLeft()) {
110       Predicate<Throwable> predicate = resPred.getLeft();
111       return Either.left(new TimedSupplier<PartialExceptionRetryPredicate<T, C>>() {
112         @Override
113         public PartialExceptionRetryPredicate<T, C> get(final long startTimeNanos, final long deadlineNanos) {
114           RetryDelaySupplier ds = new JitteredDelaySupplier(
115                   new FibonacciRetryDelaySupplier(rp.getNrInitialImmediateRetries(),
116                           rp.getStartDelay().toNanos(), rp.getMaxDelay().toNanos()), rp.getRetryDelayJitter());
117           CountLimitedPartialRetryPredicate clp
118                   = new CountLimitedPartialRetryPredicate<T, Throwable, C>(rp.getMaxNrRetries(),
119                           new PartialExceptionRetryPredicate<T, C>() {
120                     @Override
121                     public RetryDecision getExceptionDecision(final Throwable value, final Callable what) {
122 
123                       if (predicate.test(value)) {
124                         return RetryDecision.retry(ds.nextDelay(), what);
125                       }
126                       return null;
127                     }
128 
129                   });
130           TimeLimitedPartialRetryPredicate tlp
131                   = new TimeLimitedPartialRetryPredicate<T, Throwable, C>(startTimeNanos, deadlineNanos,
132                           rp.getMaxTimeToRetry().toNanos(), TimeUnit.NANOSECONDS, rp.getMaxTimeToRetryFactor(), clp);
133 
134           return (Throwable value, C what) -> tlp.apply(value, what);
135 
136         }
137       });
138     } else {
139       Predicate<Object> predicate = resPred.getRight();
140       return Either.right(new TimedSupplier<PartialResultRetryPredicate<T, C>>() {
141         @Override
142         public PartialResultRetryPredicate<T, C> get(final long startTimeNanos, final long deadlineNanos) {
143           RetryDelaySupplier ds = new JitteredDelaySupplier(
144                   new FibonacciRetryDelaySupplier(rp.getNrInitialImmediateRetries(),
145                           rp.getStartDelay().toNanos(), rp.getMaxDelay().toNanos()), rp.getRetryDelayJitter());
146           CountLimitedPartialRetryPredicate clp
147                   = new CountLimitedPartialRetryPredicate<T, T, C>(rp.getMaxNrRetries(),
148                           new PartialResultRetryPredicate<T, C>() {
149                     @Override
150                     public RetryDecision getDecision(final Object value, final Callable what) {
151 
152                       if (predicate.test(value)) {
153                         return RetryDecision.retry(ds.nextDelay(), what);
154                       }
155                       return null;
156                     }
157 
158                   });
159           TimeLimitedPartialRetryPredicate tlp
160                   = new TimeLimitedPartialRetryPredicate<T, T, C>(startTimeNanos, deadlineNanos,
161                           rp.getMaxTimeToRetry().toNanos(), TimeUnit.NANOSECONDS, rp.getMaxTimeToRetryFactor(), clp);
162 
163           return (T value, C what) -> tlp.apply(value, what);
164 
165         }
166       });
167     }
168

171   public static <T, C extends Callable<? extends T>> RetryPolicy<T, C> create(
172           final org.spf4j.failsafe.avro.RetryPolicy policy) throws InvalidRetryPolicyException {
173     return create(policy, ResultMatchers.toSupplier());
174   }

176   /**
177    * Create a retry policy with a custom result MatcherSupplier.
178    * this allows use of custom resolution to result matcher names.
179    * @param <T>
180    * @param <C>
181    * @param policy the retry policy serializable representation.
182    * @param resultMatcherSupplier - a resolvers that resolvers the referenced named matchers in the policy
183    * @return
184    * @throws InvalidRetryPolicyException
185    */
186   public static <T, C extends Callable<? extends T>> RetryPolicy<T, C> create(
187           final org.spf4j.failsafe.avro.RetryPolicy policy,
188           final ResultMatchers.Supplier resultMatcherSupplier) throws InvalidRetryPolicyException {
189     return ((RetryPolicy.Builder<T, C>) createBuilder(policy, resultMatcherSupplier)).build();
190   }

192   public static <T, C extends Callable<? extends T>> RetryPolicy.Builder<T, C> createBuilder(
193           final org.spf4j.failsafe.avro.RetryPolicy policy,
194           final ResultMatchers.Supplier resultMatcherSupplier) throws InvalidRetryPolicyException {
195     RetryPolicy.Builder<T, C> builder = RetryPolicy.newBuilder();
196     addRetryPolicy(builder, policy, resultMatcherSupplier);
197     return builder;
198   }

200   public static <C extends Callable<? extends T>, T> void addRetryPolicy(final RetryPolicy.Builder<T, C> builder,
201           final org.spf4j.failsafe.avro.RetryPolicy policy,
202           final ResultMatchers.Supplier resultMatcherSupplier) throws InvalidRetryPolicyException {
203     builder.withMaxExceptionChain(policy.getMaxSupressedExceptions());
204     for (RetryRule rule : policy.getRetryRules()) {
205       ScriptedRetryPredicateSupplier ps = rule.getPredicateSupplier();
206       if (ps == null) {
207         Either<PartialExceptionRetryPredicateSupplier, PartialResultRetryPredicateSupplier> ups
208                 = getRetryPredicateSupplier(rule.getName());
209         if (ups.isLeft()) {
210           @SuppressWarnings("unchecked")
211           TimedSupplier<PartialExceptionRetryPredicate<T, C>> ets = ups.getLeft();
212           builder.withExceptionPartialPredicateSupplier(ets);
213         } else {
214           @SuppressWarnings("unchecked")
215           TimedSupplier<PartialResultRetryPredicate<T, C>> rts = ups.getRight();
216           builder.withResultPartialPredicateSupplier(rts);
217         }
218       } else {
219         if (!"java".equals(ps.getTargetLanguage())) {
220           continue;
221         }
222         String language = ps.getLanguage();
223         ScriptEngine engine = SCRIPT_ENGINE_MANAGER.getEngineByName(language);
224         if (engine == null) {
225           throw new InvalidRetryPolicyException("No engine named: " + language);
226         }
227         Bindings bindings = engine.createBindings();
228         bindings.put("decision", Reflections.implementStatic(RetryDecisionFactory.class, RetryDecision.class));
229         engine.setBindings(bindings, ScriptContext.ENGINE_SCOPE);
230         String rps = ps.getReturnPredicateSupplier();
231         if (!rps.isEmpty()) {
232           Invocable invocable;
233           try {
234             invocable = toInvocable(engine, rps);
235           } catch (ScriptException ex) {
236             throw new InvalidRetryPolicyException("Invalid Script: " + rps, ex);
237           }
238           builder.withResultPartialPredicateSupplier(
239                   (start, deadline) -> (object, callable)
240                           -> {
241                     try {
242                       return (RetryDecision) invocable.invokeFunction(null, start, deadline, object, callable);
243                     } catch (ScriptException | NoSuchMethodException ex) {
244                       Logger.getLogger(RetryPolicies.class.getName()).log(Level.SEVERE,
245                               "Failed predicate {0}", new Object[]{rps, ex});
246                       return RetryDecision.ABORT;
247                     }
248                   });
249         }
250         String tps = ps.getThrowablePredicateSupplier();
251         if (!tps.isEmpty()) {
252           Invocable invocable;
253           try {
254             invocable = toInvocable(engine, tps);
255           } catch (ScriptException ex) {
256             throw new InvalidRetryPolicyException("Invalid Script: " + tps, ex);
257           }
258           builder.withResultPartialPredicateSupplier(
259                   (start, deadline) -> (object, callable)
260                           -> {
261                     try {
262                       return (RetryDecision) invocable.invokeFunction(null, start, deadline, object, callable);
263                     } catch (ScriptException | NoSuchMethodException ex) {
264                       Logger.getLogger(RetryPolicies.class.getName()).log(Level.SEVERE,
265                               "Failed predicate {0}", new Object[]{tps, ex});
266                       return RetryDecision.ABORT;
267                     }
268                   });
269         }
270       }
271     }
272     for (Map.Entry<String, RetryParams> entry : policy.getResponse2RetryParams().entrySet()) {
273       String reasonName = entry.getKey();
274       Either<TimedSupplier<PartialExceptionRetryPredicate<T, C>>,
275               TimedSupplier<PartialResultRetryPredicate<T, C>>> result
276               = getResultRetryPredicateSupplier(reasonName, entry.getValue(), resultMatcherSupplier);
277       if (result == null) {
278         throw new InvalidRetryPolicyException("No reason matcher defined for: " + reasonName);
279       }
280       if (result.isLeft()) {
281         TimedSupplier<PartialExceptionRetryPredicate<T, C>> ets = result.getLeft();
282         builder.withExceptionPartialPredicateSupplier(ets);
283       } else {
284         TimedSupplier<PartialResultRetryPredicate<T, C>> rts = result.getRight();
285         builder.withResultPartialPredicateSupplier(rts);
286       }
287     }

290   /**
291    * Scripted predicates are not there for application users to modify.
292    * Scripted predicates should be written by Operation engineers.
293    * The script engine choice needs to be carefully done as to not allow even ops to do bad stuff by mistake.
294    * (calling system.exit, etc...)
295    * The main reason for allowing scripted predicates is deployment speed.
296    * (a config change will propagates to your fleet several order of magnitudes faster than binaries,
297    * and will not require a process restart)
298    * Now this speed means also any stupid stuff can propagate faster, so canarying a config change is also a must.
299    * @param engine
300    * @param script
301    * @return
302    * @throws ScriptException
303    */
304   @SuppressFBWarnings("SCRIPT_ENGINE_INJECTION")
305   public static Invocable toInvocable(final ScriptEngine engine, final String script) throws ScriptException {
306     final Invocable invocable;
307     if (engine instanceof Compilable) {
308       Compilable ceng = (Compilable) engine;
309       final CompiledScript predicateScript = ceng.compile(script);
310       if (predicateScript instanceof Invocable) {
311         invocable = (Invocable) predicateScript;
312       } else {
313         Object result = predicateScript.eval();
314         if (result instanceof Invocable) {
315           invocable = (Invocable) result;
316         } else {
317           throw new ScriptException("Script must evaluate to a Invocable/function, not: " + result);
318         }
319       }
320     } else {
321       Object result = engine.eval(script);
322       if (result instanceof Invocable) {
323         invocable = (Invocable) result;
324       } else {
325         throw new ScriptException("Script must evaluate to a Invocable/function, not: " + result);
326       }
327     }
328     return invocable;
329   }
