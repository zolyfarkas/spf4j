49  /**
50   * Token bucket algorithm base call rate limiter. see https://en.wikipedia.org/wiki/Token_bucket for more detail.
51   * Unlike the Guava implementation, the token replenishment is done in a separate thread.
52   * As such permit acquisition methods have lower overhead
53   * (no System.nanotime invocation is made when permits available),
54   * This lower overhead comes at the cost of increasing the cost of RateLimiter object instance. (a scheduled runnable)
55   * This implementation also uses CAS to update the permit bucket which should yield better concurrency characteristics.
56   * there is a backoff to handle high contention better as well
57   * (see https://dzone.com/articles/wanna-get-faster-wait-bit)
58   *
59   * The different in performance can be observed in a benchmark where we try to limit 10000000 ops/s:
60   *
61   * Benchmark                                                                 Mode  Cnt        Score        Error  Units
62   * com.google.common.util.concurrent.GuavaRateLimiterBenchmark.acquire      thrpt   10  8197739.576 ± 163437.214  ops/s
63   * org.spf4j.failsafe.Spf4jRateLimiterBenchmark.acquire                     thrpt   10  9791489.540 ±  89480.010  ops/s
64   *
65   * The Guava implementation cannot really get as close to the desired rate as the spf4j implementation.
66   *
67   * The spf4j implementation tops of at about:
68   *
69   * Benchmark                               Mode  Cnt         Score        Error  Units
70   * Spf4jRateLimiterBenchmark.acquire      thrpt   10  15270385.653 ± 507217.444  ops/s
71   *
72   * these benchmarks are artificial and really measure the overhead of permit acquisition alone without doing anything
73   * else, real use cases will accompany a permit acquision with some work which impacts the CAS/lock acquisition perf...
74   *
75   * Rate Limiter also implements the PermitSupplier abstraction along with GuavaRateLimiter allowing interchangeability
76   * between the 2 implementations based on what trade-of work better for you.
77   * PermitSupplier allows interchangeability and combination with Semaphore (extends PermitSupplier) implementations.
78   *
79   * @author Zoltan Farkas
80   */
81  @Beta
82  public final class RateLimiter
83          implements AutoCloseable, PermitSupplier {

85    private static final int RE_READ_TIME_AFTER_RETRIES = Integer.getInteger("spf4j.rateLimiter.reReadTimeAfterTries", 5);

87    private static final long DEFAULT_MIN_REPLENISH_INTERVAL_MS
88            = Long.getLong("spf4j.rateLimiter.defaultMinReplenishIntervalMs", 10L);

122   public RateLimiter(final double maxReqPerSecond,
123           final int maxBurstSize,
124           final ScheduledExecutorService scheduler,
125           final LongSupplier nanoTimeSupplier) {
126     this(maxReqPerSecond, maxBurstSize, DEFAULT_MIN_REPLENISH_INTERVAL_MS, scheduler, nanoTimeSupplier);
127   }
128 
129   public RateLimiter(final double maxReqPerSecond,
130           final int maxBurstSize,
131           final long minReplenishIntervalMillis,
132           final ScheduledExecutorService scheduler,
133           final LongSupplier nanoTimeSupplier) {
134     this.sync = new Object();
135     this.nanoTimeSupplier = nanoTimeSupplier;
136     double msPerReq = 1000d / maxReqPerSecond;
137     if (msPerReq < minReplenishIntervalMillis) {
138       msPerReq = minReplenishIntervalMillis;
139     }
140     this.permitReplenishIntervalMillis = (long) msPerReq;
141     this.permitsPerReplenishInterval = maxReqPerSecond * msPerReq / 1000;
142     assert permitsPerReplenishInterval >= 1;
143     if (maxBurstSize < permitsPerReplenishInterval) {
144       throw new IllegalArgumentException("Invalid max burst size: " + maxBurstSize
145               + ",  increase maxBurstSize to something larger than " + permitsPerReplenishInterval
146               + " we assume a clock resolution of " + permitReplenishIntervalMillis
147               + " and that is the minimum replenish interval");
148     }
149     this.permits = new AtomicLong(Double.doubleToRawLongBits(permitsPerReplenishInterval));
150     lastReplenishmentNanos = nanoTimeSupplier.getAsLong();
151     this.replenisher = scheduler.scheduleAtFixedRate(() -> {
152       synchronized (sync) {
153         Atomics.accumulate(permits, permitsPerReplenishInterval, (left, right) -> {
154           double result = left + right;
155           return (result > maxBurstSize) ? maxBurstSize : result;
156         });
157         lastReplenishmentNanos = nanoTimeSupplier.getAsLong();
158       }
159     }, permitReplenishIntervalMillis, permitReplenishIntervalMillis, TimeUnit.MILLISECONDS);
160   }
161 
162   /**
163    * Try to acquire a permit if available.
164    *
165    * @return true if permit acquired. false otherwise.
166    */
167   public boolean tryAcquire() {
168     return tryAcquire(1);
169   }
170 
171   public boolean tryAcquire(final int nrPermits) {
172     if (replenisher.isCancelled()) {
173       throw new IllegalStateException("RateLimiter is closed: " + this);
174     }
175     return Atomics.maybeAccumulate(permits, (double) nrPermits, (prev, x) -> {
176       double dif = prev - x;
177       // right will be -1d.
178       return (dif < 0) ? prev : dif;
179     });
180   }
181 
182   private final class ReservationHandler implements DoubleUnaryOperator {
183 
184     private long currTimeNanos;
185 
186     private final long deadlineNanos;
187 
188     private final int permits;
189 
190     private long msUntilResourcesAvailable;
191 
192     private int reTimeCount;
193 
194     ReservationHandler(final long currTimeNanos, final long timeoutMillis, final int permits) {
195       this.currTimeNanos = currTimeNanos;
196       this.deadlineNanos = currTimeNanos + timeoutMillis;
197       this.permits = permits;
198       this.msUntilResourcesAvailable = -1;
199       this.reTimeCount = RE_READ_TIME_AFTER_RETRIES;
200     }
201 
202     @Override
203     public double applyAsDouble(final double prev) {
204       double remaimingPermits = prev - permits;
205       if (remaimingPermits >= 0) {
206         return remaimingPermits;
207       }
208       if (reTimeCount > 0) {
209         reTimeCount--;
210       } else {
211         // re-reading current time on every try is pointless since System.nanotime can take 30ns to exec...
212         reTimeCount = RE_READ_TIME_AFTER_RETRIES;
213         currTimeNanos = nanoTimeSupplier.getAsLong();
214       }
215       long timeoutMs = TimeUnit.NANOSECONDS.toMillis(deadlineNanos - currTimeNanos);
216       if (timeoutMs <= 0) {
217         return prev;
218       }
219       long msUntilNextReplenishment
220               = permitReplenishIntervalMillis - TimeUnit.NANOSECONDS.toMillis(currTimeNanos - lastReplenishmentNanos);
221       if (timeoutMs < msUntilNextReplenishment) {
222           return prev; // not enough time to wait.
223       }
224       double permitsNeeded = -remaimingPermits;
225       if (permitsNeeded <= permitsPerReplenishInterval) {
226         msUntilResourcesAvailable = msUntilNextReplenishment;
227         return remaimingPermits;
228       } else {
229         int numberOfReplenishMentsNeed = (int) Math.ceil(permitsNeeded / permitsPerReplenishInterval);
230         long msNeeded = msUntilNextReplenishment
231                 + (numberOfReplenishMentsNeed - 1) * permitReplenishIntervalMillis;
232         if (msNeeded <= timeoutMs) {
233           msUntilResourcesAvailable = msNeeded;
234           return remaimingPermits;
235         } else {
236           return prev;
237         }
238       }
239     }
240 
241     public long getMsUntilResourcesAvailable() {
242       return msUntilResourcesAvailable;
243     }
244 
245     @Override
246     public String toString() {
247       return "ReservationHandler{" + "deadlineNanos=" + deadlineNanos + ", permits=" + permits
248               + ", msUntilResourcesAvailable=" + msUntilResourcesAvailable + '}';
249     }
250   }
251 
252   public double getNrPermits() {
253     return Double.longBitsToDouble(permits.get());
254   }
255 
256   @Override
257   @SuppressFBWarnings("MDM_THREAD_YIELD") //fb has a point here...
258   public boolean tryAcquire(final int nrPermits, final long timeout, final TimeUnit unit) throws InterruptedException {
259     long tryAcquireGetDelayMillis = tryAcquireGetDelayMillis(nrPermits, timeout, unit);
260     if (tryAcquireGetDelayMillis == 0) {
261       return true;
262     } else if (tryAcquireGetDelayMillis > 0) {
263       Thread.sleep(tryAcquireGetDelayMillis);
264       return true;
265     } else {
266       return false;
267     }
268   }
269 
270   /**
271    * @param nrPermits nr of permits to acquire
272    * @param timeout the maximum time to wait to acquire the permits.
273    * @param unit the unit of time to wait.
274    * @return negative value if cannot acquire number of permits within time, or if positive it is the amount of ms
275    * we can wait to act of the acquired permissions.
276    * the user of this method needs to be trusted, since it can violate the contract.
277    * @throws InterruptedException
278    */
279   @Signed
280   long tryAcquireGetDelayMillis(final int nrPermits, final long timeout, final TimeUnit unit)
281           throws InterruptedException {
282     boolean tryAcquire = tryAcquire(nrPermits);
283     if (tryAcquire) {
284       return 0L;
285     } else { // no curent permits available, reserve some slots and wait for them.
286       ReservationHandler rh = new ReservationHandler(nanoTimeSupplier.getAsLong(), unit.toNanos(timeout), nrPermits);
287       boolean accd;
288       synchronized (sync) {
289         accd = Atomics.maybeAccumulate(permits, rh);
290       }
291       if (accd) {
292         return rh.getMsUntilResourcesAvailable();
293       } else {
294         return -1L;
295       }
296     }
297   }
298 
299   @Override
300   public void close() {
301     replenisher.cancel(false);
302   }

309     return permitReplenishIntervalMillis;

312   public long getLastReplenishmentNanos() {
313     synchronized (sync) {
314       return lastReplenishmentNanos;
315     }
316   }

318   @Override
319   public String toString() {
320     return "RateLimiter{" + "permits=" + Double.longBitsToDouble(permits.get())

322             + permitsPerReplenishInterval + ", permitReplenishIntervalMillis=" + permitReplenishIntervalMillis + '}';
323   }
