32  package org.spf4j.failsafe;
33  
34  import com.google.common.util.concurrent.UncheckedExecutionException;
35  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
36  import java.util.concurrent.Callable;
37  import java.util.concurrent.TimeUnit;
38  import java.util.concurrent.TimeoutException;
39  import org.spf4j.base.Either;
40  import org.spf4j.base.Throwables;
41  
42  /**
43   * @author Zoltan Farkas
44   */
45  public final class SyncRetry {
46  
47    private SyncRetry() { }
48  
49    /**
50     * Naive implementation of execution with retry logic. a callable will be executed and retry attempted in current
51     * thread if the result and exception predicates. before retry, a callable can be executed that can abort the retry
52     * and finish the function with the previous result.
53     *
54     * @param <T> - The type of callable to retry result;
55     * @param <EX> - the exception thrown by the callable to retry.
56     * @param pwhat - the callable to retry.
57     * @return the result of the retried callable if successful.
58     * @throws java.lang.InterruptedException - thrown if retry interrupted.
59     * @throws EX - the exception thrown by callable.
60     */
61    @SuppressFBWarnings({ "MDM_THREAD_YIELD", "ITC_INHERITANCE_TYPE_CHECKING" })
62    public static <T, EX extends Exception, C extends Callable<? extends T>> T call(
63            final C pwhat,
64            final RetryPredicate<T, C> retryPredicate,
65            final Class<EX> exceptionClass,
66            final int maxExceptionChain)
67            throws InterruptedException, TimeoutException, EX {
68      C what = pwhat;
69      T result;
70      Exception lastEx; // last exception
71      try {
72        result = what.call();
73        lastEx = null;
74      } catch (InterruptedException ex1) {
75        throw ex1;
76      } catch (Exception e) { // only EX and RuntimeException
77        lastEx = e;
78        result = null;
79      }
80      Exception lastExChain = lastEx; // last exception chained with all previous exceptions
81      RetryDecision<T, C> decision;
82      //CHECKSTYLE IGNORE InnerAssignment FOR NEXT 5 LINES
83      while ((lastEx != null)
84              ? (decision = retryPredicate.getExceptionDecision(lastEx, what)).getDecisionType()
85                == RetryDecision.Type.Retry
86              : (decision = retryPredicate.getDecision(result, what)).getDecisionType() == RetryDecision.Type.Retry) {
87        if (Thread.interrupted()) {
88          Thread.currentThread().interrupt();
89          InterruptedException ex = new InterruptedException();
90          if (lastExChain != null) {
91            ex.addSuppressed(lastExChain);
92          }
93          throw ex;
94        }
95        long delayNanos = decision.getDelayNanos();
96        if (delayNanos > 0) {
97          TimeUnit.NANOSECONDS.sleep(delayNanos);
98        } else if (delayNanos < 0) {
99          throw new IllegalStateException("Invalid retry decision delay: " + delayNanos);
100       }
101       what = decision.getNewCallable();
102       try {
103         result = what.call();
104         lastEx = null;
105       } catch (InterruptedException ex1) {
106         if (lastExChain != null) {
107           ex1.addSuppressed(lastExChain);
108         }
109         throw ex1;
110       } catch (Exception e) { // only EX and RuntimeException
111         lastEx = e;
112         result = null;
113         if (lastExChain != null) {
114           lastExChain = Throwables.suppress(lastEx, lastExChain, maxExceptionChain);
115         } else {
116           lastExChain = lastEx;
117         }
118       }
119     }
120     if (decision.getDecisionType() == RetryDecision.Type.Abort) {
121         Either<Exception, T> r = decision.getResult();
122         if (r != null) {
123           if (r.isLeft()) {
124             lastEx = r.getLeft();
125             if (lastExChain != null) {
126               lastExChain = Throwables.suppress(lastEx, lastExChain, maxExceptionChain);
127             } else {
128               lastExChain = lastEx;
129             }
130           } else {
131             result = r.getRight();
132             lastEx = null;
133           }
134         }
135     } else {
136       throw new IllegalStateException("Should not happen, decision =  " + decision);
137     }
138     if (lastEx != null) {
139       if (lastExChain == null) {
140         lastExChain = lastEx;
141       }
142       if (lastExChain instanceof RuntimeException) {
143         throw (RuntimeException) lastExChain;
144       } else if (lastExChain instanceof TimeoutException) {
145         throw (TimeoutException) lastExChain;
146       } else if (exceptionClass.isAssignableFrom(lastExChain.getClass())) {
147         throw (EX) lastExChain;
148       } else {
149         throw new UncheckedExecutionException(lastExChain);
150       }
151     }
152     return result;
153   }
