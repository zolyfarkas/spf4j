34  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
35  import java.util.concurrent.Callable;
36  import java.util.concurrent.TimeUnit;
37  import java.util.concurrent.TimeoutException;
38  import javax.annotation.CheckReturnValue;
39  import org.spf4j.base.Either;
40  import org.spf4j.base.ExecutionContexts;
41  import org.spf4j.base.Throwables;
42  import org.spf4j.base.TimeSource;

45  /**
46   * @author Zoltan Farkas
47   */
48  public interface SyncRetryExecutor<T, C extends Callable<? extends T>> {
49  
50    @CheckReturnValue
51    <R extends T, W extends C, EX extends Exception> R call(W pwhat, Class<EX> exceptionClass,
52            long startNanos, long deadlineNanos)
53            throws InterruptedException, TimeoutException, EX;
54  
55    @CheckReturnValue
56    default <R extends T, W extends C, EX extends Exception> R call(W pwhat, Class<EX> exceptionClass,
57            long deadlineNanos)
58            throws InterruptedException, TimeoutException, EX {
59      return call(pwhat, exceptionClass, TimeSource.nanoTime(), deadlineNanos);
60    }
61  
62    @CheckReturnValue
63    default <R extends T, W extends C, EX extends Exception> R call(W pwhat, Class<EX> exceptionClass)
64            throws InterruptedException, TimeoutException, EX {
65      long nanoTime = TimeSource.nanoTime();
66      return call(pwhat, exceptionClass, nanoTime, ExecutionContexts.getContextDeadlineNanos());
67    }
68  
69    @CheckReturnValue
70    default <R extends T, W extends C, EX extends Exception> R call(W pwhat, Class<EX> exceptionClass,
71            long timeout, TimeUnit tu)
72            throws InterruptedException, TimeoutException, EX {
73      long nanoTime = TimeSource.nanoTime();
74      return call(pwhat, exceptionClass, nanoTime,
75              ExecutionContexts.computeDeadline(nanoTime, ExecutionContexts.current(), tu, timeout));
76    }
77  
78    default <W extends C, EX extends Exception> void run(W pwhat, Class<EX> exceptionClass)
79            throws InterruptedException, TimeoutException, EX {
80      T res = call(pwhat, exceptionClass);
81      if (res != null) {
82        throw new IllegalStateException("result must be null not " + res);
83      }
84    }
85  
86    default <W extends C, EX extends Exception> void run(W pwhat, Class<EX> exceptionClass, long deadlineNanos)
87            throws InterruptedException, TimeoutException, EX {
88      T res = call(pwhat, exceptionClass, deadlineNanos);
89      if (res != null) {
90        throw new IllegalStateException("result must be null not " + res);
91      }
92    }
93  
94    default <W extends C, EX extends Exception> void run(W pwhat, Class<EX> exceptionClass,
95            long timeout, TimeUnit tu)
96            throws InterruptedException, TimeoutException, EX {
97       run(pwhat, exceptionClass, ExecutionContexts.computeDeadline(ExecutionContexts.current(), tu, timeout));
98    }

101   /**
102    * Naive implementation of execution with retry logic. a callable will be executed and retry attempted in current
103    * thread if the result and exception predicates. before retry, a callable can be executed that can abort the retry
104    * and finish the function with the previous result.
105    *
106    * @param <T> - The type of callable to retry result;
107    * @param <EX> - the exception thrown by the callable to retry.
108    * @param pwhat - the callable to retry.
109    * @return the result of the retried callable if successful.
110    * @throws java.lang.InterruptedException - thrown if retry interrupted.
111    * @throws EX - the exception thrown by callable.
112    */
113   @SuppressFBWarnings({ "MDM_THREAD_YIELD", "ITC_INHERITANCE_TYPE_CHECKING" })
114   static <T, EX extends Exception, C extends Callable<? extends T>> T call(
115           final C pwhat,
116           final RetryPredicate<T, C> retryPredicate,
117           final Class<EX> exceptionClass,
118           final int maxExceptionChain)
119           throws InterruptedException, TimeoutException, EX {
120     C what = pwhat;
121     T result;
122     Exception lastEx; // last exception
123     try {
124       result = what.call();
125       lastEx = null;
126     } catch (InterruptedException ex1) {
127       throw ex1;
128     } catch (Exception e) { // only EX and RuntimeException
129       lastEx = e;
130       result = null;
131     }
132     Exception lastExChain = lastEx; // last exception chained with all previous exceptions
133     RetryDecision<T, C> decision;
134     //CHECKSTYLE IGNORE InnerAssignment FOR NEXT 5 LINES
135     while ((lastEx != null)
136             ? (decision = retryPredicate.getExceptionDecision(lastEx, what)).getDecisionType()
137               == RetryDecision.Type.Retry
138             : (decision = retryPredicate.getDecision(result, what)).getDecisionType() == RetryDecision.Type.Retry) {
139       if (Thread.interrupted()) {
140         Thread.currentThread().interrupt();
141         InterruptedException ex = new InterruptedException();
142         if (lastExChain != null) {
143           ex.addSuppressed(lastExChain);
144         }
145         throw ex;
146       }
147       long delayNanos = decision.getDelayNanos();
148       if (delayNanos > 0) {
149         TimeUnit.NANOSECONDS.sleep(delayNanos);
150       } else if (delayNanos < 0) {
151         throw new IllegalStateException("Invalid retry decision delay: " + delayNanos);
152       }
153       what = decision.getNewCallable();
154       try {
155         result = what.call();
156         lastEx = null;
157       } catch (InterruptedException ex1) {
158         if (lastExChain != null) {
159           ex1.addSuppressed(lastExChain);
160         }
161         throw ex1;
162       } catch (Exception e) { // only EX and RuntimeException
163         lastEx = e;
164         result = null;
165         if (lastExChain != null) {
166           lastExChain = Throwables.suppress(lastEx, lastExChain, maxExceptionChain);
167         } else {
168           lastExChain = lastEx;
169         }
170       }
171     }
172     if (decision.getDecisionType() == RetryDecision.Type.Abort) {
173         Either<Exception, T> r = decision.getResult();
174         if (r != null) {
175           if (r.isLeft()) {
176             lastEx = r.getLeft();
177             if (lastExChain != null) {
178               lastExChain = Throwables.suppress(lastEx, lastExChain, maxExceptionChain);
179             } else {
180               lastExChain = lastEx;
181             }
182           } else {
183             result = r.getRight();
184             lastEx = null;
185           }
186         }
187     } else {
188       throw new IllegalStateException("Should not happen, decision =  " + decision);
189     }
190     if (lastEx != null) {
191       if (lastExChain == null) {
192         lastExChain = lastEx;
193       }
194       if (lastExChain instanceof RuntimeException) {
195         throw (RuntimeException) lastExChain;
196       } else if (lastExChain instanceof TimeoutException) {
197         throw (TimeoutException) lastExChain;
198       } else if (exceptionClass.isAssignableFrom(lastExChain.getClass())) {
199         throw (EX) lastExChain;
200       } else {
201         throw new UncheckedExecutionException(lastExChain);
202       }
203     }
204     return result;
205   }
