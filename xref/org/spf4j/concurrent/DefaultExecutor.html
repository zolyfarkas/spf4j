18  package org.spf4j.concurrent;
19  
20  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
21  import java.util.List;
22  import java.util.concurrent.ExecutorService;
23  import java.util.concurrent.ForkJoinPool;
24  import java.util.concurrent.SynchronousQueue;
25  import java.util.concurrent.ThreadPoolExecutor;
26  import java.util.concurrent.TimeUnit;
27  import org.spf4j.base.AbstractRunnable;
28  import static org.spf4j.base.Runtime.WAIT_FOR_SHUTDOWN_MILLIS;
29  
30  /**
31   * This executor aims to be a general purpose executor for async tasks.
32   * (equivalent to ForkJoinPool.commonPool())
33   * @author zoly
34   */
35  @SuppressFBWarnings("HES_EXECUTOR_NEVER_SHUTDOWN") // THere is a shutdownhook being registered which FB does not see
36  public final class DefaultExecutor {
37  
38      private DefaultExecutor() {
39      }
40      
41      public static final ExecutorService INSTANCE;
42  
43      static {
44          final int coreThreads = Integer.getInteger("defaultExecutor.coreThreads", 0);
45          final int maxIdleMillis = Integer.getInteger("defaultExecutor.maxIdleMillis", 60000);
46          final boolean isDaemon = Boolean.getBoolean("defaultExecutor.daemon");
47          final String impParam = "defaultExecutor.implementation";
48          final String value = System.getProperty(impParam, "spf4j");
49          switch (value) {
50            case "spf4j":
51              LifoThreadPoolExecutorSQP lifoExec = new LifoThreadPoolExecutorSQP("defaultExecutor", coreThreads,
52                      Integer.MAX_VALUE, maxIdleMillis, 0, isDaemon, Integer.getInteger("defaultExecutor.spinlockCount",
53                              1024));
54              lifoExec.exportJmx();
55              INSTANCE = lifoExec;
56              break;
57            case "fjp": // EXPERIMENTAL! canceling with interrupt a future of taks submited does not seem to work!
58              INSTANCE = new ForkJoinPool(32767);
59              break;
60            case "legacy":
61              new ThreadPoolExecutor(coreThreads, Integer.MAX_VALUE, maxIdleMillis, TimeUnit.MILLISECONDS,
62                  new SynchronousQueue<Runnable>(), new CustomThreadFactory("DefaultExecutor", isDaemon));
63            default:
64              throw new IllegalArgumentException("Ivalid setting for " + impParam + " = " + value);
65          }
66  
67          org.spf4j.base.Runtime.queueHookAtEnd(new AbstractRunnable(true) {
68  
69              @Override
70              public void doRun() throws InterruptedException {
71                  INSTANCE.shutdown();
72                  INSTANCE.awaitTermination(WAIT_FOR_SHUTDOWN_MILLIS, TimeUnit.MILLISECONDS);
73                  List<Runnable> remaining = INSTANCE.shutdownNow();
74                  if (remaining.size() > 0) {
75                      System.err.println("Remaining tasks: " + remaining);
76                  }
77              }
78          });
79      }
80  
81  }
