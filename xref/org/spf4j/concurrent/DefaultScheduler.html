19  package org.spf4j.concurrent;
20  
21  import com.google.common.util.concurrent.MoreExecutors;
22  import java.util.concurrent.ScheduledExecutorService;
23  import java.util.concurrent.ScheduledFuture;
24  import java.util.concurrent.ScheduledThreadPoolExecutor;
25  import java.util.concurrent.TimeUnit;
26  
27  
28  /**
29   *
30   * @author zoly
31   */
32  public final class DefaultScheduler {
33  
34      private DefaultScheduler() { }
35      
36      
37      public static final ScheduledExecutorService INSTANCE =
38              MoreExecutors.getExitingScheduledExecutorService(
39              new ScheduledThreadPoolExecutor(2,
40              new CustomThreadFactory("DefaultScheduler", false)));
41  
42      
43      private static final long HOUR_MILLIS = 3600000;
44      
45      private static final long DAY_MILLIS = HOUR_MILLIS * 24;
46      
47      /**
48       * this will schedule a runnable aligned to the hour or day.
49       * @param command
50       * @param millisInterval
51       * @return Future that allows to cancel the schedule.
52       */
53      public static ScheduledFuture<?> scheduleAllignedAtFixedRateMillis(
54              final Runnable command, final long millisInterval) {
55         long currentTime = System.currentTimeMillis();
56         long nextScheduleTime;
57         if (millisInterval < HOUR_MILLIS) {
58              long millisPastHour = currentTime % HOUR_MILLIS;
59              nextScheduleTime = (millisPastHour / millisInterval + 1) * millisInterval + currentTime - millisPastHour;
60         } else {
61             long millisPastDay = currentTime % DAY_MILLIS;
62             nextScheduleTime = (millisPastDay / millisInterval + 1) * millisInterval + currentTime - millisPastDay;
63         }
64         return INSTANCE.scheduleAtFixedRate(
65                 command, nextScheduleTime - currentTime, millisInterval, TimeUnit.MILLISECONDS);
66      }
67  }
