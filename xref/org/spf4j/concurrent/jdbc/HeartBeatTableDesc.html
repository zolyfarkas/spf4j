3   import java.io.Serializable;
4   import org.spf4j.jdbc.DbType;

6   /*
7    * CREATE TABLE HEARTBEATS (
8    *  OWNER VARCHAR(255) NOT NULL,
9    *  INTERVAL_MILLIS bigint(20) NOT NULL,
10   *  LAST_HEARTBEAT_INSTANT_MILLIS bigint(20) NOT NULL,
11   *  PRIMARY KEY (OWNER),
12   *  UNIQUE KEY HEARTBEATS_PK (OWNER)
13   * );
14   *
15   * Table description for storing heartbeats for processes. (OWNER)
16   * The main purpose of thistable is to detect dead OWNERS (when their heart stops beating)
17   * OWNER = String column uniquely identifying a process.
18   * INTERVAL_MILLIS - the delay between heartbeats.
19   * LAST_HEARTBEAT_INSTANT_MILLIS - the millis since epoch when the last heartbeat happened.
20   */
21  public final class HeartBeatTableDesc  implements Serializable {
22  
23    private static final long serialVersionUID = 1L;
24  
25    private final String tableName;
26    private final String ownerColumn;
27    private final String intervalColumn;
28    private final String lastHeartbeatColumn;
29  
30    /**
31     * MSSQL = DATEDIFF(ms, '1970-01-01 00:00:00', GETUTCDATE())
32     * ORACLE = (SYSDATE - TO_DATE('01-01-1970 00:00:00', 'DD-MM-YYYY HH24:MI:SS')) * 24 * 3600000
33     * H2 = TIMESTAMPDIFF('MILLISECOND', timestamp '1970-01-01 00:00:00', CURRENT_TIMESTAMP())
34     */
35    private final String currentTimeMillisFunc;
36  
37    public HeartBeatTableDesc(final String tableName, final String ownerColun,
38            final String intervalColumn, final String lastHeartbeatColumn, final String currentTimeMillisFunc) {
39      this.tableName = tableName;
40      this.ownerColumn = ownerColun;
41      this.intervalColumn = intervalColumn;
42      this.lastHeartbeatColumn = lastHeartbeatColumn;
43      this.currentTimeMillisFunc = currentTimeMillisFunc;
44    }
45  
46    public String getTableName() {
47      return tableName;
48    }
49  
50    public String getOwnerColumn() {
51      return ownerColumn;
52    }
53  
54    public String getIntervalColumn() {
55      return intervalColumn;
56    }
57  
58    public String getLastHeartbeatColumn() {
59      return lastHeartbeatColumn;
60    }
61  
62    public String getCurrentTimeMillisFunc() {
63      return currentTimeMillisFunc;
64    }
65  
66    @Override
67    public String toString() {
68      return "HeartbeatTableDesc{" + "tableName=" + tableName + ", ownerColun=" + ownerColumn + ", intervalColumn="
69              + intervalColumn + ", lastHeartbeatColumn=" + lastHeartbeatColumn + '}';
70    }
71  
72  
73    /**
74     * Return the SQL for a current time millis since a EPOCH...
75     *
76     * @param dbType - the database type.
77     * @return - the sql fragment taht returns the current sql millis.
78     * @throws ExceptionInInitializerError
79     */
80    public static String getCurrTSSqlFn(final DbType dbType) throws ExceptionInInitializerError {
81      switch (dbType) {
82        case H2:
83          return "TIMESTAMPDIFF('MILLISECOND', timestamp '1970-01-01 00:00:00', CURRENT_TIMESTAMP())";
84        case ORACLE:
85          return "(SYSDATE - TO_DATE('01-01-1970 00:00:00', 'DD-MM-YYYY HH24:MI:SS')) * 24 * 3600000";
86        case MSSQL:
87          return "DATEDIFF(ms, '1970-01-01 00:00:00', GETUTCDATE())";
88        default:
89          throw new ExceptionInInitializerError("Database not supported");
90      }
91    }
92  
93  
94    public static final HeartBeatTableDesc DEFAULT = new HeartBeatTableDesc("HEARTBEATS",
95              "OWNER", "INTERVAL_MILLIS", "LAST_HEARTBEAT_INSTANT_MILLIS",
96              getCurrTSSqlFn(DbType.DEFAULT));
97  
98  }
