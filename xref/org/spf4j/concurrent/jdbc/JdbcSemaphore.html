49  /**
50   * A jdbc table based semaphore implementation. Similar with a semaphore implemented with zookeeper, we rely on
51   * heartbeats to detect dead members. If you have a zookeeper instance accessible you should probably use a semaphore
52   * implemented with it... If you are already connecting to a database. this should be a reliable and low overhead (no
53   * calls from DBA) implementation. (at leat that is my goal) Using a crappy database will give you crappy results.
54   *
55   * There are 3 tables involved:
56   *
57   * SEMAPHORES - keep track of available and total permits by semaphore. PERMITS_BY_OWNER - keeps track of all permits by
58   * owner. HEARTBEATS - keeps heartbeats by owner to detect - dead owners.
59   *
60   * All table names and columns are customizable to adapt this implementation to different naming conventions.
61   *
62   *
63   * @author zoly
64   */

153   @SuppressFBWarnings({"CBX_CUSTOM_BUILT_XML", "STT_TOSTRING_STORED_IN_FIELD"}) // no sql builder (yet)
154   public JdbcSemaphore(final DataSource dataSource, final SemaphoreTablesDesc semTableDesc,
155           final String semaphoreName, final int nrPermits, final int jdbcTimeoutSeconds,
156           final boolean strictReservations, final int acquirePollMillis) throws InterruptedException {
157     if (nrPermits < 0) {
158       throw new IllegalArgumentException("Permits must be positive and not " + nrPermits);
159     }
160     this.acquirePollMillis = acquirePollMillis;
161     this.semName = INTERNER.intern(semaphoreName);
162     this.jdbcTimeoutSeconds = jdbcTimeoutSeconds;
163     this.jdbc = new JdbcTemplate(dataSource);
164     this.semTableDesc = semTableDesc;
165     this.rnd = new IntMath.XorShift32();
166     this.isHealthy = true;
167     this.ownedReservations = 0;
168     this.failureHook = new JdbcHeartBeat.LifecycleHook() {
169       @Override
170       public void onError(final Error error) {
171         heartBeatFailure = error;
172         isHealthy = false;
173       }
174 
175       @Override
176       public void onClose() {
177           close();
178       }
179     };
180     this.heartBeat = JdbcHeartBeat.getHeartBeatAndSubscribe(dataSource,
181             semTableDesc.getHeartBeatTableDesc(), failureHook);
182     final String semaphoreTableName = semTableDesc.getSemaphoreTableName();
183     String availablePermitsColumn = semTableDesc.getAvailablePermitsColumn();
184     String lastModifiedByColumn = semTableDesc.getLastModifiedByColumn();
185     String lastModifiedAtColumn = semTableDesc.getLastModifiedAtColumn();
186     String ownerColumn = semTableDesc.getOwnerColumn();
187     String semaphoreNameColumn = semTableDesc.getSemNameColumn();
188     String totalPermitsColumn = semTableDesc.getTotalPermitsColumn();
189     String ownerPermitsColumn = semTableDesc.getOwnerReservationsColumn();
190     String permitsByOwnerTableName = semTableDesc.getPermitsByOwnerTableName();
191     HeartBeatTableDesc hbTableDesc = heartBeat.getHbTableDesc();
192     String heartBeatTableName = hbTableDesc.getTableName();
193     String heartBeatOwnerColumn = hbTableDesc.getOwnerColumn();
194     String currentTimeMillisFunc = hbTableDesc.getCurrentTimeMillisFunc();
195 
196     this.reducePermitsSql = "UPDATE " + semaphoreTableName + " SET "
197             + totalPermitsColumn + " = " + totalPermitsColumn + " - ?, "
198             + availablePermitsColumn + " = " + availablePermitsColumn + " - ? , "
199             + lastModifiedByColumn + " = ?, " + lastModifiedAtColumn + " = " + currentTimeMillisFunc + " WHERE "
200             + semaphoreNameColumn + " = ? AND "
201             + totalPermitsColumn + " >= ?";
202 
203     this.increasePermitsSql = "UPDATE " + semaphoreTableName + " SET "
204             + totalPermitsColumn + " = " + totalPermitsColumn + " + ?, "
205             + availablePermitsColumn + " = " + availablePermitsColumn + " + ?, "
206             + lastModifiedByColumn + " = ?, " + lastModifiedAtColumn + " = " + currentTimeMillisFunc + " WHERE "
207             + semaphoreNameColumn + " = ? ";
208 
209     this.updatePermitsSql = "UPDATE " + semaphoreTableName + " SET "
210             + totalPermitsColumn + " =  ?, "
211             + availablePermitsColumn + " =  " + availablePermitsColumn + " + ? - " + totalPermitsColumn + ','
212             + lastModifiedByColumn + " = ?, " + lastModifiedAtColumn + " = " + currentTimeMillisFunc + " WHERE "
213             + semaphoreNameColumn + " = ?";
214 
215     this.acquireSql = "UPDATE " + semaphoreTableName + " SET "
216             + availablePermitsColumn + " = " + availablePermitsColumn + " - ?, "
217             + lastModifiedByColumn + " = ?, " + lastModifiedAtColumn + " = " + currentTimeMillisFunc + " WHERE "
218             + semaphoreNameColumn + " = ? AND "
219             + availablePermitsColumn + " >= ?";
220     this.acquireByOwnerSql = "UPDATE " + permitsByOwnerTableName
221             + " SET " + ownerPermitsColumn + " = " + ownerPermitsColumn + " + ?, "
222             + lastModifiedAtColumn + " = " + currentTimeMillisFunc + " WHERE "
223             + ownerColumn + " = ? AND " + semaphoreNameColumn + " = ?";
224 
225     this.releaseSql = "UPDATE " + semaphoreTableName + " SET "
226             + availablePermitsColumn + " = CASE WHEN "
227             + availablePermitsColumn + " + ? > " + totalPermitsColumn
228             + " THEN " + totalPermitsColumn + " ELSE " + availablePermitsColumn + " + ? END, "
229             + lastModifiedByColumn + " = ?, " + lastModifiedAtColumn + " = " + currentTimeMillisFunc + " WHERE "
230             + semaphoreNameColumn + " = ?";
231 
232     this.releaseByOwnerSql = "UPDATE " + permitsByOwnerTableName
233             + " SET " + ownerPermitsColumn + " = " + ownerPermitsColumn
234             + " - ?, " + lastModifiedAtColumn + " = " + currentTimeMillisFunc + " WHERE "
235             + ownerColumn + " = ? AND " + semaphoreNameColumn + " = ? and " + ownerPermitsColumn + " >= ?";
236 
237     this.availablePermitsSql = "SELECT " + semTableDesc.getAvailablePermitsColumn()
238             + ',' + totalPermitsColumn + " FROM " + semTableDesc.getSemaphoreTableName()
239             + " WHERE " + semTableDesc.getSemNameColumn() + " = ?";
240 
241     this.totalPermitsSql = "SELECT " + totalPermitsColumn
242             + ',' + totalPermitsColumn + " FROM " + semTableDesc.getSemaphoreTableName()
243             + " WHERE " + semTableDesc.getSemNameColumn() + " = ?";
244 
245     this.ownedPermitsSql = "SELECT " + ownerPermitsColumn + " FROM "
246             + permitsByOwnerTableName + "WHERE "
247             + ownerColumn + " = ? AND " + semaphoreNameColumn + " = ?";
248 
249     this.deleteDeadOwnerRecordsSql = "DELETE FROM " + permitsByOwnerTableName + " RO "
250             + "WHERE RO." + semaphoreNameColumn + " = ? AND " + ownerPermitsColumn + " = 0 AND "
251             + "NOT EXISTS (select H." + heartBeatOwnerColumn + " from " + heartBeatTableName
252             + " H where H." + heartBeatOwnerColumn + " = RO." + ownerColumn + ')';
253 
254     this.getDeadOwnerPermitsSql = "SELECT " + ownerColumn + ", " + ownerPermitsColumn
255             + " FROM " + permitsByOwnerTableName + " RO "
256             + "WHERE RO." + semaphoreNameColumn + " = ? AND  " + ownerPermitsColumn + " > 0 AND "
257             + "NOT EXISTS (select H." + heartBeatOwnerColumn + " from " + heartBeatTableName
258             + " H where H." + heartBeatOwnerColumn + " = RO." + ownerColumn
259             + ") ORDER BY " + ownerColumn + ',' + ownerPermitsColumn;
260 
261     this.deleteDeadOwerRecordSql = "DELETE FROM " + permitsByOwnerTableName + " WHERE "
262             + ownerColumn + " = ? AND " + semaphoreNameColumn + " = ? AND "
263             + ownerPermitsColumn + " = ?";
264 
265     try {
266       createLockRowIfNotPresent(strictReservations, nrPermits);
267     } catch (SQLIntegrityConstraintViolationException ex) {
268       try {
269         // RACE condition while creating the row, will retry to validate if everything is OK.
270         createLockRowIfNotPresent(strictReservations, nrPermits);
271       } catch (SQLException ex1) {
272         RuntimeException rx = new RuntimeException(ex1);
273         rx.addSuppressed(ex);
274         throw rx;
275       }
276     } catch (SQLException ex) {
277       throw new RuntimeException(ex);
278     }
279     try {
280       createOwnerRow();
281     } catch (SQLException ex) {
282       throw new RuntimeException(ex);
283     }
284   }

300   void createLockRowIfNotPresent(final boolean strictReservations, final int nrPermits)
301           throws SQLException, InterruptedException {
302     final String lastModifiedByColumn = semTableDesc.getLastModifiedByColumn();
303     final String lastModifiedAtColumn = semTableDesc.getLastModifiedAtColumn();
304     final String tableName = semTableDesc.getSemaphoreTableName();
305     final String semNameColumn = semTableDesc.getSemNameColumn();
306     final String availableReservationsColumn = semTableDesc.getAvailablePermitsColumn();
307     final String maxReservationsColumn = semTableDesc.getTotalPermitsColumn();
308 
309     jdbc.transactOnConnection((final Connection conn, final long deadlineNanos) -> {
310       try (final PreparedStatement stmt = conn.prepareStatement("SELECT " + availableReservationsColumn
311               + ',' + maxReservationsColumn + " FROM " + tableName
312               + " WHERE " + semNameColumn + " = ?")) {
313         stmt.setNString(1, semName);
314         stmt.setQueryTimeout((int) TimeUnit.NANOSECONDS.toSeconds(deadlineNanos - System.nanoTime()));
315         try (final ResultSet rs = stmt.executeQuery()) {
316           if (!rs.next()) {
317             try (final PreparedStatement insert = conn.prepareStatement("insert into " + tableName
318                     + " (" + semNameColumn + ',' + availableReservationsColumn + ',' + maxReservationsColumn
319                     + ',' + lastModifiedByColumn + ',' + lastModifiedAtColumn + ") VALUES (?, ?, ?, ?, "
320                     + heartBeat.getHbTableDesc().getCurrentTimeMillisFunc() + ')')) {
321               insert.setNString(1, semName);
322               insert.setInt(2, nrPermits);
323               insert.setInt(3, nrPermits);
324               insert.setNString(4, org.spf4j.base.Runtime.PROCESS_ID);
325               insert.setQueryTimeout((int) TimeUnit.NANOSECONDS.toSeconds(deadlineNanos - System.nanoTime()));
326               insert.executeUpdate();
327             }
328           } else if (strictReservations) { // there is a record already. for now blow up if different nr reservations.
329             int existingMaxReservations = rs.getInt(2);
330             if (existingMaxReservations != nrPermits) {
331               throw new IllegalArgumentException("Semaphore " + semName + " max reservations count different "
332                       + existingMaxReservations + " != " + nrPermits + " use different semaphore");
333             }
334             if (rs.next()) {
335               throw new IllegalStateException("Cannot have mutiple semaphores with the same name " + semName);
336             }
337           } else if (rs.next()) {
338             throw new IllegalStateException("Cannot have mutiple semaphores with the same name " + semName);
339           }
340         }
341       }
342       return null;
343     }, jdbcTimeoutSeconds, TimeUnit.SECONDS);
344   }

396   @SuppressFBWarnings("UW_UNCOND_WAIT")
397   @CheckReturnValue
398   @Override
399   public boolean tryAcquire(final int nrPermits, final long timeout, final TimeUnit unit)
400           throws InterruptedException {
401     if (nrPermits < 1) {
402       throw new IllegalArgumentException("You should try to acquire something! not " + nrPermits);
403     }
404     synchronized (semName) {
405       long deadlineNanos = System.nanoTime() + unit.toNanos(timeout);
406       boolean acquired = false;
407       final MutableHolder<Boolean> beat = MutableHolder.of(Boolean.FALSE);
408       do {
409         validate();
410         try {
411           acquired = jdbc.transactOnConnection(new HandlerNano<Connection, Boolean, SQLException>() {
412             @Override
413             public Boolean handle(final Connection conn, final long deadlineNanos) throws SQLException {
414               try (PreparedStatement stmt = conn.prepareStatement(acquireSql)) {
415                 stmt.setQueryTimeout(nanosToSeconds(deadlineNanos - System.nanoTime()));
416                 stmt.setInt(1, nrPermits);
417                 stmt.setNString(2, org.spf4j.base.Runtime.PROCESS_ID);
418                 stmt.setNString(3, semName);
419                 stmt.setInt(4, nrPermits);
420                 int rowsUpdated = stmt.executeUpdate();
421                 Boolean acquired;
422                 if (rowsUpdated == 1) {
423                   try (PreparedStatement ostmt = conn.prepareStatement(acquireByOwnerSql)) {
424                     ostmt.setInt(1, nrPermits);
425                     ostmt.setNString(2, org.spf4j.base.Runtime.PROCESS_ID);
426                     ostmt.setNString(3, semName);
427                     ostmt.setQueryTimeout(nanosToSeconds(deadlineNanos - System.nanoTime()));
428                     int nrUpdated = ostmt.executeUpdate();
429                     if (nrUpdated != 1) {
430                       throw new IllegalStateException("Updated " + nrUpdated + " is incorrect for " + ostmt);
431                     }
432                   }
433                   acquired = Boolean.TRUE;
434                 } else {
435                   if (rowsUpdated > 1) {
436                     throw new IllegalStateException("Too many rows updated! when trying to acquire " + nrPermits);
437                   }
438                   acquired = Boolean.FALSE;
439                 }
440                 if (deadlineNanos - System.nanoTime() > heartBeat.getBeatDurationNanos()) {
441                   // do a heartbeat if have time, and if it makes sense.
442                   beat.setValue(heartBeat.tryBeat(conn, deadlineNanos));
443                 }
444                 return acquired;
445               }
446             }
447           }, timeout, unit);
448         } catch (SQLException ex) {
449           throw new RuntimeException(ex);
450         }
451         if (beat.getValue()) { // we did a heartbeat as part of the acquisition.
452           heartBeat.updateLastRun(System.currentTimeMillis());
453         }
454         if (!acquired) {
455           Future<Integer> fut = DefaultExecutor.INSTANCE.submit(new Callable<Integer>() {
456             @Override
457             public Integer call() throws Exception {
458               return removeDeadHeartBeatAndNotOwnerRows(60);
459             }
460           });
461           try {
462             fut.get(deadlineNanos - System.nanoTime(), TimeUnit.NANOSECONDS);
463           } catch (TimeoutException ex) {
464             //removing dead entries did not finish in time, but continues in the background.
465             break;
466           } catch (ExecutionException ex) {
467             throw new RuntimeException(ex);
468           }
469           if (releaseDeadOwnerPermits(nrPermits) <= 0) { //wait of we did not find anything dead to release.
470             long wtimeMilis = Math.min(TimeUnit.NANOSECONDS.toMillis(deadlineNanos - System.nanoTime()),
471                     Math.abs(rnd.nextInt()) % acquirePollMillis);
472             if (wtimeMilis > 0) {
473               semName.wait(wtimeMilis);
474             } else {
475               break;
476             }
477           }
478 
479         }
480       } while (!acquired && deadlineNanos > System.nanoTime());
481       if (acquired) {
482         ownedReservations += nrPermits;
483       }
484 
485       return acquired;
486     }
487   }

494   @Override
495   @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CHECKED")
496   public void release(final int nrReservations) {
497     synchronized (semName) {
498       try {
499         jdbc.transactOnConnectionNonInterrupt(new HandlerNano<Connection, Void, SQLException>() {
500           @Override
501           public Void handle(final Connection conn, final long deadlineNanos) throws SQLException {
502             releaseReservations(conn, deadlineNanos, nrReservations);
503             try (PreparedStatement ostmt = conn.prepareStatement(releaseByOwnerSql)) {
504               ostmt.setInt(1, nrReservations);
505               ostmt.setNString(2, org.spf4j.base.Runtime.PROCESS_ID);
506               ostmt.setNString(3, semName);
507               ostmt.setInt(4, nrReservations);
508               ostmt.setQueryTimeout(nanosToSeconds(deadlineNanos - System.nanoTime()));
509               int nrUpdated = ostmt.executeUpdate();
510               if (nrUpdated != 1) {
511                 throw new IllegalStateException("Trying to release more than you own! " + ostmt);
512               }
513             }
514             return null;
515           }
516         }, jdbcTimeoutSeconds, TimeUnit.SECONDS);
517       } catch (SQLException ex) {
518         throw new RuntimeException(ex);
519       }
520       ownedReservations -= nrReservations;
521       if (ownedReservations < 0) {
522         throw new IllegalStateException("Should not be trying to release more than you acquired!" + nrReservations);
523       }
524       semName.notifyAll();
525     }
526   }
527 
528   public void releaseAll() {
529     synchronized (semName) {
530       release(ownedReservations);
531     }
532   }

534   private void releaseReservations(final Connection conn, final long deadlineNanos, final int nrReservations)
535           throws SQLException {
536     try (PreparedStatement stmt = conn.prepareStatement(releaseSql)) {
537       stmt.setQueryTimeout(nanosToSeconds(deadlineNanos - System.nanoTime()));
538       stmt.setInt(1, nrReservations);
539       stmt.setInt(2, nrReservations);
540       stmt.setNString(3, org.spf4j.base.Runtime.PROCESS_ID);
541       stmt.setNString(4, semName);
542       stmt.executeUpdate(); // Since a release might or might not update a row.
543     }
544   }

547   public int availablePermits() throws SQLException, InterruptedException {
548     return jdbc.transactOnConnection((final Connection conn, final long deadlineNanos) -> {
549       try (final PreparedStatement stmt = conn.prepareStatement(availablePermitsSql)) {
550         stmt.setNString(1, semName);
551         stmt.setQueryTimeout((int) TimeUnit.NANOSECONDS.toSeconds(deadlineNanos - System.nanoTime()));
552         try (final ResultSet rs = stmt.executeQuery()) {
553           if (!rs.next()) {
554             throw new IllegalStateException();
555           } else {
556             int result = rs.getInt(1);
557             if (rs.next()) {
558               throw new IllegalStateException();
559             }
560             return result;
561           }
562         }
563       }
564     }, jdbcTimeoutSeconds, TimeUnit.SECONDS);

567   @JmxExport(description = "get the number of permits owned by this process")

613   List<OwnerPermits> getDeadOwnerPermits(final Connection conn, final long deadlineNanos, final int wishPermits)
614           throws SQLException {
615     List<OwnerPermits> result = new ArrayList<>();
616     try (final PreparedStatement stmt = conn.prepareStatement(getDeadOwnerPermitsSql)) {
617       stmt.setNString(1, semName);
618       stmt.setQueryTimeout((int) TimeUnit.NANOSECONDS.toSeconds(deadlineNanos - System.nanoTime()));
619       try (final ResultSet rs = stmt.executeQuery()) {
620         int nrPermits = 0;
621         while (rs.next()) {
622           OwnerPermits ownerPermit = new OwnerPermits(rs.getNString(1), rs.getInt(2));
623           result.add(ownerPermit);
624           nrPermits += ownerPermit.getNrPermits();
625           if (nrPermits >= wishPermits) {
626             break;
627           }
628         }
629       }
630     }
631     return result;
632   }

634   /**
635    * Attempts to release permits for this semaphore owned by dead owners.
636    *
637    * @param wishPermits - How many permits we would like to get released.
638    * @return - the number of permits we actually released.
639    * @throws SQLException - something went wrong with the db.
640    * @throws InterruptedException - thrown if thread is interrupted.
641    */
642   @JmxExport(description = "release dead owner permits")
643   @CheckReturnValue
644   public int releaseDeadOwnerPermits(@JmxExport(value = "wishPermits",
645           description = "how many we whish to release") final int wishPermits)
646           throws InterruptedException {
647     try {
648       return jdbc.transactOnConnection((final Connection conn, final long deadlineNanos) -> {
649         List<OwnerPermits> deadOwnerPermits = getDeadOwnerPermits(conn, deadlineNanos, wishPermits);
650         int released = 0;
651         for (OwnerPermits permit : deadOwnerPermits) {
652           try (final PreparedStatement stmt = conn.prepareStatement(deleteDeadOwerRecordSql)) {
653             String owner = permit.getOwner();
654             stmt.setNString(1, owner);
655             stmt.setNString(2, semName);
656             int nrPermits = permit.getNrPermits();
657             stmt.setInt(3, nrPermits);
658             stmt.setQueryTimeout((int) TimeUnit.NANOSECONDS.toSeconds(deadlineNanos - System.nanoTime()));
659             if (stmt.executeUpdate() == 1) { // I can release! if not somebody else is doing it.
660               released += nrPermits;
661               releaseReservations(conn, deadlineNanos, nrPermits);
662               LOG.warn("Released {} reservations from dead owner {}", nrPermits, owner);
663             }
664           }
665         }
666         return released;
667       }, jdbcTimeoutSeconds, TimeUnit.SECONDS);
668     } catch (SQLException ex) {
669       throw new RuntimeException(ex);
670     }
671   }
