50  /**
51   * A JMX managed bean Builder.
52   *
53   * @author Zoltan Farkas
54   */
55  @ParametersAreNonnullByDefault
56  @SuppressFBWarnings("FCCD_FIND_CLASS_CIRCULAR_DEPENDENCY")
57  public final class DynamicMBeanBuilder {
58  
59    private final Map<String, ExportedValue<?>> exportedAttributes;
60  
61    private final Map<String, ExportedOperation> exportedOps;
62  
63    public DynamicMBeanBuilder() {
64      exportedAttributes = new HashMap<>(8);
65      exportedOps = new HashMap<>(4);
66    }
67  
68    public static DynamicMBeanBuilder newBuilder() {
69      return new DynamicMBeanBuilder();
70    }
71  
72    public DynamicMBeanBuilder withOperation(final ExportedOperation operation) {
73      if (exportedOps.put(operation.getName(), operation) != null) {
74         throw new IllegalArgumentException("Duplicate operation: " + operation);
75      }
76      return this;
77    }
78  
79    public DynamicMBeanBuilder withAttribute(final ExportedValue<?> val) {
80      if (exportedAttributes.put(val.getName(), val) != null) {
81        throw new IllegalArgumentException("Duplicate attribute: " + val);
82      }
83      return this;
84    }
85  
86    public DynamicMBeanBuilder withAttributes(final ExportedValue<?>... vals) {
87      for (ExportedValue<?> ev : vals) {
88        withAttribute(ev);
89      }
90      return this;
91    }
92  
93    public DynamicMBeanBuilder withAttributes(final Map<String, Object> mapAttributes) {
94      for (Map.Entry<String, Object> entry : mapAttributes.entrySet()) {
95        String key = entry.getKey();
96        if (key == null || key.isEmpty()) {
97          continue; // do not exportAgg crap namep mapAttributes.
98        }
99        try {
100         if (exportedAttributes.put(key, new MapExportedValue(mapAttributes, null, key, entry.getValue())) != null) {
101           throw new IllegalArgumentException("Duplicate attribute: " + key);
102         }
103       } catch (NotSerializableException ex) {
104         throw new IllegalArgumentException("Unable to export map entry via JMX: " + entry, ex);
105       }
106     }
107     return this;
108   }

110   /**
111    * add all exported (with @JmxExport) attributes and operations of the object.
112    * @param object
113    * @return
114    */
115   public DynamicMBeanBuilder withJmxExportObject(final Object object) {
116     final Map<String, ExportedValue<?>> attrs = new HashMap<>(4);
117     if (object instanceof Class) {
118       for (Method method : ((Class<?>) object).getMethods()) {
119         if (method.isSynthetic()) {
120           continue;
121         }
122         if (Modifier.isStatic(method.getModifiers())) {
123           JmxExport annot = method.getAnnotation(JmxExport.class);
124           if (annot != null) {
125             exportMethod(method, null, attrs, exportedOps, annot);
126           }
127         }
128       }
129     } else {
130       final Class<? extends Object> oClass = object.getClass();
131       for (Method method : oClass.getMethods()) {
132         if (method.isSynthetic()) {
133           continue;
134         }
135         JmxExport annot = method.getAnnotation(JmxExport.class);
136         if (annot != null) {
137           exportMethod(method, object, attrs, exportedOps, annot);
138         }
139       }
140     }
141     Collection<ExportedValue<?>> values = attrs.values();
142     for (ExportedValue expVal : values) {
143       if (!((BeanExportedValue) expVal).isValid()) {
144         throw new IllegalArgumentException("If setter is exported, getter must be exported as well " + expVal);
145       }
146     }
147     exportedAttributes.putAll(attrs);
148     return this;
149   }
150 
151   /**
152    * add all exported (with @JmxExport) attributes and operations of the objects.
153    * @param objects
154    * @return
155    */
156   public DynamicMBeanBuilder withJmxExportObjects(final Object... objects) {
157     for (Object object : objects) {
158       withJmxExportObject(object);
159     }
160     return this;
161   }

163   /**
164    * Build the dynamic mbean.
165    * @param packageName
166    * @param mbeanName
167    * @return dynamic mbean or null if no exportable attributes or operations are present.
168    */
169   @Nullable
170   public ExportedValuesMBean build(final String packageName, final String mbeanName) {
171     if (exportedAttributes.isEmpty() && exportedOps.isEmpty()) {
172       return null;
173     }
174     ObjectName objectName = ExportedValuesMBean.createObjectName(packageName, mbeanName);
175     return new ExportedValuesMBean(objectName, exportedAttributes, exportedOps);
176   }
177 
178   /**
179    * Create a dynamic bean with extends the toExtend mbean with the attributes and operations from thsi builder.
180    * @param toExtend
181    * @return
182    */
183   @Nonnull
184   public ExportedValuesMBean extend(@Nonnull final ExportedValuesMBean toExtend) {
185     if (exportedAttributes.isEmpty() && exportedOps.isEmpty()) {
186       return toExtend;
187     }
188     return new ExportedValuesMBean(toExtend, exportedAttributes, exportedOps);
189   }
190 
191 
192   /**
193    * extend existing Mbean registered with provided packageName and mbeanName
194    * @param packageName
195    * @param mbeanName
196    * @return null is nothing was registered.
197    */
198   @Nullable
199   public ExportedValuesMBean extend(final String packageName, final String mbeanName) {
200     if (exportedAttributes.isEmpty() && exportedOps.isEmpty()) {
201       return null;
202     }
203     ObjectName objectName = ExportedValuesMBean.createObjectName(packageName, mbeanName);
204     synchronized (Registry.class) {
205       ExportedValuesMBean existing = (ExportedValuesMBean) Registry.unregister(objectName);
206       ExportedValuesMBean mbean;
207       if (existing == null) {
208         mbean = new ExportedValuesMBean(objectName, exportedAttributes, exportedOps);
209       } else {
210         mbean = new ExportedValuesMBean(existing, exportedAttributes, exportedOps);
211       }
212       Registry.registerMBean(objectName, mbean);
213       return mbean;
214     }
215 
216   }

219     return exportedAttributes.isEmpty() && exportedOps.isEmpty();
220   }

222   /**
223    * Replace mbean registered with packageName and mbeanName with a mbean constructed by this builder.
224    * @param packageName
225    * @param mbeanName
226    * @return the MBean that was registered.
227    */
228   @Nonnull
229   public ExportedValuesMBean replace(final String packageName, final String mbeanName) {
230     if (isEmpty()) {
231       throw new IllegalArgumentException("Nothing to register " + this);
232     }
233     ObjectName objectName = ExportedValuesMBean.createObjectName(packageName, mbeanName);
234     ExportedValuesMBean mbean = new ExportedValuesMBean(objectName, exportedAttributes, exportedOps);
235     Registry.registerMBean(objectName, mbean);
236     return mbean;
237   }

245     ExportedValuesMBean mbean = new ExportedValuesMBean(objectName, exportedAttributes, exportedOps);
246     Registry.registerMBean(objectName, mbean);
247     return mbean;
248   }

283 static void exportMethod(final Method method,
284           @Nullable final Object object, final Map<String, ExportedValue<?>> exportedAttributes,
285           final Map<String, ExportedOperation> exportedOps, final JmxExport annot) {
286     AccessController.doPrivileged((PrivilegedAction<Void>) () -> {
287       method.setAccessible(true);
288       return null;
289     });
290     String methodName = method.getName();
291     int nrParams = method.getParameterCount();
292     if (nrParams == 0 && methodName.startsWith("get")) {
293       String valueName = methodName.substring("get".length());
294       valueName = Strings.withFirstCharLower(valueName);
295       addGetter(valueName, exportedAttributes, annot, method, object);
296     } else if (nrParams == 0 && methodName.startsWith("is")) {
297       String valueName = methodName.substring("is".length());
298       valueName = Strings.withFirstCharLower(valueName);
299       addGetter(valueName, exportedAttributes, annot, method, object);
300     } else if (nrParams == 1 && methodName.startsWith("set")) {
301       addSetter(methodName, exportedAttributes, method, object, annot);
302     } else {
303       String opName = methodName;
304       String nameOverwrite = annot.value();
305       if (!"".equals(nameOverwrite)) {
306         opName = nameOverwrite;
307       }
308       ExportedOperation existing = exportedOps.put(opName, new ExportedOperationImpl(opName,
309               annot.description(), method, object, annot.mapOpenType()));
310       if (existing != null) {
311         throw new IllegalArgumentException("exporting operations with same name not supported: " + opName);
312       }
313     }
314   }
315 
316   private static void addSetter(final String methodName, final Map<String, ExportedValue<?>> exportedAttributes,
317           final Method method, final Object object, final JmxExport annot) {
318     String customName = annot.value();
319     String valueName;
320     if ("".equals(customName)) {
321       valueName = methodName.substring("set".length());
322       valueName = Strings.withFirstCharLower(valueName);
323     } else {
324       valueName = customName;
325     }
326 
327     BeanExportedValue existing = (BeanExportedValue) exportedAttributes.get(valueName);
328     Class<?> parameterType = Reflections.getParameterTypes(method)[0];
329     if (existing == null) {
330       existing = new BeanExportedValue(valueName, null,
331               null, method, object, parameterType, annot.mapOpenType());
332     } else {
333       if (existing.getValueType() != parameterType) {
334         throw new IllegalArgumentException(
335                 "Getter and setter icorrectly defined " + existing + ' ' + method);
336       }
337       existing = existing.withSetter(method);
338     }
339     exportedAttributes.put(valueName, existing);
340   }
341 
342   private static void addGetter(final String pvalueName,
343           final Map<String, ExportedValue<?>> exported,
344           final JmxExport annot, final Method method, final Object object) {
345     String customName = annot.value();
346     String valueName = "".equals(customName) ? pvalueName : customName;
347     BeanExportedValue existing = (BeanExportedValue) exported.get(valueName);
348     if (existing == null) {
349       existing = new BeanExportedValue(
350               valueName, annot.description(),
351               method, null, object, method.getGenericReturnType(), annot.mapOpenType());
352     } else {
353       if (existing.getValueType() != method.getReturnType()) {
354         throw new IllegalArgumentException(
355                 "Getter and setter icorrectly defined " + existing + ' ' + method);
356       }
357       existing = existing.withGetter(method, annot.description());
358     }
359     exported.put(valueName, existing);
360   }

362   @Override
363   public String toString() {
364     return "MBeanBuilder{" + "exportedAttributes=" + exportedAttributes
365             + ", exportedOps=" + exportedOps + '}';
366   }
