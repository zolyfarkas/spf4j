42      private static final MBeanServer MBEAN_SERVER = ManagementFactory.getPlatformMBeanServer();
43  
44      private static void register(final ObjectName objectName, final Object mbean) {
45          if (MBEAN_SERVER.isRegistered(objectName)) {
46              try {
47                  MBEAN_SERVER.unregisterMBean(objectName);
48              } catch (InstanceNotFoundException ex) {
49                  throw new RuntimeException(ex);
50              } catch (MBeanRegistrationException ex) {
51                  throw new RuntimeException(ex);
52              }
53          }
54          try {
55              MBEAN_SERVER.registerMBean(mbean, objectName);
56          } catch (InstanceAlreadyExistsException ex) {
57              throw new RuntimeException(ex);
58          } catch (MBeanRegistrationException ex) {
59              throw new RuntimeException(ex);
60          } catch (NotCompliantMBeanException ex) {
61              throw new RuntimeException(ex);
62          }
63      }
64  
65      static void register(final String domain, final String name, final Object object) {
66          register(ExportedValuesMBean.createObjectName(domain, name), object);
67      }

69      public static void unregister(final Object object) {
70          final Class<? extends Object> aClass = object.getClass();
71          unregister(aClass.getPackage().getName(), aClass.getSimpleName());
72      }
73  
74      public static void unregister(final Class<?> object) {
75          unregister(object.getPackage().getName(), object.getSimpleName());
76      }

78      private static void unregister(final String packageName, final String mbeanName) {
79          ObjectName objectName = ExportedValuesMBean.createObjectName(packageName, mbeanName);
80          if (MBEAN_SERVER.isRegistered(objectName)) {
81              try {
82                  MBEAN_SERVER.unregisterMBean(objectName);
83              } catch (InstanceNotFoundException ex) {
84                  throw new RuntimeException(ex);
85              } catch (MBeanRegistrationException ex) {
86                  throw new RuntimeException(ex);
87              }
88          }
89          try {
90              MBEAN_SERVER.unregisterMBean(objectName);
91          } catch (InstanceNotFoundException ex) {
92              throw new RuntimeException(ex);
93          } catch (MBeanRegistrationException ex) {
94              throw new RuntimeException(ex);
95          }
96      }

107     public static void export(final String packageName, final String mbeanName, final Object ... objects) {
108         
109         Map<String, ExportedValueImpl> exportedAttributes = new HashMap<String, ExportedValueImpl>();
110         Map<String, ExportedOperationImpl> exportedOps = new HashMap<String, ExportedOperationImpl>();
111         boolean haveToPrependClass = objects.length > 1;
112         for (Object object : objects) {
113  
114             if (object instanceof Class) {
115                 for (Method method : ((Class<?>) object).getMethods()) {
116                     if (Modifier.isStatic(method.getModifiers())) {
117                         Annotation [] annotations = method.getAnnotations();
118                         for (Annotation annot : annotations) {
119                             if (annot.annotationType() == JmxExport.class) {
120                                 exportMethod(method, haveToPrependClass ? ((Class) object).getSimpleName() : null,
121                                         null, exportedAttributes, exportedOps, annot);
122                             }
123                         }
124                     }
125                 }
126             } else {
127                 final Class<? extends Object> oClass = object.getClass();
128                 String oClassName = oClass.getSimpleName();
129                 for (Method method : oClass.getMethods()) {
130                     Annotation [] annotations = method.getAnnotations();
131                     for (Annotation annot : annotations) {
132                         if (annot.annotationType() == JmxExport.class) {
133                             exportMethod(method, haveToPrependClass ? oClassName : null,
134                                     object, exportedAttributes, exportedOps, annot);
135                         }
136                     }
137                 }
138             }
139             
140         }
141         if (exportedAttributes.isEmpty() && exportedOps.isEmpty()) {
142             return;
143         }
144         ExportedValue<?> [] values = new ExportedValue[exportedAttributes.size()];
145         int i = 0;
146         for (ExportedValueImpl expVal : exportedAttributes.values()) {
147             if (expVal.isValid()) {
148                 values[i++] = expVal;
149             } else {
150                 throw new IllegalArgumentException("If setter is exported, getter must be exported as well " + expVal);
151             }
152         }
153         
154         ExportedValuesMBean mbean = new ExportedValuesMBean(packageName, mbeanName, values,
155                         exportedOps.values().toArray(new ExportedOperation [exportedOps.size()]));
156         register(mbean.getObjectName(), mbean);
157     }

159     private static void exportMethod(final Method method, @Nullable final String prependClass,
160             @Nullable final Object object, final Map<String, ExportedValueImpl> exportedAttributes,
161             final Map<String, ExportedOperationImpl> exportedOps, final Annotation annot) {
162         method.setAccessible(true); // this is to speed up invocation
163         String methodName = method.getName();
164         int nrParams = method.getParameterTypes().length;
165         if (methodName.startsWith("get") && nrParams == 0) {
166             String valueName = methodName.substring("get".length());
167             valueName = Strings.withFirstCharLower(valueName);
168             if (prependClass != null) {
169                 valueName = prependClass + "." + valueName;
170             }
171             addGetter(valueName, exportedAttributes, annot, method, object);
172         } else if (methodName.startsWith("is") && nrParams == 0) {
173             String valueName = methodName.substring("is".length());
174             valueName = Strings.withFirstCharLower(valueName);
175             if (prependClass != null) {
176                 valueName = prependClass + "." + valueName;
177             }
178             addGetter(valueName, exportedAttributes, annot, method, object);
179         } else if (methodName.startsWith("set") && nrParams == 1) {
180             String valueName = methodName.substring("set".length());
181             valueName = Strings.withFirstCharLower(valueName);
182             if (prependClass != null) {
183                 valueName = prependClass + "." + valueName;
184             }
185             ExportedValueImpl existing = exportedAttributes.get(valueName);
186             if (existing == null) {
187                 existing = new ExportedValueImpl(valueName, null,
188                         null, method, object, method.getParameterTypes()[0]);
189             } else {
190                 if (existing.getValueClass() != method.getParameterTypes()[0]) {
191                     throw new IllegalArgumentException(
192                             "Getter and setter icorrectly defined " + existing + " " + method);
193                 }
194                 existing = existing.withSetter(method);
195             }
196             exportedAttributes.put(valueName, existing);
197         } else {
198             String opName = methodName;
199             String nameOverwrite = (String) Reflections.getAnnotationAttribute(annot, "name");
200             if (!"".equals(nameOverwrite)) {
201                 opName = nameOverwrite;
202             }
203             if (prependClass != null) {
204                 opName = prependClass + "." + opName;
205             }
206             ExportedOperationImpl existing = exportedOps.put(opName, new ExportedOperationImpl(opName,
207                     (String) Reflections.getAnnotationAttribute(annot, "description"), method, object));
208             if (existing != null) {
209                 throw new IllegalArgumentException("exporting operations with same name not supported: " + opName);
210             }
211         }
212     }
213 
214     private static void addGetter(final String valueName,
215             final Map<String, ExportedValueImpl> exported,
216             final Annotation annot, final Method method, final Object object) {
217         ExportedValueImpl existing = exported.get(valueName);
218         if (existing == null) {
219             existing = new ExportedValueImpl(
220                     valueName, (String) Reflections.getAnnotationAttribute(annot, "description"),
221                     method, null, object, method.getReturnType());
222         } else {
223             if (existing.getValueClass() != method.getReturnType()) {
224                 throw new IllegalArgumentException(
225                         "Getter and setter icorrectly defined " + existing + " " + method);
226             }
227             existing = existing.withGetter(method,
228                     (String) Reflections.getAnnotationAttribute(annot, "description"));
229         }
230         exported.put(valueName, existing);
231     }
