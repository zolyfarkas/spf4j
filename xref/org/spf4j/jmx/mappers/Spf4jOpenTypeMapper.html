74  /**
75   * @author Zoltan Farkas
76   */
77  public final class Spf4jOpenTypeMapper implements JMXBeanMappingSupplier {
78  
79    private static final Logger LOG = LoggerFactory.getLogger(Spf4jOpenTypeMapper.class);
80  
81    private static final ThreadLocal<Set<Type>> IN_PROGRESS = new ThreadLocal<Set<Type>>() {
82      @Override
83      protected Set<Type> initialValue() {
84        return new HashSet<>(2);
85      }
86  
87    };
88  
89    private static final Pair<OpenType<?>, Class<?>[]>[] SIMPLE_TYPES = new Pair[]{
90      Pair.of(BIGDECIMAL, new Class[]{BigDecimal.class}),
91      Pair.of(BIGINTEGER, new Class[]{BigInteger.class}),
92      Pair.of(BOOLEAN, new Class[]{Boolean.class, boolean.class}),
93      Pair.of(BYTE, new Class[]{Byte.class, byte.class}),
94      Pair.of(CHARACTER, new Class[]{Character.class, char.class}),
95      Pair.of(DATE, new Class[]{Date.class}),
96      Pair.of(DOUBLE, new Class[]{Double.class, double.class}),
97      Pair.of(FLOAT, new Class[]{Float.class, float.class}),
98      Pair.of(INTEGER, new Class[]{Integer.class, int.class}),
99      Pair.of(LONG, new Class[]{Long.class, long.class}),
100     Pair.of(SHORT, new Class[]{Short.class, short.class}),
101     Pair.of(OBJECTNAME, new Class[]{ObjectName.class}),
102     Pair.of(STRING, new Class[]{String.class}),
103     Pair.of(VOID, new Class[]{Void.class})
104   };
105 
106 
107   private final CachingTypeMapSupplierWrapper<JMXBeanMapping, NotSerializableException> cache;
108 
109   public Spf4jOpenTypeMapper() {
110     cache = new CachingTypeMapSupplierWrapper<>(new GraphTypeMap(), NotSerializableException.class);
111       for (final Pair<OpenType<?>, Class<?>[]> t : SIMPLE_TYPES) {
112         for (Class<?> c : t.getSecond()) {
113           cache.safePut(c, (type) -> new Spf4jJMXBeanMapping.BasicMXBeanType(c, t.getFirst()));
114         }
115       }
116       cache.safePut(Enum.class, (type) -> new Spf4jJMXBeanMapping.EnumMXBeanType((Class) type));
117       cache.safePut(SpecificRecordBase.class, (type) -> new SpecificRecordOpenTypeMapping((Class) type, this));
118       cache.safePut(Properties.class, (type) -> new Spf4jJMXBeanMapping.MapMXBeanType((Class) type, this));
119       cache.safePut(Iterable.class, (type) -> {
120         if (type instanceof ParameterizedType) {
121           return new Spf4jJMXBeanMapping.ListMXBeanType((ParameterizedType) type, this);
122         } else {
123           return Spf4jJMXBeanMapping.defaultHandler(type, this);
124         }
125       });
126       cache.safePut(Map.class, (type) -> {
127         if (type instanceof ParameterizedType) {
128           return new Spf4jJMXBeanMapping.MapMXBeanType((ParameterizedType) type, this);
129         } else {
130           throw new IllegalArgumentException("Only parameterized maps can be converted to opentype, not " + type);
131         }
132       });
133       cache.safePut(Map.Entry.class, (type) -> {
134         if (type instanceof ParameterizedType) {
135           return new MapEntryOpenTypeMapping((ParameterizedType) type, this);
136         } else {
137           return Spf4jJMXBeanMapping.defaultHandler(type, this);
138         }
139       });
140       cache.safePut(Object.class, (type) -> {
141         if (type instanceof GenericArrayType) {
142           return new Spf4jJMXBeanMapping.GenericArrayMXBeanType((GenericArrayType) type, this);
143         } else if (type instanceof Class && ((Class) type).isArray()) {
144           return new ArrayMXBeanType(((Class) type), this);
145         } else {
146           return Spf4jJMXBeanMapping.defaultHandler(type, this);
147         }
148       });
149   }
150 
151 
152   @Override
153   public JMXBeanMapping get(final Type t) throws NotSerializableException {
154 
155         Set<Type> ip = IN_PROGRESS.get();
156         if (ip.contains(t)) {
157            LOG.debug("No openType mapping for {} recorsive data structure", t);
158            return null;
159         }
160         try {
161         ip.add(t);
162           return cache.get(t);
163         } catch (RuntimeException ex) {
164           LOG.debug("No mapping for type {}", t, ex);
165           TypeToken<?> tt = TypeToken.of(t);
166           if (tt.isSubtypeOf(Serializable.class) || tt.getRawType().isInterface()) {
167             return null;
168           } else {
169             NotSerializableException nsex = new NotSerializableException("Type " + t + "  must be serializable");
170             nsex.addSuppressed(ex);
171             throw nsex;
172           }
173         } finally {
174           ip.remove(t);
175         }
176   }
177 
178   public CachingTypeMapSupplierWrapper<JMXBeanMapping, NotSerializableException> getCache() {
179     return cache;
180   }
181 
182   @Override
183   public String toString() {
184     return "Spf4jOpenTypeMapper{" + "convertedTypes=" + cache + '}';
185   }
