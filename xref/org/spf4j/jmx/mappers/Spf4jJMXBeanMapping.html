95    @SuppressFBWarnings("LEST_LOST_EXCEPTION_STACK_TRACE")
96    public static JMXBeanMapping defaultHandler(final Type javaType, final JMXBeanMappingSupplier mappings)
97            throws NotSerializableException {
98      //falling back to MXBeanMappingFactory.DEFAULT
99      try {
100       MXBeanMapping mapping = MXBeanMappingFactory.DEFAULT.mappingForType(javaType, new MXBeanMappingFactory() {
101         @Override
102         public MXBeanMapping mappingForType(final Type t, final MXBeanMappingFactory f) throws OpenDataException {
103           JMXBeanMapping m;
104           try {
105             m = mappings.get(t);
106           } catch (NotSerializableException ex) {
107             OpenDataException tex = new OpenDataException(t + " is not seriablizable ");
108             tex.initCause(ex);
109             throw tex;
110           }
111           if (m != null) {
112             return MXBeanMappings.convert(m);
113           }
114           return MXBeanMappingFactory.DEFAULT.mappingForType(t, f);
115         }
116       });
117      return MXBeanMappings.convert(mapping);
118     } catch (OpenDataException ex) {
119       NotSerializableException nsex = Throwables.first(ex, NotSerializableException.class);
120       if (nsex != null) {
121         throw nsex;
122       } else {
123         throw new IllegalArgumentException("No type mapping for " + javaType, ex);
124       }
125     }
126   }

214   /**
215    * Array E[]
216    *   Mapped open type - Array with element of OpenType for E
217    *
218    * Data Mapping:
219    *   E[] <-> openTypeData(E)[]
220    */
221    static class ArrayMXBeanType extends Spf4jJMXBeanMapping {
222 
223     private final Class<?> arrayClass;
224     protected JMXBeanMapping componentType;
225     protected JMXBeanMapping baseElementType;
226 
227     ArrayMXBeanType(final Class<?> c, final JMXBeanMappingSupplier mappings)
228             throws NotSerializableException {
229       this.arrayClass = c;
230       this.componentType = mappings.get(c.getComponentType());
231 
232       StringBuilder className = new StringBuilder();
233       Class<?> et = c;
234       int dim;
235       for (dim = 0; et.isArray(); dim++) {
236         className.append('[');
237         et = et.getComponentType();
238       }
239       baseElementType = mappings.get(et);
240       if (et.isPrimitive()) {
241         className = new StringBuilder(c.getName());
242       } else {
243         className.append('L').append(baseElementType.getMappedType().getTypeName()).append(';');
244       }
245       try {
246         mappedTypeClass = Class.forName(className.toString());
247       } catch (ClassNotFoundException e) {
248         throw new IllegalArgumentException("Cannot obtain array class " + className, e);
249       }
250       try {
251         openType = new ArrayType<>(dim, baseElementType.getOpenType());
252       } catch (OpenDataException ex) {
253         throw new IllegalArgumentException("Unsupported type " + c, ex);
254       }
255     }
256 
257     protected ArrayMXBeanType() {
258       arrayClass = null;
259     }
260 
261     @Override
262     public Type getJavaType() {
263       return arrayClass;
264     }
265 
266     @Override
267     @SuppressFBWarnings("URV_INHERITED_METHOD_WITH_RELATED_TYPES")
268     public Object toOpenValue(final Object data) throws OpenDataException {
269       if (baseElementType.isSimpleType()) {
270         return data;
271       }
272       final Object[] array = (Object[]) data;
273       final Object[] openArray = (Object[]) Array.newInstance(componentType.getMappedType(),
274               array.length);
275       int i = 0;
276       for (Object o : array) {
277         if (o == null) {
278           openArray[i] = null;
279         } else {
280           openArray[i] = componentType.toOpenValue(o);
281         }
282         i++;
283       }
284       return openArray;
285     }
286 
287     @Override
288     @SuppressFBWarnings("URV_INHERITED_METHOD_WITH_RELATED_TYPES")
289     public Object fromOpenValue(final Object data)
290             throws InvalidObjectException {
291       if (baseElementType.isSimpleType()) {
292         return data;
293       }
294 
295       final Object[] openArray = (Object[]) data;
296       final Object[] array = (Object[]) Array.newInstance((Class) componentType.getJavaType(),
297               openArray.length);
298       int i = 0;
299       for (Object o : openArray) {
300         if (o == null) {
301           array[i] = null;
302         } else {
303           array[i] = componentType.fromOpenValue(o);
304         }
305         i++;
306       }
307       return array;
308     }

312   static class GenericArrayMXBeanType extends ArrayMXBeanType {

314     private final GenericArrayType gtype;
315 
316     GenericArrayMXBeanType(final GenericArrayType gat, final JMXBeanMappingSupplier mappings)
317             throws NotSerializableException {
318       this.gtype = gat;
319       this.componentType = mappings.get(gat.getGenericComponentType());
320 
321       StringBuilder className = new StringBuilder();
322       Type elementType = gat;
323       int dim;
324       for (dim = 0; elementType instanceof GenericArrayType; dim++) {
325         className.append('[');
326         GenericArrayType et = (GenericArrayType) elementType;
327         elementType = et.getGenericComponentType();
328       }
329       baseElementType = mappings.get(elementType);
330       if (elementType instanceof Class && ((Class) elementType).isPrimitive()) {
331         className = new StringBuilder(gat.toString());
332       } else {
333         className.append('L').append(baseElementType.getMappedType().getTypeName()).append(';');
334       }
335       try {
336         mappedTypeClass = Class.forName(className.toString());
337       } catch (ClassNotFoundException e) {
338         throw new IllegalArgumentException("Cannot obtain array class " + className, e);
339       }
340       try {
341         openType = new ArrayType<>(dim, baseElementType.getOpenType());
342       } catch (OpenDataException ex) {
343         throw new IllegalArgumentException(ex);
344       }
345     }
346 
347     @Override
348     public Type getJavaType() {
349       return gtype;
350     }

354   /**
355    * List<E>
356    *   Mapped open type - Array with element of OpenType for E
357    *
358    * Data Mapping:
359    *   List<E> <-> openTypeData(E)[]
360    */
361   static class ListMXBeanType extends Spf4jJMXBeanMapping {
362 
363     private final ParameterizedType javaType;
364     private final JMXBeanMapping paramMapping;
365     private final Class<?> rawType;
366 
367     ListMXBeanType(final ParameterizedType pt, final JMXBeanMappingSupplier mappings)
368             throws NotSerializableException {
369       this.javaType = pt;
370       this.rawType = (Class) pt.getRawType();
371       final Type[] argTypes = pt.getActualTypeArguments();
372       assert (argTypes.length == 1);
373       Type argType = argTypes[0];
374 
375       TypeToken<?> tt = TypeToken.of(argType);
376       final Class<?> et = tt.getRawType();
377       if (et.isArray()) {
378         throw new IllegalArgumentException("Element Type for " + pt
379                 + " not supported");
380       }
381       paramMapping = mappings.get(argType);
382       String cname = "[L" + paramMapping.getMappedType().getName() + ";";
383       try {
384         mappedTypeClass = Class.forName(cname);
385       } catch (ClassNotFoundException e) {
386         throw new IllegalArgumentException("Array class not found " + cname, e);
387       }
388       try {
389         openType = new ArrayType<>(1, paramMapping.getOpenType());
390       } catch (OpenDataException ex) {
391         throw new IllegalArgumentException("Invalid arg " + pt, ex);
392       }
393     }
394 
395     @Override
396     public Type getJavaType() {
397       return javaType;
398     }
399 
400     @Override
401     public Object toOpenValue(final Object data) throws OpenDataException {
402       if (data instanceof Collection) {
403         final Collection<Object> list = (Collection<Object>) data;
404         final Object[] openArray = (Object[]) Array.newInstance(paramMapping.getMappedType(),
405                 list.size());
406         int i = 0;
407         for (Object o : list) {
408           openArray[i++] = paramMapping.toOpenValue(o);
409         }
410         return openArray;
411       } else {
412         final Iterable<Object> list = (Iterable<Object>) data;
413         List result = new ArrayList(16);
414         for (Object o : list) {
415           result.add(paramMapping.toOpenValue(o));
416         }
417         return result.toArray((Object[]) Array.newInstance(paramMapping.getMappedType(),
418                 result.size()));
419 
420       }
421     }
422 
423     @Override
424     public Object fromOpenValue(final Object data)
425             throws InvalidObjectException {
426 
427       final Object[] openArray = (Object[]) data;
428       Collection<Object> result;
429       if (rawType.isInterface()) {
430         if (Set.class == rawType) {
431           result = new HashSet<>(openArray.length);
432         } else if (Deque.class == rawType) {
433           result = new ArrayDeque<>(openArray.length);
434         } else {
435           result = new ArrayList<>(openArray.length);
436         }
437       } else {
438         try {
439           result = (Collection) rawType.newInstance();
440         } catch (InstantiationException | IllegalAccessException ex) {
441           InvalidObjectException iox = new InvalidObjectException("Cannot instantiate " + rawType);
442           iox.addSuppressed(ex);
443           throw iox;
444         }
445       }
446       for (Object o : openArray) {
447         result.add(paramMapping.fromOpenValue(o));
448       }
449       return result;
450     }

453   /**
454    * Map<K,V>
455    *   Mapped open type - TabularType with row type:
456    *                        CompositeType:
457    *                          "key"   of openDataType(K)
458    *                          "value" of openDataType(V)
459    *                        "key" is the index name
460    *
461    * Data Mapping:
462    *   Map<K,V> <-> TabularData
463    */
464    static class MapMXBeanType extends Spf4jJMXBeanMapping {
465 
466     private final Type javaType;
467     private final Class<?> rawType;
468     private final JMXBeanMapping keyType;
469     private final JMXBeanMapping valueType;
470 
471     @SuppressFBWarnings({ "CLI_CONSTANT_LIST_INDEX", "ITC_INHERITANCE_TYPE_CHECKING" })
472     MapMXBeanType(final Type pt, final JMXBeanMappingSupplier mappings)
473       throws  NotSerializableException {
474       try {
475         this.javaType = pt;
476 
477         final Type[] argTypes;
478         if (pt instanceof ParameterizedType) {
479           argTypes = ((ParameterizedType) pt).getActualTypeArguments();
480           rawType = (Class) ((ParameterizedType) pt).getRawType();
481         } else if (pt instanceof Class && Properties.class.isAssignableFrom((Class) pt)) {
482           argTypes = STR_STR_TYPES;
483           rawType = (Class) pt;
484         } else {
485           throw new IllegalArgumentException("Unsupported type " + pt);
486         }
487         assert (argTypes.length == 2);
488         this.keyType = mappings.get(argTypes[0]);
489         if (this.keyType == null) {
490           throw new IllegalArgumentException("Key of " + pt + " cannot be converted to open type");
491         }
492         this.valueType = mappings.get(argTypes[1]);
493         if (this.valueType == null) {
494           throw new IllegalArgumentException("Value of " + pt + " cannot be converted to open type");
495         }
496 
497         // FIXME: generate typeName for generic
498         String typeName = pt.getTypeName();
499         final OpenType<?>[] mapItemTypes = new OpenType<?>[]{
500           keyType.getOpenType(),
501           valueType.getOpenType()
502         };
503         final CompositeType rowType
504                 = new CompositeType(typeName,
505                         typeName,
506                         MAP_ITEM_NAMES,
507                         MAP_ITEM_NAMES,
508                         mapItemTypes);
509 
510         openType = new TabularType(typeName, typeName, rowType, MAP_INDEX_NAMES);
511         mappedTypeClass = TabularData.class;
512       } catch (OpenDataException ex) {
513         throw new IllegalArgumentException("Unsupported type " + pt, ex);
514       }
515     }
516 
517     @Override
518     public Type getJavaType() {
519       return javaType;
520     }
521 
522     @Override
523     public Object toOpenValue(final Object data) throws OpenDataException {
524       final Map<Object, Object> map = (Map<Object, Object>) data;
525       final TabularType tabularType = (TabularType) openType;
526       final TabularData table = new TabularDataSupport(tabularType);
527       final CompositeType rowType = tabularType.getRowType();
528 
529       for (Map.Entry<Object, Object> entry : map.entrySet()) {
530         final Object key = keyType.toOpenValue(entry.getKey());
531         final Object value = valueType.toOpenValue(entry.getValue());
532         final CompositeData row
533                 = new CompositeDataSupport(rowType,
534                         MAP_ITEM_NAMES,
535                         new Object[]{key, value});
536         table.put(row);
537       }
538       return table;
539     }
540 
541     @Override
542     public Object fromOpenValue(final Object data)
543             throws InvalidObjectException {
544 
545       final TabularData td = (TabularData) data;
546       Map<Object, Object> result;
547       if (rawType.isInterface()) {
548         result = Maps.newHashMapWithExpectedSize(td.values().size());
549       } else {
550         try {
551           result = (Map) rawType.newInstance();
552         } catch (InstantiationException | IllegalAccessException ex) {
553           InvalidObjectException iox = new InvalidObjectException("Cannot instantiate " + rawType);
554           iox.addSuppressed(ex);
555           throw iox;
556         }
557       }
558       for (CompositeData row : (Collection<CompositeData>) td.values()) {
559         Object key = keyType.fromOpenValue(row.get(KEY));
560         Object value = valueType.fromOpenValue(row.get(VALUE));
561         result.put(key, value);
562       }
563       return result;
564     }
