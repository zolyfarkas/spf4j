19  package org.spf4j.aspects;
20  
21  import org.aspectj.lang.ProceedingJoinPoint;
22  import org.aspectj.lang.annotation.Around;
23  import org.aspectj.lang.annotation.Aspect;
24  import org.spf4j.annotations.Retry;
25  import org.spf4j.annotations.VoidPredicate;
26  import org.spf4j.base.Callables;
27  import org.spf4j.base.Callables.TimeoutCallable;
28  
29  /**
30   * Aspect that measures execution time and does performance logging for all methods annotated with: PerformanceMonitor
31   * annotation.
32   *
33   * @author zoly
34   */
35  @Aspect
36  public final class RetryAspect {
37  
38      
39      private static final ThreadLocal<Long> DEADLINE = new ThreadLocal<Long>() {
40  
41          @Override
42          protected Long initialValue() {
43              return Long.MAX_VALUE;
44          }
45  
46      };
47      
48      
49      public static long getDeadline() {
50          return DEADLINE.get();
51      }
52      
53      @Around(value = "execution(@org.spf4j.annotations.Retry * *(..)) && @annotation(annot)",
54              argNames = "pjp,annot")
55      public Object performanceMonitoredMethod(final ProceedingJoinPoint pjp, final Retry annot)
56              throws Throwable {
57  
58          return Callables.executeWithRetry(new TimeoutCallable<Object, Exception>(annot.timeoutMillis()) {
59  
60              @Override
61              public Object call(final long dealine) throws Exception {
62                  DEADLINE.set(dealine);
63                  try {
64                      return pjp.proceed();
65                  } catch (Exception e) {
66                    throw e;
67                  } catch (Throwable ex) {
68                      throw new Error(ex);
69                  } finally {
70                      DEADLINE.set(Long.MAX_VALUE);
71                  }
72              }
73          }, annot.immediateRetries(), annot.retryDelayMillis(),
74                  annot.exRetry() == VoidPredicate.class ? Callables.DEFAULT_EXCEPTION_RETRY
75                          : annot.exRetry().newInstance());
76  
77      }
78  }
