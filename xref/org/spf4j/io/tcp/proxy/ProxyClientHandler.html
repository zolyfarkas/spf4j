18  package org.spf4j.io.tcp.proxy;
19  
20  import com.google.common.net.HostAndPort;
21  import java.io.IOException;
22  import java.net.InetSocketAddress;
23  import java.nio.channels.Selector;
24  import java.nio.channels.SocketChannel;
25  import java.util.concurrent.BlockingQueue;
26  import java.util.concurrent.ExecutorService;
27  import javax.annotation.Nullable;
28  import javax.annotation.ParametersAreNonnullByDefault;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  import org.spf4j.base.AbstractRunnable;
32  import org.spf4j.base.Closeables;
33  import org.spf4j.ds.UpdateablePriorityQueue;
34  import org.spf4j.io.tcp.ClientHandler;
35  import org.spf4j.io.tcp.DeadlineAction;
36  
37  /**
38   *
39   * @author zoly
40   */
41  @ParametersAreNonnullByDefault
42  public final class ProxyClientHandler implements ClientHandler {
43  
44      private static final Logger LOG = LoggerFactory.getLogger(ProxyClientHandler.class);
45  
46      private final HostAndPort fwdDestination;
47      private final int proxyBufferSize;
48      private final int connectTimeoutMillis;
49      private final SnifferFactory c2sSnifferFact;
50      private final SnifferFactory s2cSnifferFact;
51  
52      /**
53       * TCP proxy client handler.
54       * @param fwdDestination - the destination all connections will be forwarded to.
55       * @param c2sSnifferFact - create sniffer to be invoked when data is received from client.
56       * @param s2cSnifferFact - create sniffer to be invoked when data is received from server.
57       * @param proxyBufferSize - the transmission buffer sizes.
58       * @param connectTimeoutMillis - The connection timeout.
59       */
60  
61      public ProxyClientHandler(final HostAndPort fwdDestination,
62          @Nullable final SnifferFactory c2sSnifferFact, @Nullable final SnifferFactory s2cSnifferFact,
63          final int proxyBufferSize, final int connectTimeoutMillis) {
64          this.fwdDestination = fwdDestination;
65          this.proxyBufferSize = proxyBufferSize;
66          this.connectTimeoutMillis = connectTimeoutMillis;
67          this.c2sSnifferFact = c2sSnifferFact;
68          this.s2cSnifferFact = s2cSnifferFact;
69      }
70  
71      @Override
72      public void handle(final Selector serverSelector, final SocketChannel clientChannel,
73              final ExecutorService exec, final BlockingQueue<Runnable> tasksToRunBySelector,
74              final UpdateablePriorityQueue<DeadlineAction> deadlineActions)  throws IOException {
75          final InetSocketAddress socketAddress = new InetSocketAddress(
76                  fwdDestination.getHostText(), fwdDestination.getPort());
77          final SocketChannel proxyChannel = SocketChannel.open();
78          try {
79              proxyChannel.configureBlocking(false);
80              proxyChannel.connect(socketAddress);
81              TransferBuffer c2s = new TransferBuffer(proxyBufferSize);
82              if (c2sSnifferFact != null) {
83                  c2s.setIncomingSniffer(c2sSnifferFact.get(clientChannel));
84              }
85              TransferBuffer s2c = new TransferBuffer(proxyBufferSize);
86              final long connectDeadline = System.currentTimeMillis() + connectTimeoutMillis;
87              UpdateablePriorityQueue.ElementRef daction = deadlineActions.add(new DeadlineAction(connectDeadline,
88                      new CloseChannelsOnTimeout(proxyChannel, clientChannel)));
89              new ProxyBufferTransferHandler(c2s, s2c, null, clientChannel,
90                      serverSelector, exec, tasksToRunBySelector, daction).initialInterestRegistration();
91              new ProxyBufferTransferHandler(s2c, c2s, s2cSnifferFact, proxyChannel,
92                      serverSelector, exec, tasksToRunBySelector, daction).initialInterestRegistration();
93          } catch (IOException ex) {
94              Exception exs = Closeables.closeAll(proxyChannel, clientChannel);
95              ex.addSuppressed(exs);
96              throw ex;
97          }
98  
99      }
100 
101     static final class CloseChannelsOnTimeout extends AbstractRunnable {
102 
103         private final SocketChannel proxyChannel;
104         private final SocketChannel clientChannel;
105 
106         CloseChannelsOnTimeout(final SocketChannel proxyChannel, final SocketChannel clientChannel) {
107             super(true);
108             this.proxyChannel = proxyChannel;
109             this.clientChannel = clientChannel;
110         }
111 
112         @Override
113         public void doRun() throws IOException {
114             LOG.warn("Timed out connecting to {}", proxyChannel);
115             try {
116                 clientChannel.close();
117             } finally {
118                 proxyChannel.close();
119             }
120         }
121     }
122 
123     @Override
124     public String toString() {
125         return "ProxyClientHandler{" + "fwdDestination=" + fwdDestination + ", proxyBufferSize="
126                 + proxyBufferSize + ", connectTimeoutMillis=" + connectTimeoutMillis + ", c2sSnifferFact="
127                 + c2sSnifferFact + ", s2cSnifferFact=" + s2cSnifferFact + '}';
128     }
