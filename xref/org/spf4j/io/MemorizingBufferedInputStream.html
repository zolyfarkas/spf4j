17  /**
18   * Why another buffered input stream?
19   * Main use case if for troubleshooting.
20   * Allows you to get more detail on where your stream processing failed.
21   *
22   * Implementation is a circular byte buffer, where you have 2 sizes to control the behavior:
23   *
24   * buffer size - the total sie of the buffer.
25   * read size - the maximum number of read bytes kept in the buffer.
26   *
27   *
28   * @author zoly
29   */
30  @ParametersAreNonnullByDefault
31  @CleanupObligation
32  public final class MemorizingBufferedInputStream extends FilterInputStream {
33  
34      private byte[] memory;
35  
36      private final  SizedRecyclingSupplier<byte[]> bufferProvider;

38      private final int readSize;

54      public MemorizingBufferedInputStream(final InputStream in) {
55          this(in, 16384, 8192, ArraySuppliers.Bytes.GL_SUPPLIER, Charsets.UTF_8);
56      }

58      public MemorizingBufferedInputStream(final InputStream in, final int size) {
59          this(in, size, size / 2, ArraySuppliers.Bytes.GL_SUPPLIER, Charsets.UTF_8);
60      }

62      public MemorizingBufferedInputStream(final InputStream in, final Charset charset) {
63          this(in, 16384, 8192, ArraySuppliers.Bytes.GL_SUPPLIER, charset);
64      }

66      public MemorizingBufferedInputStream(final InputStream in,
67              final int size, final int readSize,
68              final SizedRecyclingSupplier<byte[]> bufferProvider,
69              @Nullable final Charset charset) {
70          super(in);
71          if (readSize > size) {
72              throw new IllegalArgumentException("Read size " + readSize + " cannot be greater than " + size);
73          }
74          if (size < 2) {
75              throw new IllegalArgumentException("Buffer size " + size + " cannot be smaler than 2");
76          }
77          memory = bufferProvider.get(size);
78          this.bufferProvider = bufferProvider;
79          this.charset = charset;
80          this.startIdx = 0;
81          this.endIdx = 0;
82          this.readSize = readSize;
83          this.isEof = false;
84          this.isClosed = false;
85          this.readBytes = 0;
86      }
87  
88      @Override
89      public synchronized void close() throws IOException {
90          if (!isClosed) {
91              isClosed = true;
92              bufferProvider.recycle(memory);
93              memory = null;
94              super.close();
95          }
96      }
97  
98      private int availableToWrite() {
99          if (memIdx <= endIdx) {
100             return memIdx + memory.length - endIdx - 1;
101         } else {
102             return memIdx - endIdx - 1;
103         }
104     }
105 
106     private int availableToRead() {
107         if (startIdx <= endIdx) {
108             return endIdx - startIdx;
109         } else {
110             return memory.length - startIdx + endIdx;
111         }
112     }
113 
114     private int availableInMemory() {
115         if (memIdx <= startIdx) {
116             return startIdx - memIdx;
117         } else {
118             return memory.length - memIdx + startIdx;
119         }
120     }

123     public synchronized byte[] getReadBytesFromBuffer() {
124         final int availableInMemory = availableInMemory();
125         if (availableInMemory == 0) {
126             return Arrays.EMPTY_BYTE_ARRAY;
127         }
128         byte[] result = new byte[availableInMemory];
129         if (memIdx < startIdx) {
130             System.arraycopy(memory, memIdx, result, 0, result.length);
131         } else {
132             final int toEnd = memory.length - memIdx;
133             System.arraycopy(memory, memIdx, result, 0, toEnd);
134             System.arraycopy(memory, 0, result, toEnd, startIdx);
135         }
136         return result;
137     }
138 
139     public synchronized byte[] getUnreadBytesFromBuffer() {
140         final int availableToRead = availableToRead();
141         if (availableToRead == 0) {
142             return Arrays.EMPTY_BYTE_ARRAY;
143         }
144         byte[] result = new byte[availableToRead];
145         if (startIdx < endIdx) {
146             System.arraycopy(memory, startIdx, result, 0, result.length);
147         } else {
148             final int toEnd = memory.length - startIdx;
149             System.arraycopy(memory, startIdx, result, 0, toEnd);
150             System.arraycopy(memory, 0, result, toEnd, endIdx);
151         }
152         return result;
153     }

156     private int tryCleanup(final int size) {
157         int canWrite = availableToWrite();
158         if (canWrite < size) {
159             int toFree = size - canWrite;
160             int availableToFree = availableInMemory();
161             if (toFree > availableToFree) {
162                 toFree = availableToFree;
163             }
164             memIdx = memIdx + toFree;
165             if (memIdx >= memory.length) {
166                 memIdx = memIdx - memory.length;
167             }
168             return toFree + canWrite;
169         } else {
170             return size;
171         }
172     }
173 
174     private void fill() throws IOException {
175         int size = tryCleanup(readSize);
176         if (size < readSize) {
177             throw new IllegalStateException("Illegal state " + this);
178         }
179         int canWriteInBulk = Math.min(size, memory.length - endIdx);
180         int read = super.read(memory, endIdx, canWriteInBulk);
181         if (read < 0) {
182             isEof = true;
183             return;
184         }
185         endIdx += read;
186         if (read < canWriteInBulk) {
187             return;
188         }
189         int wrapArround = size - canWriteInBulk;
190         if (wrapArround > 0) {
191             read = super.read(memory, 0, wrapArround);
192             if (read < 0) {
193                 isEof = true;
194                 return;
195             }
196             endIdx = read;
197         } else if (endIdx >= memory.length) {
198             endIdx = 0;
199         }
200     }

202     @Override
203     public synchronized int read(final byte[] b, final int off, final int len) throws IOException {
204         if (isClosed) {
205             throw new IOException("Stream is closed " + this);
206         }
207         int availableToRead = availableToRead();
208         if (availableToRead <= 0) {
209             fill();
210             availableToRead = availableToRead();
211         }
212         if (availableToRead == 0) {
213             if (isEof) {
214                 return -1;
215             } else {
216                 throw new IllegalStateException("State=" + this);
217             }
218         }
219         int toRead = Math.min(availableToRead, len);
220         int readToEnd = Math.min(toRead, memory.length - startIdx);
221         System.arraycopy(memory, startIdx, b, off, readToEnd);
222         startIdx += readToEnd;
223         int wrapArround = toRead - readToEnd;
224         if (wrapArround > 0) {
225             System.arraycopy(memory, 0, b, off + readToEnd, wrapArround);
226             startIdx = wrapArround;
227         } else if (startIdx >= memory.length) {
228             startIdx = 0;
229         }
230         this.readBytes += toRead;
231         return toRead;
232     }

235     @Override
236     public synchronized int read() throws IOException {
237         if (isClosed) {
238             throw new IOException("Stream is closed " + this);
239         }
240         int availableToRead = availableToRead();
241         if (availableToRead <= 0) {
242             fill();
243             availableToRead = availableToRead();
244         }
245         if (availableToRead == 0) {
246             if (isEof) {
247                 return -1;
248             } else {
249                 throw new IllegalStateException("State=" + this);
250             }
251         }
252         int result = memory[startIdx++] & 0xff;
253         if (startIdx >= memory.length) {
254             startIdx = 0;
255         }
256         this.readBytes++;
257         return result;
258     }
259 
260     @Override
261     public synchronized int available() throws IOException {
262         if (isClosed) {
263             throw new IOException("Stream is closed " + this);
264         }
265         return availableToRead();
266     }

268     @Override
269     public synchronized String toString() {
270         StringBuilder result;
271         if (isClosed) {
272            result = new StringBuilder(64);
273         } else {
274            result = new StringBuilder((availableToRead() + availableInMemory()) * 2 + 128);
275         }
276         result.append("MemorizingBufferedInputStream{\n");
277         if (isClosed) {
278             result.append("closed=true\n");
279         } else if (charset == null) {
280             final BaseEncoding base64 = BaseEncoding.base64();
281             result.append("readBytes=\"").append(base64.encode(getReadBytesFromBuffer())).append("\",\n");
282             result.append("unreadBytes=\"").append(base64.encode(getUnreadBytesFromBuffer())).append("\"\n");
283         } else {
284             result.append("readStr=\"").append(
285                     Strings.fromUtf8(getReadBytesFromBuffer())).append("\",\n");
286             result.append("unreadStr=\"").append(
287                     Strings.fromUtf8(getUnreadBytesFromBuffer())).append("\"\n");
288         }
289         result.append("memIdx=").append(this.memIdx).append("\"\n");
290         result.append("startIdx=").append(this.startIdx).append("\"\n");
291         result.append("endIdx=").append(this.endIdx).append("\"\n");
292         result.append('}');
293         return result.toString();
294     }
295 
296     public synchronized long getReadBytes() {
297         return readBytes;
298     }

300     @Override
301     public synchronized long skip(final long n) throws IOException {
302         long nrSkipped = 0;
303         for (int i = 0; i < n; i++) {
304             int read = read();
305             if (read < 0) {
306                 break;
307             } else {
308                 nrSkipped++;
309             }
310         }
311         return nrSkipped;
312     }
