2   package org.spf4j.io;
3   
4   import com.google.common.io.BaseEncoding;
5   import edu.umd.cs.findbugs.annotations.CleanupObligation;
6   import java.io.InputStream;
7   
8   
9   /**
10   *
11   * @author zoly
12   */
13  @CleanupObligation
14  public final class PushbackInputStreamEx  extends java.io.PushbackInputStream {
15  
16      public PushbackInputStreamEx(final InputStream in, final int size) {
17          super(in, size);
18      }
19  
20      public PushbackInputStreamEx(final InputStream in) {
21          super(in);
22      }
23  
24      @Override
25      public String toString() {
26          return "PushbackInputStream{"
27                  + ((buf == null) ? "buf=null" : "buf=" + BaseEncoding.base64().encode(buf))
28                  + ", pos=" + pos
29                  + ", wrapped=" + this.in + '}';
30      }
31  
32      public InputStream getUnderlyingStream() {
33          return this.in;
34      }
35  
36  }
