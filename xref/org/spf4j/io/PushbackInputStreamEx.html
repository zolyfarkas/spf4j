2   package org.spf4j.io;
3   
4   import com.google.common.io.BaseEncoding;
5   import edu.umd.cs.findbugs.annotations.CleanupObligation;
6   import java.io.IOException;
7   import java.io.InputStream;
8   
9   
10  /**
11   * a extended implementation that improves the safety of unread and provides a better toString.
12   * @author zoly
13   */
14  @CleanupObligation
15  public final class PushbackInputStreamEx  extends java.io.PushbackInputStream {
16  
17      public PushbackInputStreamEx(final InputStream in, final int size) {
18          super(in, size);
19      }
20  
21      public PushbackInputStreamEx(final InputStream in) {
22          super(in);
23      }
24  
25      @Override
26      public String toString() {
27          return "PushbackInputStream{"
28                  + ((buf == null) ? "buf=null" : "buf=" + BaseEncoding.base64().encode(buf))
29                  + ", pos=" + pos
30                  + ", wrapped=" + this.in + '}';
31      }
32  
33      public InputStream getUnderlyingStream() {
34          return this.in;
35      }
36  
37    @Override
38    public void unread(final int b) throws IOException {
39      if (b < 0) {
40        throw new IllegalArgumentException("Unread int must be positive and not " + b);
41      }
42      super.unread(b);
43    }

48  }
