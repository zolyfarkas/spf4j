4   import com.google.common.io.BaseEncoding;
5   import java.io.InputStream;

7   
8   /**
9    *
10   * @author zoly
11   */
12  public final class PushbackInputStreamEx  extends java.io.PushbackInputStream {
13  
14      public PushbackInputStreamEx(final InputStream in, final int size) {
15          super(in, size);
16      }
17  
18      public PushbackInputStreamEx(final InputStream in) {
19          super(in);
20      }
21  
22      @Override
23      public String toString() {
24          return "PushbackInputStream{"
25                  + ((buf == null) ? "" : "buf=" + BaseEncoding.base64().encode(buf))
26                  + ", pos=" + pos
27                  + ", wrapped=" + this.in + '}';
28      }
29  
30      public InputStream getUnderlyingStream() {
31          return this.in;
32      }
33  
34  }
