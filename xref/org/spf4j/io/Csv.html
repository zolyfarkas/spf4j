149   private static final char[] TO_ESCAPE = new char[]{',', '\n', '\r', '"'};
150 
151   public static void writeCsvElement(final CharSequence elem, final Appendable writer) throws IOException {
152     if (Strings.contains(elem, TO_ESCAPE)) {
153       writeQuotedCsvElement(elem, writer);
154     } else {
155       writer.append(elem);
156     }
157   }
158 
159   public static void writeQuotedCsvElement(final CharSequence elem, final Appendable writer) throws IOException {
160     int length = elem.length();
161     writer.append('"');
162     for (int i = 0; i < length; i++) {
163       char c = elem.charAt(i);
164       if (c == '"') {
165         writer.append("\"\"");
166       } else {
167         writer.append(c);
168       }
169     }
170     writer.append('"');
171   }
172 
173   public static CharSequence toCsvElement(final CharSequence elem) {
174     if (Strings.contains(elem, TO_ESCAPE)) {
175       StringWriter sw = new StringWriter(elem.length() - 1);
176       try {
177         writeQuotedCsvElement(elem, sw);
178       } catch (IOException ex) {
179         throw new RuntimeException(ex);
180       }
181       return sw.toString();
182     } else {
183       return elem;
184     }
185   }

187   /**
188    * returns next character.
189    *
190    * @param reader
191    * @param addElemTo
192    * @return - next character or -1 if eof has been reached.
193    * @throws IOException
194    */
195   @CheckReturnValue
196   public static int readCsvElement(final Reader reader, final StringBuilder addElemTo) throws IOException {
197     int c = reader.read();
198     if (c < 0) {
199       return c;
200     }
201     if (c == '"') {
202       c = reader.read();
203       while (c >= 0) {
204         if (c == '"') {
205           int c2 = reader.read();
206           if (c2 >= 0) {
207             if (c2 == '"') {
208               addElemTo.append((char) c);
209             } else {
210               return c2;
211             }
212           } else {
213             return c2;
214           }
215         } else {
216           addElemTo.append((char) c);
217         }
218         c = reader.read();
219       }
220     } else {
221       while (c != ',' && c != '\n' && c != '\r' && c >= 0) {
222         addElemTo.append((char) c);
223         c = reader.read();
224       }
225     }
226     return c;
227   }
