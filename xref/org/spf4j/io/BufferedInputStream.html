18  package org.spf4j.io;
19  
20  import edu.umd.cs.findbugs.annotations.CleanupObligation;
21  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
22  import java.io.FilterInputStream;
23  import java.io.IOException;
24  import java.io.InputStream;
25  import java.util.concurrent.atomic.AtomicReferenceFieldUpdater;
26  import javax.annotation.WillClose;
27  import org.spf4j.recyclable.SizedRecyclingSupplier;
28  import org.spf4j.recyclable.impl.ArraySuppliers;
29  
30  @SuppressFBWarnings("VO_VOLATILE_REFERENCE_TO_ARRAY")
31  @CleanupObligation
32  public final class BufferedInputStream extends FilterInputStream {
33  
34      private static int defaultBufferSize = 8192;
35  
36      private volatile byte[] buf;
37  
38      private static final AtomicReferenceFieldUpdater<BufferedInputStream, byte[]> BUF_UPDATER
39              = AtomicReferenceFieldUpdater.newUpdater(BufferedInputStream.class, byte[].class, "buf");
40  
41      private int count;
42  
43      private int pos;
44  
45      private int markpos = -1;
46  
47      private int marklimit;
48  
49      private final SizedRecyclingSupplier<byte[]> bufferProvider;
50  
51      private InputStream getInIfOpen() throws IOException {
52          InputStream input = in;
53          if (input == null) {
54              throw new IOException("Stream closed " + in);
55          }
56          return input;
57      }
58  
59      private byte[] getBufIfOpen() throws IOException {
60          byte[] buffer = buf;
61          if (buffer == null) {
62              throw new IOException("Stream closed " + in);
63          }
64          return buffer;
65      }
66  
67      public BufferedInputStream(final InputStream in) {
68          this(in, defaultBufferSize);
69      }
70  
71      public BufferedInputStream(final InputStream in, final int size) {
72          this(in, size, ArraySuppliers.Bytes.JAVA_NEW);
73      }
74  
75      public BufferedInputStream(final InputStream in, final int size,
76              final SizedRecyclingSupplier<byte[]> bufferProvider) {
77          super(in);
78          if (size <= 0) {
79              throw new IllegalArgumentException("Buffer size <= 0 : " + size);
80          }
81          this.bufferProvider = bufferProvider;
82          buf = bufferProvider.get(size);
83      }
84  
85      private void fill() throws IOException {
86          byte[] buffer = getBufIfOpen();
87          if (markpos < 0) {
88              pos = 0;
89          } else if (pos >= buffer.length) {
90              if (markpos > 0) {
91                  int sz = pos - markpos;
92                  System.arraycopy(buffer, markpos, buffer, 0, sz);
93                  pos = sz;
94                  markpos = 0;
95              } else if (buffer.length >= marklimit) {
96                  markpos = -1;
97                  pos = 0;
98              } else {
99                  int nsz = pos * 2;
100                 if (nsz > marklimit) {
101                     nsz = marklimit;
102                 }
103                 byte[] nbuf = bufferProvider.get(nsz);
104                 if (!BUF_UPDATER.compareAndSet(this, buffer, nbuf)) {
105                     bufferProvider.recycle(nbuf);
106                     throw new IOException("Stream closed " + in);
107                 }
108                 System.arraycopy(buffer, 0, nbuf, 0, pos);
109                 bufferProvider.recycle(buffer);
110                 buffer = nbuf;
111             }
112         }
113         count = pos;
114         int n = getInIfOpen().read(buffer, pos, buffer.length - pos);
115         if (n > 0) {
116             count = n + pos;
117         }
118     }
119 
120     public synchronized int read() throws IOException {
121         if (pos >= count) {
122             fill();
123             if (pos >= count) {
124                 return -1;
125             }
126         }
127         return getBufIfOpen()[pos++] & 0xff;
128     }
129 
130     private int read1(final byte[] b, final int off, final int len) throws IOException {
131         int avail = count - pos;
132         if (avail <= 0) {
133             if (len >= getBufIfOpen().length && markpos < 0) {
134                 return getInIfOpen().read(b, off, len);
135             }
136             fill();
137             avail = count - pos;
138             if (avail <= 0) {
139                 return -1;
140             }
141         }
142         int cnt = (avail < len) ? avail : len;
143         System.arraycopy(getBufIfOpen(), pos, b, off, cnt);
144         pos += cnt;
145         return cnt;
146     }
147 
148     @Override
149     public synchronized int read(final byte[] b, final int off, final int len)
150             throws IOException {
151         getBufIfOpen(); // Check for closed stream
152         if ((off | len | (off + len) | (b.length - (off + len))) < 0) {
153             throw new IndexOutOfBoundsException();
154         } else if (len == 0) {
155             return 0;
156         }
157 
158         int n = 0;
159         for (;;) {
160             int nread = read1(b, off + n, len - n);
161             if (nread <= 0) {
162                 return (n == 0) ? nread : n;
163             }
164             n += nread;
165             if (n >= len) {
166                 return n;
167             }
168             // if not closed but no bytes available, return
169             InputStream input = in;
170             if (input != null && input.available() <= 0) {
171                 return n;
172             }
173         }
174     }
175 
176     @Override
177     public synchronized long skip(final long n) throws IOException {
178         getBufIfOpen(); // Check for closed stream
179         if (n <= 0) {
180             return 0;
181         }
182         long avail = count - pos;
183 
184         if (avail <= 0) {
185             // If no mark position set then don't keep in buffer
186             if (markpos < 0) {
187                 return getInIfOpen().skip(n);
188             }
189 
190             // Fill in buffer to save bytes for reset
191             fill();
192             avail = count - pos;
193             if (avail <= 0) {
194                 return 0;
195             }
196         }
197 
198         long skipped = (avail < n) ? avail : n;
199         pos += skipped;
200         return skipped;
201     }
202 
203     @Override
204     public synchronized int available() throws IOException {
205         return getInIfOpen().available() + (count - pos);
206     }
207 
208     @Override
209     public synchronized void mark(final int readlimit) {
210         marklimit = readlimit;
211         markpos = pos;
212     }
213 
214     @Override
215     public synchronized void reset() throws IOException {
216         getBufIfOpen(); // Cause exception if closed
217         if (markpos < 0) {
218             throw new IOException("Resetting to invalid mark");
219         }
220         pos = markpos;
221     }
222 
223     @Override
224     public boolean markSupported() {
225         return true;
226     }
227 
228     @Override
229     @WillClose
230     public void close() throws IOException {
231         byte[] buffer;
232         while ((buffer = buf) != null) {
233             if (BUF_UPDATER.compareAndSet(this, buffer, null)) {
234                 bufferProvider.recycle(buffer);
235                 InputStream input = in;
236                 in = null;
237                 if (input != null) {
238                     input.close();
239                 }
240                 return;
241             }
242         }
243     }
244 }
