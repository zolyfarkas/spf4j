42  /**
43   * An {@link InputStream} that converts characters from a {@link Reader} into bytes using an
44   * arbitrary Charset.
45   *
46   * <p>This is an alternative to copying the data to an {@code OutputStream} via a {@code Writer},
47   * which is necessarily blocking. By implementing an {@code InputStream} it allows consumers to
48   * "pull" as much data as they can handle, which is more convenient when dealing with flow
49   * controlled, async APIs.
50   *
51   * @author Chris Nokleberg
52   *
53   * Notes by Z: THis class in its current state is not ready for prime time. Here is why:
54   * 1) Since it buffers, there needs to be capability to access the unconsumed bytes/chars.
55   * 2) Buffer sizing could be smarter? char buffer and byte buffers are not "byte size equivalent"(just made this up :-))
56   */
57  @Beta
58  public final class ReaderInputStream extends InputStream {
59    private final Reader reader;
60    private final CharsetEncoder encoder;
61    private final byte[] singleByte = new byte[1];
62  
63    /**
64     * charBuffer holds characters that have been read from the Reader but not encoded yet. The buffer
65     * is perpetually "flipped" (unencoded characters between position and limit).
66     */
67    private CharBuffer charBuffer;
68  
69    /**
70     * byteBuffer holds encoded characters that have not yet been sent to the caller of the input
71     * stream. When encoding it is "unflipped" (encoded bytes between 0 and position) and when
72     * draining it is flipped (undrained bytes between position and limit).
73     */
74    private ByteBuffer byteBuffer;
75  
76    /** Whether we've finished reading the reader. */
77    private boolean endOfInput;
78    /** Whether we're copying encoded bytes to the caller's buffer. */
79    private boolean draining;
80    /** Whether we've successfully flushed the encoder. */
81    private boolean doneFlushing;
82  
83    /**
84     * Creates a new input stream that will encode the characters from {@code reader} into bytes using
85     * the given character set. Malformed input and unmappable characters will be replaced.
86     *
87     * @param reader input source
88     * @param charset character set used for encoding chars to bytes
89     * @param bufferSize size of internal input and output buffers
90     * @throws IllegalArgumentException if bufferSize is non-positive
91     */
92    public ReaderInputStream(final Reader reader, final Charset charset, final int bufferSize) {
93      this(
94          reader,
95          charset
96              .newEncoder()
97              .onMalformedInput(CodingErrorAction.REPLACE)
98              .onUnmappableCharacter(CodingErrorAction.REPLACE),
99          bufferSize);
100   }
101 
102   /**
103    * Creates a new input stream that will encode the characters from {@code reader} into bytes using
104    * the given character set encoder.
105    *
106    * @param reader input source
107    * @param encoder character set encoder used for encoding chars to bytes
108    * @param bufferSize size of internal input and output buffers
109    * @throws IllegalArgumentException if bufferSize is non-positive
110    */
111   public ReaderInputStream(final Reader reader, final CharsetEncoder encoder, final int bufferSize) {
112     this.reader = checkNotNull(reader);
113     this.encoder = checkNotNull(encoder);
114     checkArgument(bufferSize > 0, "bufferSize must be positive: %s", bufferSize);
115     encoder.reset();
116 
117     charBuffer = CharBuffer.allocate(bufferSize);
118     charBuffer.flip();
119 
120     byteBuffer = ByteBuffer.allocate(bufferSize);
121   }
122 
123   @Override
124   public void close() throws IOException {
125     reader.close();
126   }
127 
128   @Override
129   public int read() throws IOException {
130     return (read(singleByte) == 1) ? UnsignedBytes.toInt(singleByte[0]) : -1;
131   }
132 
133   // TODO(chrisn): Consider trying to encode/flush directly to the argument byte
134   // buffer when possible.
135   @Override
136   public int read(final byte[] b, final int off, final int len) throws IOException {
137     // Obey InputStream contract.
138     checkPositionIndexes(off, off + len, b.length);
139     if (len == 0) {
140       return 0;
141     }
142 
143     // The rest of this method implements the process described by the CharsetEncoder javadoc.
144     int totalBytesRead = 0;
145     boolean doneEncoding = endOfInput;
146 
147     DRAINING:
148     while (true) {
149       // We stay in draining mode until there are no bytes left in the output buffer. Then we go
150       // back to encoding/flushing.
151       if (draining) {
152         totalBytesRead += drain(b, off + totalBytesRead, len - totalBytesRead);
153         if (totalBytesRead == len || doneFlushing) {
154           return (totalBytesRead > 0) ? totalBytesRead : -1;
155         }
156         draining = false;
157         byteBuffer.clear();
158       }
159 
160       while (true) {
161         // We call encode until there is no more input. The last call to encode will have endOfInput
162         // == true. Then there is a final call to flush.
163         CoderResult result;
164         if (doneFlushing) {
165           result = CoderResult.UNDERFLOW;
166         } else if (doneEncoding) {
167           result = encoder.flush(byteBuffer);
168         } else {
169           result = encoder.encode(charBuffer, byteBuffer, endOfInput);
170         }
171 
172         if (result.isOverflow()) {
173           // Not enough room in output buffer--drain it, creating a bigger buffer if necessary.
174           startDraining(true);
175           continue DRAINING;
176         } else if (result.isUnderflow()) {
177           // If encoder underflows, it means either:
178           // a) the final flush() succeeded; next drain (then done)
179           // b) we encoded all of the input; next flush
180           // c) we ran of out input to encode; next read more input
181           if (doneEncoding) { // (a)
182             doneFlushing = true;
183             startDraining(false);
184             continue DRAINING;
185           } else if (endOfInput) { // (b)
186             doneEncoding = true;
187           } else { // (c)
188             readMoreChars();
189           }
190         } else if (result.isError()) {
191           // Only reach here if a CharsetEncoder with non-REPLACE settings is used.
192           result.throwException();
193           return 0; // Not called.
194         }
195       }
196     }
197   }

199   /** Returns a new CharBuffer identical to buf, except twice the capacity. */
200   private static CharBuffer grow(final CharBuffer buf) {
201     char[] copy = Arrays.copyOf(buf.array(), buf.capacity() * 2);
202     CharBuffer bigger = CharBuffer.wrap(copy);
203     bigger.position(buf.position());
204     bigger.limit(buf.limit());
205     return bigger;
206   }
207 
208   /** Handle the case of underflow caused by needing more input characters. */
209   private void readMoreChars() throws IOException {
210     // Possibilities:
211     // 1) array has space available on right hand side (between limit and capacity)
212     // 2) array has space available on left hand side (before position)
213     // 3) array has no space available
214     //
215     // In case 2 we shift the existing chars to the left, and in case 3 we create a bigger
216     // array, then they both become case 1.
217 
218     if (availableCapacity(charBuffer) == 0) {
219       if (charBuffer.position() > 0) {
220         // (2) There is room in the buffer. Move existing bytes to the beginning.
221         charBuffer.compact().flip();
222       } else {
223         // (3) Entire buffer is full, need bigger buffer.
224         charBuffer = grow(charBuffer);
225       }
226     }
227 
228     // (1) Read more characters into free space at end of array.
229     int limit = charBuffer.limit();
230     int numChars = reader.read(charBuffer.array(), limit, availableCapacity(charBuffer));
231     if (numChars == -1) {
232       endOfInput = true;
233     } else {
234       charBuffer.limit(limit + numChars);
235     }
236   }
237 
238   /** Returns the number of elements between the limit and capacity. */
239   private static int availableCapacity(final Buffer buffer) {
240     return buffer.capacity() - buffer.limit();
241   }
242 
243   /**
244    * Flips the buffer output buffer so we can start reading bytes from it. If we are starting to
245    * drain because there was overflow, and there aren't actually any characters to drain, then the
246    * overflow must be due to a small output buffer.
247    */
248   private void startDraining(final boolean overflow) {
249     byteBuffer.flip();
250     if (overflow && byteBuffer.remaining() == 0) {
251       byteBuffer = ByteBuffer.allocate(byteBuffer.capacity() * 2);
252     } else {
253       draining = true;
254     }
255   }
256 
257   /**
258    * Copy as much of the byte buffer into the output array as possible, returning the (positive)
259    * number of characters copied.
260    */
261   private int drain(final byte[] b, final int off, final int len) {
262     int remaining = Math.min(len, byteBuffer.remaining());
263     byteBuffer.get(b, off, remaining);
264     return remaining;
265   }
266 
267   @Override
268   public String toString() {
269     return "ReaderInputStream{" + "reader=" + reader + ", encoder=" + encoder + ", charBuffer=" + charBuffer
270             + ", endOfInput=" + endOfInput + ", draining=" + draining + ", doneFlushing=" + doneFlushing + '}';
271   }
