31  /**
32   * Simple NTP client. Inspired by Android Sntp client.
33   * For how to use, see SntpClientTest.java.
34   * @author zoly
35   */
36  
37  public final class SntpClient {
38  
39      private SntpClient() { }
40  
41      private static final int ORIGINATE_TIME_OFFSET = 24;
42      private static final int RECEIVE_TIME_OFFSET = 32;
43      private static final int TRANSMIT_TIME_OFFSET = 40;
44      private static final int NTP_PACKET_SIZE = 48;
45  
46      private static final int NTP_PORT = 123;
47      private static final int NTP_MODE_CLIENT = 3;
48      private static final int NTP_VERSION = 3;
49  
50      // Number of seconds between Jan 1, 1900 and Jan 1, 1970
51      // 70 years plus 17 leap days
52      private static final long OFFSET_1900_TO_1970 = ((365L * 70L) + 17L) * 24L * 60L * 60L;
53  
54      private static final int MAX_SOCKET_TIMEOUT = 1000; // 5 seconds
55      public static Timing requestTimeHA(final int timeoutMillis, final String ... hosts)
56              throws IOException, InterruptedException {
57          return requestTimeHA(timeoutMillis, MAX_SOCKET_TIMEOUT, hosts);
58      }
59  
60      /**
61       * Request NTP time with retries.
62       *
63       * @param hosts - NTP server hosts.
64       * @param timeoutMillis - Max time to attempt to get  NTP time
65       * @param ntpResponseTimeout - the time after which if we do not receive a response from the NTP server,
66       *                              we consider the call failed (and will retry until timeoutMillis.
67       * @return Ntp timing info.
68       * @throws IOException - thrown in case of time server connectivity issues.
69       * @throws InterruptedException - thrown if exec interrupted.
70       */
71      @SuppressFBWarnings("BED_BOGUS_EXCEPTION_DECLARATION") //findbugs nonsense
72      public static Timing requestTimeHA(final int timeoutMillis, final int ntpResponseTimeout, final String ... hosts)
73              throws IOException, InterruptedException {
74          return Callables.executeWithRetry(new Callables.TimeoutCallable<Timing, IOException>(timeoutMillis) {
75  
76              private int i = 0;
77  
78              @Override
79              @SuppressFBWarnings("BED_BOGUS_EXCEPTION_DECLARATION") //findbugs nonsense
80              public Timing call(final long deadline) throws IOException {
81                  int hostIdx = Math.abs(i++) % hosts.length;
82                  return requestTime(hosts[hostIdx],
83                          Math.min((int) (deadline - System.currentTimeMillis()), ntpResponseTimeout));
84              }
85          }, 3, 1000);
86      }
87  
88      /**
89       * Get NTP time.
90       * @param host - NTP server host name.
91       * @param timeoutMillis - the socket timeout.
92       * @return - NTP server timing info.
93       * @throws IOException - thrown in case of time server connectivity issues.
94       */
95      @SuppressFBWarnings("NP_LOAD_OF_KNOWN_NULL_VALUE") // false positive
96      public static Timing requestTime(final String host, final int timeoutMillis) throws IOException {
97  
98          try (DatagramSocket socket = new DatagramSocket()) {
99              socket.setSoTimeout(timeoutMillis);
100             InetAddress address = InetAddress.getByName(host);
101             byte[] buffer = new byte[NTP_PACKET_SIZE];
102             DatagramPacket request = new DatagramPacket(buffer, buffer.length, address, NTP_PORT);
103 
104             // set mode = 3 (client) and version = 3
105             // mode is in low 3 bits of first byte
106             // version is in bits 3-5 of first byte
107             buffer[0] = NTP_MODE_CLIENT | (NTP_VERSION << 3);
108             // Allocate response.
109             DatagramPacket response = new DatagramPacket(buffer, buffer.length);
110             // get current nanotime.
111             long requestTicks = System.nanoTime() / 1000000L;
112             // get current time and write it to the request packet
113             long requestTime = System.currentTimeMillis();
114             writeTimeStamp(buffer, TRANSMIT_TIME_OFFSET, requestTime);
115             socket.send(request);
116             // read the response
117             socket.receive(response);
118             long responseTicks = System.nanoTime() / 1000000L;
119             long responseTime = requestTime + (responseTicks - requestTicks);
120 
121             // extract the results
122             long originateTime = readTimeStamp(buffer, ORIGINATE_TIME_OFFSET);
123             long receiveTime = readTimeStamp(buffer, RECEIVE_TIME_OFFSET);
124             long transmitTime = readTimeStamp(buffer, TRANSMIT_TIME_OFFSET);
125             long roundTripTime = responseTicks - requestTicks - (transmitTime - receiveTime);
126             // receiveTime = originateTime + transit + skew
127             // responseTime = transmitTime + transit - skew
128             // clockOffset = ((receiveTime - originateTime) + (transmitTime - responseTime))/2
129             //             = ((originateTime + transit + skew - originateTime) +
130             //                (transmitTime - (transmitTime + transit - skew)))/2
131             //             = ((transit + skew) + (transmitTime - transmitTime - transit + skew))/2
132             //             = (transit + skew - transit + skew)/2
133             //             = (2 * skew)/2 = skew
134             long clockOffset = ((receiveTime - originateTime) + (transmitTime - responseTime)) / 2;
135             return new Timing(responseTime + clockOffset, responseTicks, roundTripTime);
136         }
137     }

139     /**
140      * Reads an unsigned 32 bit big endian number from the given offset in the buffer.
141      */
142     private static long read32(final byte[] buffer, final int offset) {
143         byte b0 = buffer[offset];
144         byte b1 = buffer[offset + 1];
145         byte b2 = buffer[offset + 2];
146         byte b3 = buffer[offset + 3];
147 
148         // convert signed bytes to unsigned values
149         int i0 = ((b0 & 0x80) == 0x80 ? (b0 & 0x7F) + 0x80 : b0);
150         int i1 = ((b1 & 0x80) == 0x80 ? (b1 & 0x7F) + 0x80 : b1);
151         int i2 = ((b2 & 0x80) == 0x80 ? (b2 & 0x7F) + 0x80 : b2);
152         int i3 = ((b3 & 0x80) == 0x80 ? (b3 & 0x7F) + 0x80 : b3);
153 
154         return ((long) i0 << 24) + ((long) i1 << 16) + ((long) i2 << 8) + (long) i3;
155     }
156 
157     /**
158      * Reads the NTP time stamp at the given offset in the buffer and returns
159      * it as a system time (milliseconds since January 1, 1970).
160      */
161     private static long readTimeStamp(final byte[] buffer, final int offset) {
162         long seconds = read32(buffer, offset);
163         long fraction = read32(buffer, offset + 4);
164         return ((seconds - OFFSET_1900_TO_1970) * 1000) + ((fraction * 1000L) / 0x100000000L);
165     }
166 
167 
168     /**
169      * Writes system time (milliseconds since January 1, 1970) as an NTP time stamp
170      * at the given offset in the buffer.
171      */
172     private static void writeTimeStamp(final byte[] buffer, final int poffset, final long time) {
173         int offset = poffset;
174         long seconds = time / 1000L;
175         long milliseconds = time - seconds * 1000L;
176         seconds += OFFSET_1900_TO_1970;
177 
178         // write seconds in big endian format
179         buffer[offset++] = (byte) (seconds >> 24);
180         buffer[offset++] = (byte) (seconds >> 16);
181         buffer[offset++] = (byte) (seconds >> 8);
182         buffer[offset++] = (byte) (seconds);
183 
184         long fraction = milliseconds * 0x100000000L / 1000L;
185         // write fraction in big endian format
186         buffer[offset++] = (byte) (fraction >> 24);
187         buffer[offset++] = (byte) (fraction >> 16);
188         buffer[offset++] = (byte) (fraction >> 8);
189         // low order bits should be random data
190         buffer[offset] = (byte) (IntMath.XorShift32ThreadSafe.Singleton.INSTANCE.nextInt() % 256);
191     }
