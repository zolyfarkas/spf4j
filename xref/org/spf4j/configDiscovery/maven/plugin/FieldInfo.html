3   import javax.annotation.Nullable;
4   import org.spf4j.base.asm.UnknownValue;
5   
6   /**
7    *
8    * @author zoly
9    */
10  public final class FieldInfo<T> {
11  
12    private final String namespace;
13  
14    private final Class<T> type;
15  
16    private final Object defaultValue;
17  
18    private final String doc;
19  
20    public FieldInfo(final String namespace, final String doc, final Class<T> type, final T defaultValue) {
21      this.namespace = namespace;
22      this.type = type;
23      if (defaultValue != null && defaultValue.getClass() == UnknownValue.class) {
24        this.defaultValue = null;
25      } else {
26        this.defaultValue = defaultValue;
27      }
28      this.doc = doc;
29    }
30  
31    public Class<?> getType() {
32      return type;
33    }
34  
35    @Nullable
36    public Object getDefaultValue() {
37      return defaultValue;
38    }
39  
40    public String getDoc() {
41      return doc;
42    }
43  
44    public String getNamespace() {
45      return namespace;
46    }
47  
48  
49    @Override
50    public String toString() {
51      return "FieldInfo{" + "type=" + type + ", defaultValue=" + defaultValue + ", doc=" + doc + '}';
52    }
53  
54  
55  }
