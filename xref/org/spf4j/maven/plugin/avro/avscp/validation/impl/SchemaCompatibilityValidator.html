55  /**
56   * Validates previously released schemas for backward compatibility.
57   *
58   * <p>The following configurations are available for this "compatibility" validator:</p>
59   * <ul>
60   * <li>versionRange - maven version range to check compatibility against. (defaults to
61   * "[," + mavenProject.getVersion() +  ')' )</li>
62   *
63   * <li>maxNrOfVersionsToCheckForCompatibility = max number oof versions to check against.
64   * (defaults to 30)</li>
65   *
66   * <li>maxNrOfDaysBackCheckForCompatibility - max released time to check against (defaults to 1 year)</li>
67   *
68   * <li>deprecationRemoval - validates  that a removed schema
69   * has been previously deprecated (false by default)</li>
70   *
71   * <li>schemaArtifactClassifier - the classifier of the artifact that contains the schema files.
72   * defaults to null (the main jar)</li>
73   *
74   * <li>schemaArtifactExtension - the extension of the artifact that contains the schema files.
75   * defaults to jar </li>
76   *
77   * </ul>
78   *
79   * <p>The following validations are performed:</p>
80   *
81   * <ul>
82   * <li>1) We check previously released schema for (reader to writer) and (writer to reader) compatibility
83   * with the current schema. Unless compatibility is not desired ( via "beta" schema annotation,
84   * or only partial compatibility is desired with  "noNewToOldCompatibility" "noOldToNewCompatibility"
85   * schema annotations. (objects written with old schema don't need to be converted to new objects)
86   * or "noNewToOldCompatibility" if (objects written with new schema don't need to be converted to old schema objects)
87   * </li>
88   *
89   * <li>
90   * 2) Schema deprecation and removal policy can also enabled with compatibiliy.deprecationRemoval = true,
91   * this will validate that a schema has been deprecated during the entire compatibility interval.
92   * </li>
93   * </ul>
94   * @author Zoltan Farkas
95   */
96  public final class SchemaCompatibilityValidator implements Validator<Void> {

98    @Override
99    public String getName() {

103   @Override
104   @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
105   public Result validate(final Void nv, final ValidatorMojo mojo) throws IOException {
106     // loop through dependencies.
107     MavenProject mavenProject = mojo.getMavenProject();
108     Map<String, String> validatorConfigs = mojo.getValidatorConfigs();
109     Log log = mojo.getLog();
110     String versionRange = validatorConfigs.get("versionRange");
111     if (versionRange == null) {
112       versionRange = "[," + mavenProject.getVersion() +  ')';
113     }
114     int maxNrVersToCheck = 30;
115     String strNrVer = validatorConfigs.get("maxNrOfVersionsToCheckForCompatibility");
116     if (strNrVer != null) {
117       maxNrVersToCheck = Integer.parseInt(strNrVer);
118     }
119     Instant instantToGoBack;
120     String strNrDays = validatorConfigs.get("maxNrOfDaysBackCheckForCompatibility");
121     if (strNrDays == null) {
122       instantToGoBack = Instant.now().atOffset(ZoneOffset.UTC).minus(1,  ChronoUnit.YEARS).toInstant();
123     } else {
124       instantToGoBack = Instant.now().atOffset(ZoneOffset.UTC)
125               .minus(Integer.parseInt(strNrDays), ChronoUnit.DAYS).toInstant();
126     }
127     String groupId = mavenProject.getGroupId();
128     String artifactId = mavenProject.getArtifactId();
129     List<RemoteRepository> remoteProjectRepositories = mavenProject.getRemoteProjectRepositories();
130     RepositorySystem repoSystem = mojo.getRepoSystem();
131     RepositorySystemSession repositorySession = mojo.getMavenSession().getRepositorySession();
132     List<Version> rangeVersions;
133     try {
134       rangeVersions = MavenRepositoryUtils.getVersions(groupId, artifactId, versionRange,
135               remoteProjectRepositories, repoSystem, repositorySession);
136     } catch (VersionRangeResolutionException ex) {
137       throw new RuntimeException("Invalid compatibiliy.versionRange = " + versionRange + " setting", ex);
138     }
139     rangeVersions = rangeVersions.stream().filter((v) -> !v.toString().endsWith("SNAPSHOT"))
140             .collect(Collectors.toList());
141     int tSize = rangeVersions.size();
142     rangeVersions = rangeVersions.subList(Math.max(tSize - maxNrVersToCheck, 0), tSize);
143     log.info("Validating compatibility with previous versions " + rangeVersions + " newer than " + instantToGoBack);
144     if (rangeVersions.isEmpty()) {
145       return Result.valid();
146     }
147     String schemaArtifactClassifier = validatorConfigs.get("schemaArtifactClassifier");
148     if (schemaArtifactClassifier != null && schemaArtifactClassifier.trim().isEmpty()) {
149       schemaArtifactClassifier = null;
150     }
151     String schemaArtifactExtension = validatorConfigs.get("schemaArtifactExtension");
152     if (schemaArtifactExtension == null || schemaArtifactExtension.trim().isEmpty()) {
153       schemaArtifactExtension = "jar";
154     }
155     List<String> issues = new ArrayList<>(4);
156     int size = rangeVersions.size();
157     for (int  i = size - 1; i >= 0; i--) {
158       Version version  = rangeVersions.get(i);
159       validateCompatibility(groupId, artifactId, schemaArtifactClassifier, schemaArtifactExtension, version,
160               remoteProjectRepositories, repoSystem, repositorySession, mojo,
161               Boolean.parseBoolean(validatorConfigs.getOrDefault("deprecationRemoval", "false")),
162               instantToGoBack, issues::add);
163     }
164     if (issues.isEmpty()) {
165       return Result.valid();
166     } else {
167       return Result.failed("Schema compatibility issues:\n" + String.join("\n", issues));
168     }
169   }

171   @SuppressFBWarnings("PCAIL_POSSIBLE_CONSTANT_ALLOCATION_IN_LOOP") // Schema.Parser is mutable, false positive.
172   @SuppressWarnings("checkstyle:ParameterNumber") // valid will address later.
173   public void validateCompatibility(final String groupId, final String artifactId,
174           @Nullable final String classifier, final String extension,
175           final Version version,
176           final List<RemoteRepository> remoteProjectRepositories, final RepositorySystem repoSystem,
177           final RepositorySystemSession repositorySession,
178           final ValidatorMojo mojo,
179           final boolean deprecationRemoval, final Instant instantToGoBack, final Consumer<String> issues)
180           throws IOException {
181     Log log = mojo.getLog();
182     log.info("Validating compatibility with version: " + version + ", " + groupId + ':' + artifactId
183             + ':' + classifier + ':' + extension);
184     Path targetPath = mojo.getTarget().toPath();
185     Path currSchemasPath = mojo.getGeneratedAvscTarget().toPath();
186     File prevSchemaArchive;
187     try {
188       prevSchemaArchive = MavenRepositoryUtils.resolveArtifact(
189               groupId, artifactId, classifier, extension, version.toString(),
190               remoteProjectRepositories, repoSystem, repositorySession);
191     } catch (ArtifactResolutionException ex) {
192       throw new RuntimeException("Cannot resolve previous version "  + version, ex);
193     }
194     Path dest = targetPath.resolve("prevSchemas").resolve(version.toString());
195     Files.createDirectories(dest);
196     log.debug("Unzipping " + prevSchemaArchive + " to " + dest);
197     List<Path> prevSchemas = Compress.unzip2(prevSchemaArchive.toPath(), dest, (Path p) -> {
198       Path fileName = p.getFileName();
199       if (fileName == null) {
200         return false;
201       }
202       String fname = fileName.toString();
203       return (fname.endsWith("avsc")
204               || SchemaCompileMojo.SCHEMA_MANIFEST.equals(fname)
205               || "MANIFEST.MF".equals(fname));
206     });
207     Instant dependencyBuildTime = getDependencyBuildTime(dest, log);
208     if (dependencyBuildTime == null) {
209       log.info("Package " + dest + " build time missing from manifest");
210     } else if (dependencyBuildTime.isBefore(instantToGoBack)) {
211       return;
212     }
213     for (Path prevSchemaPath : prevSchemas) {
214       Path fileName = prevSchemaPath.getFileName();
215       if (fileName == null) {
216         continue;
217       }
218       if (!fileName.toString().endsWith("avsc")) {
219         continue;
220       }
221       Path relPath = dest.relativize(prevSchemaPath);
222       Path newSchemaPath = currSchemasPath.resolve(relPath);
223       Schema previousSchema = new Schema.Parser().parse(prevSchemaPath.toFile());
224       String previousSchemaName = previousSchema.getFullName();
225       if (previousSchema.getProp("beta") != null) {
226         log.debug("Skipping beta schema " + previousSchemaName);
227         continue;
228       }
229       log.debug("Validating compatibility for " + previousSchemaName + " "
230               + prevSchemaPath + " -> "  + newSchemaPath);
231       if (!Files.exists(newSchemaPath)) {
232         if (deprecationRemoval && previousSchema.getProp("deprecated") == null) {
233           issues.accept(previousSchemaName + " is being removed without being deprecated first");
234         }
235       } else {
236         Schema newSchema = new Schema.Parser().parse(newSchemaPath.toFile());
237         if (newSchema.getProp("beta") != null) {
238           log.debug("Skipping beta schema " + newSchema.getFullName());
239         } else {
240           if (newSchema.getProp("noOldToNewCompatibility") == null) {
241             SchemaCompatibility.SchemaPairCompatibility o2n
242                     = SchemaCompatibility.checkReaderWriterCompatibility(newSchema, previousSchema);
243             if (o2n.getType() == SchemaCompatibility.SchemaCompatibilityType.INCOMPATIBLE) {
244               issues.accept(newSchema.getFullName() + " cannot convert previous versions " + version + " to current"
245                     + " detail: " + o2n);
246             }
247           }
248           if (newSchema.getProp("noNewToOldCompatibility") == null) {
249             SchemaCompatibility.SchemaPairCompatibility n2o
250                     = SchemaCompatibility.checkReaderWriterCompatibility(previousSchema, newSchema);
251             if (n2o.getType() == SchemaCompatibility.SchemaCompatibilityType.INCOMPATIBLE) {
252               issues.accept(newSchema.getFullName() + " cannot convert current to previos version " + version
253               + " detail: " + n2o);
254             }
255           }
256         }
257       }
258     }
259   }

261   @Nullable
262   private static Instant getDependencyBuildTime(final Path location, final Log log) throws IOException {
263     Path jarManifest = location.resolve("META-INF/MANIFEST.MF");
264     if (Files.exists(jarManifest)) {
265       try (BufferedInputStream bis = new BufferedInputStream(
266               Files.newInputStream(jarManifest))) {
267         Manifest manifest = new Manifest(bis);
268         Attributes mainAttributes = manifest.getMainAttributes();
269         String buildTime = mainAttributes.getValue("Build-Time");
270         if (buildTime != null) {
271           try {
272             return Instant.parse(buildTime);
273           } catch (DateTimeParseException ex) {
274             log.warn("Cannot parse manifest build time " + buildTime, ex);
275           }
276         }
277       }
278     }
279     Path codegenManifest = location.resolve(SchemaCompileMojo.SCHEMA_MANIFEST);
280     if (Files.exists(codegenManifest)) {
281       try (BufferedReader br = Files.newBufferedReader(codegenManifest, StandardCharsets.UTF_8)) {
282         Properties props = new Properties();
283         props.load(br);
284         String buildTime = props.getProperty("Build-Time");
285         if (buildTime != null) {
286           try {
287             return Instant.parse(buildTime);
288           } catch (DateTimeParseException ex) {
289             log.warn("Cannot parse manifest build time " + buildTime, ex);
290           }
291         }
292       }
293     }
294     return null;
295   }
