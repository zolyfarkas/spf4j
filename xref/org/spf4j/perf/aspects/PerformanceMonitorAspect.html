19  package org.spf4j.perf.aspects;
20  
21  import com.google.common.cache.CacheLoader;
22  import com.google.common.cache.LoadingCache;
23  import org.spf4j.perf.MeasurementRecorderSource;
24  import org.aspectj.lang.ProceedingJoinPoint;
25  import org.aspectj.lang.annotation.Around;
26  import org.aspectj.lang.annotation.Aspect;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  import org.spf4j.concurrent.UnboundedLoadingCache;
30  import org.spf4j.annotations.PerformanceMonitor;
31  import org.spf4j.annotations.RecorderSourceInstance;
32  
33  /**
34   * Aspect that measures execution time and does performance logging
35   * for all methods annotated with: PerformanceMonitor annotation.
36   * 
37   * @author zoly
38   */
39  @Aspect
40  public final class PerformanceMonitorAspect {
41  
42      private static final Logger LOG = LoggerFactory.getLogger(PerformanceMonitorAspect.class);
43      private static final LoadingCache<Class<? extends RecorderSourceInstance>, MeasurementRecorderSource> REC_SOURCES =
44              new UnboundedLoadingCache<Class<? extends RecorderSourceInstance>, MeasurementRecorderSource>(
45                      32, new CacheLoader<Class<? extends RecorderSourceInstance>, MeasurementRecorderSource>() {
46          @Override
47          public MeasurementRecorderSource load(final Class<? extends RecorderSourceInstance> key) throws Exception {
48              return (MeasurementRecorderSource) key.getField("INSTANCE").get(null);
49          }
50      });
51  
52      @Around(value = "execution(@org.spf4j.annotations.PerformanceMonitor * *(..)) && @annotation(annot)",
53              argNames = "pjp,annot")
54      public Object performanceMonitoredMethod(final ProceedingJoinPoint pjp, final PerformanceMonitor annot)
55              throws Throwable {
56          final long start = System.currentTimeMillis();
57          Object result = pjp.proceed();
58          final long elapsed = System.currentTimeMillis() - start;
59          MeasurementRecorderSource mrs = REC_SOURCES.getUnchecked(annot.recorderSource());
60          mrs.getRecorder(pjp.toLongString()).record(elapsed);
61          final long warnThresholdMillis = annot.warnThresholdMillis();
62          if (elapsed > warnThresholdMillis) {
63              final long errorThresholdMillis = annot.errorThresholdMillis();
64              if (elapsed > errorThresholdMillis) {
65                  LOG.error("Execution time  {} ms for {} exceeds error threshold of {} ms, arguments {}",
66                              elapsed, pjp.toShortString(), errorThresholdMillis, pjp.getArgs());
67              } else {
68                  LOG.warn("Execution time  {} ms for {} exceeds warning threshold of {} ms, arguments {}",
69                              elapsed, pjp.toShortString(), warnThresholdMillis, pjp.getArgs());
70              }
71          } else {
72              if (annot.defaultInfoLog()) {
73                  LOG.info("Execution time {} ms for {}, arguments {}",
74                          elapsed, pjp.toShortString(), pjp.getArgs());
75              } else {
76                  LOG.debug("Execution time {} ms for {}, arguments {}",
77                          elapsed, pjp.toShortString(), pjp.getArgs());
78              }
79          }
80          return result;
81      }
82  }
