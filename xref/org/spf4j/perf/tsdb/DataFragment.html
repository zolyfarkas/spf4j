18  package org.spf4j.perf.tsdb;
19  
20  import gnu.trove.list.array.TIntArrayList;
21  import java.io.ByteArrayInputStream;
22  import java.io.DataInput;
23  import java.io.DataInputStream;
24  import java.io.DataOutput;
25  import java.io.DataOutputStream;
26  import java.io.IOException;
27  import java.io.RandomAccessFile;
28  import java.util.ArrayList;
29  import java.util.List;
30  import org.spf4j.io.ByteArrayBuilder;
31  
32  /**
33   *
34   * @author zoly
35   */
36  final class DataFragment {
37  
38      private long location;
39      private long nextDataFragment;
40      private final long startTimeMillis;
41      private List<long[]> data;
42      private TIntArrayList timestamps;
43  
44      public DataFragment(final long startTimeMillis) {
45          this.location = 0;
46          this.nextDataFragment = 0;
47          this.startTimeMillis = startTimeMillis;
48          data = new ArrayList<>();
49          timestamps = new TIntArrayList();
50      }
51  
52      public DataFragment(final RandomAccessFile raf) throws IOException {
53          location = raf.getFilePointer();
54          this.nextDataFragment = raf.readLong();
55          this.startTimeMillis = raf.readLong();
56          int nrSamples = raf.readInt();
57          int samplesLength = raf.readInt();
58          int bufferSize = nrSamples * (samplesLength * 8 + 4);
59          byte[] buffer = new byte[bufferSize];
60          raf.readFully(buffer);
61          loadData(nrSamples, samplesLength, new DataInputStream(new ByteArrayInputStream(buffer)));
62      }
63  
64      public void writeTo(final DataOutput dos) throws IOException {
65          dos.writeLong(nextDataFragment);
66          dos.writeLong(startTimeMillis);
67          dos.writeInt(data.size());
68          dos.writeInt(data.get(0).length);
69          for (int i = 0; i < timestamps.size(); i++) {
70              dos.writeInt(timestamps.get(i));
71              for (long value : data.get(i)) {
72                  dos.writeLong(value);
73              }
74          }
75      }
76  
77      public void writeTo(final RandomAccessFile raf) throws IOException {
78          ByteArrayBuilder bos = new ByteArrayBuilder();
79          DataOutput dos = new DataOutputStream(bos);
80          writeTo(dos);
81          raf.seek(location);
82          raf.write(bos.getBuffer(), 0, bos.size());
83      }
84  
85      public void addData(final long timestamp, final long[] dataRow) {
86          data.add(dataRow);
87          timestamps.add((int) (timestamp - startTimeMillis));
88      }
89  
90      public void setNextDataFragment(final long pnextDataFragment, final RandomAccessFile raf) throws IOException {
91          this.nextDataFragment = pnextDataFragment;
92          setNextDataFragment(location, nextDataFragment, raf);
93      }
94  
95      public static void setNextDataFragment(final long dataFragmentPosition, final long nextDataFragment,
96              final RandomAccessFile raf) throws IOException {
97          raf.seek(dataFragmentPosition);
98          raf.writeLong(nextDataFragment);
99      }
100 
101     public long getNextDataFragment() {
102         return nextDataFragment;
103     }
104 
105     public long getLocation() {
106         return location;
107     }
108 
109     private void loadData(final int nrSamples, final int samplesLength, final DataInput raf) throws IOException {
110         data = new ArrayList(nrSamples);
111         timestamps = new TIntArrayList(nrSamples);
112         for (int i = 0; i < nrSamples; i++) {
113             timestamps.add(raf.readInt());
114             long[] row = new long[samplesLength];
115             for (int j = 0; j < samplesLength; j++) {
116                 row[j] = raf.readLong();
117             }
118             data.add(row);
119         }
120     }
121 
122     public long getStartTimeMillis() {
123         return startTimeMillis;
124     }
125 
126     public int getNrRows() {
127         return data.size();
128     }
129 
130     public List<long[]> getData() {
131         return data;
132     }
133 
134     public TIntArrayList getTimestamps() {
135         return timestamps;
136     }
137 
138     public void setLocation(final long location) {
139         this.location = location;
140     }
141 
142     @Override
143     public String toString() {
144         return "DataFragment{" + "location=" + location + ", nextDataFragment=" + nextDataFragment
145                 + ", startTimeMillis=" + startTimeMillis + ", data=" + data + ", timestamps=" + timestamps + '}';
146     }
147 
148 }
