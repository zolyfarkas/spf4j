60  /**
61   * Yet another time series database. Why? because all the other ts databases had various constraints that restrict the
62   * functionality I can add to spf4j.
63   *
64   * Initial Features:
65   *
66   * 1. measurements can be added dynamically anytime to a database.
67   * 2. long measurement names.
68   * 3. the stored interval is not known from the beginning.
69   * 4. implementation biased towards write performance.
70   *
71   * Format:
72   *
73   * Header: TSDB[version:int][metadata:bytes]
74   * Table of Contents: firstTableInfoPtr, lastTableInfoPtr
75   * TableInfo:
76   * DataFragment:
77   * ......
78   * ......
79   * TableInfo:
80   * DataFragment:
81   * EOF
82   * 
83   * @deprecated please use org.spf4j.tsdb2
84   * @author zoly
85   */

110     public TimeSeriesDatabase(final String pathToDatabaseFile, final boolean isWrite, final byte... metaData)
111             throws IOException {
112         file = new RandomAccessFile(pathToDatabaseFile, isWrite ? "rw" : "r");
113         // uniques per process string for sync purposes.
114         this.path = INTERNER.intern(new File(pathToDatabaseFile).getPath());
115         tables = new ConcurrentHashMap<>();
116         writeDataFragments = new HashMap<>();
117         // read or create header
118         synchronized (path) {
119             this.ch = file.getChannel();
120             FileLock lock;
121             if (isWrite) {
122                 lock = ch.lock();
123             } else {
124                 lock = ch.lock(0, Long.MAX_VALUE, true);
125             }
126             try {
127                 if (file.length() == 0) {
128                     this.header = new Header(VERSION, metaData);
129                     this.header.writeTo(file);
130                     this.toc = new TableOfContents(file.getFilePointer());
131                     this.toc.writeTo(file);
132                 } else {
133                     this.header = new Header(file);
134                     this.toc = new TableOfContents(file);
135                 }
136             } catch (IOException | RuntimeException e) {
137                 try {
138                     lock.release();
139                     throw e;
140                 } catch (IOException ex) {
141                     ex.addSuppressed(e);
142                     throw ex;
143                 }
144             }
145             lock.release();
146             lock = ch.lock(0, Long.MAX_VALUE, true);
147             try {
148                 readTableInfos();
149             } catch (IOException | RuntimeException e) {
150                 try {
151                     lock.release();
152                     throw e;
153                 } catch (IOException ex) {
154                     ex.addSuppressed(e);
155                     throw ex;
156                 }
157             }
158             lock.release();
159         }
160     }

162     public void reReadTableInfos() throws IOException {
163         synchronized (path) {
164             FileLock lock = ch.lock(0, Long.MAX_VALUE, true);
165             try {
166                 toc = new TableOfContents(file, toc.getLocation()); // reread toc
167                 readTableInfos();
168             } catch (IOException | RuntimeException e) {
169                 try {
170                     lock.release();
171                     throw e;
172                 } catch (IOException ex) {
173                     ex.addSuppressed(e);
174                     throw ex;
175                 }
176             }
177             lock.release();
178         }
179     }

181     private void readTableInfos() throws IOException {
182         final long firstColumnInfo = toc.getFirstTableInfoPtr();
183         if (firstColumnInfo > 0) {
184             file.seek(firstColumnInfo);
185             TSTable colInfo = new TSTable(file);
186             tables.put(colInfo.getTableName(), colInfo);
187             lastTableInfo = colInfo;
188             while (colInfo.getNextTSTable() > 0) {
189                 file.seek(colInfo.getNextTSTable());
190                 colInfo = new TSTable(file);
191                 tables.put(colInfo.getTableName(), colInfo);
192                 lastTableInfo = colInfo;
193             }
194         }
195     }
196 
197     private void readLastTableInfo() throws IOException {
198         toc = new TableOfContents(file, toc.getLocation()); // reread toc
199         if (toc.getLastTableInfoPtr() == 0) {
200             return;
201         }
202         lastTableInfo = new TSTable(file, toc.getLastTableInfoPtr()); // update last table info
203     }

231     public void addTSTable(final String tableName,
232             final byte[] tableMetaData, final int sampleTime, final String[] columnNames,
233             final byte[][] columnMetaData) throws IOException {
234         synchronized (path) {
235             if (hasTSTable(tableName)) {
236                 throw new IllegalArgumentException("group already exists " + tableName);
237             }
238             flush();
239             FileLock lock = ch.lock();
240             TSTable colInfo;
241             try {
242                 readLastTableInfo();
243                 //write column information at the end of the file.
244                 file.seek(file.length());
245                 colInfo = new TSTable(tableName, tableMetaData, columnNames,
246                         columnMetaData, sampleTime, file.getFilePointer());
247                 colInfo.writeTo(file);
248                 //update refferences to this new TableInfo.
249                 if (lastTableInfo != null) {
250                     lastTableInfo.setNextColumnInfo(colInfo.getLocation(), file);
251                 } else {
252                     toc.setFirstTableInfo(colInfo.getLocation(), file);
253                 }
254                 toc.setLastTableInfo(colInfo.getLocation(), file);
255             } catch (IOException | RuntimeException e) {
256                 try {
257                     lock.release();
258                     throw e;
259                 } catch (IOException ex) {
260                     ex.addSuppressed(e);
261                     throw ex;
262                 }
263             }
264             lock.release();
265 
266             lastTableInfo = colInfo;
267             tables.put(tableName, colInfo);
268         }
269     }

271     public void write(final long time, final String tableName, final long[] values) {
272         if (!hasTSTable(tableName)) {
273             throw new IllegalArgumentException("Unknown table name" + tableName);
274         }
275         synchronized (writeDataFragments) {
276             DataFragment writeDataFragment = writeDataFragments.get(tableName);
277             if (writeDataFragment == null) {
278                 writeDataFragment = new DataFragment(time);
279                 writeDataFragments.put(tableName, writeDataFragment);
280             }
281             writeDataFragment.addData(time, values);
282         }
283     }

285     public void flush() throws IOException {
286         synchronized (path) {
287             List<Map.Entry<String, DataFragment>> lwriteDataFragments;
288             synchronized (writeDataFragments) {
289                 if (writeDataFragments.isEmpty()) {
290                     return;
291                 }
292                 lwriteDataFragments = new ArrayList<>(writeDataFragments.entrySet());
293                 writeDataFragments.clear();
294             }
295             FileLock lock = ch.lock();
296             try {
297                 for (Map.Entry<String, DataFragment> entry : lwriteDataFragments) {
298                     DataFragment writeDataFragment = entry.getValue();
299                     String groupName = entry.getKey();
300                     file.seek(file.length());
301                     writeDataFragment.setLocation(file.getFilePointer());
302                     writeDataFragment.writeTo(file);
303                     TSTable colInfo = tables.get(groupName);
304                     colInfo = new TSTable(file, colInfo.getLocation()); // reread colInfo
305                     tables.put(groupName, colInfo); // update colInfo
306                     final long lastDataFragment = colInfo.getLastDataFragment();
307                     final long location = writeDataFragment.getLocation();
308                     if (lastDataFragment != 0) {
309                         DataFragment.setNextDataFragment(lastDataFragment, location, file);
310                     } else {
311                         colInfo.setFirstDataFragment(location, file);
312                     }
313                     colInfo.setLastDataFragment(location, file);
314                 }
315                 sync();
316             } catch (IOException | RuntimeException e) {
317                 try {
318                     lock.release();
319                     throw e;
320                 } catch (IOException ex) {
321                     ex.addSuppressed(e);
322                     throw ex;
323                 }
324             }
325             lock.release();
326         }
327     }

391     /**
392      * Read measurements from table.
393      *
394      * @param tableName
395      * @param startTime start time including
396      * @param endTime end time including
397      * @return
398      * @throws IOException
399      */
400     private TimeSeries read(
401             final long startTime, final long endTime,
402             final long startAtFragment, final long endAtFragment,
403             final boolean skipFirst)
404             throws IOException {
405         synchronized (path) {
406             TLongArrayList timeStamps = new TLongArrayList();
407             List<long[]> data = new ArrayList<>();
408             if (startAtFragment > 0) {
409                 FileLock lock = ch.lock(0, Long.MAX_VALUE, true);
410                 try {
411                     DataFragment frag;
412                     long nextFragmentLocation = startAtFragment;
413                     boolean last = false;
414                     boolean psFirst = skipFirst;
415                     do {
416                         if (nextFragmentLocation == endAtFragment) {
417                             last = true;
418                         }
419                         file.seek(nextFragmentLocation);
420                         frag = new DataFragment(file);
421                         if (psFirst) {
422                             psFirst = false;
423                         } else {
424                         long fragStartTime = frag.getStartTimeMillis();
425                         if (fragStartTime >= startTime) {
426                             TIntArrayList fragTimestamps = frag.getTimestamps();
427                             int nr = 0;
428                             for (int i = 0; i < fragTimestamps.size(); i++) {
429                                 long ts = fragStartTime + fragTimestamps.get(i);
430                                 if (ts <= endTime) {
431                                     timeStamps.add(ts);
432                                     nr++;
433                                 } else {
434                                     break;
435                                 }
436                             }
437                             int i = 0;
438                             for (long[] d : frag.getData()) {
439                                 if (i < nr) {
440                                     data.add(d);
441                                 } else {
442                                     break;
443                                 }
444                                 nr++;
445                             }
446                             if (fragTimestamps.size() > nr) {
447                                 break;
448                             }
449                         }
450                         }
451                         nextFragmentLocation = frag.getNextDataFragment();
452                     } while (nextFragmentLocation > 0 && !last);
453                 } catch (IOException | RuntimeException e) {
454                     try {
455                         lock.release();
456                         throw e;
457                     } catch (IOException ex) {
458                         ex.addSuppressed(e);
459                         throw ex;
460                     }
461                 }
462                 lock.release();
463             }
464             return new TimeSeries(timeStamps.toArray(), data.toArray(new long[data.size()][]));
465         }
466     }

521         Pair<long[], double[][]> mData = fillGaps(data.getTimeStamps(), data.getValues(),
522                 info.getSampleTime(), info.getColumnNames().length);
523         final int totalColumnIndex = info.getColumnIndex("total");

525                 mData.getSecond(), data.getTimeStamps()[0], info.getSampleTime(),
526                 new String(info.getColumnMetaData()[totalColumnIndex], Charsets.UTF_8), "Measurements distribution for "
527                 + info.getTableName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j");

530     public static JFreeChart createMinMaxAvgJFreeChart(final TimeSeries data, final TSTable info) {
531         long[][] vals = data.getValues();
532         double[] min = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("min"));
533         double[] max = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("max"));
534         final int totalColumnIndex = info.getColumnIndex("total");
535         double[] total = Arrays.getColumnAsDoubles(vals, totalColumnIndex);
536         double[] count = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("count"));
537         for (int i = 0; i < count.length; i++) {
538             if (count[i] == 0) {
539                 min[i] = 0;
540                 max[i] = 0;
541             }
542         }
543         long[] timestamps = data.getTimeStamps();
544         return Charts.createTimeSeriesJFreeChart("Min,Max,Avg chart for "
545                 + info.getTableName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j", timestamps,
546                 new String[]{"min", "max", "avg"}, new String(info.getColumnMetaData()[totalColumnIndex],
547                         Charsets.UTF_8), new double[][]{min, max, Arrays.divide(total, count)});
548     }

559     public static List<JFreeChart> createJFreeCharts(final TimeSeries data, final TSTable info) {
560         long[][] vals = data.getValues();
561         List<JFreeChart> result = new ArrayList<>();
562         Map<String, Pair<List<String>, List<double[]>>> measurementsByUom = new HashMap<>();
563         String[] columnMetaData = info.getColumnMetaDataAsStrings();
564         for (int i = 0; i < info.getColumnNumber(); i++) {
565             String uom = columnMetaData[i];
566             Pair<List<String>, List<double[]>> meas = measurementsByUom.get(uom);
567             if (meas == null) {
568                 meas = Pair.of((List<String>) new ArrayList<String>(), (List<double[]>) new ArrayList<double[]>());
569                 measurementsByUom.put(uom, meas);
570             }
571             meas.getFirst().add(info.getColumnName(i));
572             meas.getSecond().add(Arrays.getColumnAsDoubles(vals, i));
573         }
574         long[] timestamps = data.getTimeStamps();
575         for (Map.Entry<String, Pair<List<String>, List<double[]>>> entry : measurementsByUom.entrySet()) {
576             Pair<List<String>, List<double[]>> p = entry.getValue();
577             final List<String> measurementNames = p.getFirst();
578             final List<double[]> measurements = p.getSecond();
579             result.add(Charts.createTimeSeriesJFreeChart("chart for "
580                     + info.getTableName() + ", sampleTime " + info.getSampleTime()
581                     + " ms, generated by spf4j", timestamps,
582                     measurementNames.toArray(new String[measurementNames.size()]), entry.getKey(),
583                     measurements.toArray(new double[measurements.size()][])));
584         }
585         return result;
586     }

592     public void writeCsvTable(final String tableName, final File output) throws IOException {
593         TSTable table = getTSTable(tableName);
594         TimeSeries data = readAll(tableName);
595         DateTimeFormatter formatter = ISODateTimeFormat.dateTime();
596         try (Writer writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(output), Charsets.UTF_8))) {
597             Csv.writeCsvElement("timestamp", writer);
598             for (String colName : table.getColumnNames()) {
599                 writer.append(',');
600                 Csv.writeCsvElement(colName, writer);
601             }
602             writer.write('\n');
603             long[] timestamps = data.getTimeStamps();
604             long[][] values = data.getValues();
605             for (int i = 0; i < timestamps.length; i++)  {
606                 Csv.writeCsvElement(formatter.print(timestamps[i]), writer);
607                 for (long val : values[i]) {
608                     writer.append(',');
609                     Csv.writeCsvElement(Long.toString(val), writer);
610                 }
611                 writer.write('\n');
612             }
613         }
614     }

616     public void writeCsvTables(final List<String> tableNames, final File output) throws IOException {
617         DateTimeFormatter formatter = ISODateTimeFormat.dateTime();
618         try (Writer writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(output), Charsets.UTF_8))) {
619             String firstTable = tableNames.get(0);
620             TSTable table = getTSTable(firstTable);
621             Csv.writeCsvElement("table", writer);
622             writer.append(',');
623             Csv.writeCsvElement("timestamp", writer);
624             for (String colName : table.getColumnNames()) {
625                 writer.append(',');
626                 Csv.writeCsvElement(colName, writer);
627             }
628             writer.write('\n');
629 
630             for (String tableName : tableNames) {
631                 TimeSeries data = readAll(tableName);
632                 long[] timestamps = data.getTimeStamps();
633                 long[][] values = data.getValues();
634                 for (int i = 0; i < timestamps.length; i++)  {
635                     Csv.writeCsvElement(tableName, writer);
636                     writer.append(',');
637                     Csv.writeCsvElement(formatter.print(timestamps[i]), writer);
638                     for (long val : values[i]) {
639                         writer.append(',');
640                         Csv.writeCsvElement(Long.toString(val), writer);
641                     }
642                     writer.write('\n');
643                 }
644             }
645         }

654     public void tail(final long pollMillis, final long from, final TSDataHandler handler)
655             throws IOException {
656         Map<String, TSTable> lastState = new HashMap<>();
657         long lastSize = 0;
658         while (!Thread.interrupted() && !handler.finish()) {
659             long currSize = this.file.length();
660             if (currSize > lastSize) {
661                 // see if we have new Tables;
662                 reReadTableInfos();
663                 Map<String, TSTable> currState = getTsTables();
664                 for (String tableName : Sets.difference(currState.keySet(), lastState.keySet())) {
665                     handler.newTable(tableName, currState.get(tableName).getColumnNames());
666                 }
667                 for (TSTable table : currState.values()) {
668                     final String tableName = table.getTableName();
669                     TSTable prevTableState = lastState.get(tableName);
670                     final long currLastDataFragment = table.getLastDataFragment();
671 
672                     if (prevTableState == null) {
673                         long lastDataFragment = table.getFirstDataFragment();
674                         if (lastDataFragment > 0) {
675                             TimeSeries data = read(from, Long.MAX_VALUE, lastDataFragment, currLastDataFragment,
676                                     false);
677                             handler.newData(tableName, data);
678                         }
679                     } else {
680                         long lastDataFragment = prevTableState.getLastDataFragment();
681                         if (lastDataFragment == 0) {
682                             lastDataFragment = table.getFirstDataFragment();
683                             if (lastDataFragment > 0) {
684                                 TimeSeries data = read(from, Long.MAX_VALUE, lastDataFragment, currLastDataFragment,
685                                         false);
686                                 handler.newData(tableName, data);
687                             }
688                         } else if (currLastDataFragment > lastDataFragment) {
689                             TimeSeries data = read(from, Long.MAX_VALUE, lastDataFragment, currLastDataFragment,
690                                     true);
691                             handler.newData(tableName, data);
692                         }
693                     }
694                 }
695                 lastState = currState;
696             }
697             lastSize = currSize;
698             try {
699                 Thread.sleep(pollMillis);
700             } catch (InterruptedException ex) {
701                 Thread.currentThread().interrupt();
702                 break;
703             }
704         }
705     }

713     /**
714      * Quantized recorders will have min, max avg charts and distribution charts
715      * generated. Counting recorders will have simple charts generated.
716      *
717      * @param startTimeMillis
718      * @param endTimeMillis
719      * @param width
720      * @param height
721      * @return
722      * @throws IOException
723      */
724     @JmxExport(value = "generateChartsInterval",
725             description = "generate charts for all measurements in specified interval")
726     public List<String> generateCharts(final long startTimeMillis, final long endTimeMillis,
727             final int width, final int height) throws IOException {
728         try {
729             this.flush();
730             List<String> result = new ArrayList<>();
731             Collection<TSTable> columnsInfo = this.getTSTables();
732             for (TSTable info : columnsInfo) {
733                 TimeSeries data = this.read(info.getTableName(), startTimeMillis, endTimeMillis);
734                 if (data.getTimeStamps().length > 0) {
735                     if (canGenerateMinMaxAvgCount(info)) {
736                         result.add(generateMinMaxAvgCountChart(info, data, width, height));
737                     }
738                     if (canGenerateHeatChart(info)) {
739                         result.add(generateHeatChart(info, data, width, height));
740                     }
741                 }
742             }
743             Multimap<String, TSTable> counters = getCounters(columnsInfo);
744             for (Map.Entry<String, Collection<TSTable>> entry : counters.asMap().entrySet()) {
745                 Collection<TSTable> ltables = entry.getValue();
746                 int l = ltables.size();
747                 long[][] timestamps = new long[l][];
748                 double[][] cdata = new double[l][];
749                 double[][] cdata2 = new double[l][];
750                 int i = 0;
751                 String[] measurementNames = new String[cdata.length];
752                 String[] measurementNames2 = new String[cdata2.length];
753                 String uom1 = "count";
754                 String uom2 = "";
755                 for (TSTable colInfo : ltables) {
756                     TimeSeries data = this.read(colInfo.getTableName(), startTimeMillis, endTimeMillis);
757                     timestamps[i] = data.getTimeStamps();
758                     final long[][] values = data.getValues();
759                     cdata[i] = Arrays.getColumnAsDoubles(values, colInfo.getColumnIndex("count"));
760                     cdata2[i] = Arrays.getColumnAsDoubles(values, colInfo.getColumnIndex("total"));
761                     measurementNames[i] = colInfo.getTableName() + ".count";
762                     measurementNames2[i] = colInfo.getTableName() + ".total";
763                     uom2 = new String(colInfo.getTableMetaData(), Charsets.UTF_8);
764                     i++;
765                 }
766                 result.add(generateCountChart(entry.getKey(), timestamps, measurementNames,
767                         measurementNames2, uom1, uom2, cdata, cdata2, width, height));
768             }
769             return result;
770         } catch (IOException | RuntimeException ex) {
771             throw ex;
772         }
773     }

775     private static Multimap<String, TSTable> getCounters(final Collection<TSTable> columnInfos) {
776         Multimap<String, TSTable> result = HashMultimap.create();
777         for (TSTable info : columnInfos) {
778             if (isCounterOnly(info)) {
779                 String groupName = info.getTableName();
780                 if (groupName.startsWith("(")) {
781                     int cidx = groupName.indexOf(',');
782                     if (cidx > 0) {
783                         groupName = groupName.substring(1, cidx);
784                     }
785                 }
786                 result.put(groupName, info);
787             }
788         }
789         return result;
790     }

820     private String generateMinMaxAvgCountChart(
821             final TSTable info, final TimeSeries data,
822             final int width, final int height) throws IOException {
823         long[][] vals = data.getValues();
824         double[] min = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("min"));
825         double[] max = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("max"));
826         double[] total = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("total"));
827         double[] count = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("count"));
828         for (int i = 0; i < count.length; i++) {
829             if (count[i] == 0) {
830                 min[i] = 0;
831                 max[i] = 0;
832             }
833         }
834         long[] timestamps = data.getTimeStamps();
835         BufferedImage combined = Charts.createMinMaxAvgCountImg("Measurements for "
836                 + info.getTableName() + " generated by spf4j",
837                 timestamps, min, max, total, count, new String(info.getTableMetaData(), Charsets.UTF_8), width, height);
838         File dbFile = new File(this.getDBFilePath());
839         File graphicFile = File.createTempFile(dbFile.getName() + '_' + fixName(info.getTableName()), ".mmac.png",
840                 dbFile.getParentFile());
841         ImageIO.write(combined, "png", graphicFile);
842         return graphicFile.getPath();
843     }

874     private static String fixName(final String name) {
875         final int length = name.length();
876         StringBuilder result = new StringBuilder(length);
877         for (int i = 0; i < length; i++) {
878             char c = name.charAt(i);
879             if (Character.isJavaIdentifierPart(c)) {
880                 result.append(c);
881             }
882         }
883         return result.toString();
884     }
