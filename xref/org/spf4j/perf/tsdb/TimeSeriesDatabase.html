53  /**
54   * Yet another time series database. Why? because all the other ts databases had various constraints that restrict the
55   * functionality I can add to spf4j.
56   *
57   * Initial Features:
58   *
59   * 1. measurements can be added dynamically anytime to a database.
60   * 2. long measurement names.
61   * 3. the stored interval is not known from the beginning.
62   * 4. implementation biased towards write performance.
63   *
64   * Format:
65   * 
66   * Header: TSDB[version:int][metadata:bytes]
67   * Table of Contents: firstTableInfoPtr, lastTableInfoPtr
68   * TableInfo:
69   * DataFragment:
70   * ......
71   * ......
72   * TableInfo:
73   * DataFragment:
74   * EOF
75   * 
76   * @author zoly
77   */
78  public final class TimeSeriesDatabase implements Closeable {

101     public TimeSeriesDatabase(final String pathToDatabaseFile, final boolean isWrite, final byte... metaData)
102             throws IOException {
103         file = new RandomAccessFile(pathToDatabaseFile, isWrite ? "rw" : "r");
104         // uniques per process string for sync purposes.
105         this.path = INTERNER.intern(new File(pathToDatabaseFile).getPath());
106         tables = new ConcurrentHashMap<>();
107         writeDataFragments = new HashMap<>();
108         // read or create header
109         synchronized (path) {
110             this.ch = file.getChannel();
111             FileLock lock;
112             if (isWrite) {
113                 lock = ch.lock();
114             } else {
115                 lock = ch.lock(0, Long.MAX_VALUE, true);
116             }
117             try {
118                 if (file.length() == 0) {
119                     this.header = new Header(VERSION, metaData);
120                     this.header.writeTo(file);
121                     this.toc = new TableOfContents(file.getFilePointer());
122                     this.toc.writeTo(file);
123                 } else {
124                     this.header = new Header(file);
125                     this.toc = new TableOfContents(file);
126                 }
127             } catch (IOException | RuntimeException e) {
128                 try {
129                     lock.release();
130                     throw e;
131                 } catch (IOException ex) {
132                     ex.addSuppressed(e);
133                     throw ex;
134                 }
135             }
136             lock.release();
137             lock = ch.lock(0, Long.MAX_VALUE, true);
138             try {
139                 readTableInfos();
140             } catch (IOException | RuntimeException e) {
141                 try {
142                     lock.release();
143                     throw e;
144                 } catch (IOException ex) {
145                     ex.addSuppressed(e);
146                     throw ex;
147                 }
148             }
149             lock.release();
150         }
151     }

153     public void reReadTableInfos() throws IOException {
154         synchronized (path) {
155             FileLock lock = ch.lock(0, Long.MAX_VALUE, true);
156             try {
157                 toc = new TableOfContents(file, toc.getLocation()); // reread toc
158                 readTableInfos();
159             } catch (IOException | RuntimeException e) {
160                 try {
161                     lock.release();
162                     throw e;
163                 } catch (IOException ex) {
164                     ex.addSuppressed(e);
165                     throw ex;
166                 }
167             }
168             lock.release();
169         }
170     }
171     
172     private void readTableInfos() throws IOException {
173         final long firstColumnInfo = toc.getFirstTableInfoPtr();
174         if (firstColumnInfo > 0) {
175             file.seek(firstColumnInfo);
176             TSTable colInfo = new TSTable(file);
177             tables.put(colInfo.getTableName(), colInfo);
178             lastTableInfo = colInfo;
179             while (colInfo.getNextTSTable() > 0) {
180                 file.seek(colInfo.getNextTSTable());
181                 colInfo = new TSTable(file);
182                 tables.put(colInfo.getTableName(), colInfo);
183                 lastTableInfo = colInfo;
184             }
185         }
186     }
187     
188     private void readLastTableInfo() throws IOException {
189         toc = new TableOfContents(file, toc.getLocation()); // reread toc
190         if (toc.getLastTableInfoPtr() == 0) {
191             return;
192         }
193         lastTableInfo = new TSTable(file, toc.getLastTableInfoPtr()); // update last table info
194     }

213             final byte[] tableMetaData, final int sampleTime, final String[] columnNames,

215         byte[][] metadata = new byte[columnMetaData.length][];

218         }

222     public void addTSTable(final String tableName,
223             final byte[] tableMetaData, final int sampleTime, final String[] columnNames,
224             final byte[][] columnMetaData) throws IOException {
225         synchronized (path) {
226             if (hasTSTable(tableName)) {
227                 throw new IllegalArgumentException("group already exists " + tableName);
228             }
229             flush();
230             FileLock lock = ch.lock();
231             TSTable colInfo;
232             try {
233                 readLastTableInfo();
234                 //write column information at the end of the file.
235                 file.seek(file.length());
236                 colInfo = new TSTable(tableName, tableMetaData, columnNames,
237                         columnMetaData, sampleTime, file.getFilePointer());
238                 colInfo.writeTo(file);
239                 //update refferences to this new TableInfo.
240                 if (lastTableInfo != null) {
241                     lastTableInfo.setNextColumnInfo(colInfo.getLocation(), file);
242                 } else {
243                     toc.setFirstTableInfo(colInfo.getLocation(), file);
244                 }
245                 toc.setLastTableInfo(colInfo.getLocation(), file);
246             } catch (IOException | RuntimeException e) {
247                 try {
248                     lock.release();
249                     throw e;
250                 } catch (IOException ex) {
251                     ex.addSuppressed(e);
252                     throw ex;
253                 }
254             }
255             lock.release();
256             
257             lastTableInfo = colInfo;
258             tables.put(tableName, colInfo);
259         }
260     }

262     public void write(final long time, final String tableName, final long[] values) throws IOException {
263         if (!hasTSTable(tableName)) {
264             throw new IllegalArgumentException("Unknown table name" + tableName);
265         }
266         synchronized (writeDataFragments) {
267             DataFragment writeDataFragment = writeDataFragments.get(tableName);
268             if (writeDataFragment == null) {
269                 writeDataFragment = new DataFragment(time);
270                 writeDataFragments.put(tableName, writeDataFragment);
271             }
272             writeDataFragment.addData(time, values);
273         }
274     }

276     public void flush() throws IOException {
277         synchronized (path) {
278             List<Map.Entry<String, DataFragment>> lwriteDataFragments;
279             synchronized (writeDataFragments) {
280                 if (writeDataFragments.isEmpty()) {
281                     return;
282                 }
283                 lwriteDataFragments = new ArrayList<>(writeDataFragments.entrySet());
284                 writeDataFragments.clear();
285             }
286             FileLock lock = ch.lock();
287             try {
288                 for (Map.Entry<String, DataFragment> entry : lwriteDataFragments) {
289                     DataFragment writeDataFragment = entry.getValue();
290                     String groupName = entry.getKey();
291                     file.seek(file.length());
292                     writeDataFragment.setLocation(file.getFilePointer());
293                     writeDataFragment.writeTo(file);
294                     TSTable colInfo = tables.get(groupName);
295                     colInfo = new TSTable(file, colInfo.getLocation()); // reread colInfo
296                     tables.put(groupName, colInfo); // update colInfo
297                     final long lastDataFragment = colInfo.getLastDataFragment();
298                     final long location = writeDataFragment.getLocation();
299                     if (lastDataFragment != 0) {
300                         DataFragment.setNextDataFragment(lastDataFragment, location, file);
301                     } else {
302                         colInfo.setFirstDataFragment(location, file);
303                     }
304                     colInfo.setLastDataFragment(location, file);
305                 }
306                 sync();
307             } catch (IOException | RuntimeException e) {
308                 try {
309                     lock.release();
310                     throw e;
311                 } catch (IOException ex) {
312                     ex.addSuppressed(e);
313                     throw ex;
314                 }
315             }
316             lock.release();
317         }

359         synchronized (path) {
360             TSTable info = tables.get(tableName);
361             long nextFragmentLocation = info.getFirstDataFragment();
362             if (nextFragmentLocation > 0) {
363                 DataFragment frag;
364                 file.seek(nextFragmentLocation);
365                 frag = new DataFragment(file);
366                 return frag.getStartTimeMillis();
367             } else {
368                 return -1;
369             }

382     /**
383      * Read measurements from table.
384      *
385      * @param tableName
386      * @param startTime start time including
387      * @param endTime end time including
388      * @return
389      * @throws IOException
390      */
391     private TimeSeries read(
392             final long startTime, final long endTime,
393             final long startAtFragment, final long endAtFragment,
394             final boolean skipFirst)
395             throws IOException {
396         synchronized (path) {
397             TLongArrayList timeStamps = new TLongArrayList();
398             List<long[]> data = new ArrayList<>();
399             if (startAtFragment > 0) {
400                 FileLock lock = ch.lock(0, Long.MAX_VALUE, true);
401                 try {
402                     DataFragment frag;
403                     long nextFragmentLocation = startAtFragment;
404                     boolean last = false;
405                     boolean psFirst = skipFirst;
406                     do {
407                         if (nextFragmentLocation == endAtFragment) {
408                             last = true;
409                         }
410                         file.seek(nextFragmentLocation);
411                         frag = new DataFragment(file);
412                         if (psFirst) {
413                             psFirst = false;
414                         } else {
415                         long fragStartTime = frag.getStartTimeMillis();
416                         if (fragStartTime >= startTime) {
417                             TIntArrayList fragTimestamps = frag.getTimestamps();
418                             int nr = 0;
419                             for (int i = 0; i < fragTimestamps.size(); i++) {
420                                 long ts = fragStartTime + fragTimestamps.get(i);
421                                 if (ts <= endTime) {
422                                     timeStamps.add(ts);
423                                     nr++;
424                                 } else {
425                                     break;
426                                 }
427                             }
428                             int i = 0;
429                             for (long[] d : frag.getData()) {
430                                 if (i < nr) {
431                                     data.add(d);
432                                 } else {
433                                     break;
434                                 }
435                                 nr++;
436                             }
437                             if (fragTimestamps.size() > nr) {
438                                 break;
439                             }
440                         }
441                         }
442                         nextFragmentLocation = frag.getNextDataFragment();
443                     } while (nextFragmentLocation > 0 && !last);
444                 } catch (IOException | RuntimeException e) {
445                     try {
446                         lock.release();
447                         throw e;
448                     } catch (IOException ex) {
449                         ex.addSuppressed(e);
450                         throw ex;
451                     }
452                 }
453                 lock.release();
454             }
455             return new TimeSeries(timeStamps.toArray(), data.toArray(new long[data.size()][]));
456         }
457     }

511     public static JFreeChart createHeatJFreeChart(final TimeSeries data, final TSTable info) {
512         Pair<long[], double[][]> mData = fillGaps(data.getTimeStamps(), data.getValues(),
513                 info.getSampleTime(), info.getColumnNames().length);
514         final int totalColumnIndex = info.getColumnIndex("total");
515         return Charts.createHeatJFreeChart(info.getColumnNames(),
516                 mData.getSecond(), data.getTimeStamps()[0], info.getSampleTime(),
517                 new String(info.getColumnMetaData()[totalColumnIndex], Charsets.UTF_8), "Measurements distribution for "
518                 + info.getTableName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j");
519     }

521     public static JFreeChart createMinMaxAvgJFreeChart(final TimeSeries data, final TSTable info) {
522         long[][] vals = data.getValues();
523         double[] min = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("min"));
524         double[] max = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("max"));
525         final int totalColumnIndex = info.getColumnIndex("total");
526         double[] total = Arrays.getColumnAsDoubles(vals, totalColumnIndex);
527         double[] count = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("count"));
528         for (int i = 0; i < count.length; i++) {
529             if (count[i] == 0) {
530                 min[i] = 0;
531                 max[i] = 0;
532             }
533         }
534         long[] timestamps = data.getTimeStamps();
535         return Charts.createTimeSeriesJFreeChart("Min,Max,Avg chart for "
536                 + info.getTableName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j", timestamps,
537                 new String[]{"min", "max", "avg"}, new String(info.getColumnMetaData()[totalColumnIndex],
538                         Charsets.UTF_8), new double[][]{min, max, Arrays.divide(total, count)});
539     }

550     public static List<JFreeChart> createJFreeCharts(final TimeSeries data, final TSTable info) {
551         long[][] vals = data.getValues();
552         List<JFreeChart> result = new ArrayList<>();
553         Map<String, Pair<List<String>, List<double[]>>> measurementsByUom = new HashMap<>();
554         String[] columnMetaData = info.getColumnMetaDataAsStrings();
555         for (int i = 0; i < info.getColumnNumber(); i++) {
556             String uom = columnMetaData[i];
557             Pair<List<String>, List<double[]>> meas = measurementsByUom.get(uom);
558             if (meas == null) {
559                 meas = Pair.of((List<String>) new ArrayList<String>(), (List<double[]>) new ArrayList<double[]>());
560                 measurementsByUom.put(uom, meas);
561             }
562             meas.getFirst().add(info.getColumnName(i));
563             meas.getSecond().add(Arrays.getColumnAsDoubles(vals, i));
564         }
565         long[] timestamps = data.getTimeStamps();
566         for (Map.Entry<String, Pair<List<String>, List<double[]>>> entry : measurementsByUom.entrySet()) {
567             Pair<List<String>, List<double[]>> p = entry.getValue();
568             final List<String> measurementNames = p.getFirst();
569             final List<double[]> measurements = p.getSecond();
570             result.add(Charts.createTimeSeriesJFreeChart("chart for "
571                     + info.getTableName() + ", sampleTime " + info.getSampleTime()
572                     + " ms, generated by spf4j", timestamps,
573                     measurementNames.toArray(new String[measurementNames.size()]), entry.getKey(),
574                     measurements.toArray(new double[measurements.size()][])));
575         }
576         return result;
577     }

583     public void writeCsvTable(final String tableName, final File output) throws IOException {
584         TSTable table = getTSTable(tableName);
585         TimeSeries data = readAll(tableName);
586         DateTimeFormatter formatter = ISODateTimeFormat.dateTime();
587         try (Writer writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(output), Charsets.UTF_8))) {
588             Csv.writeCsvElement("timestamp", writer);
589             for (String colName : table.getColumnNames()) {
590                 writer.append(',');
591                 Csv.writeCsvElement(colName, writer);
592             }
593             writer.write('\n');
594             long[] timestamps = data.getTimeStamps();
595             long[][] values = data.getValues();
596             for (int i = 0; i < timestamps.length; i++)  {
597                 Csv.writeCsvElement(formatter.print(timestamps[i]), writer);
598                 for (long val : values[i]) {
599                     writer.append(',');
600                     Csv.writeCsvElement(Long.toString(val), writer);
601                 }
602                 writer.write('\n');
603             }
604         }
605     }

607     public void writeCsvTables(final List<String> tableNames, final File output) throws IOException {
608         DateTimeFormatter formatter = ISODateTimeFormat.dateTime();
609         try (Writer writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(output), Charsets.UTF_8))) {
610             String firstTable = tableNames.get(0);
611             TSTable table = getTSTable(firstTable);
612             Csv.writeCsvElement("table", writer);
613             writer.append(',');
614             Csv.writeCsvElement("timestamp", writer);
615             for (String colName : table.getColumnNames()) {
616                 writer.append(',');
617                 Csv.writeCsvElement(colName, writer);
618             }
619             writer.write('\n');
620             
621             for (String tableName : tableNames) {
622                 TimeSeries data = readAll(tableName);
623                 long[] timestamps = data.getTimeStamps();
624                 long[][] values = data.getValues();
625                 for (int i = 0; i < timestamps.length; i++)  {
626                     Csv.writeCsvElement(tableName, writer);
627                     writer.append(',');
628                     Csv.writeCsvElement(formatter.print(timestamps[i]), writer);
629                     for (long val : values[i]) {
630                         writer.append(',');
631                         Csv.writeCsvElement(Long.toString(val), writer);
632                     }
633                     writer.write('\n');
634                 }
635             }
636         }

644     @SuppressFBWarnings("MDM_THREAD_YIELD")
645     public void tail(final long pollMillis, final long from, final TSDataHandler handler)
646             throws IOException {
647         Map<String, TSTable> lastState = new HashMap<>();
648         long lastSize = 0;
649         while (!Thread.interrupted() && !handler.finish()) {
650             long currSize = this.file.length();
651             if (currSize > lastSize) {
652                 // see if we have new Tables;
653                 reReadTableInfos();
654                 Map<String, TSTable> currState = getTsTables();
655                 for (String tableName : Sets.difference(currState.keySet(), lastState.keySet())) {
656                     handler.newTable(tableName, currState.get(tableName).getColumnNames());
657                 }
658                 for (TSTable table : currState.values()) {
659                     final String tableName = table.getTableName();
660                     TSTable prevTableState = lastState.get(tableName);
661                     final long currLastDataFragment = table.getLastDataFragment();
662 
663                     if (prevTableState == null) {
664                         long lastDataFragment = table.getFirstDataFragment();
665                         if (lastDataFragment > 0) {
666                             TimeSeries data = read(from, Long.MAX_VALUE, lastDataFragment, currLastDataFragment,
667                                     false);
668                             handler.newData(tableName, data);
669                         }
670                     } else {
671                         long lastDataFragment = prevTableState.getLastDataFragment();
672                         if (lastDataFragment == 0) {
673                             lastDataFragment = table.getFirstDataFragment();
674                             if (lastDataFragment > 0) {
675                                 TimeSeries data = read(from, Long.MAX_VALUE, lastDataFragment, currLastDataFragment,
676                                         false);
677                                 handler.newData(tableName, data);
678                             }
679                         } else if (currLastDataFragment > lastDataFragment) {
680                             TimeSeries data = read(from, Long.MAX_VALUE, lastDataFragment, currLastDataFragment,
681                                     true);
682                             handler.newData(tableName, data);
683                         }
684                     }
685                 }
686                 lastState = currState;
687             }
688             lastSize = currSize;
689             try {
690                 Thread.sleep(pollMillis);
691             } catch (InterruptedException ex) {
692                 Thread.currentThread().interrupt();
693                 break;
694             }
695         }
696     }
