18  package org.spf4j.perf.tsdb;
19  
20  import com.google.common.base.Charsets;
21  import java.io.DataOutput;
22  import java.io.DataOutputStream;
23  import java.io.IOException;
24  import java.io.RandomAccessFile;
25  import java.nio.channels.FileChannel;
26  import java.nio.channels.FileLock;
27  import java.util.Arrays;
28  import java.util.HashMap;
29  import java.util.Map;
30  import org.spf4j.io.ByteArrayBuilder;
31  
32  /**
33   *
34   * @author zoly
35   */
36  public final class TSTable {
37      
38      private final long location;
39      private long nextColumnInfo;
40      private long firstDataFragment;
41      private long lastDataFragment;
42      private final String tableName;
43      private final int sampleTime;
44      private final byte[] tableMetaData;
45      private final String [] columnNames;
46      private final byte [][] columnMetaData;
47      private final Map<String, Integer> nameToIndex;
48      
49  
50      TSTable(final String tableName, final byte [] tableMetaData,
51              final String [] columnNames, final byte [][] columnMetaData,
52              final int sampleTime, final long location) {
53          this.location = location;
54          this.nextColumnInfo = 0;
55          this.firstDataFragment = 0;
56          this.lastDataFragment = 0;
57          this.tableName = tableName;
58          this.sampleTime = sampleTime;
59          this.tableMetaData = tableMetaData;
60          this.columnNames = columnNames;
61          this.columnMetaData = columnMetaData;
62          this.nameToIndex = new HashMap<String, Integer>(columnNames.length + columnNames.length / 3);
63          for (int i = 0; i < columnNames.length; i++) {
64              this.nameToIndex.put(columnNames[i], i);
65          }
66      }
67  
68      TSTable(final RandomAccessFile raf) throws IOException {
69          location = raf.getFilePointer();
70          FileChannel ch = raf.getChannel();
71          FileLock lock = ch.lock(location, 8, true);
72          try {
73              this.nextColumnInfo = raf.readLong();
74              this.firstDataFragment = raf.readLong();
75              this.lastDataFragment = raf.readLong();
76              this.tableName = raf.readUTF();
77              this.sampleTime = raf.readInt();
78              int grMetaSize = raf.readInt();
79              this.tableMetaData = new byte [grMetaSize];
80              raf.readFully(tableMetaData);
81              int nrColumns = raf.readShort();
82              columnNames = new String[nrColumns];
83              this.nameToIndex = new HashMap<String, Integer>(nrColumns + nrColumns / 3);
84              for (int i = 0; i < columnNames.length; i++) {
85                  String colName = raf.readUTF();
86                  columnNames[i] = colName;
87                  this.nameToIndex.put(colName, i);
88              }
89              columnMetaData = new byte[raf.readInt()][];
90              for (int i = 0; i < columnMetaData.length; i++) {
91                  int metaLength = raf.readInt();
92                  byte [] colMetaData = new byte[metaLength];
93                  raf.readFully(colMetaData);
94                  columnMetaData[i] = colMetaData;
95              }
96              
97          } finally {
98              lock.release();
99          }
100     }
101 
102     void writeTo(final DataOutput dos) throws IOException {
103         dos.writeLong(nextColumnInfo);
104         dos.writeLong(firstDataFragment);
105         dos.writeLong(lastDataFragment);
106         dos.writeUTF(tableName);
107         dos.writeInt(sampleTime);
108         dos.writeInt(tableMetaData.length);
109         dos.write(tableMetaData);
110         dos.writeShort(columnNames.length);
111         for (String columnName : columnNames) {
112             dos.writeUTF(columnName);
113         }
114         dos.writeInt(columnMetaData.length);
115         for (byte[] colMeta : columnMetaData) {
116             dos.writeInt(colMeta.length);
117             dos.write(colMeta);
118         }
119         
120     }
121     
122     void writeTo(final RandomAccessFile raf) throws IOException {
123         ByteArrayBuilder bos = new ByteArrayBuilder();
124         DataOutput dos = new DataOutputStream(bos);
125         writeTo(dos);
126         FileChannel ch = raf.getChannel();
127         FileLock lock = ch.lock(raf.getFilePointer(), 8, false);
128         try {
129             raf.seek(location);
130             raf.write(bos.getBuffer(), 0, bos.size());
131         } finally {
132             lock.release();
133         }
134     }
135 
136     public String [] getColumnNames() {
137         return columnNames.clone();
138     }
139 
140     void setNextColumnInfo(final long pnextColumnInfo, final RandomAccessFile raf) throws IOException {
141         this.nextColumnInfo = pnextColumnInfo;
142         FileChannel ch = raf.getChannel();
143         FileLock lock = ch.lock(location, 8, false);
144         try {
145             raf.seek(location);
146             raf.writeLong(nextColumnInfo);
147         } finally {
148             lock.release();
149         }
150     }
151 
152     void setFirstDataFragment(final long pfirstDataFragment, final RandomAccessFile raf) throws IOException {
153         this.firstDataFragment = pfirstDataFragment;
154         FileChannel ch = raf.getChannel();
155         FileLock lock = ch.lock(location + 8, 8, false);
156         try {
157             raf.seek(location + 8);
158             raf.writeLong(firstDataFragment);
159         } finally {
160             lock.release();
161         }
162     }
163     
164     
165     void setLastDataFragment(final long plastDataFragment, final RandomAccessFile raf) throws IOException {
166         this.lastDataFragment = plastDataFragment;
167         FileChannel ch = raf.getChannel();
168         FileLock lock = ch.lock(raf.getFilePointer() + 16, 8, false);
169         try {
170             raf.seek(location + 16);
171             raf.writeLong(lastDataFragment);
172         } finally {
173             lock.release();
174         }
175     }
176     
177     
178     public long getNextTSTable() {
179         return nextColumnInfo;
180     }
181 
182     public long getLocation() {
183         return location;
184     }
185 
186     public String getTableName() {
187         return tableName;
188     }
189 
190     public long getFirstDataFragment() {
191         return firstDataFragment;
192     }
193 
194     public long getLastDataFragment() {
195         return lastDataFragment;
196     }
197 
198     public byte[][] getColumnMetaData() {
199         return columnMetaData.clone();
200     }
201     
202     public String[] getColumnMetaDataAsStrings() {
203         String [] result = new String [columnMetaData.length];
204         for (int i = 0; i < columnMetaData.length; i++) {
205             result[i] = new String(columnMetaData[i], Charsets.UTF_8);
206         }
207         return result;
208     }
209     
210     public int getColumnIndex(final String columnName) {
211         Integer result = this.nameToIndex.get(columnName);
212         if (result == null) {
213             return -1;
214         } else {
215             return result;
216         }
217     }
218 
219     public int getSampleTime() {
220         return sampleTime;
221     }
222 
223     public byte[] getTableMetaData() {
224         return tableMetaData.clone();
225     }
226     
227     public int getColumnNumber() {
228         return columnNames.length;
229     }
230     
231     public String getColumnName(final int index) {
232         return columnNames[index];
233     }
234 
235     @Override
236     public String toString() {
237         return "TSTable{" + "location=" + location + ", nextColumnInfo=" + nextColumnInfo
238                 + ", firstDataFragment=" + firstDataFragment + ", lastDataFragment="
239                 + lastDataFragment + ", groupName=" + tableName + ", sampleTime=" + sampleTime
240                 + ", columnNames=" + Arrays.toString(columnNames) + ", nameToIndex=" + nameToIndex + '}';
241     }
