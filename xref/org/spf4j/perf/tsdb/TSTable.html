36  
37      private final long location;
38      private long nextTableInfo;
39      private long firstDataFragment;
40      private long lastDataFragment;
41      private final String tableName;
42      private final int sampleTime;
43      private final byte[] tableMetaData;
44      private final String[] columnNames;
45      private final byte[][] columnMetaData;
46      private final Map<String, Integer> nameToIndex;
47  
48      TSTable(final String tableName, final byte[] tableMetaData,
49              final String[] columnNames, final byte[][] columnMetaData,
50              final int sampleTime, final long location) {
51          this.location = location;
52          this.nextTableInfo = 0;
53          this.firstDataFragment = 0;
54          this.lastDataFragment = 0;
55          this.tableName = tableName;
56          this.sampleTime = sampleTime;
57          this.tableMetaData = tableMetaData;
58          this.columnNames = columnNames;
59          this.columnMetaData = columnMetaData;
60          this.nameToIndex = new HashMap<>(columnNames.length + columnNames.length / 3);
61          for (int i = 0; i < columnNames.length; i++) {
62              this.nameToIndex.put(columnNames[i], i);
63          }
64      }
65  
66      /**
67       * This is a shalow copy.
68       *
69       * @param copyOf
70       */
71  
72      TSTable(final TSTable copyOf) {
73          this.location = copyOf.location;
74          this.nextTableInfo = copyOf.nextTableInfo;
75          this.firstDataFragment = copyOf.firstDataFragment;
76          this.lastDataFragment = copyOf.lastDataFragment;
77          this.tableName = copyOf.tableName;
78          this.sampleTime = copyOf.sampleTime;
79          this.tableMetaData = copyOf.tableMetaData;
80          this.columnNames = copyOf.columnNames;
81          this.columnMetaData = copyOf.columnMetaData;
82          this.nameToIndex = copyOf.nameToIndex;
83      }
84  
85  
86      /**
87       * Read table structure from file.
88       *
89       * @param raf
90       * @throws IOException
91       */
92      TSTable(final RandomAccessFile raf) throws IOException {
93          this(raf, raf.getFilePointer());
94      }
95  
96      TSTable(final RandomAccessFile raf, final long location) throws IOException {
97          this.location = location;
98          raf.seek(location);
99          this.nextTableInfo = raf.readLong();
100         this.firstDataFragment = raf.readLong();
101         this.lastDataFragment = raf.readLong();
102         this.tableName = raf.readUTF();
103         this.sampleTime = raf.readInt();
104         int grMetaSize = raf.readInt();
105         this.tableMetaData = new byte[grMetaSize];
106         raf.readFully(tableMetaData);
107         int nrColumns = raf.readShort();
108         columnNames = new String[nrColumns];
109         this.nameToIndex = new HashMap<>(nrColumns + nrColumns / 3);
110         for (int i = 0; i < columnNames.length; i++) {
111             String colName = raf.readUTF();
112             columnNames[i] = colName;
113             this.nameToIndex.put(colName, i);
114         }
115         columnMetaData = new byte[raf.readInt()][];
116         for (int i = 0; i < columnMetaData.length; i++) {
117             int metaLength = raf.readInt();
118             byte[] colMetaData = new byte[metaLength];
119             raf.readFully(colMetaData);
120             columnMetaData[i] = colMetaData;
121         }
122     }

126     void writeTo(final DataOutput dos) throws IOException {
127         dos.writeLong(nextTableInfo);
128         dos.writeLong(firstDataFragment);
129         dos.writeLong(lastDataFragment);
130         dos.writeUTF(tableName);
131         dos.writeInt(sampleTime);
132         dos.writeInt(tableMetaData.length);
133         dos.write(tableMetaData);
134         dos.writeShort(columnNames.length);
135         for (String columnName : columnNames) {
136             dos.writeUTF(columnName);
137         }
138         dos.writeInt(columnMetaData.length);
139         for (byte[] colMeta : columnMetaData) {
140             dos.writeInt(colMeta.length);
141             dos.write(colMeta);
142         }
143 
144     }
145 
146     void writeTo(final RandomAccessFile raf) throws IOException {
147         try (ByteArrayBuilder bos = new ByteArrayBuilder()) {
148             DataOutput dos = new DataOutputStream(bos);
149             writeTo(dos);
150 
151             raf.seek(location);
152             raf.write(bos.getBuffer(), 0, bos.size());
153         }
154     }

167     void setFirstDataFragment(final long pfirstDataFragment, final RandomAccessFile raf) throws IOException {
168         this.firstDataFragment = pfirstDataFragment;
169         raf.seek(location + 8);

171     }

173     void setLastDataFragment(final long plastDataFragment, final RandomAccessFile raf) throws IOException {
174         this.lastDataFragment = plastDataFragment;
175         raf.seek(location + 16);
176         raf.writeLong(lastDataFragment);
177     }

203     public String[] getColumnMetaDataAsStrings() {
204         String[] result = new String[columnMetaData.length];

207         }

211     public int getColumnIndex(final String columnName) {
212         Integer result = this.nameToIndex.get(columnName);
213         if (result == null) {
214             return -1;
215         } else {
216             return result;
217         }
218     }
