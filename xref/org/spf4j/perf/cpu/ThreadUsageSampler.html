33      private static final ThreadMXBean TH_BEAN = ManagementFactory.getThreadMXBean();

35      private static ScheduledFuture<?> samplingFuture;

37      private static final List<String> PEAK_THREAD_NAMES = new ArrayList<>();
38      private static final List<StackTraceElement []> PEAK_THREAD_TRACES = new ArrayList<>();
39      private static final BitSet PEAK_THREAD_DAEMON = new BitSet();
40  
41      public static void writePeakThreadInfo(final PrintStream out) throws IOException {
42          out.println("Peak Threads:");
43          int i = 0;
44          boolean haveStacktraces = PEAK_THREAD_TRACES.size() > 0;
45          for (String tname : PEAK_THREAD_NAMES) {
46              out.print(tname);
47              out.print(", daemon =");
48              out.print(PEAK_THREAD_DAEMON.get(i));
49              out.print(',');
50              if (haveStacktraces) {
51                  out.print(' ');
52                  out.print(Arrays.toString(PEAK_THREAD_TRACES.get(i)));
53              }
54              out.println();
55              i++;
56          }
57      }
58  
59      @JmxExport
60      @SuppressFBWarnings("DM_DEFAULT_ENCODING")
61      public static String getPeakThreadInfo() {
62          try (ByteArrayBuilder bab = new ByteArrayBuilder()) {
63              PrintStream ps = new PrintStream(bab);
64              writePeakThreadInfo(ps);
65              return bab.toString(Charset.defaultCharset());
66          } catch (IOException ex) {
67             throw new RuntimeException(ex);
68          }
69      }

72      @JmxExport
73      public static String getCurrentAliveThreadInfo() {
74          StringBuilder sb = new StringBuilder(1024);
75          Thread[] threads = FastStackCollector.getThreads();
76          StackTraceElement[][] stackTraces = FastStackCollector.getStackTraces(threads);
77          for (int i = 0; i < threads.length; i++) {
78              Thread t = threads[i];
79              if (t.isAlive()) {
80                  sb.append(t.getName());
81                  sb.append(", daemon =");
82                  sb.append(t.isDaemon());
83                  sb.append(',');
84                  StackTraceElement [] straces = stackTraces[i];
85                  if (straces != null && straces.length > 0) {
86                      sb.append(' ');
87                      sb.append(Arrays.toString(straces));
88                  }
89                  sb.append('\n');
90              }
91          }
92          return sb.toString();
93      }

111        start(sampleTime, true);

114     @JmxExport
115     public static synchronized void start(@JmxExport("sampleTimeMillis") final int sampleTime,
116             @JmxExport("withStackTraces") final boolean withStackTraces) {
117         if (samplingFuture == null) {
118             final MeasurementRecorder cpuUsage
119                     = RecorderFactory.createDirectRecorder("peak-thread-count", "count", sampleTime);
120             samplingFuture = DefaultScheduler.INSTANCE.scheduleWithFixedDelay(new AbstractRunnable() {
121 
122                 private int maxThreadsNr = 0;
123 
124                 @Override
125                 public void doRun() throws Exception {
126                     final int peakThreadCount = TH_BEAN.getPeakThreadCount();
127                     cpuUsage.record(peakThreadCount);
128                     if (peakThreadCount > maxThreadsNr) {
129                         Thread[] ths = FastStackCollector.getThreads();
130                         if (ths.length > PEAK_THREAD_NAMES.size()) {
131                             if (withStackTraces) {
132                                 StackTraceElement[][] stackTraces = FastStackCollector.getStackTraces(ths);
133                                 PEAK_THREAD_TRACES.clear();
134                                 PEAK_THREAD_TRACES.addAll(Arrays.asList(stackTraces));
135                             }
136 
137                             PEAK_THREAD_NAMES.clear();
138                             int i = 0;
139                             for (Thread th : ths) {
140                                 PEAK_THREAD_NAMES.add(th.getName());
141                                 if (th.isDaemon()) {
142                                     PEAK_THREAD_DAEMON.set(i);
143                                 } else {
144                                     PEAK_THREAD_DAEMON.clear(i);
145                                 }
146                                 i++;
147                             }
148                         }
149                         maxThreadsNr = peakThreadCount;
150                     }
151                     TH_BEAN.resetPeakThreadCount();
152                 }
153             }, sampleTime, sampleTime, TimeUnit.MILLISECONDS);
154         } else {
155             throw new IllegalStateException("Thread sampling already started " + samplingFuture);
156         }
157     }
158 
159     @JmxExport
160     public static synchronized void stop() {
161         if (samplingFuture != null) {
162             samplingFuture.cancel(false);
163             samplingFuture = null;
164         }
165     }
166 
167     @JmxExport
168     public static synchronized boolean isStarted() {
169         return samplingFuture != null;
170     }
