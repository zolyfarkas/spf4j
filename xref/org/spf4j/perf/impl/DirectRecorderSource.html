29  public final class DirectRecorderSource implements MeasurementRecorderSource {
30  
31  
32      private final LoadingCache<Object, MeasurementRecorder> recorders;
33  
34      public DirectRecorderSource(final Object forWhat, final String description,
35              final String uom, final int sampleTimeMillis, final MeasurementStore store) {
36          recorders = new UnboundedLoadingCache(16,
37                  new CreateDirectRecorder(forWhat, description, uom, sampleTimeMillis, store));
38      }
39  
40      @Override
41      public MeasurementRecorder getRecorder(final Object forWhat) {
42          return recorders.getUnchecked(forWhat);
43      }
44  
45      @Override
46      public void close() {
47      }
48  
49      private static final class CreateDirectRecorder extends CacheLoader<Object, MeasurementRecorder> {
50  
51          private final Object forWhat;
52          private final String description;
53          private final String uom;
54          private final int sampleTimeMillis;
55          private final MeasurementStore store;
56  
57          public CreateDirectRecorder(final Object forWhat, final String description,
58                  final String uom, final int sampleTimeMillis, final MeasurementStore store) {
59              this.forWhat = forWhat;
60              this.description = description;
61              this.uom = uom;
62              this.sampleTimeMillis = sampleTimeMillis;
63              this.store = store;
64          }
65  
66          @Override
67          public MeasurementRecorder load(final Object key) throws Exception {
68              return new DirectStoreAccumulator(Pair.of(forWhat, key), description,
69                      uom, sampleTimeMillis, store);
70          }
71      }
72  
73  }
