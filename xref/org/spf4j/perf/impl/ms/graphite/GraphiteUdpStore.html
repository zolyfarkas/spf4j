67  public final class GraphiteUdpStore implements MeasurementStore {
68  
69    public static final int MAX_UDP_MSG_SIZE = 512;
70  
71    private final RecyclingSupplier<DatagramChannel> datagramChannelSupplier;
72  
73    private final InetSocketAddress address;
74  
75    private static class DatagramChannelSupplierFactory implements RecyclingSupplier.Factory<DatagramChannel> {
76  
77      private final InetSocketAddress address;
78  
79      DatagramChannelSupplierFactory(final InetSocketAddress address) {
80        this.address = address;
81      }
82  
83      @Override
84      public DatagramChannel create() throws ObjectCreationException {
85        DatagramChannel datagramChannel;
86        try {
87          datagramChannel = DatagramChannel.open();
88        } catch (IOException ex) {
89          throw new ObjectCreationException(ex);
90        }
91        try {
92          datagramChannel.connect(address);
93          return datagramChannel;
94        } catch (IOException ex) {
95          try {
96            datagramChannel.close();
97          } catch (IOException ex1) {
98            ex1.addSuppressed(ex);
99            throw new ObjectCreationException(ex1);
100         }
101         throw new ObjectCreationException(ex);
102       }
103     }
104 
105     @Override
106     public void dispose(final DatagramChannel object) throws ObjectDisposeException {
107       try {
108         object.close();
109       } catch (IOException ex) {
110         throw new ObjectDisposeException(ex);
111       }
112     }
113 
114     @Override
115     public boolean validate(final DatagramChannel object, final Exception e) throws Exception {
116       return e == null || !(Throwables.getRootCause(e) instanceof IOException);
117     }
118 
119   }
120 
121   public GraphiteUdpStore(final String hostPort) throws ObjectCreationException, URISyntaxException {
122     this(new URI("graphiteUdp://" + hostPort));
123   }
124 
125   public GraphiteUdpStore(final URI uri) throws ObjectCreationException {
126     this(uri.getHost(), uri.getPort());
127   }
128 
129   public GraphiteUdpStore(final String hostName, final int port) throws ObjectCreationException {
130     address = new InetSocketAddress(hostName, port);
131     datagramChannelSupplier = new RecyclingSupplierBuilder<>(1,
132             new DatagramChannelSupplierFactory(address)).build();
133   }
134 
135   @Override
136   public void flush() {
137     // No buffering yet
138   }
139 
140   @Override
141   public long alocateMeasurements(final MeasurementsInfo measurement, final int sampleTimeMillis) {
142     return Id2Info.getId(measurement);
143   }
144 
145   @Override
146   @SuppressFBWarnings("BED_BOGUS_EXCEPTION_DECLARATION") // fb nonsense
147   public void saveMeasurements(final long tableId,
148           final long timeStampMillis, final long... measurements) throws IOException {
149 
150     try {
151       Template.doOnSupplied(new HandlerImpl(measurements, Id2Info.getInfo(tableId), timeStampMillis),
152               1, TimeUnit.MINUTES,
153               datagramChannelSupplier, RetryPolicy.defaultPolicy(), IOException.class);
154     } catch (TimeoutException ex) {
155       throw new UncheckedTimeoutException(ex);
156     } catch (InterruptedException ex) {
157       Thread.currentThread().interrupt();
158     }
159 
160   }
161 
162   /**
163    * Write with the plaintext protocol: https://graphite.readthedocs.io/en/0.9.10/feeding-carbon.html
164    *
165    * @param measurementInfo measuremrnt info
166    * @param measurementName measurement name
167    * @param measurement measurement value
168    * @param timeStampMillis timestamp millis since epoch.
169    * @param os the output writer to write to.
170    * @throws IOException
171    */
172   public static void writeMetric(final MeasurementsInfo measurementInfo, final String measurementName,
173           final long measurement, final long timeStampMillis, final Writer os)
174           throws IOException {
175     Strings.writeReplaceWhitespaces(measurementInfo.getMeasuredEntity().toString(), '-', os);
176     os.append('/');
177     Strings.writeReplaceWhitespaces(measurementName, '-', os);
178     os.append(' ');
179     os.append(Long.toString(measurement));
180     os.append(' ');
181     os.append(Long.toString(timeStampMillis));
182     os.append('\n');
183   }
184 
185   @Override
186   public String toString() {
187     return "GraphiteUdpStore{address=" + address + '}';
188   }
189 
190   @Override
191   public void close() {
192     try {
193       datagramChannelSupplier.dispose();
194     } catch (ObjectDisposeException | InterruptedException ex) {
195       throw new UncheckedExecutionException(ex);
196     }
197   }
198 
199   private static class HandlerImpl implements HandlerNano<DatagramChannel, Void, IOException> {
200 
201     private final long[] measurements;
202     private final MeasurementsInfo measurementInfo;
203     private final long timeStampMillis;
204 
205     HandlerImpl(final long[] measurements, final MeasurementsInfo measurementInfo,
206             final long timeStampMillis) {
207       this.measurements = measurements;
208       this.measurementInfo = measurementInfo;
209       this.timeStampMillis = timeStampMillis;
210     }
211 
212     @Override
213     @Nullable
214     public Void handle(final DatagramChannel datagramChannel, final long deadline) throws IOException {
215       try (ByteArrayBuilder bos = new ByteArrayBuilder();
216               OutputStreamWriter os = new OutputStreamWriter(bos, StandardCharsets.UTF_8)) {
217 
218         int msgStart = 0;
219         int msgEnd = 0;
220         int prevEnd = 0;
221 
222         for (int i = 0; i < measurements.length; i++) {
223           writeMetric(measurementInfo, measurementInfo.getMeasurementName(i),
224                   measurements[i], timeStampMillis, os);
225           os.flush();
226           msgEnd = bos.size();
227           int length = msgEnd - msgStart;
228           if (length > MAX_UDP_MSG_SIZE) {
229             ByteBuffer byteBuffer = ByteBuffer.wrap(bos.getBuffer(), msgStart, prevEnd - msgStart);
230             datagramChannel.write(byteBuffer);
231             msgStart = prevEnd;
232           }
233           prevEnd = msgEnd;
234         }
235         if (msgEnd > msgStart) {
236           ByteBuffer byteBuffer = ByteBuffer.wrap(bos.getBuffer(), msgStart, msgEnd - msgStart);
237           datagramChannel.write(byteBuffer);
238         }
239       }
240       return null;
241     }
242   }
