57      public TSDBMeasurementStore(final String databaseName) throws IOException {
58          this.database = new TimeSeriesDatabase(databaseName, new byte[]{});
59      }

62      @Override
63      public void alocateMeasurements(final EntityMeasurementsInfo measurement,
64                                      final int sampleTimeMillis) throws IOException {
65          String groupName = measurement.getMeasuredEntity().toString();
66          alocateMeasurements(groupName, measurement, sampleTimeMillis);
67      }
68      
69      private void alocateMeasurements(final String groupName, final EntityMeasurementsInfo measurement,
70              final int sampleTimeMillis) throws IOException {
71          synchronized (database) {
72              if (!database.hasTSTable(groupName)) {
73                  String[] measurementNames = measurement.getMeasurementNames();
74                  byte[] description = measurement.getDescription().getBytes(Charsets.UTF_8);
75                  String [] uoms = measurement.getMeasurementUnits();
76                  database.addTSTable(groupName, description, sampleTimeMillis, measurementNames, uoms);
77              }
78          }
79      }
80  
81  
82      @Override
83      public void saveMeasurements(final EntityMeasurementsInfo measurementInfo,
84              final long timeStampMillis, final int sampleTimeMillis, final long ... measurements)
85              throws IOException {
86          String groupName = measurementInfo.getMeasuredEntity().toString();
87          alocateMeasurements(groupName, measurementInfo, sampleTimeMillis);
88          database.write(timeStampMillis, groupName, measurements);
89      }
90  
91      @Override
92      public void close() throws IOException {
93          database.close();
94      }

96      private static String fixName(final String name) {
97          final int length = name.length();
98          StringBuilder result = new StringBuilder(length);
99          for (int i = 0; i < length; i++) {
100             char c = name.charAt(i);
101             if (Character.isJavaIdentifierPart(c)) {
102                 result.append(c);
103             }
104         }
105         return result.toString();
106     }

108     @JmxExport(description = "generate charts for all measurements")
109     public List<String> generateCharts(final int width, final int height) throws IOException {
110         long startTime = ManagementFactory.getRuntimeMXBean().getStartTime();
111         long endTime = System.currentTimeMillis();
112         return generateCharts(startTime, endTime, width, height);
113     }

115     /**
116      * Quantized recorders will have min, max avg charts and distribution charts
117      * generated. Counting recorders will have simple charts generated.
118      *
119      * @param startTimeMillis
120      * @param endTimeMillis
121      * @param width
122      * @param height
123      * @return
124      * @throws IOException
125      */
126     @JmxExport(name = "generateChartsInterval",
127             description = "generate charts for all measurements in specified interval")
128     public List<String> generateCharts(final long startTimeMillis, final long endTimeMillis,
129             final int width, final int height) throws IOException {
130         try {
131             database.flush();
132             List<String> result = new ArrayList<>();
133             Collection<TSTable> columnsInfo = database.getTSTables();
134             for (TSTable info : columnsInfo) {
135                 TimeSeries data = database.read(info.getTableName(), startTimeMillis, endTimeMillis);
136                 if (data.getTimeStamps().length > 0) {
137                     if (canGenerateMinMaxAvgCount(info)) {
138                         result.add(generateMinMaxAvgCountChart(info, data, width, height));
139                     }
140                     if (canGenerateHeatChart(info)) {
141                         result.add(generateHeatChart(info, data, width, height));
142                     }
143                 }
144             }
145             Multimap<String, TSTable> counters = getCounters(columnsInfo);
146             for (Map.Entry<String, Collection<TSTable>> entry : counters.asMap().entrySet()) {
147                 Collection<TSTable> tables = entry.getValue();
148                 int l = tables.size();
149                 long[][] timestamps = new long[l][];
150                 double[][] cdata = new double[l][];
151                 double[][] cdata2 = new double[l][];
152                 int i = 0;
153                 String[] measurementNames = new String[cdata.length];
154                 String[] measurementNames2 = new String[cdata2.length];
155                 String uom1 = "count";
156                 String uom2 = "";
157                 for (TSTable colInfo : tables) {
158                     TimeSeries data = database.read(colInfo.getTableName(), startTimeMillis, endTimeMillis);
159                     timestamps[i] = data.getTimeStamps();
160                     final long[][] values = data.getValues();
161                     cdata[i] = Arrays.getColumnAsDoubles(values, colInfo.getColumnIndex("count"));
162                     cdata2[i] = Arrays.getColumnAsDoubles(values, colInfo.getColumnIndex("total"));
163                     measurementNames[i] = colInfo.getTableName() + ".count";
164                     measurementNames2[i] = colInfo.getTableName() + ".total";
165                     uom2 = new String(colInfo.getTableMetaData(), Charsets.UTF_8);
166                     i++;
167                 }
168                 result.add(generateCountChart(entry.getKey(), timestamps, measurementNames,
169                         measurementNames2, uom1, uom2, cdata, cdata2, width, height));
170             }
171             LOG.info("Generated charts {}", result);
172             return result;
173         } catch (IOException | RuntimeException ex) {
174             LOG.error("Error while generating charts", ex);
175             throw ex;
176         }
177     }

179     private static Multimap<String, TSTable> getCounters(final Collection<TSTable> columnInfos) {
180         Multimap<String, TSTable> result = HashMultimap.create();
181         for (TSTable info : columnInfos) {
182             if (isCounterOnly(info)) {
183                 String groupName = info.getTableName();
184                 if (groupName.startsWith("(")) {
185                     int cidx = groupName.indexOf(',');
186                     if (cidx > 0) {
187                         groupName = groupName.substring(1, cidx);
188                     }
189                 }
190                 result.put(groupName, info);
191             }
192         }
193         return result;
194     }

196     @edu.umd.cs.findbugs.annotations.SuppressWarnings("CLI_CONSTANT_LIST_INDEX")
197     public static boolean isCounterOnly(final TSTable info) {
198         String[] columns = info.getColumnNames();
199         return columns.length == 2 && "count".equals(columns[0])
200                 && "total".equals(columns[1]);
201     }
202 
203     public static boolean canGenerateMinMaxAvgCount(final TSTable info) {
204         return ((info.getColumnIndex("min") >= 0)
205                 && (info.getColumnIndex("max") >= 0)
206                 && (info.getColumnIndex("total") >= 0)
207                 && (info.getColumnIndex("count") >= 0));
208     }
209 
210     public static boolean canGenerateCount(final TSTable info) {
211         return ((info.getColumnIndex("count") >= 0));
212     }
213     
214     
215     public static boolean canGenerateHeatChart(final TSTable info) {
216         for (String mname : info.getColumnNames()) {
217             if (mname.startsWith("Q") && mname.contains("_")) {
218                 return true;
219             }
220         }
221         return false;
222     }

224     private String generateMinMaxAvgCountChart(
225             final TSTable info, final TimeSeries data,
226             final int width, final int height) throws IOException {
227         long[][] vals = data.getValues();
228         double[] min = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("min"));
229         double[] max = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("max"));
230         double[] total = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("total"));
231         double[] count = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("count"));
232         for (int i = 0; i < count.length; i++) {
233             if (count[i] == 0) {
234                 min[i] = 0;
235                 max[i] = 0;
236             }
237         }
238         long[] timestamps = data.getTimeStamps();
239         BufferedImage combined = Charts.createMinMaxAvgCountImg("Measurements for "
240                 + info.getTableName() + " generated by spf4j",
241                 timestamps, min, max, total, count, new String(info.getTableMetaData(), Charsets.UTF_8), width, height);
242         File dbFile = new File(database.getDBFilePath());
243         File graphicFile = File.createTempFile(dbFile.getName() + "_" + fixName(info.getTableName()), ".mmac.png",
244                 dbFile.getParentFile());
245         ImageIO.write(combined, "png", graphicFile);
246         return graphicFile.getPath();
247     }
248 
249     private String generateCountChart(
250             final String groupName, final long[][] timestamps,
251             final String[] measurementNames, final String[] measurementNames2,
252             final String uom1, final String uom2,
253             final double[][] measurements, final double[][] measurements2,
254             final int width, final int height) throws IOException {
255         BufferedImage combined = Charts.generateCountTotalChart(
256                 groupName, timestamps, measurementNames, uom1, measurements, width, height,
257                 measurementNames2, uom2, measurements2);
258         File dbFile = new File(database.getDBFilePath());
259         File graphicFile = File.createTempFile(dbFile.getName() + "_" + fixName(groupName), ".count.png",
260                 dbFile.getParentFile());
261         ImageIO.write(combined, "png", graphicFile);
262         return graphicFile.getPath();
263     }

267     private String generateHeatChart(final TSTable info, final TimeSeries data,
268             final int width, final int height) throws IOException {
269         JFreeChart chart = TimeSeriesDatabase.createHeatJFreeChart(data, info);
270         BufferedImage img = chart.createBufferedImage(width, height);
271         File dbFile = new File(database.getDBFilePath());
272         File graphicFile = File.createTempFile(dbFile.getName() + "_" + fixName(info.getTableName()), ".dist.png",
273                 dbFile.getParentFile());
274         ImageIO.write(img, "png", graphicFile);
275         return graphicFile.getAbsolutePath();
276     }
