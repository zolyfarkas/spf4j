18  package org.spf4j.perf.impl.chart;
19  
20  import java.awt.Color;
21  import java.awt.Paint;
22  import java.io.Serializable;
23  import org.jfree.chart.renderer.PaintScale;
24  import org.jfree.util.PublicCloneable;
25  
26  /**
27   * A paint scale that returns shades of gray.
28   *
29   * @since 1.0.4
30   */
31  public final class InverseGrayScale
32          implements PaintScale, PublicCloneable, Serializable {
33  
34  
35      /** The lower bound. */
36      private final double lowerBound;
37  
38      /** The upper bound. */
39      private final double upperBound;
40  
41      /**
42       * The alpha transparency (0-255).
43       *
44       * @since 1.0.13
45       */
46      private final int alpha;
47  
48      /**
49       * Creates a new <code>GrayPaintScale</code> instance with default values.
50       */
51      public InverseGrayScale() {
52          this(0.0, 1.0);
53      }
54  
55      /**
56       * Creates a new paint scale for values in the specified range.
57       *
58       * @param lowerBound  the lower bound.
59       * @param upperBound  the upper bound.
60       *
61       * @throws IllegalArgumentException if <code>lowerBound</code> is not
62       *       less than <code>upperBound</code>.
63       */
64      public InverseGrayScale(final double lowerBound, final double upperBound) {
65          this(lowerBound, upperBound, 255);
66      }
67  
68      /**
69       * Creates a new paint scale for values in the specified range.
70       *
71       * @param lowerBound  the lower bound.
72       * @param upperBound  the upper bound.
73       * @param alpha  the alpha transparency (0-255).
74       *
75       * @throws IllegalArgumentException if <code>lowerBound</code> is not
76       *       less than <code>upperBound</code>, or <code>alpha</code> is not in
77       *       the range 0 to 255.
78       *
79       * @since 1.0.13
80       */
81      public InverseGrayScale(final double lowerBound, final double upperBound, final int alpha) {
82          if (lowerBound >= upperBound) {
83              throw new IllegalArgumentException(
84                      "Requires " + lowerBound + " < " + upperBound + ".");
85          }
86          if (alpha < 0 || alpha > 255) {
87              throw new IllegalArgumentException(
88                      "Requires alpha in the range 0 to 255. and not " + alpha);
89  
90          }
91          this.lowerBound = lowerBound;
92          this.upperBound = upperBound;
93          this.alpha = alpha;
94      }
95  
96      /**
97       * Returns the lower bound.
98       *
99       * @return The lower bound.
100      *
101      * @see #getUpperBound()
102      */
103     @Override
104     public double getLowerBound() {
105         return this.lowerBound;
106     }
107 
108     /**
109      * Returns the upper bound.
110      *
111      * @return The upper bound.
112      *
113      * @see #getLowerBound()
114      */
115     @Override
116     public double getUpperBound() {
117         return this.upperBound;
118     }
119 
120     /**
121      * Returns the alpha transparency that was specified in the constructor.
122      * 
123      * @return The alpha transparency (in the range 0 to 255).
124      * 
125      * @since 1.0.13
126      */
127     public int getAlpha() {
128         return this.alpha;
129     }
130 
131     /**
132      * Returns a paint for the specified value.
133      *
134      * @param value  the value (must be within the range specified by the
135      *         lower and upper bounds for the scale).
136      *
137      * @return A paint for the specified value.
138      */
139     @Override
140     public Paint getPaint(final double value) {
141         double v = Math.max(value, this.lowerBound);
142         v = Math.min(v, this.upperBound);
143         int g;
144         if (value >= 1) {
145             g = 240 -  (int) ((v - this.lowerBound) / (this.upperBound
146                 - this.lowerBound) * 240.0);
147         } else {
148             g = 255 -  (int) ((v - this.lowerBound) / (this.upperBound
149                 - this.lowerBound) * 15.0);
150         }
151         // FIXME:  it probably makes sense to allocate an array of 256 Colors
152         // and lazily populate this array...
153 
154         return new Color(g, g, g, this.alpha);
155     }
156 
157     @Override
158     public boolean equals(final Object obj) {
159         if (obj == null) {
160             return false;
161         }
162         if (getClass() != obj.getClass()) {
163             return false;
164         }
165         final InverseGrayScale other = (InverseGrayScale) obj;
166         if (Double.doubleToLongBits(this.lowerBound) != Double.doubleToLongBits(other.lowerBound)) {
167             return false;
168         }
169         if (Double.doubleToLongBits(this.upperBound) != Double.doubleToLongBits(other.upperBound)) {
170             return false;
171         }
172         if (this.alpha != other.alpha) {
173             return false;
174         }
175         return true;
176     }
177 
178     @Override
179     public int hashCode() {
180         int hash = 3;
181         hash = 97 * hash
182                 + (int) (Double.doubleToLongBits(this.lowerBound) ^ (Double.doubleToLongBits(this.lowerBound) >>> 32));
183         hash = 97 * hash
184                 + (int) (Double.doubleToLongBits(this.upperBound) ^ (Double.doubleToLongBits(this.upperBound) >>> 32));
185         return  97 * hash + this.alpha;
186     }
187 
188 
189 
190     /**
191      * Returns a clone of this <code>GrayPaintScale</code> instance.
192      *
193      * @return A clone.
194      *
195      * @throws CloneNotSupportedException if there is a problem cloning this
196      *     instance.
197      */
198     @Override
199     public InverseGrayScale clone() {
200         try {
201             return (InverseGrayScale) super.clone();
202         } catch (CloneNotSupportedException ex) {
203            throw new RuntimeException(ex);
204         }
205     }
206     
207     private static final long serialVersionUID = 1L;
208 
209 }
