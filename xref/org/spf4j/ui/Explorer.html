153     private void openMenuItemActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_openMenuItemActionPerformed
154         JFileChooser chooser = new JFileChooser();
155         chooser.setDialogType(JFileChooser.OPEN_DIALOG);
156         chooser.setFileFilter(new Spf4jFileFilter());
157         if (folder != null) {
158             chooser.setCurrentDirectory(folder);
159         }
160 
161         int returnVal = chooser.showOpenDialog(this);
162 
163         if (returnVal == JFileChooser.APPROVE_OPTION) {
164             File file = chooser.getSelectedFile();
165             folder = file.getParentFile();
166             try {
167                 openFile(file);
168             } catch (IOException ex) {
169                 throw new RuntimeException(ex);
170             }
171         }
172     }//GEN-LAST:event_openMenuItemActionPerformed
173 
174     private void openFile(final File file) throws IOException {
175         String fileName = file.getName();
176         JInternalFrame frame;
177         final String absolutePath = file.getAbsolutePath();
178         if (fileName.endsWith("tsdb")) {
179             frame = new TSDBViewJInternalFrame(absolutePath);
180         } else if (fileName.endsWith("ssdump")) {
181             frame = new StackDumpJInternalFrame(absolutePath, true);
182             frame.setVisible(true);
183             desktopPane.add(frame, javax.swing.JLayeredPane.DEFAULT_LAYER);
184             frame = new StackDumpJInternalFrame(absolutePath, false);
185         } else {
186             throw new RuntimeException("Unsupported file format " + fileName);
187         }
188 
189         frame.setVisible(true);
190         desktopPane.add(frame, javax.swing.JLayeredPane.DEFAULT_LAYER);
191     }

244     private static class Spf4jFileFilter extends FileFilter {
245 
246         @Override
247         public boolean accept(File f) {
248             if (f.isDirectory()) {
249                 return true;
250             } else if (f.isFile()) {
251                 String name = f.getName();
252                 return (name.endsWith("tsdb") || name.endsWith("ssdump"));
253             } else {
254                 return false;
255             }
256         }
257 
258         @Override
259         public String getDescription() {
260             return "spf4j dumps";
261         }
262     }
