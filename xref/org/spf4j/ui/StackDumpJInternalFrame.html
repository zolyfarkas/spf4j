56    public StackDumpJInternalFrame(final SampleNode samples,
57            final String title, final boolean isgraph) {
58      super(title);
59      setName(title);
60      initComponents();
61      if (samples == null) {
62        this.samples = new SampleNode(new StackTraceElement[]{new StackTraceElement("NO SAMPLES", "", "", -1)}, 0);
63      } else {
64        this.samples = samples;
65      }
66      setViewType(isgraph);
67      ssScrollPanel.setVisible(true);
68      pack();
69    }
70  
71    private void setViewType(final boolean isgraph) {
72      if (isgraph) {
73        graphToggle.setSelected(true);
74        //ssScrollPanel.setViewportView(new ZStackPanel(this.samples));
75        ssScrollPanel.setViewportView(new HotFlameStackPanel(this.samples));
76      } else {
77        graphToggle.setSelected(false);
78        ssScrollPanel.setViewportView(new FlameStackPanel(this.samples));
79      }
80    }
