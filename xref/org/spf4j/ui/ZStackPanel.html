46      private Graph<Method, SampleNode.InvocationCount> graph;
47      private Graph<Method, SampleNode.InvocationCount> completeGraph;
48      private Map<Method, Rectangle2D> methodLocations;
49      private double totalHeight = 0;

51      public ZStackPanel(final SampleNode samples) {
52          super(samples);
53          completeGraph = SampleNode.toGraph(samples);
54          graph = null;
55      }

57      @Override
58      public int paint(final Graphics2D gr, final double width, final double rowHeight) {
59          paintGraph(gr, 0, 0, (int) width, rowHeight);
60          return (int) totalHeight;
61      }
62  
63      private void paintGraph(
64              final Graphics2D g2, final int x, final int y, final double areaWidth, final double rowHeight) {
65  
66          graph = completeGraph.copy();
67          int rootSamples = graph.getEdges(Method.ROOT).getIncomming().keySet().iterator().next().getValue();
68          final double pps = areaWidth / rootSamples;
69          methodLocations = new HashMap<Method, Rectangle2D>();
70          final Traversals.TraversalCallback<Method, SampleNode.InvocationCount> traversalCallback =
71                  new Traversals.TraversalCallback<Method, SampleNode.InvocationCount>() {
72              private int counter = 0;
73  
74              @Override
75              public void handle(final Method vertex, final Map<SampleNode.InvocationCount, Method> edges) {
76                  if (edges.size() == 1) {
77                      if (vertex.equals(Method.ROOT)) {
78                          int nrSamples = edges.keySet().iterator().next().getValue();
79                          drawMethod(vertex, nrSamples, (double) x, (double) y, (double) areaWidth,
80                                  (double) rowHeight);
81                      } else {
82                          Map.Entry<SampleNode.InvocationCount, Method> fromEntry =
83                                  edges.entrySet().iterator().next();
84                          Method fromMethod = fromEntry.getValue();
85                          Rectangle2D fromMethodLocation = methodLocations.get(fromMethod);
86                          int relativeSamples = 0;
87                          for (Map.Entry<SampleNode.InvocationCount, Method> ens
88                                  : graph.getEdges(fromMethod).getOutgoing().entrySet()) {
89                              Method slm = ens.getValue();
90                              if (methodLocations.containsKey(slm) && graph.getEdges(slm).getIncomming().size() == 1
91                                      && !fromMethod.equals(slm)) {
92                                  relativeSamples += ens.getKey().getValue();
93                              }
94                          }
95  
96                          int nrSamples = fromEntry.getKey().getValue();
97                          double width = nrSamples * pps;
98                          double newX = fromMethodLocation.getX() + relativeSamples * pps;
99                          drawMethod(vertex, nrSamples, newX, (fromMethodLocation.getY() + rowHeight),
100                                 width, rowHeight);
101                     }
102                 } else if (edges.size() > 1) {
103                     renderMethodLinked(edges, vertex);
104                 } else {
105                     throw new IllegalStateException("Invalid state, there must be a way to get to node " + vertex);
106                 }
107             }
108 
109             private void drawMethod(final Method vertex, final int nrSamples,
110                     final double x, final double y, final double width, final double height,
111                     final Point... fromLinks) {
112                 Rectangle2D.Double location = new Rectangle2D.Double(x, y, width, height);
113                 methodLocations.put(vertex, location);
114                 tooltipDetail.insert(new float[]{(float) x, (float) y}, new float[]{(float) width, (float) height},
115                         Pair.of(vertex, nrSamples));
116                 double newHeight = y + height;
117                 if (totalHeight < newHeight) {
118                     totalHeight = newHeight;
119                 }
120                 FlameStackPanel.setElementColor(counter++, g2);
121                 g2.setClip((int) x, (int) y, (int) width, (int) height);
122                 g2.fillRect((int) x, (int) y, (int) width, (int) height);
123                 String val = vertex + "-" + nrSamples;
124 
125                 g2.setPaint(Color.BLACK);
126                 g2.drawString(val, (int) x, (int) (y + height - 1));
127                 g2.setClip(null);
128                 g2.setPaint(LINK_COLOR);
129                 for (Point divLoc : fromLinks) {
130                     g2.drawLine((int) divLoc.getX(), (int) divLoc.getY(),
131                             (int) (x + width / 2), (int) y);
132                 }
133                 g2.drawRect((int) x, (int) y, (int) width, (int) height);
134             }
135 
136             private void renderMethodLinked(final Map<SampleNode.InvocationCount, Method> edges, final Method vertex) {
137                 List<Point> fromPoints = new ArrayList<Point>(edges.size());
138                 double newYBase = 0;
139                 double newXBase = Double.MAX_VALUE;
140                 double newWidth = 0;
141                 double maxX = Double.MIN_VALUE;
142                 int nrSamples = 0;
143                 for (Map.Entry<SampleNode.InvocationCount, Method> fromEntry : edges.entrySet()) {
144                     Rectangle2D fromRect = methodLocations.get(fromEntry.getValue());
145                     newWidth += fromEntry.getKey().getValue() * pps;
146                     nrSamples += fromEntry.getKey().getValue();
147                     if (fromRect == null) {
148                         continue;
149                     }
150                     double fromX = fromRect.getX();
151                     if (fromX < newXBase) {
152                         newXBase = fromX;
153                     }
154                     double mx = fromRect.getMaxX();
155                     if (mx > maxX) {
156                         maxX = mx;
157                     }
158                     double newY = fromRect.getMaxY() + rowHeight;
159                     if (newY > newYBase) {
160                         newYBase = newY;
161                     }
162                     fromPoints.add(new Point((int) fromRect.getCenterX(), (int) fromRect.getMaxY()));
163                 }
164 
165                 Pair<List<Pair<Method, Integer>>, Double> result = findEmptySpace(newXBase, newYBase, newWidth, maxX);
166                 while (!result.getFirst().isEmpty()) {
167                     // TODO: optimize this to increment with a better value
168                     newYBase += rowHeight;
169                     result = findEmptySpace(newXBase, newYBase, newWidth, maxX);
170                 }
171                 drawMethod(vertex, nrSamples, result.getSecond(), newYBase,
172                         newWidth, rowHeight, fromPoints.toArray(new Point[fromPoints.size()]));
173             }
174 
175             @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
176             private Pair<List<Pair<Method, Integer>>, Double> findEmptySpace(
177                     final double newXBase, final double newYBase,
178                     final double newWidth, final double maxX) {
179                 double tryx = newXBase + (maxX - newXBase) / 2 - newWidth / 2;
180                 tryx = fitToViewableArea(tryx, newWidth);
181                 final float[] oc = new float[]{(float) newWidth, Float.MAX_VALUE};
182                 List<Pair<Method, Integer>> methods =
183                         tooltipDetail.search(new float[]{(float) tryx, (float) newYBase}, oc);
184                 if (!methods.isEmpty()) {
185                     tryx = newXBase;
186                     tryx = fitToViewableArea(tryx, newWidth);
187                     methods = tooltipDetail.search(new float[]{(float) tryx, (float) newYBase}, oc);
188                     if (!methods.isEmpty()) {
189                         tryx = maxX - newWidth;
190                         tryx = fitToViewableArea(tryx, newWidth);
191                         methods = tooltipDetail.search(new float[]{(float) tryx, (float) newYBase}, oc);
192                     }
193                 }
194                 return Pair.of(methods, tryx);
195             }
196 
197             private double fitToViewableArea(final double ptryx, final double newWidth) {
198                 double tryx = ptryx;
199                 if (tryx < 0) {
200                     tryx = 0;
201                 } else if (tryx > areaWidth - newWidth) {
202                     tryx = areaWidth - newWidth;
203                 }
204                 return tryx;
205             }
206         };
207 
208         Traversals.customTraverse(graph, Method.ROOT, traversalCallback);
209 
210     }

212     @Override
213     public String getDetail(final Point location) {
214         List<Pair<Method, Integer>> tips = tooltipDetail.search(new float[]{location.x, location.y}, new float[]{0, 0});
215         if (tips.size() >= 1) {
216             final Pair<Method, Integer> node = tips.get(0);
217             final Method method = node.getFirst();
218             final Map<SampleNode.InvocationCount, Method> incomming =
219                     graph.getEdges(method).getIncomming();
220             StringBuilder sb = new StringBuilder();
221             sb.append(method).append('-').append(node.getSecond())
222                     .append("\n invoked from: ");
223             appendEdgeInfo(incomming, sb);
224             sb.append("\n invoking: ");
225             final Map<SampleNode.InvocationCount, Method> outgoing =
226                     graph.getEdges(method).getOutgoing();
227             appendEdgeInfo(outgoing, sb);
228             return sb.toString();
229         } else {
230             return null;
231         }
232     }

234     @Override
235     public void filter() {
236         List<Pair<Method, Integer>> tips = tooltipDetail.search(new float[]{xx, yy}, new float[]{0, 0});
237         if (tips.size() >= 1) {
238             final Method value = tips.get(0).getFirst().withId(0);
239             samples = samples.filteredBy(new EqualsPredicate<Method>(value));
240             if (samples != null) {
241                 this.completeGraph = SampleNode.toGraph(samples);
242             } else {
243                 this.completeGraph = new HashMapGraph<>();
244             }
245             repaint();
246         }
247     }
248 
249     private void appendEdgeInfo(final Map<SampleNode.InvocationCount, Method> incomming, final StringBuilder sb) {
250         for (Map.Entry<SampleNode.InvocationCount, Method> entry : incomming.entrySet()) {
251             int ic = entry.getKey().getValue();
252             Method method = entry.getValue();
253             sb.append(method).append('-').append(ic).append("; ");
254         }
255     }
