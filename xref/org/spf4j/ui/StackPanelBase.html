51   */
52  public abstract class StackPanelBase extends JPanel
53          implements ActionListener, MouseListener {
54      private static final long serialVersionUID = 1L;
55      //CHECKSTYLE:OFF
56      protected SampleNode samples;
57      protected RTree<Pair<Method, Integer>> tooltipDetail = new RTree<Pair<Method, Integer>>();
58      protected int xx;
59      protected int yy;
60      //CHECKSTYLE:ON
61      private final JPopupMenu menu;
62  
63      public static final Color LINK_COLOR = new Color(128, 128, 128, 128);
64  
65      public StackPanelBase(final SampleNode samples) {
66          this.samples = samples;
67          setPreferredSize(new Dimension(400, 20 * samples.height() + 10));
68          final ToolTipManager sharedInstance = ToolTipManager.sharedInstance();
69          sharedInstance.registerComponent(this);
70          sharedInstance.setDismissDelay(30000);
71          menu = buildPopupMenu(this);
72          addMouseListener(this);
73      }
74  
75  
76          // disable finbugs since I don't care about internationalization for now.
77      @edu.umd.cs.findbugs.annotations.SuppressWarnings
78      private static JPopupMenu buildPopupMenu(final ActionListener listener) {
79          JPopupMenu result = new JPopupMenu("Actions");
80          JMenuItem filter = new JMenuItem("Filter");
81          filter.setActionCommand("FILTER");
82          filter.addActionListener(listener);
83          result.add(filter);
84          JMenuItem copy = new JMenuItem("Copy");
85          copy.setActionCommand("COPY");
86          copy.addActionListener(listener);
87          result.add(copy);
88          return result;
89      }
90  
91  
92      @Override
93      public final String getToolTipText(final MouseEvent event) {
94          Point location = event.getPoint();
95          return getDetail(location);
96      }
97  
98      @Override
99      public final void paintComponent(final Graphics g) {
100         super.paintComponent(g);
101         Dimension size = getSize();
102         Insets insets = getInsets();
103         Rectangle2D available = new Rectangle2D.Double(insets.left, insets.top,
104                 size.getWidth() - insets.left - insets.right,
105                 size.getHeight() - insets.top - insets.bottom);
106         Graphics2D g2 = (Graphics2D) g.create();
107         try {
108             double rowHeight =  g2.getFont().getStringBounds("ROOT", g2.getFontRenderContext()).getHeight();
109 
110             GraphicsConfiguration gc = g2.getDeviceConfiguration();
111             BufferedImage img = gc.createCompatibleImage(
112                     (int) available.getWidth(), (int) available.getHeight(),
113                     Transparency.TRANSLUCENT);
114 
115             double width = available.getWidth();
116             tooltipDetail.clear();
117             Graphics2D gr = img.createGraphics();
118             gr.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING, RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
119             int height = paint(gr, width, rowHeight);
120             g2.drawImage(img, insets.left, insets.top, this);
121             final Dimension dimension = new Dimension((int) size.getWidth(), height + 10);
122             setPreferredSize(dimension);
123             setSize(dimension);
124 
125         } finally {
126             g2.dispose();
127         }
128     }
129 
130 
131     public abstract int paint(Graphics2D gr, double width, double rowHeight);
132 
133     @Override
134     public final void actionPerformed(final ActionEvent e) {
135         final String actionCommand = e.getActionCommand();
136         if ("FILTER".equals(actionCommand)) {
137             filter();
138         } else if ("COPY".equals(actionCommand)) {
139             final String detail = getDetail(new Point(xx, yy));
140             java.awt.Toolkit.getDefaultToolkit().getSystemClipboard().setContents(
141                     new Transferable() {
142                 @Override
143                 public DataFlavor[] getTransferDataFlavors() {
144                     return new DataFlavor[]{DataFlavor.stringFlavor};
145                 }
146 
147                 @Override
148                 public boolean isDataFlavorSupported(final DataFlavor flavor) {
149                     return flavor.equals(DataFlavor.stringFlavor);
150                 }
151 
152                 @Override
153                 public Object getTransferData(final DataFlavor flavor) {
154                     return detail;
155                 }
156             }, new ClipboardOwner() {
157                 @Override
158                 public void lostOwnership(final Clipboard clipboard, final Transferable contents) {
159                 }
160             });
161         }
162     }
163 
164 
165 
166     @Override
167     public final void mousePressed(final MouseEvent e) {
168         if (e.isPopupTrigger()) {
169             xx = e.getX();
170             yy = e.getY();
171             menu.show(this, e.getX(), e.getY());
172         }
173     }
174 
175     @Override
176     public final void mouseReleased(final MouseEvent e) {
177         if (e.isPopupTrigger()) {
178             xx = e.getX();
179             yy = e.getY();
180             menu.show(this, e.getX(), e.getY());
181         }
182     }
183

185     @edu.umd.cs.findbugs.annotations.SuppressWarnings
186     public  void mouseClicked(final MouseEvent e) {
187     }
188 
189     @Override
190     @edu.umd.cs.findbugs.annotations.SuppressWarnings
191     public void mouseEntered(final MouseEvent e) {
192     }
193 
194     @Override
195     @edu.umd.cs.findbugs.annotations.SuppressWarnings
196     public void mouseExited(final MouseEvent e) {
197     }
198 
199     public static void setElementColor(final int depth, final Graphics2D g2) {
200         if (depth % 2 == 0) {
201             g2.setPaint(Color.YELLOW);
202             g2.setBackground(Color.YELLOW);
203         } else {
204             g2.setPaint(Color.ORANGE);
205             g2.setBackground(Color.ORANGE);
206         }
207     }
208 
209     public abstract String getDetail(Point location);
210 
211     public abstract void filter();
212 }
