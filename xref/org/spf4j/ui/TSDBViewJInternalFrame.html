60      public TSDBViewJInternalFrame(final String databaseFile) throws IOException {
61          super(databaseFile);
62          initComponents();
63          tsDb = new TimeSeriesDatabase(databaseFile, null);
64          Collection<TSTable> columnsInfo = tsDb.getTSTables();
65          Map<String, DefaultMutableTreeNode> gNodes = new HashMap<String, DefaultMutableTreeNode>();
66          DefaultMutableTreeNode root = new DefaultMutableTreeNode(databaseFile);
67          long startDate = System.currentTimeMillis();
68          for (TSTable info : columnsInfo) {
69              String groupName = info.getTableName();
70              long tableStart = tsDb.readStartDate(groupName);
71              if (tableStart < startDate) {
72                  startDate = tableStart;
73              }
74              Pair<String, String> pair = Pair.from(groupName);
75              if (pair == null) {
76                  DefaultMutableTreeNode child = new DefaultMutableTreeNode(groupName);
77                  for (String colName : info.getColumnNames()) {
78                      child.add(new DefaultMutableTreeNode(colName));
79                  }
80                  root.add(child);
81              } else {
82                  groupName = pair.getFirst();
83                  DefaultMutableTreeNode gNode = gNodes.get(groupName);
84                  if (gNode == null) {
85                      gNode = new DefaultMutableTreeNode(groupName);
86                      gNodes.put(groupName, gNode);
87                      root.add(gNode);
88                  }
89                  DefaultMutableTreeNode child = new DefaultMutableTreeNode(pair.getSecond());
90                  for (String colName : info.getColumnNames()) {
91                      child.add(new DefaultMutableTreeNode(colName));
92                  }
93                  gNode.add(child);
94              }
95          }
96          measurementTree.setModel(new DefaultTreeModel(root));
97          measurementTree.setVisible(true);
98          this.startDate.setValue(new Date(startDate));
99      }

236         TreePath[] selectionPaths = measurementTree.getSelectionPaths();
237         List<String> selectedTables = getSelectedTables(selectionPaths);
238         if (!selectedTables.isEmpty()) {
239             JFileChooser chooser = new JFileChooser();
240             chooser.setDialogType(JFileChooser.SAVE_DIALOG);
241             int returnVal = chooser.showOpenDialog(this);
242             if (returnVal == JFileChooser.APPROVE_OPTION) {
243                 File file = chooser.getSelectedFile();
244                 try {
245                     tsDb.writeCsvTables(selectedTables, file);
246                 } catch (IOException ex) {
247                     throw new RuntimeException(ex);
248                 }
249             }
250         }

266     @edu.umd.cs.findbugs.annotations.SuppressWarnings("CLI_CONSTANT_LIST_INDEX")
267     private static List<String> getSelectedTables(@Nullable final TreePath[] selectionPaths) {
268         if (selectionPaths == null) {
269             return Collections.EMPTY_LIST;
270         }
271         List<String> result = new ArrayList<String>();
272         for (TreePath path : selectionPaths) {
273             Object[] pathArr = path.getPath();
274             if (pathArr.length < 2) {
275                 continue;
276             }
277             DefaultMutableTreeNode colNode = (DefaultMutableTreeNode) pathArr[1];
278             int depth = colNode.getDepth();
279             String tableName;
280             if (depth == 1) {
281                 result.add((String) colNode.getUserObject());
282             } else {
283                 Enumeration childEnum = colNode.children();
284                 while (childEnum.hasMoreElements()) {
285                     DefaultMutableTreeNode child = (DefaultMutableTreeNode) childEnum.nextElement();
286                     tableName =
287                             Pair.of((String) colNode.getUserObject(), (String) child.getUserObject()).toString();
288                     result.add(tableName);
289                 }
290             }
291         }
292         return result;
293     }

295     private void addChartToPanel(final String tableName, final JPanel content) throws IOException {
296         TSTable info = tsDb.getTSTable(tableName);
297          long startTime = ((Date) startDate.getValue()).getTime();
298          long endTime = ((Date) endDate.getValue()).getTime();
299         if (TSDBMeasurementStore.canGenerateHeatChart(info)) {
300             JFreeChart chart = tsDb.createHeatJFreeChart(info.getTableName(),
301                     startTime, endTime);
302             ChartPanel pannel = new ChartPanel(chart);
303             pannel.setPreferredSize(new Dimension(600, 800));
304             pannel.setDomainZoomable(false);
305             pannel.setMouseZoomable(false);
306             pannel.setRangeZoomable(false);
307             pannel.setZoomAroundAnchor(false);
308             pannel.setZoomInFactor(1);
309             pannel.setZoomOutFactor(1);
310             content.add(pannel);
311         }
312         if (TSDBMeasurementStore.canGenerateMinMaxAvgCount(info)) {
313             JFreeChart chart = tsDb.createMinMaxAvgJFreeChart(info.getTableName(),
314                     startTime, endTime);
315             ChartPanel pannel = new ChartPanel(chart);
316             pannel.setPreferredSize(new Dimension(600, 600));
317             content.add(pannel);
318 
319         }
320         if (TSDBMeasurementStore.canGenerateCount(info)) {
321             JFreeChart chart = tsDb.createCountJFreeChart(info.getTableName(),
322                     startTime, endTime);
323             ChartPanel pannel = new ChartPanel(chart);
324             pannel.setPreferredSize(new Dimension(600, 600));
325             content.add(pannel);
326         } else {
327             List<JFreeChart> createJFreeCharts = tsDb.createJFreeCharts(info.getTableName(), startTime, endTime);
328             for (JFreeChart chart : createJFreeCharts) {
329                 ChartPanel pannel = new ChartPanel(chart);
330                 pannel.setPreferredSize(new Dimension(600, 600));
331                 content.add(pannel);
332             }
333         }
