26  @State(Scope.Benchmark)

28  @Threads(value = 8)
29  public class ZelBenchmark {
30  
31      private static final Program ZEL_PROG;
32      private static final Script GROOVY_PROG;
33      private static final Serializable MVEL_PROG;
34      private static final Expression SPRING_EXP;
35  
36      /**
37       * A few notes about this benchmark:
38       *
39       * 1) Zel does overflow check for each math operation and automatically upgrades
40       * the data representation, which comes at a overhead.
41       * 2) Zel uses reflection to do java method invocations, which is slower than byte code generation
42       * that mvel and groovy uses.
43       * 3) Spring expression language is slow like hell, and there is no excuse for it :-)
44       *
45       */
46      
47      static {
48          String testScript = "a-b+1+c.length() - d.toString().substring(0, 1).length()";
49          try {
50              ZEL_PROG = Program.compile(testScript, "a", "b", "c", "d");
51          } catch (CompileException ex) {
52              throw new RuntimeException(ex);
53          }
54          GroovyShell shell = new GroovyShell();
55          GROOVY_PROG = shell.parse(testScript);
56          MVEL_PROG = MVEL.compileExpression(testScript);
57          ExpressionParser parser = new SpelExpressionParser();
58          SPRING_EXP = parser.parseExpression(
59                  "['a']-['b']+1+['c'].length() - ['d'].toString().substring(0, 1).length()");
60      }
61  
62      @Benchmark
63      public Object testZel()
64              throws ZExecutionException, InterruptedException {
65          return ZEL_PROG.execute(3, 2, " ", "bla");
66      }
67  
68      @Benchmark
69      public Object testGroovy() {
70          Binding binding = new Binding();
71          binding.setVariable("a", 3);
72          binding.setVariable("b", 2);
73          binding.setVariable("c", " ");
74          binding.setVariable("d", "bla");
75          GROOVY_PROG.setBinding(binding);
76          return GROOVY_PROG.run();
77      }
78  
79      @Benchmark
80      public Object testMvel() {
81          Map vars = new HashMap();
82          vars.put("a", 3);
83          vars.put("b", 2);
84          vars.put("c", " ");
85          vars.put("d", "bla");
86          return MVEL.executeExpression(MVEL_PROG, vars);
87  
88      }
89      
90      @Benchmark
91      public Object testSpring() {
92          Map vars = new HashMap();
93          vars.put("a", 3);
94          vars.put("b", 2);
95          vars.put("c", " ");
96          vars.put("d", "bla");
97          return SPRING_EXP.getValue(vars, Integer.class);
98  
99      }
