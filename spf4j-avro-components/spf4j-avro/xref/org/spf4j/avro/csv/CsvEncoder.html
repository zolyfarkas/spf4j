16  package org.spf4j.avro.csv;
17  
18  import java.io.IOException;
19  import java.nio.ByteBuffer;
20  import javax.annotation.Nullable;
21  import org.apache.avro.Schema;
22  import org.apache.avro.io.Encoder;
23  import org.apache.avro.io.parsing.JsonGrammarGenerator;
24  import org.apache.avro.io.parsing.Parser;
25  import org.apache.avro.io.parsing.Symbol;
26  import org.apache.avro.util.Utf8;
27  import org.spf4j.base.Base64;
28  import org.spf4j.io.csv.CsvWriter;
29  
30  /**
31   * A Csv decoder for avro records.
32   *
33   * @author Zoltan Farkas
34   */
35  public final class CsvEncoder extends Encoder implements Parser.ActionHandler {
36  
37    private final CsvWriter csvWriter;
38  
39    private final Parser parser;
40  
41    private final Schema writerSchema;
42  
43    private boolean firstItem = true;
44  
45    public CsvEncoder(final CsvWriter csvWriter, final Schema writerSchema) throws IOException {
46      this.csvWriter = csvWriter;
47      this.parser = new Parser(new JsonGrammarGenerator().generate(writerSchema), this);
48      if (writerSchema.getType() != Schema.Type.ARRAY) {
49        throw new IllegalArgumentException("schema must be an array: " + writerSchema);
50      }
51      if (writerSchema.getElementType().getType() != Schema.Type.RECORD) {
52        throw new IllegalArgumentException("schema  array element must be record: " + writerSchema);
53      }
54      this.writerSchema = writerSchema;
55    }
56  
57    public void writeHeader() throws IOException {
58      writeHeader("", writerSchema.getElementType());
59      csvWriter.writeEol();
60    }
61  
62    private void writeHeader(final String prefix, final Schema rowSchema) throws IOException {
63      for (Schema.Field field : rowSchema.getFields()) {
64        Schema schema = field.schema();
65        Schema.Type type = schema.getType();
66        String name = prefix + field.name();
67        switch (type) {
68          case ARRAY:
69          case MAP:
70            throw new UnsupportedOperationException(type + " unsupported in CSV row" + rowSchema);
71          case UNION:
72            csvWriter.writeElement(name + ".type");
73            for (Schema us : schema.getTypes()) {
74              switch (us.getType()) {
75                case ARRAY:
76                case RECORD:
77                case MAP:
78                  throw new UnsupportedOperationException(type + " unsupported in CSV row union" + rowSchema);
79                default:
80                  continue;
81              }
82            }
83            csvWriter.writeElement(name);
84            break;
85          case RECORD:
86            writeHeader(name + '.', schema);
87            break;
88          default:
89            csvWriter.writeElement(name);
90        }
91      }
92    }
93  
94    @Override
95    public void writeNull() throws IOException {
96      parser.advance(Symbol.NULL);
97      csvWriter.writeElement("");
98    }
99  
100   @Override
101   public void writeBoolean(final boolean bln) throws IOException {
102     parser.advance(Symbol.BOOLEAN);
103     csvWriter.writeElement(Boolean.toString(bln));
104   }
105 
106   @Override
107   public void writeInt(final int i) throws IOException {
108     parser.advance(Symbol.INT);
109     csvWriter.writeElement(Integer.toString(i));
110   }
111 
112   @Override
113   public void writeLong(final long l) throws IOException {
114     parser.advance(Symbol.LONG);
115     csvWriter.writeElement(Long.toString(l));
116   }
117 
118   @Override
119   public void writeFloat(final float f) throws IOException {
120     parser.advance(Symbol.FLOAT);
121     csvWriter.writeElement(Float.toString(f));
122   }
123 
124   @Override
125   public void writeDouble(final double d) throws IOException {
126     parser.advance(Symbol.DOUBLE);
127     csvWriter.writeElement(Double.toString(d));
128   }
129 
130   @Override
131   public void writeString(final Utf8 utf8) throws IOException {
132     parser.advance(Symbol.STRING);
133     csvWriter.writeElement(utf8);
134   }
135 
136   @Override
137   public void writeBytes(final ByteBuffer bb) throws IOException {
138     parser.advance(Symbol.BYTES);
139     int position = bb.position();
140     csvWriter.writeElement(Base64.encodeBase64(bb.array(), bb.arrayOffset() + position, bb.limit() - position));
141   }
142 
143   @Override
144   public void writeBytes(final byte[] bytes, final int i, final int length) throws IOException {
145     parser.advance(Symbol.BYTES);
146     csvWriter.writeElement(Base64.encodeBase64(bytes, i, length));
147   }
148 
149   @Override
150   public void writeFixed(final byte[] bytes, final int i, final int length) throws IOException {
151     parser.advance(Symbol.FIXED);
152     csvWriter.writeElement(Base64.encodeBase64(bytes, i, length));
153   }
154 
155   @Override
156   public void writeEnum(final int e) throws IOException {
157     parser.advance(Symbol.ENUM);
158     Symbol.EnumLabelsAction top = (Symbol.EnumLabelsAction) parser.popSymbol();
159     csvWriter.writeElement(top.getLabel(e));
160   }
161 
162   @Override
163   public void writeArrayStart() throws IOException {
164     firstItem = true;
165     parser.advance(Symbol.ARRAY_START);
166     // do nothing.
167   }
168 
169   @Override
170   public void writeArrayEnd() throws IOException {
171     parser.advance(Symbol.ITEM_END);
172     parser.advance(Symbol.ARRAY_END);
173     // do nothing.
174   }
175 
176   @Override
177   public void writeMapStart() {
178     throw new UnsupportedOperationException();
179   }
180 
181   @Override
182   public void writeMapEnd() {
183     throw new UnsupportedOperationException();
184   }
185 
186   @Override
187   public void writeIndex(final int unionIndex) throws IOException {
188     parser.advance(Symbol.UNION);
189     Symbol.Alternative top = (Symbol.Alternative) parser.popSymbol();
190     Symbol symbol = top.getSymbol(unionIndex);
191     csvWriter.writeElement(top.getLabel(unionIndex));
192     parser.pushSymbol(Symbol.UNION_END);
193     parser.pushSymbol(symbol);
194   }
195 
196   @Override
197   public void flush() throws IOException {
198     csvWriter.flush();
199   }
200 
201   @Override
202   @Nullable
203   public Symbol doAction(final Symbol symbol, final Symbol symbol1) {
204     return null;
205   }
206 
207   @Override
208   public void setItemCount(final long l) {
209     // nothing to do.
210   }
211 
212   @Override
213   public void startItem() throws IOException {
214     if (!firstItem) {
215       parser.advance(Symbol.ITEM_END);
216       csvWriter.writeEol();
217     } else {
218       firstItem = false;
219     }
220     // nothing to do
221   }
222 
223   @Override
224   public String toString() {
225     return "CsvEncoder{" + "csvWriter=" + csvWriter + ", parser=" + parser
226             + ", writerSchema=" + writerSchema + ", firstItem=" + firstItem + '}';
227   }
228 
229 }
