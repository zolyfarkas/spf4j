84            = (a, b) -> {
85              SchemaUtils.copyAliases(a, b);
86              SchemaUtils.copyLogicalTypes(a, b);
87            };

116   public static void copyAliases(final Schema from, final Schema to) {
117     //CHECKSTYLE:OFF
118     switch (from.getType()) { // only named types.
119       case RECORD:
120       case ENUM:
121       case FIXED:
122         Set<String> aliases = from.getAliases();
123         for (String alias : aliases) {
124           to.addAlias(alias);
125         }
126         break;
127       default:
128       //ignore unnamed one's
129     }
130     //CHECKSTYLE:OFF
131   }
132 
133   public static void copyAliases(final Schema.Field from, final Schema.Field to) {
134     Set<String> aliases = from.aliases();
135     for (String alias : aliases) {
136       to.addAlias(alias);
137     }
138   }

140   public static void copyLogicalTypes(final Schema from, final Schema to) {
141     LogicalType logicalType = from.getLogicalType();
142     if (logicalType != null) {
143       logicalType.addToSchema(to);
144     }
145   }

147   public static void copyProperties(final JsonProperties from, final JsonProperties to) {
148     Map<String, Object> objectProps = from.getObjectProps();
149     for (Map.Entry<String, Object> entry : objectProps.entrySet()) {
150       to.addProp(entry.getKey(), entry.getValue());
151     }
152   }

154   public static boolean isNamedType(final Schema schema) {
155     Schema.Type type = schema.getType();
156     switch (type) {
157       case ENUM:
158       case RECORD:
159       case FIXED:
160         return true;
161       default:
162         return false;
163     }
164   }

166   public static String getJavaClassName(final Schema schema) {
167     String namespace = schema.getNamespace();
168     if (namespace == null || namespace.isEmpty()) {
169       return SpecificCompiler.mangle(schema.getName());
170     } else {
171       return namespace + '.' + SpecificCompiler.mangle(schema.getName());
172     }
173   }

175   @Beta
176   public static void writeIdlProtocol(final String protocolName, final String protocolNameSpace,
177           final Appendable appendable, final Schema... schemas) throws IOException {
178     if (protocolNameSpace != null) {
179       appendable.append("@namespace(\"").append(protocolNameSpace).append("\")\n");
180     }
181     appendable.append("protocol ").append(protocolName).append(" {\n\n");
182     if (isIdlCycleSupport()) {
183       writeIdl(appendable, new HashSet<String>(4), protocolNameSpace, schemas);
184     } else {
185       writeIdlLegacy(appendable, new HashSet<String>(4), protocolNameSpace, schemas);
186     }
187     appendable.append("}\n");
188   }

190   @Beta
191   public static void writeIdl(final Appendable appendable,
192           final Set<String> alreadyDeclared, final String protocolNameSpace, final Schema ... pschemas)
193           throws IOException {
194     try(JsonGenerator jsonGen = createJsonGenerator(appendable)) {
195       final Set<Schema> toDeclare = new HashSet<>(4);
196       toDeclare.addAll(Arrays.asList(pschemas));
197       while (!toDeclare.isEmpty()) {
198         Iterator<Schema> iterator = toDeclare.iterator();
199         Schema schema = iterator.next();
200         iterator.remove();
201         writeSchema(schema, appendable, jsonGen, protocolNameSpace, alreadyDeclared, toDeclare);
202         appendable.append('\n');
203       }
204     }
205   }

207   public static JsonGenerator createJsonGenerator(final Appendable appendable) throws IOException {
208     JsonGenerator jsonGen;
209     if (appendable instanceof Writer) {
210       jsonGen = Schema.FACTORY.createGenerator((Writer) appendable);
211     } else {
212       jsonGen = Schema.FACTORY.createGenerator(new AppendableWriter(appendable));
213     }
214     return jsonGen;
215   }

217   @Beta
218   @SuppressFBWarnings("RV_RETURN_VALUE_IGNORED")
219   public static void writeIdlLegacy(final Appendable appendable,
220           final Set<String> alreadyDeclared, final String protocolNameSpace, final Schema ... pschemas)
221           throws IOException {
222     MutableGraph<Schema> schemaDeps = GraphBuilder.directed().allowsSelfLoops(false)
223             .expectedNodeCount(4).build();
224     Map<Schema, String> idlRepresentation = new HashMap<>(4);
225     final Set<Schema> toDeclare = new HashSet<>(4);
226     toDeclare.addAll(Arrays.asList(pschemas));
227     while (!toDeclare.isEmpty()) {
228       Iterator<Schema> iterator = toDeclare.iterator();
229       Schema schema = iterator.next();
230       iterator.remove();
231       StringWriter schemaIdrStr = new StringWriter();
232       JsonGenerator jsonGen = Schema.FACTORY.createGenerator(schemaIdrStr);
233       Set<Schema> orig = new HashSet<>(toDeclare);
234       writeSchema(schema, schemaIdrStr, jsonGen, protocolNameSpace, alreadyDeclared, toDeclare);
235       idlRepresentation.put(schema, schemaIdrStr.toString());
236       schemaDeps.addNode(schema);
237       Set<Schema> dependencies = Sets.difference(toDeclare, orig);
238       for (Schema dep : dependencies) {
239         schemaDeps.putEdge(schema, dep);
240       }
241      }
242      // Now process the nodes from leaves onward.
243 
244     MutableGraph<Schema> traverseGraph = Graphs.clone(schemaDeps);
245     Set<Schema> nodes = traverseGraph.nodes();
246     List<Schema> nodesToRemove = new ArrayList<>();
247     do {
248       for (Schema token : nodes) {
249         if (traverseGraph.outDegree(token) == 0) {
250           nodesToRemove.add(token);
251           appendable.append(idlRepresentation.get(token));
252           appendable.append('\n');
253         }
254       }
255       if (nodesToRemove.isEmpty() && !nodes.isEmpty()) {
256         throw new IllegalArgumentException("Schema definition cycle for" + nodes);
257       }
258       for (Schema token : nodesToRemove) {
259         traverseGraph.removeNode(token);
260       }
261       nodesToRemove.clear();
262       nodes = traverseGraph.nodes();
263     } while (!nodes.isEmpty());
264 
265 
266   }

269   private static void writeSchema(Schema schema, final Appendable appendable, JsonGenerator jsonGen,
270           final String protocolNameSpace, final Set<String> alreadyDeclared, final Set<Schema> toDeclare)
271           throws IOException {
272     Schema.Type type = schema.getType();
273     writeSchemaAttributes(schema, appendable, jsonGen, true);
274     String namespace = schema.getNamespace();
275     if (!Objects.equals(namespace, protocolNameSpace)) {
276       appendable.append("@namespace(\"").append(namespace).append("\")\n");
277     }
278     Set<String> saliases = schema.getAliases();
279     if (!saliases.isEmpty()) {
280       appendable.append("@aliases(");
281       toJson(saliases, jsonGen);
282       jsonGen.flush();
283       appendable.append(")\n");
284     }
285     switch (type) {
286       case RECORD:
287         appendable.append("record ").append(schema.getName()).append(" {\n\n");
288         alreadyDeclared.add(schema.getFullName());
289         for (Field field : schema.getFields()) {
290           String fDoc = field.doc();
291           if (fDoc != null) {
292             appendable.append("  /** ").append(fDoc).append(" */\n");
293           }
294           appendable.append("  ");
295           writeFieldSchema(field.schema(), appendable, jsonGen, alreadyDeclared, toDeclare, schema.getNamespace());
296           appendable.append(' ');
297           Set<String> faliases = field.aliases();
298           if (!faliases.isEmpty()) {
299             appendable.append("@aliases(");
300             toJson(faliases, jsonGen);
301             jsonGen.flush();
302             appendable.append(") ");
303           }
304           Field.Order order = field.order();
305           if (order != null) {
306             appendable.append(" @order(\"").append(order.name()).append("\") ");
307           }
308           writeJsonProperties(field, appendable, jsonGen, false);
309           appendable.append(' ');
310           appendable.append(field.name());
311           JsonNode defaultValue = field.defaultValue();
312           if (defaultValue != null) {
313             appendable.append(" = ");
314             toJson(field.defaultVal(), jsonGen);
315             jsonGen.flush();
316           }
317           appendable.append(";\n\n");
318         }
319         appendable.append("}\n");
320         break;
321       case ARRAY:
322       case MAP:
323       case UNION:
324       case BOOLEAN:
325       case BYTES:
326       case DOUBLE:
327       case FLOAT:
328       case INT:
329       case LONG:
330       case NULL:
331       case STRING:
332         throw new UnsupportedOperationException("No IDL entity, nust be part of a record: " + schema);
333       case ENUM:
334         appendable.append("enum ").append(schema.getName()).append(" {");
335         alreadyDeclared.add(schema.getFullName());
336         Iterator<String> i = schema.getEnumSymbols().iterator();
337         if (i.hasNext()) {
338           appendable.append(i.next());
339           while (i.hasNext()) {
340             appendable.append(',');
341             appendable.append(i.next());
342           }
343         } else {
344           throw new IllegalStateException("Enum schema must have at least a symbol " + schema);
345         }
346         appendable.append("}\n");
347         break;
348       case FIXED:
349         appendable.append("fixed ").append(schema.getName()).append('(')
350                 .append(Integer.toString(schema.getFixedSize())).append(");\n");
351         alreadyDeclared.add(schema.getFullName());
352         break;
353       default:
354         throw new IllegalStateException("Unsupported schema " + schema);
355     }
356   }

358   private static void writeFieldSchema(final Schema schema,
359           final Appendable appendable, final JsonGenerator jsonGen,
360           final Set<String> alreadyDeclared, final Set<Schema> toDeclare,
361           final String recordNameSpace) throws IOException {
362     Schema.Type type = schema.getType();
363     switch (type) {
364       case RECORD:
365       case ENUM:
366       case FIXED:
367         if (Objects.equals(recordNameSpace, schema.getNamespace())) {
368           appendable.append(schema.getName());
369         } else {
370           appendable.append(schema.getFullName());
371         }
372         if (!alreadyDeclared.contains(schema.getFullName())) {
373           toDeclare.add(schema);
374         }
375         break;
376       case ARRAY:
377         writeSchemaAttributes(schema, appendable, jsonGen, false);
378         appendable.append("array<");
379         writeFieldSchema(schema.getElementType(), appendable, jsonGen, alreadyDeclared, toDeclare, recordNameSpace);
380         appendable.append('>');
381         break;
382       case MAP:
383         writeSchemaAttributes(schema, appendable, jsonGen, false);
384         appendable.append("map<");
385         writeFieldSchema(schema.getValueType(), appendable, jsonGen, alreadyDeclared, toDeclare, recordNameSpace);
386         appendable.append('>');
387         break;
388       case UNION:
389         writeSchemaAttributes(schema, appendable, jsonGen, false);
390         appendable.append("union {");
391         List<Schema> types = schema.getTypes();
392         Iterator<Schema> iterator = types.iterator();
393         if (iterator.hasNext()) {
394           writeFieldSchema(iterator.next(), appendable, jsonGen, alreadyDeclared, toDeclare, recordNameSpace);
395           while (iterator.hasNext()) {
396             appendable.append(',');
397             writeFieldSchema(iterator.next(), appendable, jsonGen, alreadyDeclared, toDeclare, recordNameSpace);
398           }
399         } else {
400           throw new IllegalStateException("Union schmemas must have member types " + schema);
401         }
402         appendable.append('}');
403         break;
404       case BOOLEAN:
405       case BYTES:
406       case DOUBLE:
407       case FLOAT:
408       case INT:
409       case LONG:
410       case NULL:
411       case STRING:
412         writeSchemaAttributes(schema, appendable, jsonGen, false);
413         appendable.append(schema.getName());
414         break;
415       default:
416         throw new IllegalStateException("Unsupported schema " + schema);
417     }
418   }

420   public static void writeSchemaAttributes(final Schema schema,
421           final Appendable appendable, final JsonGenerator jsonGen, final boolean crBetween)
422           throws IOException {
423     String doc = schema.getDoc();
424     if (doc != null) {
425       appendable.append("/** ").append(doc).append(" */");
426       if (crBetween) {
427         appendable.append('\n');
428       } else {
429         appendable.append(' ');
430       }
431     }
432     writeJsonProperties(schema, appendable, jsonGen, crBetween);
433   }

435   public static void writeJsonProperties(final JsonProperties props,
436           final Appendable appendable, final JsonGenerator jsonGen, final boolean crBetween) throws IOException {
437     Map<String, Object> objectProps = props.getObjectProps();
438     for (Map.Entry<String, Object> entry : objectProps.entrySet()) {
439       appendable.append('@').append(entry.getKey()).append('(');
440       toJson(entry.getValue(), jsonGen);
441       jsonGen.flush();
442       appendable.append(')');
443       if (crBetween) {
444         appendable.append('\n');
445       } else {
446         appendable.append(' ');
447       }
448     }
449   }

451   @SuppressWarnings(value="unchecked")
452   @SuppressFBWarnings("ITC_INHERITANCE_TYPE_CHECKING")
453   static void toJson(final Object datum, final JsonGenerator generator) throws IOException {
454     if (datum == JsonProperties.NULL_VALUE) { // null
455       generator.writeNull();
456     } else if (datum instanceof Map) { // record, map
457       generator.writeStartObject();
458       for (Map.Entry<Object,Object> entry : ((Map<Object,Object>) datum).entrySet()) {
459         generator.writeFieldName(entry.getKey().toString());
460         toJson(entry.getValue(), generator);
461       }
462       generator.writeEndObject();
463     } else if (datum instanceof Collection) { // array
464       generator.writeStartArray();
465       for (Object element : (Collection<?>) datum) {
466         toJson(element, generator);
467       }
468       generator.writeEndArray();
469     } else if (datum instanceof byte[]) { // bytes, fixed
470       generator.writeString(new String((byte[]) datum, StandardCharsets.ISO_8859_1));
471     } else if (datum instanceof CharSequence || datum instanceof Enum<?>) { // string, enum
472       generator.writeString(datum.toString());
473     } else if (datum instanceof Double) { // double
474       generator.writeNumber((Double) datum);
475     } else if (datum instanceof Float) { // float
476       generator.writeNumber((Float) datum);
477     } else if (datum instanceof Long) { // long
478       generator.writeNumber((Long) datum);
479     } else if (datum instanceof Integer) { // int
480       generator.writeNumber((Integer) datum);
481     } else if (datum instanceof Boolean) { // boolean
482       generator.writeBoolean((Boolean) datum);
483     } else {
484       throw new AvroRuntimeException("Unknown datum class: " + datum.getClass());
485     }
486   }
