16  package org.spf4j.maven.plugin.avro.avscp;
17  
18  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
19  import java.io.File;
20  import org.apache.avro.SchemaRefWriter;
21  import org.apache.maven.execution.MavenSession;
22  import org.apache.maven.plugin.AbstractMojo;
23  import org.apache.maven.plugin.MojoExecutionException;
24  import org.apache.maven.plugin.MojoFailureException;
25  import org.apache.maven.plugins.annotations.Component;
26  import org.apache.maven.plugins.annotations.Parameter;
27  import org.apache.maven.project.MavenProject;
28  import org.eclipse.aether.RepositorySystem;
29  import org.spf4j.maven.Registerer;
30  
31  /**
32   * @author Zoltan Farkas
33   */
34  @SuppressWarnings("checkstyle:VisibilityModifier")
35  @SuppressFBWarnings("AI_ANNOTATION_ISSUES_NEEDS_NULLABLE")
36  public abstract class SchemaMojoBase extends AbstractMojo {
37  
38  
39    @Parameter(defaultValue = "${project}", required = true, readonly = true)
40    protected MavenProject mavenProject;
41  
42  
43    /**
44     * The directory where all schema dependencies (avsc, avpr, avdl) are made vailable
45     */
46    @Parameter(name = "dependenciesDirectory",
47            defaultValue = "${project.build.directory}/schema-dependencies", readonly = true)
48    protected File dependenciesDirectory;
49  
50    /**
51     * The source directory of avro files. This directory is added to the classpath at schema compiling time. All files
52     * can therefore be referenced as classpath resources following the directory structure under the source directory.
53     */
54    @Parameter(name = "sourceDirectory", defaultValue = "${basedir}/src/main/avro")
55    protected File sourceDirectory;
56  
57  
58    /**
59     *  the destination for the java generated files.
60     */
61    @Parameter(name = "generatedJavaTarget",
62            defaultValue = "${project.build.directory}/generated-sources/avro")
63    protected File generatedJavaTarget;
64  
65  
66    /**
67     * the destination for the generated avro schema json files (will be published along with the java code).
68     */
69    @Parameter(name = "generatedAvscTarget",
70            defaultValue = "${project.build.directory}/generated-sources/avsc")
71    protected File generatedAvscTarget;
72  
73  
74    /**
75     *  the target folder.
76     */
77    @Parameter(name = "target", defaultValue = "${project.build.directory}")
78    protected File target;
79  
80    /**
81     * The current build mavenSession instance.
82     */
83    @Parameter(defaultValue = "${session}", required = true, readonly = true)
84    protected MavenSession mavenSession;
85  
86    /**
87     * This option will use schema references when writing schemas that depend of schemas from other projects,
88     * instead of baking them in.
89     * by default (false) all schema references will be inlined.
90     */
91    @Parameter(name = "useSchemaReferencesForAvsc",
92            defaultValue = "false")
93    protected boolean useSchemaReferencesForAvsc = false;
94  
95    /**
96     *  the schema artifact classifier.
97     */
98    @Parameter(name = "schemaArtifactClassifier", defaultValue = "avsc")
99    protected String schemaArtifactClassifier = "avsc";
100 
101   /**
102    *  the schema artifact extension.
103    */
104   @Parameter(name = "schemaArtifactExtension", defaultValue = "jar")
105   protected String schemaArtifactExtension = "jar";
106 
107 
108   /**
109    * The entry point to Aether, i.e. the component doing all the work.
110    */
111   @Component
112   protected  RepositorySystem repoSystem;
113 
114   public final RepositorySystem getRepoSystem() {
115     return repoSystem;
116   }
117 
118   public final MavenSession getMavenSession() {
119     return mavenSession;
120   }
121 
122   public final MavenProject getMavenProject() {
123     return mavenProject;
124   }
125 
126   public final File getGeneratedAvscTarget() {
127     return generatedAvscTarget;
128   }
129 
130   public final File getTarget() {
131     return target;
132   }
133 
134   /**
135    * Children must call this.
136    * @throws MojoExecutionException
137    * @throws MojoFailureException
138    */
139   @Override
140   public void execute() throws MojoExecutionException, MojoFailureException {
141     if (useSchemaReferencesForAvsc && SchemaRefWriter.isSchemaRefsSupported()) {
142       Registerer.register(repoSystem, getMavenSession().getRepositorySession(),
143               mavenProject.getRemoteProjectRepositories(),
144               schemaArtifactClassifier, schemaArtifactExtension);
145     }
146   }
147 
148 
149 
150 
151   /**
152    * will be overwritten as needed, and override will include this result.
153    */
154   @Override
155   public String toString() {
156     return "SchemaMojoBase{" + "dependenciesDirectory=" + dependenciesDirectory
157             + ", sourceDirectory=" + sourceDirectory + ", generatedJavaTarget="
158             + generatedJavaTarget + ", generatedAvscTarget=" + generatedAvscTarget + ", target=" + target + '}';
159   }
