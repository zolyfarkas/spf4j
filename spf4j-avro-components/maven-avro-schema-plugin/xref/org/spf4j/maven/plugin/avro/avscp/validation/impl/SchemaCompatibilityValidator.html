57  /**
58   * Validates previously released schemas for backward compatibility.
59   *
60   * <p>The following configurations are available for this "compatibility" validator:</p>
61   * <ul>
62   * <li>versionRange - maven version range to check compatibility against. (defaults to
63   * "[," + mavenProject.getVersion() +  ')' )</li>
64   *
65   * <li>maxNrOfVersionsToCheckForCompatibility = max number oof versions to check against.
66   * (defaults to 30)</li>
67   *
68   * <li>maxNrOfDaysBackCheckForCompatibility - max released time to check against (defaults to 1 year)</li>
69   *
70   * <li>deprecationRemoval - validates  that a removed schema
71   * has been previously deprecated (false by default)</li>
72   *
73   * <li>schemaArtifactClassifier - the classifier of the artifact that contains the schema files.
74   * defaults to null (the main jar)</li>
75   *
76   * <li>schemaArtifactExtension - the extension of the artifact that contains the schema files.
77   * defaults to jar </li>
78   *
79   * </ul>
80   *
81   * <p>The following validations are performed:</p>
82   *
83   * <ul>
84   * <li>1) We check previously released schema for (reader to writer) and (writer to reader) compatibility
85   * with the current schema. Unless compatibility is not desired ( via "beta" schema annotation,
86   * or only partial compatibility is desired with  "noNewToOldCompatibility" "noOldToNewCompatibility"
87   * schema annotations. (objects written with old schema don't need to be converted to new objects)
88   * or "noNewToOldCompatibility" if (objects written with new schema don't need to be converted to old schema objects)
89   * </li>
90   *
91   * <li>
92   * 2) Schema deprecation and removal policy can also enabled with compatibiliy.deprecationRemoval = true,
93   * this will validate that a schema has been deprecated during the entire compatibility interval.
94   * </li>
95   * </ul>
96   * @author Zoltan Farkas
97   */
98  public final class SchemaCompatibilityValidator implements Validator<Void> {
99  
100   @Override
101   public String getName() {
102     return "compatibility";
103   }

105   @Override
106   @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
107   public Result validate(final Void nv, final ValidatorMojo mojo) throws IOException {
108     // loop through dependencies.
109     MavenProject mavenProject = mojo.getMavenProject();
110     Map<String, String> validatorConfigs = mojo.getValidatorConfigs();
111     Log log = mojo.getLog();
112     String versionRange = validatorConfigs.get("versionRange");
113     if (versionRange == null) {
114       versionRange = "[," + mavenProject.getVersion() +  ')';
115     }
116     int maxNrVersToCheck = 30;
117     String strNrVer = validatorConfigs.get("maxNrOfVersionsToCheckForCompatibility");
118     if (strNrVer != null) {
119       maxNrVersToCheck = Integer.parseInt(strNrVer);
120     }
121     Instant instantToGoBack;
122     String strNrDays = validatorConfigs.get("maxNrOfDaysBackCheckForCompatibility");
123     if (strNrDays == null) {
124       instantToGoBack = Instant.now().atOffset(ZoneOffset.UTC).minus(1,  ChronoUnit.YEARS).toInstant();
125     } else {
126       instantToGoBack = Instant.now().atOffset(ZoneOffset.UTC)
127               .minus(Integer.parseInt(strNrDays), ChronoUnit.DAYS).toInstant();
128     }
129     String groupId = mavenProject.getGroupId();
130     String artifactId = mavenProject.getArtifactId();
131     List<RemoteRepository> remoteProjectRepositories = mavenProject.getRemoteProjectRepositories();
132     RepositorySystem repoSystem = mojo.getRepoSystem();
133     RepositorySystemSession repositorySession = mojo.getMavenSession().getRepositorySession();
134     List<Version> rangeVersions;
135     try {
136       rangeVersions = MavenRepositoryUtils.getVersions(groupId, artifactId, versionRange,
137               remoteProjectRepositories, repoSystem, repositorySession);
138     } catch (VersionRangeResolutionException ex) {
139       throw new RuntimeException("Invalid compatibiliy.versionRange = " + versionRange + " setting", ex);
140     }
141     rangeVersions = rangeVersions.stream().filter((v) -> !v.toString().endsWith("SNAPSHOT"))
142             .collect(Collectors.toList());
143     int tSize = rangeVersions.size();
144     rangeVersions = rangeVersions.subList(Math.max(tSize - maxNrVersToCheck, 0), tSize);
145     log.info("Validating compatibility with previous versions " + rangeVersions + " newer than " + instantToGoBack);
146     if (rangeVersions.isEmpty()) {
147       return Result.valid();
148     }
149     String schemaArtifactClassifier = validatorConfigs.get("schemaArtifactClassifier");
150     if (schemaArtifactClassifier != null && schemaArtifactClassifier.trim().isEmpty()) {
151       schemaArtifactClassifier = null;
152     }
153     String schemaArtifactExtension = validatorConfigs.get("schemaArtifactExtension");
154     if (schemaArtifactExtension == null || schemaArtifactExtension.trim().isEmpty()) {
155       schemaArtifactExtension = "jar";
156     }
157     List<String> issues = new ArrayList<>(4);
158     int size = rangeVersions.size();
159     for (int  i = size - 1; i >= 0; i--) {
160       Version version  = rangeVersions.get(i);
161       validateCompatibility(groupId, artifactId, schemaArtifactClassifier, schemaArtifactExtension, version,
162               remoteProjectRepositories, repoSystem, repositorySession, mojo,
163               Boolean.parseBoolean(validatorConfigs.getOrDefault("deprecationRemoval", "false")),
164               instantToGoBack, issues::add);
165     }
166     if (issues.isEmpty()) {
167       return Result.valid();
168     } else {
169       return Result.failed("Schema compatibility issues:\n" + String.join("\n", issues));
170     }
171   }

173   @SuppressFBWarnings("PCAIL_POSSIBLE_CONSTANT_ALLOCATION_IN_LOOP") // Schema.Parser is mutable, false positive.
174   @SuppressWarnings("checkstyle:ParameterNumber") // valid will address later.
175   public void validateCompatibility(final String groupId, final String artifactId,
176           @Nullable final String classifier, final String extension,
177           final Version version,
178           final List<RemoteRepository> remoteProjectRepositories, final RepositorySystem repoSystem,
179           final RepositorySystemSession repositorySession,
180           final ValidatorMojo mojo,
181           final boolean deprecationRemoval, final Instant instantToGoBack, final Consumer<String> issues)
182           throws IOException {
183     Log log = mojo.getLog();
184     log.info("Validating compatibility with version: " + version + ", " + groupId + ':' + artifactId
185             + ':' + classifier + ':' + extension);
186     Path targetPath = mojo.getTarget().toPath();
187     Path currSchemasPath = mojo.getGeneratedAvscTarget().toPath();
188     File prevSchemaArchive;
189     try {
190       prevSchemaArchive = MavenRepositoryUtils.resolveArtifact(
191               groupId, artifactId, classifier, extension, version.toString(),
192               remoteProjectRepositories, repoSystem, repositorySession);
193     } catch (ArtifactResolutionException ex) {
194       throw new RuntimeException("Cannot resolve previous version "  + version, ex);
195     }
196     Path dest = targetPath.resolve("prevSchemas").resolve(version.toString());
197     Files.createDirectories(dest);
198     log.debug("Unzipping " + prevSchemaArchive + " to " + dest);
199     List<Path> prevSchemas = Compress.unzip2(prevSchemaArchive.toPath(), dest, (Path p) -> {
200       Path fileName = p.getFileName();
201       if (fileName == null) {
202         return false;
203       }
204       String fname = fileName.toString();
205       return (fname.endsWith("avsc")
206               || SchemaCompileMojo.SCHEMA_MANIFEST.equals(fname)
207               || "MANIFEST.MF".equals(fname));
208     });
209     Instant dependencyBuildTime = getDependencyBuildTime(dest, log);
210     if (dependencyBuildTime == null) {
211       log.info("Package " + dest + " build time missing from manifest");
212     } else if (dependencyBuildTime.isBefore(instantToGoBack)) {
213       return;
214     }
215     for (Path prevSchemaPath : prevSchemas) {
216       Path fileName = prevSchemaPath.getFileName();
217       if (fileName == null) {
218         continue;
219       }
220       if (!fileName.toString().endsWith("avsc")) {
221         continue;
222       }
223       Path relPath = dest.relativize(prevSchemaPath);
224       Path newSchemaPath = currSchemasPath.resolve(relPath);
225       Schema previousSchema = new Schema.Parser().parse(prevSchemaPath.toFile());
226       String previousSchemaName = previousSchema.getFullName();
227       if (previousSchema.getProp("beta") != null) {
228         log.debug("Skipping beta schema " + previousSchemaName);
229         continue;
230       }
231       log.debug("Validating compatibility for " + previousSchemaName + " "
232               + prevSchemaPath + " -> "  + newSchemaPath);
233       if (!Files.exists(newSchemaPath)) {
234         if (deprecationRemoval && previousSchema.getProp("deprecated") == null) {
235           issues.accept(previousSchemaName + " is being removed without being deprecated first");
236         }
237       } else {
238         Schema newSchema = new Schema.Parser().parse(newSchemaPath.toFile());
239         if (newSchema.getProp("beta") != null) {
240           log.debug("Skipping beta schema " + newSchema.getFullName());
241         } else {
242           if (newSchema.getProp("noOldToNewCompatibility") == null) {
243             SchemaCompatibility.SchemaPairCompatibility o2n
244                     = SchemaCompatibility.checkReaderWriterCompatibility(newSchema, previousSchema);
245             if (o2n.getType() == SchemaCompatibility.SchemaCompatibilityType.INCOMPATIBLE) {
246               issues.accept(newSchema.getFullName() + " cannot convert from previous version " + version
247                     + ", diff: \n" + diff(previousSchema, newSchema));
248             }
249           }
250           if (newSchema.getProp("noNewToOldCompatibility") == null) {
251             SchemaCompatibility.SchemaPairCompatibility n2o
252                     = SchemaCompatibility.checkReaderWriterCompatibility(previousSchema, newSchema);
253             if (n2o.getType() == SchemaCompatibility.SchemaCompatibilityType.INCOMPATIBLE) {
254               issues.accept(newSchema.getFullName() + " cannot convert current to previos version " + version
255                     + ", diff: \n" + diff(previousSchema, newSchema));
256             }
257           }
258         }
259       }
260     }
261   }

263   @Nullable
264   private static Instant getDependencyBuildTime(final Path location, final Log log) throws IOException {
265     Path jarManifest = location.resolve("META-INF/MANIFEST.MF");
266     if (Files.exists(jarManifest)) {
267       try (BufferedInputStream bis = new BufferedInputStream(
268               Files.newInputStream(jarManifest))) {
269         Manifest manifest = new Manifest(bis);
270         Attributes mainAttributes = manifest.getMainAttributes();
271         String buildTime = mainAttributes.getValue("Build-Time");
272         if (buildTime != null) {
273           try {
274             return Instant.parse(buildTime);
275           } catch (DateTimeParseException ex) {
276             log.warn("Cannot parse manifest build time " + buildTime, ex);
277           }
278         }
279       }
280     }
281     Path codegenManifest = location.resolve(SchemaCompileMojo.SCHEMA_MANIFEST);
282     if (Files.exists(codegenManifest)) {
283       try (BufferedReader br = Files.newBufferedReader(codegenManifest, StandardCharsets.UTF_8)) {
284         Properties props = new Properties();
285         props.load(br);
286         String buildTime = props.getProperty("Build-Time");
287         if (buildTime != null) {
288           try {
289             return Instant.parse(buildTime);
290           } catch (DateTimeParseException ex) {
291             log.warn("Cannot parse manifest build time " + buildTime, ex);
292           }
293         }
294       }
295     }
296     return null;
297   }

299   static String diff(final Schema reader, final Schema writer) {
300     String s1 = reader.toString(true);
301     String s2 = writer.toString(true);
302     DiffMatchPatch dmp = new DiffMatchPatch();
303     LinkedList<DiffMatchPatch.Diff> diffs = dmp.diffMain(s1, s2, false);
304     dmp.diffCleanupSemantic(diffs);
305     StringBuilder result = new StringBuilder();
306     for (DiffMatchPatch.Diff aDiff : diffs) {
307       String text = aDiff.text;
308       switch (aDiff.operation) {
309         case INSERT:
310           result.append("+<<<").append(text).append(">>>");
311           break;
312         case DELETE:
313           result.append("-<<<").append(text).append(">>>");
314           break;
315         case EQUAL:
316           result.append(text);
317           break;
318         default:
319           throw new UnsupportedOperationException("Not supported " + aDiff.operation);
320       }
321     }
322     return result.toString();
323   }
