package org.spf4j.base;

import java.io.IOException;
import java.io.UncheckedIOException;
import javax.annotation.ParametersAreNonnullByDefault;

/**
 * Encoder that encodes an arbitrary string to an ID.
 *
 * @author Zoltan Farkas
 */
@ParametersAreNonnullByDefault
public final class LangIdEncDec {

  private LangIdEncDec() {
  }

  public static void lossyEncode(final CharSequence what, final Appendable to) throws IOException {
    int length = what.length();
    if (length == 0) {
      return;
    }
    char first = what.charAt(0);
    if (Character.isLetter(first) || first == '_') {
      to.append(first);
    } else {
      to.append('_');
    }
    for (int i = 1; i < length; i++) {
      char c = what.charAt(i);
      if (Character.isLetterOrDigit(c) || c == '_') {
        to.append(c);
      } else {
        to.append('_');
      }

    }
  }

  public static String lossyEncode(final String what) {
    if (CharSequences.isValidJavaId(what)) {
      return what;
    } else {
      int length = what.length();
      if (length == 0) {
        throw new IllegalArgumentException("Invalid id: '" + what + '\'');
      }
      StringBuilder sb = new StringBuilder(length);
      try {
        lossyEncode(what, sb);
      } catch (IOException ex) {
        throw new UncheckedIOException(ex);
      }
      return sb.toString();
    }
  }

}
