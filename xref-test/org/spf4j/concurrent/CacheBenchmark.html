2   package org.spf4j.concurrent;
3   
4   import com.google.common.cache.CacheBuilder;
5   import com.google.common.cache.CacheLoader;
6   import com.google.common.cache.LoadingCache;
7   import org.openjdk.jmh.annotations.Benchmark;
8   import org.openjdk.jmh.annotations.Fork;
9   import org.openjdk.jmh.annotations.Scope;
10  import org.openjdk.jmh.annotations.State;
11  import org.openjdk.jmh.annotations.Threads;
12  import org.spf4j.base.IntMath;
13  
14  @State(Scope.Benchmark)
15  @Fork(2)
16  @Threads(value = 8)
17  public class CacheBenchmark {
18  
19  
20      private static final CacheLoader<String, String> TEST_LOADER
21              = new CacheLoader<String, String>() {
22  
23          @Override
24          public String load(String key) throws Exception {
25              return "TEST";
26          }
27      };
28  
29      private static final LoadingCache<String, String> GUAVA = CacheBuilder.newBuilder()
30              .initialCapacity(16)
31              .concurrencyLevel(16)
32              .build(TEST_LOADER);
33      private static final LoadingCache<String, String> SPF4J = new UnboundedLoadingCache<>(16, 16, TEST_LOADER);
34  
35      private static final LoadingCache<String, String> SPF4J_RACY =
36              new UnboundedRacyLoadingCache<>(16, 16, TEST_LOADER);
37  
38  
39      private static final IntMath.XorShift32 RND = new IntMath.XorShift32();
40  
41      @Benchmark
42      public final String spf4jCache() {
43          return SPF4J.getUnchecked("key" + (RND.nextInt() % 100));
44      }
45  
46      @Benchmark
47      public final String spf4jRacyCache() {
48          return SPF4J_RACY.getUnchecked("key" + (RND.nextInt() % 100));
49      }
50  
51      @Benchmark
52      public final String guavaCache() {
53          return GUAVA.getUnchecked("key" + (RND.nextInt() % 100));
54      }
55  
56  
57  }
