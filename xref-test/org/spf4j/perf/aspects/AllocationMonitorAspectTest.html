30  /**
31   *
32   * @author zoly
33   */
34  public final class AllocationMonitorAspectTest {
35  
36      public AllocationMonitorAspectTest() {
37      }
38      private final long startTime = System.currentTimeMillis();
39  
40      private static void testAllocInStaticContext() throws InterruptedException {
41          for (int i = 0; i < 1000; i++) {
42              System.err.println("S" + i + Strings.repeat("A", i % 2 * 10));
43              if (i % 100 == 0) {
44                  Thread.sleep(100);
45              }
46          }
47      }
48  
49      /**
50       * Test of afterAllocation method, of class AllocationMonitorAspect.
51       */
52      @Test
53      public void testAfterAllocation() throws InterruptedException, IOException {
54          System.setProperty("perf.memory.sampleAggMillis", "1000");
55          System.setProperty("perf.io.openFiles.sampleAggMillis", "1000");
56          System.setProperty("perf.allocations.sampleTimeMillis", "1000");
57          MemoryUsageSampler.startMemoryUsageSampling(500);
58          OpenFilesSampler.startFileUsageSampling(500, 512, 1000);
59          for (int i = 0; i < 1000; i++) {
60              System.err.println("T" + i);
61              if (i % 100 == 0) {
62                  Thread.sleep(500);
63              }
64          }
65          testAllocInStaticContext();
66          TestClass.testAllocInStaticContext();
67          System.out.println(((TSDBMeasurementStore) RecorderFactory.MEASUREMENT_STORE).generateCharts(startTime,
68                  System.currentTimeMillis(), 1200, 600));
69      }
70  }
