25      private static final String TSTR =
26              "This is a super \u00EF cool, mega dupper test string for testing piping..........E";
27  
28      @Test
29      public void testSimpleStreamBuffering() throws IOException {
30          final byte[] array = Strings.toUtf8(TSTR);
31          ByteArrayInputStream bis = new ByteArrayInputStream(array);
32          MemorizingBufferedInputStream mis = new MemorizingBufferedInputStream(bis);
33          int val;
34          int i = 0;
35          while ((val = mis.read()) > 0)  {
36              System.out.print((char) val);
37              Assert.assertEquals(array[i], (byte) val);
38              i++;
39          }
40          Assert.assertEquals(i, array.length);
41      }

61      @Test
62      public void testStreamBuffering() throws IOException {
63          test(TSTR, 8, true);
64          final IntMath.XorShift32 random = new IntMath.XorShift32();
65          for (int i = 0; i < 100; i++) {
66              int nrChars = Math.abs(random.nextInt() % 100000);
67              StringBuilder sb = generateTestStr(nrChars);
68              test(sb.toString(), Math.abs(random.nextInt() % 10000), true);
69              testBuff(sb, 8192);
70              testBuff(sb, 32);
71          }
72      }
73  
74      private void testBuff(final StringBuilder sb, final int buffSize) throws IOException {
75          final byte[] utf8Bytes = Strings.toUtf8(sb.toString());
76          ByteArrayInputStream bis = new ByteArrayInputStream(utf8Bytes);
77          try (MemorizingBufferedInputStream mbis = new MemorizingBufferedInputStream(bis, buffSize, buffSize / 2,
78                  ArraySuppliers.Bytes.GL_SUPPLIER, Charsets.UTF_8)) {
79              int val = mbis.read();
80              Assert.assertEquals(val, mbis.getReadBytesFromBuffer()[0]);
81              byte [] buff = new byte [8];
82              int read = mbis.read(buff);
83              Assert.assertEquals(8, read);
84              Assert.assertTrue(Arrays.equals(buff, Arrays.copyOfRange(mbis.getReadBytesFromBuffer(), 1, 9)));
85              int result;
86              do {
87                  result = mbis.read(buff);
88              } while (result >= 0);
89              byte[] unredBytesEnd = mbis.getUnreadBytesFromBuffer();
90              Assert.assertEquals(0, unredBytesEnd.length);
91              final byte[] readBytesFromBuffer = mbis.getReadBytesFromBuffer();
92              Assert.assertTrue(Arrays.equals(readBytesFromBuffer,
93                      Arrays.copyOfRange(utf8Bytes, utf8Bytes.length - readBytesFromBuffer.length, utf8Bytes.length)));
94  
95          }
96      }
