25      @Test
26      public void testRw() throws IOException {
27          DataBlock data = DataBlock.newBuilder()
28                  .setBaseTimestamp(0).setValues(Collections.EMPTY_LIST).build();
29          try (ByteArrayBuilder bab = new ByteArrayBuilder()) {
30              SpecificDatumWriter<DataBlock> writer = new SpecificDatumWriter<>(data.getSchema());
31              final BinaryEncoder directBinaryEncoder = EncoderFactory.get().directBinaryEncoder(bab, null);
32              writer.write(data, directBinaryEncoder);
33              directBinaryEncoder.flush();
34              ByteArrayInputStream bis = new ByteArrayInputStream(bab.getBuffer(), 0, bab.size());
35              SpecificDatumReader<DataBlock> reader = new SpecificDatumReader<>(data.getSchema());
36              BinaryDecoder directBinaryDecoder = DecoderFactory.get().directBinaryDecoder(bis, null);
37              DataBlock read = reader.read(null, directBinaryDecoder);
38              Assert.assertEquals(read, data);
39          }
40      }
