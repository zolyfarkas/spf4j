22    @Test
23    @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
24    public void testGenericRecordCreation() throws IOException, Exception {
25      String schemaStr = Resources.toString(Resources.getResource("SchemaBuilder.avsc"), Charsets.US_ASCII);
26      Schema schema = new Schema.Parser().parse(schemaStr);
27      try (GenericRecordBuilder builder = new GenericRecordBuilder(schema)) {
28        Class<? extends SpecificRecordBase> clasz = builder.getClass(schema);
29        GenericRecord record = clasz.newInstance();
30        record.put("requiredBoolean", Boolean.TRUE);
31        Schema fieldSchema = schema.getField("optionalRecord").schema().getTypes().get(1);
32        Class<? extends SpecificRecordBase> nestedRecordClass = builder.getClass(fieldSchema);
33        SpecificRecordBase nr = nestedRecordClass.newInstance();
34        nr.put("nestedRequiredBoolean", Boolean.TRUE);
35        record.put("optionalRecord", nr);
36        Assert.assertEquals(nr, record.get("optionalRecord"));
37        System.out.println(record);
38      }
39    }
40  
41    @Test
42    @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
43    public void testMoreGenericRecordCreation() throws InstantiationException, IllegalAccessException {
44      Schema reuse = SchemaBuilder.builder().record("Reuse").fields().requiredString("field").endRecord();
45      Schema record = SchemaBuilder.builder().record("TestRecord").fields()
46              .requiredInt("number")
47              .name("record").type(reuse).noDefault()
48              .endRecord();
49      Schema record2 = SchemaBuilder.builder().record("TestRecord2").fields()
50              .requiredBoolean("isThere")
51              .name("record").type(reuse).noDefault()
52              .endRecord();
53      try (GenericRecordBuilder builder = new GenericRecordBuilder(record, record2)) {
54        Class<? extends SpecificRecordBase> clasz = builder.getClass(record);
55        Class<? extends SpecificRecordBase> clasz2 = builder.getClass(record);
56        Assert.assertSame(clasz, clasz2);
57        SpecificRecordBase myRecord = clasz.newInstance();
58        myRecord.put("number", 35);
59        Assert.assertEquals(35, (int) myRecord.get("number"));
60        Class<? extends SpecificRecordBase> aClass = builder.getClass(record2);
61        GenericRecord r2 = aClass.newInstance();
62        System.out.println(r2);
63      }
64    }
