12      public void testRandom() {
13          IntMath.XorShift32 random = new IntMath.XorShift32();
14          int sum = 0;
15          for (int i = 1; i < 100; i++) {
16              sum += Math.abs(random.nextInt()) % 100;
17          }
18          sum /= 100;
19          Assert.assertTrue(sum > 15);
20  
21          IntMath.XorShift128 random2 = new IntMath.XorShift128();
22          sum = 0;
23          for (int i = 1; i < 100; i++) {
24              sum += Math.abs(random2.nextInt()) % 100;
25          }
26          sum /= 100;
27          Assert.assertTrue(sum > 15);
28      }
