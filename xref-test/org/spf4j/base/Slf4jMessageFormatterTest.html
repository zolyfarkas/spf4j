39      @Test
40      public void testFormatter() throws IOException {
41          StringBuilder sb = new StringBuilder();
42          Slf4jMessageFormatter.format(sb, "bla bla");
43          Assert.assertEquals("bla bla", sb.toString());
44          sb.setLength(0);
45          Slf4jMessageFormatter.format(sb, "bla bla {}", "coco");
46          Assert.assertEquals("bla bla coco", sb.toString());
47          sb.setLength(0);
48          Slf4jMessageFormatter.format(sb, "\\{}bla bla {}", "coco");
49          Assert.assertEquals("{}bla bla coco", sb.toString());
50          sb.setLength(0);
51          int processed = Slf4jMessageFormatter.format(sb, "Some Message", "coco");
52          Assert.assertEquals("Some Message", sb.toString());
53          Assert.assertEquals(0, processed);
54      }

56      @Test
57      public void testFormatter2() throws IOException {
58          ConfigurableAppenderSupplier appSupp = new ConfigurableAppenderSupplier();
59          System.out.println(appSupp);
60          StringBuilder sb = new StringBuilder();
61          final long currentTimeMillis = System.currentTimeMillis();
62          Slf4jMessageFormatter.format(sb, "bla bla {}", appSupp, new java.sql.Date(currentTimeMillis));
63          Assert.assertEquals("bla bla " + LocalDateAppender.FMT.print(currentTimeMillis), sb.toString());
64          sb.setLength(0);
65          AMethod method = AMethod.newBuilder().setName("m1").setDeclaringClass("c1").build();
66          int written = Slf4jMessageFormatter.format(sb, "bla bla {}", appSupp, method);
67          Assert.assertEquals("bla bla {\"declaringClass\":\"c1\",\"name\":\"m1\"}", sb.toString());
68          Assert.assertEquals(1, written);
69          sb.setLength(0);
70          written = Slf4jMessageFormatter.format(1, sb, "bla bla {}", appSupp,  "ifff", method);
71          Assert.assertEquals("bla bla {\"declaringClass\":\"c1\",\"name\":\"m1\"}", sb.toString());
72          Assert.assertEquals(2, written);
73          sb.setLength(0);
74          written = Slf4jMessageFormatter.format(sb, "bla bla {}", appSupp, method, "yohooo");
75          System.out.println("formatted message: " + sb.toString());
76          Assert.assertEquals(1, written);
77          sb.setLength(0);
78          written = Slf4jMessageFormatter.format(sb, "bla bla {}", appSupp);
79          System.out.println("formatted message: " + sb.toString());
80          Assert.assertEquals(0, written);
81          sb.setLength(0);
82          EscapeJsonStringAppendableWrapper escaper = new EscapeJsonStringAppendableWrapper(sb);
83          Slf4jMessageFormatter.format(escaper, "bla bla {} {}", appSupp, "\n\u2013\u0010",
84                  new int [] {1, 2, 3});
85          System.out.println("formatted message: " + sb.toString());
86          Assert.assertEquals("bla bla \\nâ€“\\u0010 [1, 2, 3]", sb.toString());
87          appSupp.replace(String.class, new Function<ObjectAppender<? super String>, ObjectAppender<? super String>>() {
88            @Override
89            public ObjectAppender<? super String> apply(final ObjectAppender<? super String> input) {
90              return new ObjectAppender<String>() {
91                @Override
92                public void append(final String object, final Appendable appendTo) throws IOException {
93                  try (AppendableLimiterWithOverflow limiter =
94                          new AppendableLimiterWithOverflow(90, File.createTempFile("string", ".overflow"),
95                                  "...@", Charsets.UTF_8, appendTo)) {
96                    input.append(object, limiter);
97                  }
98                }
99              };
100           }
101         });
102         sb.setLength(0);
103         Slf4jMessageFormatter.format(sb, "bla bla {}", appSupp,
104          "01234567890123456789012345678901234567890123456789012345678901234567890123456789012345678901234567890123456789");
105         System.out.println(sb.toString());
106 
107 
108     }
