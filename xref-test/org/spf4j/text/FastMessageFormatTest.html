17    @Test
18    public void testFormatter() throws IOException {
19      StringBuilder sb = new StringBuilder();
20      MessageFormat format = new MessageFormat("{0}, {1}");
21      format.format(new Object [] {"a", "b"}, sb, null);
22      Assert.assertEquals("a, b", sb.toString());
23    }
24    
25    @Test
26    public void testFormatter5() throws IOException {
27      StringBuilder sb = new StringBuilder();
28      MessageFormat format = new MessageFormat("{0}, {1}, {2}, {3}, {4}");
29      format.format(new Object [] {"a", "b", "c", "d", "e"}, sb, null);
30      Assert.assertEquals("a, b, c, d, e", sb.toString());
31    }
