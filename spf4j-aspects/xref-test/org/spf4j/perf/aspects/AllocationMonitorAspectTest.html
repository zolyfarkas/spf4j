40  public final class AllocationMonitorAspectTest {
41  
42      public AllocationMonitorAspectTest() {
43      }
44      private final long startTime = System.currentTimeMillis();
45  
46      private static void testAllocInStaticContext() throws InterruptedException {
47          for (int i = 0; i < 1000; i++) {
48              System.err.println("S" + i + Strings.repeat("A", i % 2 * 10));
49              if (i % 100 == 0) {
50                  Thread.sleep(100);
51              }
52          }
53      }
54  
55      /**
56       * Test of afterAllocation method, of class AllocationMonitorAspect.
57       */
58      @Test
59      public void testAfterAllocation() throws InterruptedException, IOException {
60          System.setProperty("perf.memory.sampleAggMillis", "1000");
61          System.setProperty("perf.io.openFiles.sampleAggMillis", "1000");
62          System.setProperty("perf.allocations.sampleTimeMillis", "1000");
63          MemoryUsageSampler.start(500);
64          OpenFilesSampler.start(500, 512, 1000, false);
65          for (int i = 0; i < 1000; i++) {
66              System.err.println("T" + i);
67              if (i % 100 == 0) {
68                  Thread.sleep(500);
69              }
70          }
71          testAllocInStaticContext();
72          TestClass.testAllocInStaticContext();
73          final TSDBWriter dbWriter = ((TSDBMeasurementStore) RecorderFactory.MEASUREMENT_STORE).getDBWriter();
74          dbWriter.flush();
75          File file = dbWriter.getFile();
76          List<TableDef> tableDef = TSDBQuery.getTableDef(file, "heap-used");
77          Assert.assertTrue(!tableDef.isEmpty());
78      }
