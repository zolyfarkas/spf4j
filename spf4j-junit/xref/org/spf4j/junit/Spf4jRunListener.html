3   import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
4   import java.io.File;
5   import java.io.IOException;
6   import javax.annotation.concurrent.NotThreadSafe;
7   import org.junit.runner.Description;
8   import org.junit.runner.Result;
9   import org.junit.runner.notification.Failure;
10  import org.junit.runner.notification.RunListener;
11  import org.spf4j.stackmonitor.FastStackCollector;
12  import org.spf4j.stackmonitor.Sampler;
13  
14  /**
15   * A Junit run listener that will collect performance profiles for your unit tests.
16   *
17   * Works only if unit tests are NOT executed in parallel inside the same JVM.
18   *
19   * To enable this Run listener with maven, you will need to configure your maven-surefire-plugin with:
20   *
21   *       <plugin>
22   *       <groupId>org.apache.maven.plugins</groupId>
23   *       <artifactId>maven-surefire-plugin</artifactId>
24   *       <configuration>
25   *         <properties>
26   *           <property>
27   *             <name>listener</name>
28   *             <value>org.spf4j.junit.Spf4jRunListener</value>
29   *           </property>
30   *           <property>
31   *             <name>spf4j.junit.sampleTimeMillis</name>
32   *             <value>5</value>
33   *           </property>
34   *         </properties>
35   *       </configuration>
36   *     </plugin>
37   *
38   * and add spf4j-junit to your test classpath:
39   *
40   *   <dependency>
41   *     <groupId>org.spf4j</groupId>
42   *     <artifactId>spf4j-junit</artifactId>
43   *     <version>${spf4j.version}</version>
44   *     <scope>test</scope>
45   *   </dependency>
46   *
47   * @author zoly
48   */
49  @SuppressFBWarnings("PATH_TRAVERSAL_IN")
50  @NotThreadSafe
51  public final class Spf4jRunListener extends RunListener {
52  
53    private final Sampler sampler;
54  
55    private final File destinationFolder;
56  
57    private volatile File lastWrittenFile;
58  
59    public Spf4jRunListener() {
60      sampler = new Sampler(Integer.getInteger("spf4j.junit.sampleTimeMillis", 5),
61            Integer.getInteger("spf4j.junit.dumpAfterMillis", Integer.MAX_VALUE),
62            new FastStackCollector(true));
63  
64      destinationFolder = new File(System.getProperty("spf4j.junit.destinationFolder",
65            "target/junit-ssdump"));
66  
67      if (!destinationFolder.mkdirs() && !destinationFolder.canWrite()) {
68        throw new ExceptionInInitializerError("Unable to write to " + destinationFolder);
69      }
70    }
71  
72    @Override
73    public void testFailure(final Failure failure)
74            throws IOException {
75      File dumpToFile = sampler.dumpToFile(new File(destinationFolder, failure.getTestHeader() + ".ssdump2"));
76      if (dumpToFile != null) {
77        System.out.print("Profile saved to " + dumpToFile);
78      }
79    }
80  
81    @Override
82    public void testFinished(final Description description)
83            throws IOException {
84      File dump = sampler.dumpToFile(new File(destinationFolder, description.getDisplayName() + ".ssdump2"));
85      if (dump != null) {
86        System.out.print("Profile saved to " + dump);
87        lastWrittenFile = dump;
88      }
89    }
90  
91    @Override
92    public void testRunFinished(final Result result) throws InterruptedException {
93      sampler.stop();
94    }
95  
96    @Override
97    public void testRunStarted(final Description description)  {
98     sampler.start();
99    }
100 
101   public Sampler getSampler() {
102     return sampler;
103   }
104 
105   public File getDestinationFolder() {
106     return destinationFolder;
107   }
108 
109   public File getLastWrittenFile() {
110     return lastWrittenFile;
111   }
112 
113 
114   @Override
115   public String toString() {
116     return "Spf4jRunListener{" + "sampler=" + sampler + '}';
117   }
118 
119 }
