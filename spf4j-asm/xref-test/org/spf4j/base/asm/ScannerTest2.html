18  public class ScannerTest2 {
19  
20      public static final String DEFAULT_SS_DUMP_FILE_NAME_PREFIX =
21              System.getProperty("spf4j.perf.ms.defaultSsdumpFilePrefix", ManagementFactory.getRuntimeMXBean().getName());
22  
23    @Test
24    public void testScan() throws NoSuchMethodException, IOException {
25  
26      final ImmutableSet<Method> lookFor = ImmutableSet.of(System.class.getDeclaredMethod("getProperty", String.class),
27              System.class.getDeclaredMethod("getProperty", String.class, String.class),
28              Integer.class.getDeclaredMethod("getInteger", String.class),
29              Integer.class.getDeclaredMethod("getInteger", String.class, int.class),
30              Integer.class.getDeclaredMethod("getInteger", String.class, Integer.class),
31              Long.class.getDeclaredMethod("getLong", String.class),
32              Long.class.getDeclaredMethod("getLong", String.class, Long.class),
33              Long.class.getDeclaredMethod("getLong", String.class, long.class),
34              Boolean.class.getDeclaredMethod("getBoolean", String.class));
35      List<Invocation> findUsages2 = Scanner.findUsages(ScannerTest2.class, lookFor);
36      System.out.println("Scan 1 = " + findUsages2);
37      Assert.assertThat(findUsages2, CoreMatchers.hasItem(
38              Matchers.allOf(
39                      Matchers.hasProperty("invokedMethod", Matchers.hasProperty("name", Matchers.equalTo("getProperty"))),
40                      Matchers.hasProperty("parameters", Matchers.hasItemInArray("spf4j.perf.ms.defaultSsdumpFilePrefix")))));
41    }
