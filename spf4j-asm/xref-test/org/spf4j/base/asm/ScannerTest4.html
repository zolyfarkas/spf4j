32    @Test
33    public void testScan() throws NoSuchMethodException, IOException {
34  
35      final ImmutableSet<Method> lookFor = ImmutableSet.of(System.class.getDeclaredMethod("getProperty", String.class),
36              System.class.getDeclaredMethod("getProperty", String.class, String.class),
37              Integer.class.getDeclaredMethod("getInteger", String.class),
38              Integer.class.getDeclaredMethod("getInteger", String.class, int.class),
39              Integer.class.getDeclaredMethod("getInteger", String.class, Integer.class),
40              Long.class.getDeclaredMethod("getLong", String.class),
41              Long.class.getDeclaredMethod("getLong", String.class, Long.class),
42              Long.class.getDeclaredMethod("getLong", String.class, long.class),
43              Boolean.class.getDeclaredMethod("getBoolean", String.class));
44      List<Invocation> findUsages2 = Scanner.findUsages(ScannerTest4.class, lookFor);
45      System.out.println("Scan 1 = " + findUsages2);
46      Assert.assertThat(findUsages2, CoreMatchers.hasItem(
47          Matchers.allOf(
48               Matchers.hasProperty("invokedMethod", Matchers.hasProperty("name", Matchers.equalTo("getInteger"))),
49               Matchers.hasProperty("parameters", Matchers.hasItemInArray("spf4j.lifoTp.maxSpinning")))));
50    }
