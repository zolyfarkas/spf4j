74      File TEST_FILE = File.createTempFile("test", ".tsdb2");
75      long tableId;
76      try (TSDBWriter writer = new TSDBWriter(TEST_FILE, 4, "test", false)) {
77  
78      }
79      try (TSDBWriter writer = new TSDBWriter(TEST_FILE, 4, "test", false)) {
80        tableId = writer.writeTableDef(tableDef);
81        final long time = System.currentTimeMillis();
82        writer.writeDataRow(tableId, time, 0, 1, 2);
83        writer.writeDataRow(tableId, time + 10, 1, 1, 2);

93      }

103   @Test
104   public void testTailing() throws IOException, InterruptedException, ExecutionException, TimeoutException {
105     File TEST_FILE = File.createTempFile("test", ".tsdb2");
106     try (TSDBWriter writer = new TSDBWriter(TEST_FILE, 4, "test", true);
107             TSDBReader reader = new TSDBReader(TEST_FILE, 1024)) {
108       writer.flush();
109       final BlockingQueue<Either<TableDef, DataBlock>> queue = new ArrayBlockingQueue<>(100);
110       Future<Void> bgWatch = reader.bgWatch((Either<TableDef, DataBlock> object, long deadline) -> {
111         queue.put(object);
112       }, TSDBReader.EventSensitivity.HIGH);
113       long tableId = writer.writeTableDef(tableDef);
114       writer.flush();
115       final long time = System.currentTimeMillis();
116       writer.writeDataRow(tableId, time, 0, 1, 2);
117       writer.writeDataRow(tableId, time + 10, 1, 1, 2);
118       writer.flush();
119       Either<TableDef, DataBlock> td = queue.take();
120       Assert.assertEquals(tableDef, td.getLeft());
121       Either<TableDef, DataBlock> take = queue.take();
122       Assert.assertEquals(2, take.getRight().getValues().size());
123       writer.writeDataRow(tableId, time + 20, 2, 1, 2);
124       writer.writeDataRow(tableId, time + 30, 3, 1, 2);
125       writer.writeDataRow(tableId, time + 40, 4, 1, 2);
126       writer.flush();
127       Assert.assertEquals(3, queue.take().getRight().getValues().size());
128       reader.stopWatching();
129       bgWatch.get(10000, TimeUnit.MILLISECONDS);
130 
131     }
