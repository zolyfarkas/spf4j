58     * Test of createScalableQuantizedRecorder method, of class RecorderFactory.

60    @Test
61    public void testCreateScalableQuantizedRecorder() throws IOException, InterruptedException {
62      System.out.println("createScalableQuantizedRecorder");
63      String forWhat = "test1";
64      String unitOfMeasurement = "ms";
65      int sampleTime = 1000;
66      int factor = 10;
67      int lowerMagnitude = 0;
68      int higherMagnitude = 3;
69      int quantasPerMagnitude = 10;
70      MeasurementRecorder result = RecorderFactory.createScalableQuantizedRecorder(
71              forWhat, unitOfMeasurement, sampleTime, factor, lowerMagnitude, higherMagnitude, quantasPerMagnitude);
72      for (int i = 0; i < 500; i++) {
73        result.record(i);
74        Thread.sleep(20);
75      }
76      result.close();
77      assertData(forWhat, 124750);
78    }

143   @SuppressFBWarnings("CLI_CONSTANT_LIST_INDEX")
144   public static void assertData(final String forWhat, final long expectedValue) throws IOException {
145     TSDBWriter dbWriter = ((TSDBMeasurementStore) RecorderFactory.MEASUREMENT_STORE).getDBWriter();
146     dbWriter.flush();
147     final File file = dbWriter.getFile();
148     List<TableDef> tableDefs = TSDBQuery.getTableDef(file, forWhat);
149     TableDef tableDef = tableDefs.get(0);
150     TimeSeries timeSeries = TSDBQuery.getTimeSeries(file, new long[]{tableDef.id}, 0, Long.MAX_VALUE);
151     long sum = 0;
152     long[][] values = timeSeries.getValues();
153     for (long[] row : values) {
154       sum += row[0];
155     }
156     Assert.assertEquals(expectedValue, sum);
157   }
