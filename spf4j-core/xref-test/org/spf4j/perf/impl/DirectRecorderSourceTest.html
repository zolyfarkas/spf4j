16      @Test
17      public void testDirectRecorder() throws IOException {
18          MeasurementRecorderSource recorderSource =
19                  RecorderFactory.createDirectRecorderSource("test", "description");
20          MeasurementRecorder recorder = recorderSource.getRecorder("A");
21          long time = System.currentTimeMillis();
22          for (int i = 0; i < 100; i++) {
23              recorder.recordAt(time + (long) i * 1000, i);
24          }
25          MeasurementRecorder recorder2 = recorderSource.getRecorder("B");
26          for (int i = 0; i < 100; i++) {
27              recorder2.recordAt(time + (long) i * 1000, i);
28          }
29          recorderSource.close();
30          org.spf4j.perf.RecorderFactoryTest.assertData("(test,A)", 4950);
31          org.spf4j.perf.RecorderFactoryTest.assertData("(test,B)", 4950);
32      }
