7   import org.junit.Assert;
8   import org.junit.Test;
9   
10  /**
11   *
12   * @author zoly

14  public final class SampleNodeTest {
15      
16      public SampleNodeTest() {
17      }
18  
19      @Test
20      public void testSampleNode() {
21         
22          System.out.println("sample");
23          StackTraceElement[] st1 = new StackTraceElement[3];
24          st1[0] = new StackTraceElement("C1", "m1", "C1.java", 10);
25          st1[1] = new StackTraceElement("C1", "m2", "C1.java", 11);
26          st1[2] = new StackTraceElement("C1", "m3", "C1.java", 12);
27          SampleNode node1 = SampleNode.createSampleNode(st1);
28          SampleNode node2 = new SampleNode(st1, st1.length - 1);
29          System.out.println(node1);
30          System.out.println(node2);
31          
32          Assert.assertEquals(4, node1.getNrNodes());
33          Assert.assertEquals(4, node2.getNrNodes());
34  
35          
36          StackTraceElement[] st2 = new StackTraceElement[1];
37          st2[0] = new StackTraceElement("C1", "m1", "C1.java", 10);
38          SampleNode.addToSampleNode(node1, st2);
39          node2.addSample(st2, st2.length - 1);
40          System.out.println(node1);
41          System.out.println(node2);
42          Assert.assertEquals(5, node1.getNrNodes());
43          Assert.assertEquals(5, node2.getNrNodes());
44  
45          StackTraceElement[] st3 = new StackTraceElement[3];
46          st3[0] = new StackTraceElement("C2", "m1", "C2.java", 10);
47          st3[1] = new StackTraceElement("C2", "m2", "C2.java", 11);
48          st3[2] = new StackTraceElement("C2", "m3", "C2.java", 12);
49          SampleNode.addToSampleNode(node1, st3);
50          node2.addSample(st3, st3.length - 1);
51          System.out.println(node1);
52          System.out.println(node2);
53  
54          StackTraceElement[] st4 = new StackTraceElement[3];
55          st4[0] = new StackTraceElement("C1", "m1", "C1.java", 10);
56          st4[1] = new StackTraceElement("C1", "m2", "C1.java", 11);
57          st4[2] = new StackTraceElement("C1", "m4", "C1.java", 14);
58          SampleNode.addToSampleNode(node1, st4);
59          node2.addSample(st4, st4.length - 1);
60  
61          SampleNode.addToSampleNode(node1, st1);
62          node2.addSample(st1, st1.length - 1);
63          
64          System.out.println("n1 = " + node1);
65          System.out.println("n2 = " + node2);
66          Assert.assertEquals(node1.toString(), node2.toString());
67          
68          SampleNode agg = SampleNode.aggregate(node1, node2);
69          System.out.println("n1 + n2 = " + agg);
70          Assert.assertEquals(node1.getSampleCount() + node2.getSampleCount(), agg.getSampleCount());
71          final Method method = new Method("C1", "m3");
72          Assert.assertEquals(node1.getSubNodes().get(method).getSampleCount()
73                  + node2.getSubNodes().get(method).getSampleCount(),
74                  agg.getSubNodes().get(method).getSampleCount());
75          
76      }
