28  public final class StringsTest {

30      public StringsTest() {
31      }

33      @Test
34      public void testDistance() {
35          assertEquals(3, Strings.distance("abc", "abcdef"));
36          assertEquals(3, Strings.distance("def", "abcdef"));
37          assertEquals(1, Strings.distance("abc", "bc"));
38          assertEquals(3, Strings.distance("abc", "def"));
39          assertEquals(1, Strings.distance("zoltran", "zoltan"));
40      }
41  
42      @Test
43      public void testEscaping() {
44          String res = Strings.unescape("a\\n");
45          Assert.assertEquals("a\n", res);
46      }
47  
48      @Test
49      public void testUnsafeOps() {
50         String testString = "dfjgdjshfgsjdhgfskhdfkdshf\ndfs@#$%^&\u63A5\u53D7*($%^&*()(*&^%$#@!>::><>?{PLKJHGFDEWSDFG";
51         char [] chars = Strings.steal(testString);
52         String otherString = Strings.wrap(chars);
53         Assert.assertEquals(testString, otherString);
54  
55         byte [] bytes = Strings.toUtf8(testString);
56         otherString = Strings.fromUtf8(bytes);
57         Assert.assertEquals(testString, otherString);
58  
59      }
60  
61      @Test
62      public void testSubSequence() {
63          String str = "dsfhjgsdjfgwuergfedhgfjhwheriufwiueruhfguyerugfweuyrygfwueyrghfuwoeruhgfdgwsjhfg";
64          assertEquals(0, Strings.compareTo(str.subSequence(3, 15), Strings.subSequence(str, 3, 15)));
65          Assert.assertTrue(Strings.equals(str.subSequence(15, 15), Strings.subSequence(str, 15, 15)));
66          Assert.assertTrue(Strings.equals(str.subSequence(0, str.length()), Strings.subSequence(str, 0, str.length())));
67          Assert.assertTrue(str.subSequence(3, 15).toString().equals(Strings.subSequence(str, 3, 15).toString()));
68      }
69  
70  
71      public void testEndsWith() {
72          Assert.assertEquals(true, Strings.endsWith("dfjkshfks", ""));
73          Assert.assertEquals(true, Strings.endsWith("dfjkshfks", "hfks"));
74          Assert.assertEquals(false, Strings.endsWith("dfjkshfks", "hfk"));
75          Assert.assertEquals(true, Strings.endsWith("dfjkshfks", "dfjkshfks"));
76          Assert.assertEquals(false, Strings.endsWith("dfjkshfks", "dfjkshfksu"));
77      }
78  
79      @Test
80      public void testEncoding() {
81          StringBuilder sb = new StringBuilder();
82          Strings.appendUnsignedString(sb, 38, 4);
83          Assert.assertEquals(Integer.toHexString(38), sb.toString());
84          System.out.println("Encoded: " + sb);
85      }
86      
87      @Test
88      public void testEncoding2() {
89          StringBuilder sb = new StringBuilder();
90          Strings.appendUnsignedStringPadded(sb, 38, 5, 4);
91          Assert.assertEquals(4, sb.length());
92          System.out.println("Encoded: " + sb);
93      }    
94     
95      
96      @Test
97      public void testJsonEncoding() {
98          StringBuilder sb = new StringBuilder();
99          sb.append("\"");
100         Strings.escapeJsonString("\n\u0008\u0000abc\"", sb);
101         sb.append("\"");
102         System.out.println("Encoded: " + sb);
103         Assert.assertEquals("\"\\n\\b\\u0000abc\\\"\"", sb.toString());
104     }
