63    @Test
64    @SuppressFBWarnings({ "SIC_INNER_SHOULD_BE_STATIC_ANON", "RV_RETURN_VALUE_IGNORED" })
65    public void testBuffering() throws IOException {
66      byte[] bytes = new byte[16384];
67      for (int i = 0; i < bytes.length; i++) {
68        bytes[i] = (byte) ('a' + i % 10);
69      }
70      ByteArrayInputStream bis = new ByteArrayInputStream(bytes) {
71        private boolean first = true;
72  
73        @Override
74        public synchronized int read(final byte[] b, final int off, final int len) {
75          if (first && len > 260) {
76            first = false;
77            return super.read(b, off, 260);
78          } else {
79            return super.read(b, off, len);
80          }
81        }
82  
83      };
84      try (MemorizingBufferedInputStream mbis = new MemorizingBufferedInputStream(bis)) {
85        StringBuilder sb = new StringBuilder();
86        PushbackInputStreamEx pushbackInputStreamEx = new PushbackInputStreamEx(mbis);
87        int read = pushbackInputStreamEx.read();
88        pushbackInputStreamEx.unread(read);
89        CharStreams.copy(new InputStreamReader(pushbackInputStreamEx, StandardCharsets.UTF_8), sb);
90        Assert.assertEquals(16384, sb.length());
91        Assert.assertEquals(new String(bytes, StandardCharsets.US_ASCII), sb.toString());
92      }
93    }

95    @Test
96    public void testSimpleStreamBuffering() throws IOException {
97      final byte[] array = Strings.toUtf8(TSTR);
98      ByteArrayInputStream bis = new ByteArrayInputStream(array);
99      MemorizingBufferedInputStream mis = new MemorizingBufferedInputStream(bis);
100     int val;
101     int i = 0;
102     while ((val = mis.read()) > 0) {
103       System.out.print((char) val);
104       Assert.assertEquals(array[i], (byte) val);
105       i++;
106     }
107     Assert.assertEquals(i, array.length);
108   }

133   @Test
134   public void testStreamBuffering() throws IOException {
135     test(TSTR, 8, true);
136     for (int i = 0; i < 100; i++) {
137       ThreadLocalRandom rnd = ThreadLocalRandom.current();
138       int nrChars = rnd.nextInt(0, 100000);
139       int bufferSize = rnd.nextInt(2, 100000);
140       StringBuilder sb = generateTestStr(nrChars);
141       test(sb.toString(), bufferSize, true);
142       testBuff(sb, 8192);
143       testBuff(sb, 32);
144     }
145   }
146 
147   @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
148   private void testBuff(final StringBuilder sb, final int buffSize) throws IOException {
149     final byte[] utf8Bytes = Strings.toUtf8(sb.toString());
150     ByteArrayInputStream bis = new ByteArrayInputStream(utf8Bytes);
151     try (MemorizingBufferedInputStream mbis = new MemorizingBufferedInputStream(bis, buffSize, buffSize / 2,
152             ArraySuppliers.Bytes.GL_SUPPLIER, StandardCharsets.UTF_8)) {
153       int val = mbis.read();
154       Assert.assertEquals(val, mbis.getReadBytesFromBuffer()[0]);
155       byte[] buff = new byte[8];
156       if (utf8Bytes.length > 8) {
157         ByteStreams.readFully(mbis, buff);
158         Assert.assertTrue(Arrays.equals(buff, Arrays.copyOfRange(mbis.getReadBytesFromBuffer(), 1, 9)));
159       }
160       int result;
161       do {
162         result = mbis.read(buff);
163       } while (result >= 0);
164       byte[] unredBytesEnd = mbis.getUnreadBytesFromBuffer();
165       Assert.assertEquals(0, unredBytesEnd.length);
166       final byte[] readBytesFromBuffer = mbis.getReadBytesFromBuffer();
167       Assert.assertTrue(Arrays.equals(readBytesFromBuffer,
168               Arrays.copyOfRange(utf8Bytes, utf8Bytes.length - readBytesFromBuffer.length, utf8Bytes.length)));
169 
170     }
