32      @Test
33      @SuppressFBWarnings("SIC_INNER_SHOULD_BE_STATIC_ANON")
34      public void testBuffering() throws IOException {
35        byte [] bytes = new byte[16384];
36        for (int i = 0; i< bytes.length; i++) {
37          bytes[i] = (byte) ('a' + i % 10);
38        }
39        ByteArrayInputStream bis = new ByteArrayInputStream(bytes) {
40          private boolean first = true;
41          @Override
42          public synchronized int read(byte[] b, int off, int len) {
43            if (first && len > 260) {
44              first = false;
45              return super.read(b, off, 260);
46            } else {
47              return super.read(b, off, len);
48            }
49          }
50  
51        };
52        try (MemorizingBufferedInputStream mbis = new MemorizingBufferedInputStream(bis)) {
53          StringBuilder sb = new StringBuilder();
54          PushbackInputStreamEx pushbackInputStreamEx = new PushbackInputStreamEx(mbis);
55          int read = pushbackInputStreamEx.read();
56          pushbackInputStreamEx.unread(read);
57          CharStreams.copy(new InputStreamReader(pushbackInputStreamEx, StandardCharsets.UTF_8), sb);
58          Assert.assertEquals(16384, sb.length());
59          Assert.assertEquals(new String(bytes, StandardCharsets.US_ASCII), sb.toString());
60        }
61      }

67      @Test
68      public void testSimpleStreamBuffering() throws IOException {
69          final byte[] array = Strings.toUtf8(TSTR);
70          ByteArrayInputStream bis = new ByteArrayInputStream(array);
71          MemorizingBufferedInputStream mis = new MemorizingBufferedInputStream(bis);
72          int val;
73          int i = 0;
74          while ((val = mis.read()) > 0)  {
75              System.out.print((char) val);
76              Assert.assertEquals(array[i], (byte) val);
77              i++;
78          }
79          Assert.assertEquals(i, array.length);
80      }

100     @Test
101     public void testStreamBuffering() throws IOException {
102         test(TSTR, 8, true);
103         final IntMath.XorShift32 random = new IntMath.XorShift32();
104         for (int i = 0; i < 100; i++) {
105             int nrChars = Math.abs(random.nextInt() % 100000);
106             StringBuilder sb = generateTestStr(nrChars);
107             test(sb.toString(), Math.abs(random.nextInt() % 10000), true);
108             testBuff(sb, 8192);
109             testBuff(sb, 32);
110         }

113     private void testBuff(final StringBuilder sb, final int buffSize) throws IOException {
114         final byte[] utf8Bytes = Strings.toUtf8(sb.toString());
115         ByteArrayInputStream bis = new ByteArrayInputStream(utf8Bytes);
116         try (MemorizingBufferedInputStream mbis = new MemorizingBufferedInputStream(bis, buffSize, buffSize / 2,
117                 ArraySuppliers.Bytes.GL_SUPPLIER, Charsets.UTF_8)) {
118             int val = mbis.read();
119             Assert.assertEquals(val, mbis.getReadBytesFromBuffer()[0]);
120             byte [] buff = new byte [8];
121             int read = mbis.read(buff);
122             Assert.assertEquals(8, read);
123             Assert.assertTrue(Arrays.equals(buff, Arrays.copyOfRange(mbis.getReadBytesFromBuffer(), 1, 9)));
124             int result;
125             do {
126                 result = mbis.read(buff);
127             } while (result >= 0);
128             byte[] unredBytesEnd = mbis.getUnreadBytesFromBuffer();
129             Assert.assertEquals(0, unredBytesEnd.length);
130             final byte[] readBytesFromBuffer = mbis.getReadBytesFromBuffer();
131             Assert.assertTrue(Arrays.equals(readBytesFromBuffer,
132                     Arrays.copyOfRange(utf8Bytes, utf8Bytes.length - readBytesFromBuffer.length, utf8Bytes.length)));
133 
134         }
