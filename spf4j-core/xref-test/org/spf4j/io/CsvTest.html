49      @Test
50      public void testCsvReadWrite() throws IOException {
51  
52          File testFile = createTestCsv();
53  
54          List<Map<String, String>> data =
55                  Csv.read(testFile, Charsets.UTF_8, new Csv.CsvHandler<List<Map<String, String>>>() {
56  
57              private boolean firstRow = true;
58  
59              private final List<Map<String, String>> result = new ArrayList<>();
60  
61              private List<String> header = new ArrayList<>();
62              private Map<String, String> row = null;
63              private int i = 0;
64  
65              @Override
66              public void startRow() {
67                  if (!firstRow) {
68                      row = new HashMap<>(header.size());
69                      i = 0;
70                  }
71              }
72  
73              @Override
74              public void element(final CharSequence elem) {
75                  if (firstRow) {
76                      header.add(elem.toString());
77                  } else {
78                      row.put(header.get(i++), elem.toString());
79                  }
80              }
81  
82              @Override
83              public void endRow() {
84                  if (firstRow) {
85                      firstRow = false;
86                  } else {
87                      result.add(row);
88                  }
89              }
90  
91              @Override
92              public List<Map<String, String>> eof() {
93                  return result;
94              }
95          });
96        Map<String, String> d0 = data.get(0);
97        Assert.assertEquals("1,3", d0.get("c"));
98        Assert.assertEquals("1", d0.get("d"));
99        Map<String, String> d1 = data.get(1);
100       Assert.assertEquals("1,3", d1.get("d"));
101       Assert.assertEquals("\"", d1.get("b"));
102       Assert.assertEquals("0\n", d1.get("c"));
103     }

133     private File createTestCsv() throws IOException {
134         File testFile = File.createTempFile("csvTest", ".csv");
135         System.out.println("test file : " + testFile);
136         try (BufferedWriter writer = new BufferedWriter(
137                 new OutputStreamWriter(new FileOutputStream(testFile), Charsets.UTF_8))) {
138             Csv.writeCsvRow(writer, "a", "b", "c", "d");
139             Csv.writeCsvRow(writer, "1.2\r", "1", "1,3", 1);
140             Csv.writeCsvRow(writer, "0", "\"", "0\n", "1,3");
141         }
142         System.out.println("test file written");
143         return testFile;
144     }
