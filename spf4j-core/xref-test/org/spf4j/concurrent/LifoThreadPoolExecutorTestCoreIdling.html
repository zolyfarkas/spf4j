19  package org.spf4j.concurrent;
20  
21  import java.io.IOException;
22  import java.util.concurrent.TimeUnit;
23  import junit.framework.Assert;
24  import org.junit.Test;
25  import org.spf4j.perf.cpu.CpuUsageSampler;
26  
27  /**
28   *
29   * @author zoly
30   */
31  public class LifoThreadPoolExecutorTestCoreIdling {
32  
33  
34      @Test
35      public void testLifoExecSQ() throws InterruptedException, IOException {
36          LifoThreadPoolExecutorSQP executor =
37                  new LifoThreadPoolExecutorSQP("test", 2, 8, 20, 1024, 1024);
38          Thread.sleep(20);
39          long time = CpuUsageSampler.getProcessCpuTimeNanos();
40          Thread.sleep(3000);
41          long cpuTime =  CpuUsageSampler.getProcessCpuTimeNanos() - time;
42          Assert.assertTrue(cpuTime < 200000000);
43          System.out.println(cpuTime); // 6069497000 with bug  53945000 without bug
44          executor.shutdown();
45          executor.awaitTermination(1, TimeUnit.SECONDS);
46      }
47
