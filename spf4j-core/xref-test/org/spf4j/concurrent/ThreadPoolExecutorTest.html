32  package org.spf4j.concurrent;

34  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
35  import java.util.concurrent.CancellationException;
36  import java.util.concurrent.CountDownLatch;
37  import java.util.concurrent.ExecutionException;
38  import java.util.concurrent.ExecutorService;
39  import java.util.concurrent.ForkJoinPool;
40  import java.util.concurrent.Future;
41  import java.util.concurrent.TimeUnit;
42  import org.junit.Assert;
43  import org.junit.Test;
44  
45  /**
46   * test to validate the behavior between spf4j and FJP implementations...
47   * @author zoly
48   */
49  public class ThreadPoolExecutorTest {
50  
51  
52    @Test(timeout = 10000)
53    public void testInteruptionBehavior() throws InterruptedException, ExecutionException {
54      LifoThreadPoolExecutorSQP executor = new LifoThreadPoolExecutorSQP("test", 0, 16, 60000, 0, 1024);
55      testPoolTaskCancellation(executor);
56    }
57  
58    @Test(timeout = 10000)
59    public void testInteruptionBehaviorFJP() throws InterruptedException, ExecutionException {
60      boolean assertError = false;
61      try {
62        testPoolTaskCancellation(new ForkJoinPool(16));
63      } catch (AssertionError err) {
64        System.err.println("expected " + err);
65        assertError = true;
66      }
67      Assert.assertTrue("expected that FJP tasks cannot be interrupted", assertError);
68    }
69  
70    public void testPoolTaskCancellation(ExecutorService executor) throws InterruptedException, ExecutionException {
71      RunnableImpl testRunnable = new RunnableImpl();
72      Future<?> submit = executor.submit(testRunnable);
73      Assert.assertTrue("task did not start", testRunnable.getStartedlatch().await(5, TimeUnit.SECONDS));
74      submit.cancel(true);
75      try {
76        submit.get();
77        Assert.fail("expected CancellationException");
78      } catch (CancellationException ex) {
79        // expected
80      }
81      Assert.assertTrue("task was not interrupted", testRunnable.getInterruptedLatch().await(5, TimeUnit.SECONDS));
82      executor.shutdown();
83      Assert.assertTrue("executor was not shut down", executor.awaitTermination(1000, TimeUnit.MILLISECONDS));
84    }
85  
86    static class RunnableImpl implements Runnable {
87  
88      private CountDownLatch startedlatch = new CountDownLatch(1);
89  
90      private CountDownLatch interruptedLatch = new CountDownLatch(1);
91  
92      @Override
93      @SuppressFBWarnings("MDM_THREAD_YIELD")
94      public void run() {
95        startedlatch.countDown();
96        try {
97          Thread.sleep(100000);
98        } catch (InterruptedException ex) {
99          interruptedLatch.countDown();
100       }
101     }
102 
103     public CountDownLatch getInterruptedLatch() {
104       return interruptedLatch;
105     }
106 
107 
108 
109 
110     public CountDownLatch getStartedlatch() {
111       return startedlatch;
112     }

117   }
