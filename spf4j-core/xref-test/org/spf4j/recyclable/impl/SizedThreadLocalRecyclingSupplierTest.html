6   package org.spf4j.recyclable.impl;
7   
8   import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
9   import junit.framework.Assert;
10  import org.junit.Test;
11  import org.spf4j.base.ReferenceType;
12  import org.spf4j.recyclable.SizedRecyclingSupplier;
13  
14  /**
15   *
16   * @author zoly
17   */
18  public class SizedThreadLocalRecyclingSupplierTest {
19  
20  
21      @Test
22      public void testSupplier() {
23          SizedThreadLocalRecyclingSupplier<byte[]> supplier = new SizedThreadLocalRecyclingSupplier<>(
24                  new SizedRecyclingSupplier.Factory<byte[]>() {
25  
26              @Override
27              @SuppressFBWarnings("SUA_SUSPICIOUS_UNINITIALIZED_ARRAY")
28              public byte[] create(final int size) {
29                  return new byte[size];
30              }
31  
32              @Override
33              public int size(final byte[] object) {
34                  return object.length;
35              }
36          }, ReferenceType.SOFT);
37  
38          byte[] get = supplier.get(8000);
39          supplier.recycle(get);
40          byte [] get2 = supplier.get(8000);
41          supplier.recycle(get2);
42          Assert.assertTrue(get == get2);
43  
44          get = supplier.get(8000);
45          get2 = supplier.get(8000);
46          Assert.assertTrue(get != get2);
47          supplier.recycle(get);
48          supplier.recycle(get2);
49  
50          get = supplier.get(4000);
51          Assert.assertEquals(8000, get.length);
52          supplier.recycle(get);
53  
54          get = supplier.get(10000);
55          Assert.assertEquals(10000, get.length);
56          supplier.recycle(get);
57  
58          get = supplier.get(8000);
59          get2 = supplier.get(8000);
60          Assert.assertTrue(get.length == 10000 || get2.length == 10000);
61  
62      }
63  
64  }
