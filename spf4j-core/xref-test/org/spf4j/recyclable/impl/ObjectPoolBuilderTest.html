54      @Test
55      @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
56      public void testBuild() throws ObjectCreationException, InterruptedException,
57              ObjectBorrowException, TimeoutException, ObjectDisposeException {
58          System.out.println("test=build");
59          RecyclingSupplier<ExpensiveTestObject> pool =
60                  new RecyclingSupplierBuilder(10, new ExpensiveTestObjectFactory()).build();
61          System.out.println(pool);
62          ExpensiveTestObject object = pool.get();
63          System.out.println(pool);
64          pool.recycle(object, null);
65          System.out.println(pool);
66          ExpensiveTestObject object2 = pool.get();
67          Assert.assertSame(object2, object);
68  
69          pool.dispose();
70      }

171     private volatile boolean isDeadlock = false;
172 
173     private Thread startDeadlockMonitor(final RecyclingSupplier<ExpensiveTestObject> pool,
174             final long deadlockTimeout) {
175         isDeadlock = false;
176         Thread monitor = new Thread(() -> {
177           try {
178             Thread.sleep(deadlockTimeout);
179             ThreadMXBean threadMX = ManagementFactory.getThreadMXBean();
180             System.err.println(Arrays.toString(threadMX.dumpAllThreads(true, true)));
181             System.err.println(pool.toString());
182             isDeadlock = true;
183           } catch (InterruptedException ex) {
184             // terminating monitor
185             return;
186           }
187         });
188         monitor.start();
189         return monitor;
190     }
191 
192    @SuppressFBWarnings("MDM_THREAD_YIELD")
193    private void runTest(final RecyclingSupplier<ExpensiveTestObject> pool,
194             final long sleepBetweenSubmit, final long deadlockTimeout) throws InterruptedException, ExecutionException {
195         Thread monitor = startDeadlockMonitor(pool, deadlockTimeout);
196         ExecutorService execService = new LifoThreadPoolExecutorSQP("test", 10, 10,
197                 5000, 1024, true);
198         BlockingQueue<Future<?>> completionQueue = new LinkedBlockingDeque<>();
199         RetryExecutor exec = new RetryExecutor(execService, (final Callable<Object> parameter)
200                         -> new Callables.DelayPredicate<Exception>() {
201 
202                   @Override
203                   public int apply(final Exception value) {
204                     return 0;
205                   }
206                 }, completionQueue);
207         int nrTests = 1000;
208         for (int i = 0; i < nrTests; i++) {
209             exec.submit(new TestCallable(pool, i));
210             Thread.sleep(sleepBetweenSubmit);
211         }
212         for (int i = 0; i < nrTests; i++) {
213             System.out.println("Task " + completionQueue.take().get() + " finished ");
214         }
215         monitor.interrupt();
216         monitor.join();
217         Thread.sleep(100);
218         Assert.assertEquals(0, completionQueue.size());
219         if (isDeadlock) {
220             Assert.fail("deadlock detected");
221         }
222     }
