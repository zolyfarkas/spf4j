61  final class SimpleSmartObjectPool<T> implements SmartRecyclingSupplier<T> {
62  
63    private int maxSize;
64    private int waitingForReturn;
65    private final LinkedHashMultimap<ObjectBorower<T>, T> borrowedObjects;
66    private final List<T> availableObjects;
67    private final ReentrantLock lock;
68    private final Condition available;
69    private final RecyclingSupplier.Factory<T> factory;
70    private T sample;
71  
72    SimpleSmartObjectPool(final int initialSize, final int maxSize,
73            final RecyclingSupplier.Factory<T> factory, final boolean fair)
74            throws ObjectCreationException {
75      this.maxSize = maxSize;
76      this.factory = factory;
77      this.lock = new ReentrantLock(fair);
78      this.available = this.lock.newCondition();
79      this.borrowedObjects = LinkedHashMultimap.create();
80      this.availableObjects = new ArrayList<>(initialSize);
81      for (int i = 0; i < initialSize; i++) {
82        availableObjects.add(factory.create());
83      }
84      waitingForReturn = 0;
85      this.sample = factory.create();
86    }

88    @Override
89    @SuppressFBWarnings({"RV_RETURN_VALUE_IGNORED_BAD_PRACTICE", "PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS"})
90    public T get(final ObjectBorower borower) throws InterruptedException,
91            TimeoutException, ObjectCreationException {
92      lock.lock();
93      try {
94        // trying to be fair here, if others are already waiting, we will not get one.
95        if (availableObjects.size() - waitingForReturn > 0) {
96          Iterator<T> it = availableObjects.iterator();
97          T object = it.next();
98          it.remove();
99          if (!borrowedObjects.put(borower, object)) {
100           throw new IllegalStateException("Cannot borrow " + object + ", " + borrowedObjects);
101         }
102         return object;
103       } else if (borrowedObjects.size() < maxSize) {
104         T object = factory.create();
105         if (!borrowedObjects.put(borower, object)) {
106           throw new IllegalStateException("Cannot borrow " + object + ", " + borrowedObjects);
107         }
108         return object;
109       } else {
110         if (borrowedObjects.isEmpty() && availableObjects.isEmpty()) {
111           throw new IllegalStateException(
112                   "Pool is probably closing down or is missconfigured with size 0 " + this);
113         }
114         while (borrowedObjects.isEmpty()) {
115           available.await(1, TimeUnit.MILLISECONDS);
116           long millisToDeadline = ExecutionContexts.getMillisToDeadline();
117           if (millisToDeadline < 0) {
118             throw new TimeoutException("Object wait timeout expired by " + (-millisToDeadline));
119           }
120         }
121         // try to reclaim object from borrowers
122         for (ObjectBorower<T> b : borrowedObjects.keySet()) {
123           if (borower != b) {
124             T object = b.tryReturnObjectIfNotInUse();
125             if (object != null) {
126               if (!borrowedObjects.remove(b, object)) {
127                 throw new IllegalStateException("Returned Object hasn't been borrowed " + object);
128               }
129               if (!borrowedObjects.put(borower, object)) {
130                 throw new IllegalStateException("Cannot borrow " + object + ", " + borrowedObjects);
131               }
132               return object;
133             }
134           }
135         }
136         // evrything in use, try to place a return request
137         boolean requestMade = false;
138         while (!requestMade) {
139           boolean hasValidBorowers = false;
140           for (ObjectBorower<T> b : borrowedObjects.keySet()) {
141             if (borower != b) {
142               hasValidBorowers = true;
143               Either<ObjectBorower.Action, T> objOrPromise = b.tryRequestReturnObject();
144               if (objOrPromise.isRight()) {
145                 T obj = objOrPromise.getRight();
146                 if (!borrowedObjects.remove(b, obj)) {
147                   throw new IllegalStateException("Returned Object " + obj
148                           + " hasn't been borrowed: " + borrowedObjects);
149                 }
150 
151                 if (!borrowedObjects.put(borower, obj)) {
152                   throw new IllegalStateException("Cannot boroow " + obj + ", " + borrowedObjects);
153                 }
154                 return obj;
155               } else {
156                 requestMade = objOrPromise.getLeft() == Action.REQUEST_MADE;
157                 if (requestMade) {
158                   break;
159                 }
160               }
161             }
162           }
163           if (!hasValidBorowers) {
164             throw new IllegalStateException("Borrower asks for more than possible " + borower);
165           }
166           if (!requestMade) {
167             // probably was unable to acquire the locks
168             do {
169               available.await(1, TimeUnit.MILLISECONDS);
170               long millisToDeadline = ExecutionContexts.getMillisToDeadline();
171               if (millisToDeadline < 0) {
172                 throw new TimeoutException("Object wait timeout expired by " + (-millisToDeadline));
173               }
174             } while (borrowedObjects.isEmpty());
175           }
176         }
177         waitingForReturn++;
178         while (availableObjects.isEmpty()) {
179           long waitTime = ExecutionContexts.getMillisToDeadline();
180           if (waitTime <= 0) {
181             throw new TimeoutException("Object wait timeout expired, waitTime = " + waitTime);
182           }
183           if (!available.await(waitTime, TimeUnit.MILLISECONDS)) {
184             throw new TimeoutException("Object wait timeout expired, waitTime = " + waitTime);
185           }
186         }
187         waitingForReturn--;
188         Iterator<T> it = availableObjects.iterator();
189         T objectT = it.next();
190         it.remove();
191         if (!borrowedObjects.put(borower, objectT)) {
192           throw new IllegalStateException("Cannot borrow " + objectT + ", " + borrowedObjects);
193         }
194         return objectT;
195 
196       }
197     } finally {
198       lock.unlock();
199     }
200   }

202   @Override
203   public void recycle(final T object, final ObjectBorower borower) {
204     lock.lock();
205     try {
206       if (!borrowedObjects.remove(borower, object)) {
207         // returned somebody else's objOrPromise.
208         Entry<ObjectBorower<T>, T> foundEntry = null;
209         for (Entry<ObjectBorower<T>, T> entry : borrowedObjects.entries()) {
210           final ObjectBorower<T> lb = entry.getKey();
211           if (lb == borower) {
212             continue;
213           }
214           if (lb.nevermind(entry.getValue())) {
215             foundEntry = entry;
216             break;
217           }
218         }
219         if (foundEntry == null) {
220           throw new IllegalStateException("Object " + object + " has not been borrowed from this pool");
221         } else {
222           if (!borrowedObjects.remove(foundEntry.getKey(), foundEntry.getValue())) {
223             throw new IllegalStateException("Should have removed " + foundEntry);
224           }
225         }
226       }
227       availableObjects.add(object);
228       available.signalAll();
229     } finally {
230       lock.unlock();
231     }
232   }

234   @Override
235   public boolean tryDispose(final long timeoutMillis) throws ObjectDisposeException, InterruptedException {
236     factory.dispose(sample);
237     long deadline = System.currentTimeMillis() + timeoutMillis;
238     lock.lock();
239     try {
240       maxSize = 0;
241       List<Pair<ObjectBorower<T>, T>> returnedObjects = new ArrayList<>();
242       for (Entry<ObjectBorower<T>, Collection<T>> b : borrowedObjects.asMap().entrySet()) {
243         ObjectBorower<T> borrower = b.getKey();
244         final int nrObjects = b.getValue().size();
245         for (int i = 0; i < nrObjects; i++) {
246           Either<Action, T> object = borrower.tryRequestReturnObject();
247           if (object.isRight()) {
248             returnedObjects.add(Pair.of(borrower, object.getRight()));
249           }
250         }
251       }
252       for (Pair<ObjectBorower<T>, T> objectAndBorrower : returnedObjects) {
253         T object = objectAndBorrower.getSecond();
254         if (!borrowedObjects.remove(objectAndBorrower.getFirst(), object)) {
255           throw new IllegalStateException("Returned Object hasn't been borrowed " + object);
256         }
257         availableObjects.add(object);
258       }
259       ObjectDisposeException exception = disposeReturnedObjects(null);
260       while (!borrowedObjects.isEmpty()) {
261         long waitTime = deadline - System.currentTimeMillis();
262         if (waitTime <= 0) {
263           return false;
264         }
265         if (!available.await(waitTime, TimeUnit.MILLISECONDS)) {
266           return false;
267         }
268         exception = disposeReturnedObjects(exception);
269       }
270       if (exception != null) {
271         throw exception;
272       }
273       return true;
274     } catch (InterruptedException | RuntimeException e) {
275       throw e;
276     } finally {
277       lock.unlock();
278     }
279   }

281   @CheckReturnValue
282   private ObjectDisposeException disposeReturnedObjects(final ObjectDisposeException exception) {
283     ObjectDisposeException result = exception;
284     for (T obj : availableObjects) {
285       try {
286         factory.dispose(obj);
287       } catch (ObjectDisposeException ex) {
288         if (result == null) {
289           result = ex;
290         } else {
291           result = Throwables.suppress(ex, result);
292         }
293       } catch (Exception ex) {
294         if (result == null) {
295           result = new ObjectDisposeException(ex);
296         } else {
297           result = Throwables.suppress(new ObjectDisposeException(ex), result);
298         }
299       }
300     }
301     availableObjects.clear();
302     return result;
303   }

305   @edu.umd.cs.findbugs.annotations.SuppressWarnings("EXS_EXCEPTION_SOFTENING_HAS_CHECKED")
306   @Override
307   public boolean scan(final ScanHandler<T> handler) throws Exception {
308     lock.lock();
309     Exception resEx = null;
310     try {
311       for (ObjectBorower<T> objectBorower : borrowedObjects.keySet()) {
312         try {
313           if (!objectBorower.scan(handler)) {
314             return false;
315           }
316         } catch (Exception e) {
317           if (resEx == null) {
318             resEx = e;
319           } else {
320             resEx = Throwables.suppress(e, resEx);
321           }
322         }
323 
324         Collection<T> returned = objectBorower.tryReturnObjectsIfNotNeededAnymore();
325         if (returned != null) {
326           for (T ro : returned) {
327             if (!borrowedObjects.remove(objectBorower, ro)) {
328               throw new IllegalStateException("Object returned hasn't been borrowed" + ro);
329             }
330             availableObjects.add(ro);
331           }
332         }
333 
334       }
335       for (T object : availableObjects) {
336         try {
337           if (!handler.handle(object)) {
338             return false;
339           }
340         } catch (Exception e) {
341           if (resEx == null) {
342             resEx = e;
343           } else {
344             resEx = Throwables.suppress(e, resEx);
345           }
346         }
347       }
348       if (resEx != null) {
349         throw resEx;
350       }
351       return true;
352     } finally {
353       lock.unlock();
354     }
355   }

357   @SuppressFBWarnings("RV_RETURN_VALUE_IGNORED")
358   public void requestReturnFromBorrowersIfNotInUse() throws InterruptedException {
359     lock.lock();
360     try {
361       List<Pair<ObjectBorower<T>, T>> returnedObjects = new ArrayList<>();
362       for (ObjectBorower<T> b : borrowedObjects.keySet()) {
363         Collection<T> objects = b.tryReturnObjectsIfNotInUse();
364         if (objects != null) {
365           for (T object : objects) {
366             returnedObjects.add(Pair.of(b, object));
367           }
368         }
369       }
370       for (Pair<ObjectBorower<T>, T> ro : returnedObjects) {
371         T object = ro.getSecond();
372         borrowedObjects.remove(ro.getFirst(), object);
373         availableObjects.add(object);
374       }
375     } finally {
376       lock.unlock();
377     }
378   }

380   @Override
381   public String toString() {
382     lock.lock();
383     try {
384       return "SimpleSmartObjectPool{" + "maxSize=" + maxSize + ", borrowedObjects="
385               + borrowedObjects + ", returnedObjects=" + availableObjects
386               + ", factory=" + factory + '}';
387     } finally {
388       lock.unlock();
389     }
390   }
