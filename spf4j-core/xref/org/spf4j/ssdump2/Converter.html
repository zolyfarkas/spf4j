39      public static <E extends Exception> int convert(final Method method, final SampleNode node,
40              final int parentId, final int id,
41              final Handler<ASample, E> handler) throws E {
42          ASample sample = new ASample();
43          sample.id = id;
44          sample.count = node.getSampleCount();
45          AMethod m = new AMethod();
46          m.setName(method.getMethodName());
47          m.setDeclaringClass(method.getDeclaringClass());
48          sample.method = m;
49          sample.parentId = parentId;
50          handler.handle(sample, Long.MAX_VALUE);
51          final TMap<Method, SampleNode> subNodes = node.getSubNodes();
52          int nid = id + 1;
53          if (subNodes != null) {
54              final TObjectObjectProcedureImpl proc = new TObjectObjectProcedureImpl(id, handler);
55              subNodes.forEachEntry(proc);
56              nid = proc.getNid();
57          }
58          return nid;
59      }
60  
61      public static SampleNode convert(final Iterator<ASample> samples) {
62          TIntObjectMap<SampleNode> index = new TIntObjectHashMap<>();
63          while (samples.hasNext()) {
64              ASample asmp = samples.next();
65              SampleNode sn = new SampleNode(asmp.count, new THashMap<Method, SampleNode>());
66              SampleNode parent = index.get(asmp.parentId);
67              if (parent != null) {
68                  AMethod method = asmp.getMethod();
69                  Method m = Method.getMethod(method.declaringClass, method.getName());
70                  final Map<Method, SampleNode> subNodes = parent.getSubNodes();
71                  if (subNodes == null) {
72                      throw new IllegalStateException("Bug, state " + index + "; at node " + asmp);
73                  }
74                  subNodes.put(m, sn);
75              }
76              index.put(asmp.id, sn);
77          }
78          return index.get(0);
79      }

81      public static void save(final File file, final SampleNode collected) throws IOException {
82          try (BufferedOutputStream bos = new BufferedOutputStream(
83                  new FileOutputStream(file))) {
84              final SpecificDatumWriter<ASample> writer = new SpecificDatumWriter<>(ASample.SCHEMA$);
85              final BinaryEncoder encoder = EncoderFactory.get().directBinaryEncoder(bos, null);
86              Converter.convert(Method.ROOT, collected,
87                      -1, 0, new Handler<ASample, IOException>() {
88  
89                          @Override
90                          public void handle(final ASample object, final long deadline)
91                                  throws IOException {
92                              writer.write(object, encoder);
93                          }
94                      });
95              encoder.flush();
96          }
97      }

99      @SuppressFBWarnings("NP_LOAD_OF_KNOWN_NULL_VALUE")
100     public static SampleNode load(final File file) throws IOException {
101         try (MemorizingBufferedInputStream bis = new MemorizingBufferedInputStream(new FileInputStream(file))) {
102             final PushbackInputStream pis = new PushbackInputStream(bis);
103             final SpecificDatumReader<ASample> reader = new SpecificDatumReader<>(ASample.SCHEMA$);
104             final BinaryDecoder decoder = DecoderFactory.get().directBinaryDecoder(pis, null);
105             return convert(new Iterator<ASample>() {
106 
107                 @Override
108                 public boolean hasNext() {
109                     try {
110                         int read = pis.read();
111                         pis.unread(read);
112                         return read >= 0;
113                     } catch (IOException ex) {
114                         throw new RuntimeException(ex);
115                     }
116                 }
117 
118                 @Override
119                 @SuppressFBWarnings
120                 public ASample next() {
121                     try {
122                         return reader.read(null, decoder);
123                     } catch (IOException ex) {
124                         NoSuchElementException e = new NoSuchElementException();
125                         e.addSuppressed(ex);
126                         throw e;
127                     }
128                 }
129 
130                 @Override
131                 public void remove() {
132                     throw new UnsupportedOperationException("Not supported yet.");
133                 }
134             });
135         }

138     private static final class TObjectObjectProcedureImpl<E extends Exception>
139                         implements TObjectObjectProcedure<Method, SampleNode> {
140 
141         private final int id;
142         private final Handler<ASample, E> handler;
143         private int nid;
144 
145         public TObjectObjectProcedureImpl(final int id, final Handler<ASample, E> handler) {
146             this.id = id;
147             this.handler = handler;
148             nid = id + 1;
149         }
150 
151         @Override
152         public boolean execute(final Method a, final SampleNode b) {
153             try {
154                 nid = convert(a, b, id, nid, handler);
155             } catch (Exception ex) {
156                 throw new RuntimeException(ex);
157             }
158             return true;
159         }
160 
161         public int getNid() {
162             return nid;
163         }
