51  /**
52   * Second generation Time-Series database format.
53   * The linked list structure from first generation is dropped to reduce write overhead.
54   *
55   *
56   * @author zoly
57   */
58  public final class TSDBWriter implements Closeable, Flushable {
59  
60      public static final Schema FILE_RECORD_SCHEMA =
61              Schema.createUnion(Arrays.asList(TableDef.SCHEMA$, DataBlock.SCHEMA$));
62  
63  
64      private final File file;
65      private final FileChannel channel;
66      private final BinaryEncoder encoder;
67      private final Header header;
68      private final SpecificDatumWriter<Object> recordWriter = new SpecificDatumWriter<>(FILE_RECORD_SCHEMA);
69      private final DataBlock writeBlock;
70      private final int maxRowsPerBlock;
71      private final  RandomAccessFile raf;
72  
73      static final byte[] MAGIC = Strings.toUtf8("TSDB2");
74      private final ByteArrayBuilder bab;
75  
76      @CreatesObligation
77      public TSDBWriter(final File file, final int maxRowsPerBlock,
78              final String description, final boolean append) throws IOException {
79          SpecificDatumWriter<Header> headerWriter = new SpecificDatumWriter<>(Header.SCHEMA$);
80          this.file = file;
81          this.maxRowsPerBlock = maxRowsPerBlock;
82          this.writeBlock = new DataBlock();
83          this.writeBlock.baseTimestamp = System.currentTimeMillis();
84          this.writeBlock.setValues(new ArrayList<DataRow>(maxRowsPerBlock));
85          raf = new RandomAccessFile(file, "rw");
86          bab = new ByteArrayBuilder(32768, ArraySuppliers.Bytes.JAVA_NEW);
87          encoder = EncoderFactory.get().directBinaryEncoder(bab, null);
88          channel = raf.getChannel();
89          channel.lock();
90          if (!append) {
91              raf.setLength(0);
92              channel.force(true);
93          }
94          if (raf.length() <= 0) {
95              // new file or overwite, will write header;
96              bab.write(MAGIC);
97              toOutputStream(0, bab);
98              header = Header.newBuilder()
99                      .setContentSchema(FILE_RECORD_SCHEMA.toString())
100                     .setDescription(description)
101                     .build();
102             headerWriter.write(header, encoder);
103             encoder.flush();
104             byte[] buffer = bab.getBuffer();
105             final int size = bab.size();
106             toByteArray(size, buffer, MAGIC.length);
107             raf.write(buffer, 0, size);
108         } else {
109             if (description != null) {
110                 throw new IllegalArgumentException("Providing description when appending is not allowed for " + file);
111             }
112 
113             try (BufferedInputStream bis = new BufferedInputStream(new FileInputStream(file));
114                     DataInputStream dis = new DataInputStream(bis)) {
115                 validateType(dis);
116                 long size = dis.readLong();
117                 SpecificDatumReader<Header> reader = new SpecificDatumReader<>(Header.getClassSchema());
118                 BinaryDecoder directBinaryDecoder = DecoderFactory.get().directBinaryDecoder(dis, null);
119                 header = reader.read(null, directBinaryDecoder);
120                 raf.seek(size);
121             }
122         }
123     }
124 
125     static void validateType(final InputStream dis) throws IOException {
126         byte[] readMagic = new byte[MAGIC.length];
127         ByteStreams.readFully(dis, readMagic);
128         if (!Arrays.equals(MAGIC, readMagic)) {
129             throw new IOException("wrong file type, magic is " + Arrays.toString(readMagic));
130         }
131     }

133 
134 
135     public synchronized long writeTableDef(final TableDef tableDef) throws IOException {
136         final long position = raf.getFilePointer();
137         bab.reset();
138         tableDef.id = position;
139         recordWriter.write(tableDef, encoder);
140         raf.write(bab.getBuffer(), 0, bab.size());
141         return position;
142     }
143 
144 
145     public synchronized void writeDataRow(final long tableId, final long timestamp, final long... data)
146             throws IOException {
147         if (this.writeBlock.values.size() >= this.maxRowsPerBlock) {
148             flush();
149         }
150         long baseTs = writeBlock.baseTimestamp;
151         DataRow row = new DataRow();
152         row.relTimeStamp = (int) (timestamp - baseTs);
153         row.tableDefId = tableId;
154         row.setData(Longs.asList(data));
155         this.writeBlock.values.add(row);
156     }
157 
158 
159 
160     @Override
161     public synchronized void close() throws IOException {
162         try (RandomAccessFile f = raf) {
163            flush();
164         }
165     }
166 
167     public File getFile() {
168         return file;
169     }
170 
171   public static void toByteArray(final long pvalue, final byte[] bytes, final int idx) {
172     long value = pvalue;
173     for (int i = idx + 7; i >= idx; i--) {
174       bytes[i] = (byte) (value & 0xffL);
175       value >>= 8;
176     }
177   }
178 
179   public static void toOutputStream(final long pvalue, final OutputStream os) throws IOException {
180     long value = pvalue;
181     for (int i = 7; i >= 0; i--) {
182       os.write((byte) (value & 0xffL));
183       value >>= 8;
184     }
185   }
186 
187 
188     /**
189      * Commits the data to disk.
190      * @throws IOException
191      */
192 
193     @Override
194     public synchronized void flush() throws IOException {
195         if (writeBlock.getValues().size() > 0) {
196             bab.reset();
197             this.recordWriter.write(writeBlock, this.encoder);
198             raf.write(bab.getBuffer(), 0, bab.size());
199             long filePointer = raf.getFilePointer();
200             raf.seek(MAGIC.length);
201             raf.writeLong(filePointer);
202             raf.seek(filePointer);
203             writeBlock.values.clear();
204             encoder.flush();
205         }
206         channel.force(true);
207     }
208 
209     public Header getHeader() {
210         return header;
211     }
212 
213     @Override
214     public String toString() {
215         return "TSDBWriter{" + "file=" + file + ", raf=" + raf + '}';
216     }
