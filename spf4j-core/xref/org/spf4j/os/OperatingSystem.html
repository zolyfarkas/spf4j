97    private static final String OS_NAME;

99    static {
100     final String osName = System.getProperty("os.name");
101     OS_NAME = osName;
102     IS_MAC_OSX = "Mac OS X".equals(osName);
103     IS_WINDOWS = osName.startsWith("Windows");
104     OS_MBEAN = ManagementFactory.getOperatingSystemMXBean();
105     if (OS_MBEAN instanceof com.sun.management.OperatingSystemMXBean) {
106       SUN_OS_MBEAN = (com.sun.management.OperatingSystemMXBean) OS_MBEAN;
107     } else {
108       SUN_OS_MBEAN = null;
109     }
110     if (OS_MBEAN instanceof UnixOperatingSystemMXBean) {
111       UNIX_OS_MBEAN = (UnixOperatingSystemMXBean) OS_MBEAN;
112       MAX_NR_OPENFILES = UNIX_OS_MBEAN.getMaxFileDescriptorCount();
113     } else {
114       UNIX_OS_MBEAN = null;
115       if (IS_WINDOWS) {
116         MAX_NR_OPENFILES = Integer.MAX_VALUE;
117       } else if (JNA.haveJnaPlatformClib()) {
118         try {
119           MAX_NR_OPENFILES = UnixResources.RLIMIT_NOFILE.getSoftLimit();
120         } catch (UnixException ex) {
121           throw new ExceptionInInitializerError(ex);
122         }
123       } else {
124         MAX_NR_OPENFILES = Integer.MAX_VALUE;
125       }
126     }

160   public static long getOpenFileDescriptorCount() {
161     if (UNIX_OS_MBEAN != null) {
162       return UNIX_OS_MBEAN.getOpenFileDescriptorCount();
163     } else {
164       try {
165         if (isMacOsx()) {
166           try {
167             return Lsof.getNrOpenFiles();
168           } catch (ExecutionException | TimeoutException ex) {
169             LOG.log(Level.WARNING, "Unable to get nr of open files", ex);
170             return -1;
171           } catch (InterruptedException ex) {
172             LOG.log(Level.WARNING, "Unable to get nr of open files", ex);
173             Thread.currentThread().interrupt();
174             return -1;
175           }
176         } else {
177           if (Files.isDirectory(FD_FOLDER)) {
178             int result = 0;
179             try (DirectoryStream<Path> stream = Files.newDirectoryStream(FD_FOLDER)) {
180               Iterator<Path> iterator = stream.iterator();
181               while (iterator.hasNext()) {
182                 iterator.next();
183                 result++;
184               }
185             }
186             return result;
187           } else {
188             return -1;
189           }
190         }
191       } catch (IOException ex) {
192         String msg = ex.getMessage();
193         if (msg != null && msg.contains("Too many open files")) {
194           return getMaxFileDescriptorCount();
195         } else {
196           LOG.log(Level.WARNING, "Unable to get nr of open files", ex);
197           return -1;
198         }
199       }
200     }

203   public static String getHostName() {
204     if (Platform.isWindows()) {
205       return Kernel32Util.getComputerName();
206     } else {
207       // For now, we'll consider anyhting other than Windows to be unix-ish enough to have gethostname
208       // TODO - Consider http://stackoverflow.com/a/10543006 as a possibly better MacOS option
209 
210       byte[] hostnameBuffer = new byte[257];
211       int result = CLibrary.INSTANCE.gethostname(hostnameBuffer, hostnameBuffer.length);
212       if (result != 0) {
213         int lastError = Native.getLastError();
214         throw new UnixRuntimeException("Error code " + CLibrary.INSTANCE.strerror(lastError)
215               + " for gethostname(byyte[], 256)", lastError);
216       }
217       return Native.toString(hostnameBuffer);
218     }
219   }

225   public static int killProcess(final Process proc, final long terminateTimeoutMillis,
226           final long forceTerminateTimeoutMillis)
227           throws InterruptedException, TimeoutException {
228 
229     proc.destroy();
230     if (proc.waitFor(terminateTimeoutMillis, TimeUnit.MILLISECONDS)) {
231       return proc.exitValue();
232     } else {
233       proc.destroyForcibly();
234       if (!proc.waitFor(forceTerminateTimeoutMillis, TimeUnit.MILLISECONDS)) {
235         throw new TimeoutException("Cannot terminate " + proc);
236       } else {
237         return proc.exitValue();
238       }
239     }
240   }

242   /**
243    * Process execution utility.
244    * @param <T> type the stdout is reduced to.
245    * @param <E> type stderr is reduced to.
246    * @param command the command to execute.
247    * @param handler handler for child stdin, stdout and stderr. stdout and stderr handling will be done in 2 threads
248    * from the DefaultExecutor thread pool. while stdin handling will execute in the current thread.
249    * @param timeoutMillis time to wait for the process to execute.
250    * @param terminationTimeoutMillis this is the timeout used when trying to terminate the process gracefully.
251    * @return the response (respCode, stdout reduction, stderr reduction)
252    * @throws IOException
253    * @throws InterruptedException
254    * @throws ExecutionException
255    * @throws TimeoutException when timeout happens.
256    */
257   @SuppressFBWarnings({ "COMMAND_INJECTION", "CC_CYCLOMATIC_COMPLEXITY" })
258   public static <T, E> ProcessResponse<T, E> forkExec(final String[] command, final ProcessHandler<T, E> handler,
259           final long timeoutMillis, final long terminationTimeoutMillis)
260           throws IOException, InterruptedException, ExecutionException, TimeoutException {
261     LOG.log(Level.FINE, "Executing {0}", (Object) command);
262     final Process proc = java.lang.Runtime.getRuntime().exec(command);
263     handler.started(proc);
264     try (InputStream pos = proc.getInputStream();
265             InputStream pes = proc.getErrorStream();
266             OutputStream pis = proc.getOutputStream()) {
267       Future<E> esh;
268       Future<T> osh;
269       final AtomicReference<Throwable> eex = new AtomicReference<>();
270       try {
271         esh = DefaultExecutor.INSTANCE.submit(() -> {
272           try {
273             return handler.handleStdErr(pes);
274           } catch (Throwable t) {
275             eex.set(t);
276             throw t;
277           }
278         });
279       } catch (RuntimeException ex) { // Executor might Reject
280         int result = killProcess(proc, terminationTimeoutMillis, ABORT_TIMEOUT_MILLIS);
281         throw new ExecutionException("Cannot execute stderr handler, killed process returned " + result, ex);
282       }
283       try {
284         osh = DefaultExecutor.INSTANCE.submit(() -> {
285           try {
286           return handler.handleStdOut(pos);
287           } catch (Throwable t) {
288             eex.set(t);
289             throw t;
290           }
291         });
292       } catch (RuntimeException ex) { // Executor might Reject
293         RuntimeException cex = Futures.cancelAll(true, esh);
294         if (cex != null) {
295           ex.addSuppressed(cex);
296         }
297         int result = killProcess(proc, terminationTimeoutMillis, ABORT_TIMEOUT_MILLIS);
298         throw new ExecutionException("Cannot execute stdout handler, killed process returned " + result, ex);
299       }
300       long deadlineNanos = TimeSource.nanoTime() + TimeUnit.NANOSECONDS.convert(timeoutMillis, TimeUnit.MILLISECONDS);
301       try {
302         handler.writeStdIn(pis);
303       } catch (RuntimeException | IOException ex) {
304         RuntimeException cex = Futures.cancelAll(true, esh, osh);
305         if (cex != null) {
306           ex.addSuppressed(cex);
307         }
308         int result = killProcess(proc, terminationTimeoutMillis, ABORT_TIMEOUT_MILLIS);
309         throw new ExecutionException("Failure executing stdin handler, killed process returned " + result, ex);
310       }
311       try {
312         int result = waitFor(eex, proc, timeoutMillis, TimeUnit.MILLISECONDS);
313         Pair<Map<Future, Object>, Exception> results = Futures.getAllWithDeadlineNanos(deadlineNanos, osh, esh);
314         Exception hex = results.getSecond();
315         Map<Future, Object> asyncRes = results.getFirst();
316         if (hex == null) {
317           return new ProcessResponse<>(result, (T) asyncRes.get(osh), (E) asyncRes.get(esh));
318         } else {
319           Throwables.throwException(hex);
320           throw new IllegalStateException();
321         }
322       } catch (TimeoutException | ExecutionException | InterruptedException ex) {
323         killProcess(proc, terminationTimeoutMillis, ABORT_TIMEOUT_MILLIS);
324         RuntimeException cex = Futures.cancelAll(true, osh, esh);
325         if (cex != null) {
326           ex.addSuppressed(cex);
327         }
328         throw ex;
329       }
330     }
331   }

333   @SuppressFBWarnings("MDM_THREAD_YIELD") // best I can come up with so far. (same as JDK)
334   private static int waitFor(final AtomicReference<Throwable> exr,
335           final Process process,
336           final long timeout, final TimeUnit unit)
337           throws InterruptedException, ExecutionException, TimeoutException {
338     long startTime = TimeSource.nanoTime();
339     long rem = unit.toNanos(timeout);
340     do {
341       try {
342         return process.exitValue();
343       } catch (IllegalThreadStateException ex) {
344         if (rem > 0) {
345           Thread.sleep(Math.min(TimeUnit.NANOSECONDS.toMillis(rem) + 1, 100));
346         }
347       }
348       rem = unit.toNanos(timeout) - (TimeSource.nanoTime() - startTime);
349       Throwable get = exr.get();
350       if (get != null) {
351         throw new ExecutionException(get);
352       }
353     } while (rem > 0);
354     throw new TimeoutException("Process " + process + " timed out after " + timeout + " " + unit);
355   }

358   @CheckReturnValue
359   public static String forkExec(final String[] command,
360           final long timeoutMillis) throws IOException, InterruptedException, ExecutionException, TimeoutException {
361     ProcessResponse<String, String> resp
362             = forkExec(command, new StdOutToStringProcessHandler(), timeoutMillis, 60000);
363     if (resp.getResponseExitCode() != SysExits.OK) {
364       throw new ExecutionException("Failed to execute " + Arrays.toString(command)
365               + ", exitCode = " + resp.getResponseCode() + ", stderr = " + resp.getErrOutput(), null);
366     }
367     return resp.getOutput();
368   }

370   public static void forkExecLog(final String[] command,
371           final long timeoutMillis) throws IOException, InterruptedException, ExecutionException, TimeoutException {
372     ProcessResponse<Void, Void> resp
373             = forkExec(command,
374                     new LoggingProcessHandler(Logger.getLogger("fork." + command[0])),
375                     timeoutMillis, 60000);
376     if (resp.getResponseExitCode() != SysExits.OK) {
377       throw new ExecutionException("Failed to execute " + java.util.Arrays.toString(command)
378               + ", exitCode = " + resp.getResponseCode() + ", stderr = " + resp.getErrOutput(), null);
379     }
380   }
