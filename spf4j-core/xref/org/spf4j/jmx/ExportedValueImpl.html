20  package org.spf4j.jmx;
21  
22  import com.google.common.base.Converter;
23  import java.lang.reflect.InvocationTargetException;
24  import java.lang.reflect.Method;
25  import javax.annotation.Nonnull;
26  import javax.annotation.Nullable;
27  import javax.management.InvalidAttributeValueException;
28  import javax.management.openmbean.OpenDataException;
29  
30  
31  /**
32   *
33   * @author zoly
34   */
35  class ExportedValueImpl implements ExportedValue<Object> {
36      private final String name;
37      private final String description;
38      private final Method getMethod;
39      private final Method setMethod;
40      private final Object object;
41      private final Class<?> valueClass;
42      private final Converter<Object, Object> converter;
43  
44      ExportedValueImpl(@Nonnull final String name, @Nullable final String description,
45              @Nullable final Method getMethod, @Nullable final Method setMethod,
46              @Nullable final Object object, @Nonnull final Class<?> valueClass) {
47          this.name = name;
48          this.description = description;
49          this.getMethod = getMethod;
50          this.setMethod = setMethod;
51          this.object = object;
52          this.valueClass = valueClass;
53          if (valueClass.isPrimitive()
54                  || Number.class.isAssignableFrom(valueClass)
55                  || valueClass == Boolean.class
56                  || valueClass == String.class) {
57              this.converter = null;
58          } else {
59              try {
60                  this.converter = OpenTypeConverter.getConverter(valueClass);
61              } catch (OpenDataException ex) {
62                  throw new RuntimeException(ex);
63              }
64          }
65      }
66  
67      public ExportedValueImpl withSetter(@Nonnull final Method psetMethod) {
68          return new ExportedValueImpl(name, description, getMethod, psetMethod, object, valueClass);
69      }
70  
71      public ExportedValueImpl withGetter(@Nonnull final Method pgetMethod, @Nonnull final String pdescription) {
72          return new ExportedValueImpl(name, pdescription, pgetMethod, setMethod, object, valueClass);
73      }
74  
75      @Override
76      public String getName() {
77          return name;
78      }
79  
80      @Override
81      public String getDescription() {
82          return description;
83      }
84  
85      @Override
86      @edu.umd.cs.findbugs.annotations.SuppressFBWarnings(value = "EXS_EXCEPTION_SOFTENING_NO_CHECKED")
87      public Object get() {
88  
89          try {
90              if (converter != null) {
91                  return converter.reverse().convert(getMethod.invoke(object));
92              } else {
93                  return getMethod.invoke(object);
94              }
95          } catch (IllegalAccessException ex) {
96              throw new RuntimeException(ex);
97          } catch (IllegalArgumentException ex) {
98              throw new RuntimeException(ex);
99          } catch (InvocationTargetException ex) {
100             throw new RuntimeException(ex);
101         }
102     }
103 
104     @Override
105     @edu.umd.cs.findbugs.annotations.SuppressFBWarnings(value = "EXS_EXCEPTION_SOFTENING_HAS_CHECKED")
106     public void set(final Object value) throws InvalidAttributeValueException {
107         if (setMethod == null) {
108             throw new InvalidAttributeValueException(name + " is a read only attribute ");
109         }
110         try {
111             if (converter != null) {
112                 setMethod.invoke(object, converter.convert(value));
113             } else {
114                 setMethod.invoke(object, value);
115             }
116         } catch (IllegalAccessException ex) {
117             throw new RuntimeException(ex);
118         } catch (IllegalArgumentException ex) {
119             throw new RuntimeException(name + " has an invalid type ", ex);
120         } catch (InvocationTargetException ex) {
121             throw new RuntimeException(ex);
122         }
123     }
124 
125     @Override
126     public boolean isWriteable() {
127         return setMethod != null;
128     }
129 
130     @Override
131     public Class<? extends Object> getValueClass() {
132         return valueClass;
133     }
134 
135     public boolean isValid() {
136         return getMethod != null;
137     }
138 
139     @Override
140     public String toString() {
141         return "ExportedValueImpl{" + "name=" + name + ", description=" + description + ", getMethod="
142                 + getMethod + ", setMethod=" + setMethod + ", object=" + object + ", valueClass=" + valueClass + '}';
143     }
144 
145 }
