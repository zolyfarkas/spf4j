18  package org.spf4j.jmx;
19  
20  import com.google.common.base.Converter;
21  import com.sun.jmx.mbeanserver.MXBeanMapping;
22  import com.sun.jmx.mbeanserver.MXBeanMappingFactory;
23  import java.io.InvalidObjectException;
24  import javax.management.openmbean.OpenDataException;
25  import javax.management.openmbean.OpenType;
26  
27  /**
28   *
29   * @author zoly
30   */
31  // FB does not like guava Convertere equals/hashcode.
32  @edu.umd.cs.findbugs.annotations.SuppressFBWarnings("HE_INHERITS_EQUALS_USE_HASHCODE")
33  public final class OpenTypeConverter {
34  
35    private OpenTypeConverter() {
36    }
37  
38  
39    /**
40     * returns MXBeanMapping
41     */
42    private static MXBeanMapping getMXBeanMapping(final Class<?> type) {
43      try {
44        return MXBeanMappingFactory.DEFAULT.mappingForType(type, MXBeanMappingFactory.DEFAULT);
45      } catch (OpenDataException ex) {
46        throw new UnsupportedOperationException("Unsupported MX bean mapping for " + type, ex);
47      }
48    }
49  
50    public static OpenType<?> getOpenType(final Class<?> type) {
51      return getMXBeanMapping(type).getOpenType();
52    }
53  
54    /**
55     *
56     * @param type the java type to from to convert to.
57     * @return Converter - the guava converter for the type.
58     */
59    public static Converter<Object, Object> getConverter(final Class<?> type) {
60  
61      final MXBeanMapping mapping = getMXBeanMapping(type);
62      return new Converter<Object, Object>() {
63  
64        @Override
65        protected Object doForward(final Object a) {
66          try {
67            return mapping.fromOpenValue(a);
68          } catch (InvalidObjectException ex) {
69            throw new IllegalArgumentException("Open value canot be converted: " + a, ex);
70          }
71        }
72  
73        @Override
74        protected Object doBackward(final Object b) {
75          try {
76            return mapping.toOpenValue(b);
77          } catch (OpenDataException ex) {
78            throw new IllegalArgumentException("Value cannot be converted to open value: " + b, ex);
79          }
80        }
81      };
82  
83    }
84  
85  }
