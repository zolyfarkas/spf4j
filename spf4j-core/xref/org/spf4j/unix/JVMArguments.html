77    public String removeSystemProperty(final String pname) {
78      String name = "-D" + pname;
79      String nameeq = name + '=';
80      for (Iterator<String> itr = arguments.iterator(); itr.hasNext();) {
81        String s = itr.next();
82        if (s.equals(name) || s.startsWith(nameeq)) {
83          itr.remove();
84          return s.substring(nameeq.length());
85        }
86      }
87      return null;
88    }
89  
90    public int removeAllSystemPropertiesStartingWith(final String pname) {
91      String name = "-D" + pname;
92      int nrRemoved = 0;
93      for (Iterator<String> itr = arguments.iterator(); itr.hasNext();) {
94        String s = itr.next();
95        if (s.startsWith(name)) {
96          itr.remove();
97          nrRemoved++;
98        }
99      }
100     return nrRemoved;
101   }

103   public void setSystemProperty(final String name, final String value) {
104     removeSystemProperty(name);
105     // index 0 is the executable name
106     arguments.add(1, "-D" + name + '=' + value);
107   }

116   public void removeTail(final int n) {
117     int size = arguments.size();
118     arguments.removeAll(arguments.subList(size - n, size));

132   /**
133    * Gets the process argument list of the specified process ID.
134    *
135    * @param pid -1 to indicate the current process.
136    */
137   public static JVMArguments of(final int pid) throws IOException {
138     String os = Runtime.OS_NAME;
139     switch (os) {
140       case "Linux":
141         return ofLinux(pid);
142       case "SunOS":
143         return ofSolaris(pid);
144       case "Mac OS X":
145         return ofMac(pid);
146       case "FreeBSD":
147         return ofFreeBSD(pid);
148       default:
149         throw new UnsupportedOperationException("Unsupported Operating System " + os);
150     }
151   }

153   private static JVMArguments ofLinux(final int ppid) throws IOException {
154     int pid = resolvePID(ppid);
155     String cmdline = Files.asCharSource(new File("/proc/" + pid + "/cmdline"), Charset.defaultCharset()).read();
156     return new JVMArguments(java.util.Arrays.asList(cmdline.split("\0")));
157   }

159   private static int resolvePID(final int pid) {
160     if (pid == -1) {
161       return Runtime.PID;
162     } else {
163       return pid;
164     }
165   }

167   private static JVMArguments ofSolaris(final int ppid) throws IOException {
168     // /proc shows different contents based on the caller's memory model, so we need to know if we are 32 or 64.
169     // 32 JVMs are the norm, so err on the 32bit side.
170     boolean areWe64 = "64".equals(System.getProperty("sun.arch.data.model"));
171     int pid = resolvePID(ppid);
172     try (RandomAccessFile psinfo = new RandomAccessFile(new File("/proc/" + pid + "/psinfo"), "r")) {
173       // see http://cvs.opensolaris.org/source/xref/onnv/onnv-gate/usr/src/uts/common/sys/procfs.h
174       //typedef struct psinfo {
175       // int pr_flag; /* process flags */
176       // int pr_nlwp; /* number of lwps in the process */
177       // pid_t pr_pid; /* process id */
178       // pid_t pr_ppid; /* process id of parent */
179       // pid_t pr_pgid; /* process id of process group leader */
180       // pid_t pr_sid; /* session id */
181       // uid_t pr_uid; /* real user id */
182       // uid_t pr_euid; /* effective user id */
183       // gid_t pr_gid; /* real group id */
184       // gid_t pr_egid; /* effective group id */
185       // uintptr_t pr_addr; /* address of process */
186       // size_t pr_size; /* size of process image in Kbytes */
187       // size_t pr_rssize; /* resident set size in Kbytes */
188       // dev_t pr_ttydev; /* controlling tty device (or PRNODEV) */
189       // ushort_t pr_pctcpu; /* % of recent cpu time used by all lwps */
190       // ushort_t pr_pctmem; /* % of system memory used by process */
191       // timestruc_t pr_start; /* process start time, from the epoch */
192       // timestruc_t pr_time; /* cpu time for this process */
193       // timestruc_t pr_ctime; /* cpu time for reaped children */
194       // char pr_fname[PRFNSZ]; /* name of exec'ed file */
195       // char pr_psargs[PRARGSZ]; /* initial characters of arg list */
196       // int pr_wstat; /* if zombie, the wait() status */
197       // int pr_argc; /* initial argument count */
198       // uintptr_t pr_argv; /* address of initial argument vector */
199       // uintptr_t pr_envp; /* address of initial environment vector */
200       // char pr_dmodel; /* data model of the process */
201       // lwpsinfo_t pr_lwp; /* information for representative lwp */
202       //} psinfo_t;
203 
204       // see http://cvs.opensolaris.org/source/xref/onnv/onnv-gate/usr/src/uts/common/sys/types.h
205       // for the size of the various datatype.
206       // see http://cvs.opensolaris.org/source/xref/onnv/onnv-gate/usr/src/cmd/ptools/pargs/pargs.c
207       // for how to read this information
208       psinfo.seek(8);
209       if (adjust(psinfo.readInt()) != pid) {
210         throw new IOException("psinfo PID mismatch: " + pid + ", " + psinfo);   // sanity check
211       }
212       /* The following program computes the offset:
213                     #include <stdio.h>
214                     #include <sys/procfs.h>
215                     int main() {
216                       printf("psinfo_t = %d\n", sizeof(psinfo_t));
217                       psinfo_t *x;
218                       x = 0;
219                       printf("%x\n", &(x->pr_argc));
220                     }
221        */
222 
223       psinfo.seek(areWe64 ? 0xEC : 0xBC);  // now jump to pr_argc
224       int argc = adjust(psinfo.readInt());
225       long argp = areWe64 ? adjust(psinfo.readLong()) : to64(adjust(psinfo.readInt()));
226       File asFile = new File("/proc/" + pid + "/as");
227       if (areWe64) {
228         // 32bit and 64bit basically does the same thing, but because the stream position
229         // is computed with signed long, doing 64bit seek to a position bigger than Long.MAX_VALUE
230         // requres some real hacking. Hence two different code path.
231         // (RandomAccessFile uses Java long for offset, so it just can't get to anywhere beyond Long.MAX_VALUE)
232         FILE fp = CLibrary.INSTANCE.fopen(asFile.getPath(), "r");
233         try {
234           JVMArguments args = new JVMArguments(16);
235           Memory m = new Memory(8);
236           for (int n = 0; n < argc; n++) {
237             // read a pointer to one entry
238             seek64(fp, argp + ((long) n) * 8);
239             m.setLong(0, 0); // just to make sure failed read won't result in bogus value
240             CLibrary.INSTANCE.fread(m, 1, 8, fp);
241             long p = m.getLong(0);
242             args.add(readLine(fp, p));
243           }
244           return args;
245         } finally {
246           CLibrary.INSTANCE.fclose(fp);
247         }
248       } else {
249         try (RandomAccessFile as = new RandomAccessFile(asFile, "r")) {
250           JVMArguments args = new JVMArguments(16);
251           for (int n = 0; n < argc; n++) {
252             // read a pointer to one entry
253             as.seek(argp + n * 4);
254             int p = adjust(as.readInt());
255 
256             args.add(readLine(as, p));
257           }
258           return args;
259         }
260       }
261     }
262   }

264   /**
265    * Seek to the specified position. This method handles offset bigger than {@link Long#MAX_VALUE} correctly.
266    *
267    * @param upos This value is interpreted as unsigned 64bit integer (even though it's typed 'long')
268    */
269   private static void seek64(final FILE fp, final long pupos) {
270     long upos = pupos;
271     CLibrary.INSTANCE.fseek(fp, 0, 0); // start at the beginning
272     while (upos < 0) {
273       long chunk = Long.MAX_VALUE;
274       upos -= chunk;
275       CLibrary.INSTANCE.fseek(fp, chunk, 1);
276     }
277     CLibrary.INSTANCE.fseek(fp, upos, 1);
278   }

280   /**
281    * {@link DataInputStream} reads a value in big-endian, so convert it to the correct value on little-endian systems.
282    */
283   private static int adjust(final int i) {
284     if (Runtime.IS_LITTLE_ENDIAN) {
285       return (i << 24) | ((i << 8) & 0x00FF0000) | ((i >> 8) & 0x0000FF00) | (i >>> 24);
286     } else {
287       return i;
288     }
289   }
290 
291   private static long adjust(final long i) {
292     if (Runtime.IS_LITTLE_ENDIAN) {
293       return (i << 56)
294               | ((i << 40) & 0x00FF000000000000L)
295               | ((i << 24) & 0x0000FF0000000000L)
296               | ((i << 8) & 0x000000FF00000000L)
297               | ((i >> 8) & 0x00000000FF000000L)
298               | ((i >> 24) & 0x0000000000FF0000L)
299               | ((i >> 40) & 0x000000000000FF00L)
300               | (i >> 56);
301     } else {
302       return i;
303     }
304   }

306   /**
307    * int to long conversion with zero-padding.
308    */
309   private static long to64(final int i) {
310     return i & 0xFFFFFFFFL;
311   }

313   private static String readLine(final RandomAccessFile as, final int p) throws IOException {
314     as.seek(to64(p));
315     ByteArrayOutputStream buf = new ByteArrayOutputStream();
316     int ch;
317     while ((ch = as.read()) > 0) {
318       buf.write(ch);
319     }
320     return buf.toString();
321   }
322 
323   private static String readLine(final FILE as, final long p) {
324     seek64(as, p);
325     Memory m = new Memory(1);
326     ByteArrayOutputStream buf = new ByteArrayOutputStream();
327     while (true) {
328       if (CLibrary.INSTANCE.fread(m, 1, 1, as) == 0) {
329         break;
330       }
331       byte b = m.getByte(0);
332       if (b == 0) {
333         break;
334       }
335       buf.write(b);
336     }
337     return buf.toString();
338   }

340   /**
341    * Mac support
342    *
343    * See http://developer.apple.com/qa/qa2001/qa1123.html http://www.osxfaq.com/man/3/kvm_getprocs.ws
344    * http://matburt.net/?p=16 (libkvm is removed from OSX) where is kinfo_proc?
345    * http://lists.apple.com/archives/xcode-users/2008/Mar/msg00781.html
346    *
347    * This code uses sysctl to get the arg/env list:
348    * http://www.psychofx.com/psi/trac/browser/psi/trunk/src/arch/macosx/macosx_process.c which came from
349    * http://www.opensource.apple.com/darwinsource/10.4.2/top-15/libtop.c
350    *
351    * sysctl is defined in libc.
352    *
353    * PS source code for Mac: http://www.opensource.apple.com/darwinsource/10.4.1/adv_cmds-79.1/ps.tproj/
354    */
355   @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
356   private static JVMArguments ofMac(final int pid) {
357     // local constants
358     final int ctlKern = 1;
359     final int kernArgMax = 8;
360     final int kernProcArgs2 = 49;
361     final int sizeOfInt = Native.getNativeSize(int.class);
362     IntByReference ibf = new IntByReference();
363 
364     IntByReference argmaxRef = new IntByReference(0);
365     IntByReference size = new IntByReference(sizeOfInt);
366 
367     // for some reason, I was never able to get sysctlbyname work.
368 //        if(LIBC.sysctlbyname("kern.argmax", argmaxRef.getPointer(), size, NULL, ibf)!=0)
369     if (CLibrary.INSTANCE.sysctl(new int[]{ctlKern, kernArgMax}, 2, argmaxRef.getPointer(), size, NULL, ibf) != 0) {
370       throw new UnsupportedOperationException("Failed to get kernl.argmax: "
371               + CLibrary.INSTANCE.strerror(Native.getLastError()));
372     }
373 
374     int argmax = argmaxRef.getValue();
375     StringArrayMemory m = new StringArrayMemory(argmax, sizeOfInt);
376     size.setValue(argmax);
377     if (CLibrary.INSTANCE.sysctl(new int[]{ctlKern, kernProcArgs2, resolvePID(pid)}, 3, m, size, NULL, ibf) != 0) {
378       throw new UnsupportedOperationException("Failed to obtain ken.procargs2: "
379               + CLibrary.INSTANCE.strerror(Native.getLastError()));
380     }
381     /*
382          * Make a sysctl() call to get the raw argument space of the
383          * process.  The layout is documented in start.s, which is part
384          * of the Csu project.  In summary, it looks like:
385          *
386          * /---------------\ 0x00000000
387          * :               :
388          * :               :
389          * |---------------|
390          * | argc          |
391          * |---------------|
392          * | arg[0]        |
393          * |---------------|
394          * :               :
395          * :               :
396          * |---------------|
397          * | arg[argc - 1] |
398          * |---------------|
399          * | 0             |
400          * |---------------|
401          * | env[0]        |
402          * |---------------|
403          * :               :
404          * :               :
405          * |---------------|
406          * | env[n]        |
407          * |---------------|
408          * | 0             |
409          * |---------------| <-- Beginning of data returned by sysctl()
410          * | exec_path     |     is here.
411          * |:::::::::::::::|
412          * |               |
413          * | String area.  |
414          * |               |
415          * |---------------| <-- Top of stack.
416          * :               :
417          * :               :
418          * \---------------/ 0xffffffff
419      */
420     int nargs = m.readInt();
421     JVMArguments args = new JVMArguments(nargs);
422     m.readString(); // exec path
423     for (int i = 0; i < nargs; i++) {
424       m.skip0();
425       args.add(m.readString());
426     }
427     return args;
428   }

430   @SuppressFBWarnings("EQ_DOESNT_OVERRIDE_EQUALS")
431   private static final class StringArrayMemory extends Memory {
432 
433     private long offset = 0;
434     private final int sizeOfInt;
435 
436     StringArrayMemory(final long l, final int sizeOfInt) {
437       super(l);
438       this.sizeOfInt = sizeOfInt;
439     }
440 
441     int readInt() {
442       int r = getInt(offset);
443       offset += sizeOfInt;
444       return r;
445     }
446 
447     String readString() {
448       ByteArrayOutputStream baos = new ByteArrayOutputStream();
449       byte ch;
450       while ((ch = getByte(offset++)) != '\0') {
451         baos.write(ch);
452       }
453       return baos.toString();
454     }
455 
456     void skip0() {
457       // skip trailing '\0's
458       while (getByte(offset) == '\0') {
459         offset++;
460       }
461     }
462   }

464   private static JVMArguments ofFreeBSD(final int pid) {
465     // taken from sys/sysctl.h
466     final int ctlKern = 1;
467     final int kernArgMax = 8;
468     final int kernProc = 14;
469     final int kernProcArgs = 7;
470 
471     IntByReference ibr = new IntByReference();
472     IntByReference sysctlArgMax = new IntByReference();
473     IntByReference size = new IntByReference();
474 
475     size.setValue(4);
476     if (CLibrary.INSTANCE.sysctl(new int[]{ctlKern, kernArgMax},
477             2, sysctlArgMax.getPointer(), size, NULL, ibr) != 0) {
478       throw new UnsupportedOperationException("Failed to sysctl kern.argmax");
479     }
480 
481     int argmax = sysctlArgMax.getValue();
482     Memory m = new Memory(argmax);
483     size.setValue(argmax);
484 
485     if (CLibrary.INSTANCE.sysctl(new int[]{ctlKern, kernProc, kernProcArgs, resolvePID(pid)},
486             4, m, size, NULL, ibr) != 0) {
487       throw new UnsupportedOperationException();
488     }
489 
490     ByteArrayOutputStream baos = new ByteArrayOutputStream();
491     ArrayList<String> lArgs = new ArrayList<String>();
492     byte ch;
493     int offset = 0;
494     while (offset < size.getValue()) {
495       while ((ch = m.getByte(offset++)) != '\0') {
496         baos.write(ch);
497       }
498       lArgs.add(baos.toString());
499       baos.reset();
500     }
501 
502     return new JVMArguments(lArgs);
503   }
