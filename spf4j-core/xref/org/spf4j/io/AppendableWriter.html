1   package org.spf4j.io;
2   
3   import edu.umd.cs.findbugs.annotations.CleanupObligation;
4   import edu.umd.cs.findbugs.annotations.DischargesObligation;
5   import java.io.Closeable;
6   import java.io.Flushable;
7   import java.io.IOException;
8   import java.io.Writer;
9   import java.nio.CharBuffer;
10  
11  /**
12   * Utility class to adapt a Appendable to a Writer.
13   * this is a faster version of guava: CharStreams.asWriter
14   * @author zoly
15   */
16  @CleanupObligation
17  public final class AppendableWriter extends Writer {
18  
19    private final Appendable appendable;
20    
21    private final boolean flushable;
22    
23    private boolean closed;
24  
25    public AppendableWriter(final Appendable appendable) {
26      this.appendable = appendable;
27      this.flushable = appendable instanceof Flushable;
28      this.closed = false;
29    }
30  
31    @Override
32    public void write(final char[] cbuf, final int off, final int len) throws IOException {
33      checkNotClosed();
34      /* Guava claims:
35      It turns out that creating a new String is usually as fast,
36      or faster than wrapping cbuf in a light-weight CharSequence.
37      
38      Since I am suspicious of claims that do not make much sense and most developers write rubbish benchmarks
39      (Linus I start to be like you more and more),
40      I wrote a JMH benchmark, and the results are as expected the opposite:
41      
42      Benchmark                                          Mode  Cnt      Score     Error  Units
43      AppendableWriterBenchmark.guavaAppendable         thrpt   10  10731.940 ± 427.258  ops/s
44      AppendableWriterBenchmark.spf4jAppendable         thrpt   10  17613.093 ± 344.769  ops/s
45       
46       Using a light weight wrapper is more than 50% faster!
47      
48       See AppendableWriterBenchmark for more detail...
49      
50      */
51      appendable.append(CharBuffer.wrap(cbuf), off, off + len);
52    }
53  
54    @Override
55    public void write(final int c) throws IOException {
56      checkNotClosed();
57      appendable.append((char) c);
58    }
59  
60    @Override
61    public Writer append(final char c) throws IOException {
62      checkNotClosed();
63      appendable.append(c);
64      return this;
65    }
66  
67    @Override
68    public Writer append(final CharSequence csq, final int start, final int end) throws IOException {
69      checkNotClosed();
70      appendable.append(csq, start, end);
71      return this;
72    }
73  
74    @Override
75    public Writer append(final CharSequence csq) throws IOException {
76      checkNotClosed();
77      appendable.append(csq);
78      return this;
79    }
80  
81    @Override
82    public void write(final String str, final int off, final int len) throws IOException {
83      checkNotClosed();
84      appendable.append(str, off, off + len);
85    }
86  
87    @Override
88    public void write(final String str) throws IOException {
89      appendable.append(str);
90    }
91  
92    @Override
93    public void write(final char[] cbuf) throws IOException {
94      appendable.append(CharBuffer.wrap(cbuf));
95    }
96  
97    @Override
98    public void flush() throws IOException {
99      checkNotClosed();
100     if (flushable) {
101       ((Flushable) appendable).flush();
102     }
103   }
104 
105   private void checkNotClosed() throws IOException {
106     if (closed) {
107       throw new IOException("Cannot write to closed writer " + this);
108     }
109   }
110   
111   @Override
112   @DischargesObligation
113   public void close() throws IOException {
114     if (!closed) {
115       flush();
116       if (appendable instanceof Closeable) {
117         ((Closeable) appendable).close();
118       }
119       closed = true;
120     }
121   }
122 
123   @Override
124   public String toString() {
125     return "AppendableWriter{" + "appendable=" + appendable + ", flushable=" + flushable + ", closed=" + closed + '}';
126   }
127 
128 }
