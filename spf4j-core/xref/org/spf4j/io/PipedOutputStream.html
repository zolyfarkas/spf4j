53  /**
54   * Equivalent to Java piped input/output stream.
55   *
56   * This implementation supports timeouts, timeout are specified by setting in the ExecutionContext
57   *
58   * Implementation supports multiple readers and writers.
59   *
60   * Data is available to readers only after it is flushed. (happens automatically when buffer is full)
61   *
62   * This implementation should be slightly faster than the JDK implementation.
63   *
64   * @author zoly
65   */
66  @ThreadSafe
67  @CleanupObligation
68  @SuppressWarnings("checkstyle:InnerAssignment")
69  public final class PipedOutputStream extends OutputStream {
70  
71    private byte[] buffer;
72  
73    private final Object sync = new Object();
74  
75    private int startIdx;
76    private int endIdx;
77    private int readerPerceivedEndIdx;
78    private boolean writerClosed;
79    private Exception closedException;
80    private int nrReadStreams;
81    private final SizedRecyclingSupplier<byte[]> bufferProvider;
82    private final Long globalDeadlineNanos;
83  
84    public PipedOutputStream() {
85      this(8192);
86    }
87  
88    public PipedOutputStream(final int bufferSize) {
89      this(bufferSize, ArraySuppliers.Bytes.JAVA_NEW);

124   /**
125    * Create a PipedOutputStream.
126    * @param globalDeadlineNanos the deadline relative to System.nanoTime().
127    * @param bufferSize the buffer size in bytes.
128    * @param bufferProvider a buffer provider. (to allow more efficient recycling)
129    */
130   public PipedOutputStream(@Nullable final Long globalDeadlineNanos, final int bufferSize,
131           final SizedRecyclingSupplier<byte[]> bufferProvider) {
132     if (bufferSize < 2) {
133       throw new IllegalArgumentException("Illegal buffer size " + bufferSize);
134     }
135     this.bufferProvider = bufferProvider;
136     buffer = bufferProvider.get(bufferSize);
137     startIdx = 0;
138     endIdx = 0;
139     readerPerceivedEndIdx = 0;
140     writerClosed = false;
141     nrReadStreams = 0;
142     closedException = null;
143     this.globalDeadlineNanos = globalDeadlineNanos;
144   }
145 
146   @Override
147   public void write(final byte[] b, final int off, final int len) throws IOException {
148     long deadline = getNanoDeadline();
149     writeUntil(b, off, len, deadline);
150   }

152   public long getNanoDeadline() {
153     if (globalDeadlineNanos == null) {
154       return ExecutionContexts.getContextDeadlineNanos();
155     } else {
156       return globalDeadlineNanos;
157     }
158   }

160   public void writeUntil(final byte[] b, final int off, final int len, final long deadlineNanos) throws IOException {
161     int bytesWritten = 0;
162     while (bytesWritten < len) {
163       synchronized (sync) {
164         int a2w = 0;
165         while (!writerClosed && (a2w = availableToWrite()) < 1) {
166           long timeToWaitNanos = deadlineNanos - TimeSource.nanoTime();
167           if (timeToWaitNanos <= 0) {
168             throw new IOTimeoutException(deadlineNanos, -timeToWaitNanos);
169           }
170           try {
171             TimeUnit.NANOSECONDS.timedWait(sync, timeToWaitNanos);
172           } catch (InterruptedException ex) {
173             throw new IOException("Interrupted while writing " + Arrays.toString(b), ex);
174           }
175         }
176         if (writerClosed) {
177           throw new IOException("Cannot write, stream closed " + this, closedException);
178         }
179         a2w = Math.min(a2w, len - bytesWritten);
180         int wrToEnd = Math.min(a2w, buffer.length - endIdx);
181         System.arraycopy(b, off + bytesWritten, buffer, endIdx, wrToEnd);
182         endIdx += wrToEnd;
183         bytesWritten += wrToEnd;
184         int wrapArround = a2w - wrToEnd;
185         if (wrapArround > 0) {
186           System.arraycopy(b, off + bytesWritten, buffer, 0, wrapArround);
187           endIdx = wrapArround;
188           bytesWritten += wrapArround;
189         } else if (endIdx >= buffer.length) {
190           endIdx = 0;
191         }
192         if (availableToWrite() < 1) {
193           flush();
194         }
195       }
196     }
197   }

199   @Override
200   public void write(final int b) throws IOException {
201     long deadline = getNanoDeadline();
202     writeUntil(b, deadline);
203   }

205   public void writeUntil(final int b, final long deadlineNanos) throws IOException {
206     synchronized (sync) {
207       int a2w = 0;
208       while (!writerClosed && (a2w = availableToWrite()) < 1) {
209         try {
210           long timeToWaitNanos = deadlineNanos - TimeSource.nanoTime();
211           if (timeToWaitNanos <= 0) {
212             throw new IOTimeoutException(deadlineNanos, -timeToWaitNanos);
213           }
214           TimeUnit.NANOSECONDS.timedWait(sync, timeToWaitNanos);
215         } catch (InterruptedException ex) {
216           throw new IOException("Interrupted while writing " + b, ex);
217         }
218       }
219       if (writerClosed) {
220         throw new IOException("Cannot write stream closed " + this, closedException);
221       }
222       buffer[endIdx++] = (byte) b;
223       if (endIdx >= buffer.length) {
224         endIdx = 0;
225       }
226       if (a2w < 2) {
227         flush();
228       }
229     }
230   }
231 
232   private int availableToWrite() {
233     if (startIdx <= endIdx) {
234       return startIdx + buffer.length - endIdx - 1;
235     } else {
236       return startIdx - endIdx - 1;
237     }
238   }
239 
240   private int availableToRead() {
241     if (startIdx <= readerPerceivedEndIdx) {
242       return readerPerceivedEndIdx - startIdx;
243     } else {
244       return buffer.length - startIdx + readerPerceivedEndIdx;
245     }
246   }
247 
248   private int contentInBuffer() {
249     if (startIdx <= endIdx) {
250       return endIdx - startIdx;
251     } else {
252       return buffer.length - startIdx + endIdx;
253     }
254   }
255 
256   @Override
257   public void flush() {
258     synchronized (sync) {
259       if (readerPerceivedEndIdx != endIdx) {
260         readerPerceivedEndIdx = endIdx;
261         sync.notifyAll();
262       }
263     }
264   }
265 
266   @Override
267   @DischargesObligation
268   public void close() {
269     synchronized (sync) {
270       if (!writerClosed) {
271         try {
272           writerClosed = true;
273           flush();
274         } finally {
275           if (nrReadStreams == 0 && availableToRead() == 0) {
276             bufferProvider.recycle(buffer);
277             buffer = null;
278           }
279           sync.notifyAll();
280         }
281       }
282     }
283   }

286   /**
287    * Close this piped output stream, and provide a exception reason.
288    * Designed to propagate exception details from the consumer to the producer.
289    * A consumer that reads data from a InputStream connected to this PipedOutpuStream,
290    * can close this pipe, and provide a exception. This exception will be used as cause for any
291    * exceptions thrown by subsequent attempted write operations.
292    * @param ex the exception to use as root cause.
293    */
294   @DischargesObligation
295   @SuppressFBWarnings("EI_EXPOSE_REP2")
296   public void close(final Exception ex) {
297     synchronized (sync) {
298       if (closedException != null) {
299         ex.addSuppressed(closedException);
300       }
301       closedException = ex;
302       close();
303     }
304   }

307   public InputStream getInputStream() {
308     synchronized (sync) {
309       if (writerClosed && availableToRead() == 0) {
310         return EmptyInputStream.INSTANCE;
311       }
312       nrReadStreams++;
313       return new PipedInputStream();
314     }

316   }

318   public synchronized byte[] getUnreadBytesFromBuffer() {
319     final int size = contentInBuffer();
320     if (size == 0) {
321       return org.spf4j.base.Arrays.EMPTY_BYTE_ARRAY;
322     }
323     byte[] result = new byte[size];
324     if (startIdx < endIdx) {
325       System.arraycopy(buffer, startIdx, result, 0, result.length);
326     } else {
327       final int toEnd = buffer.length - startIdx;
328       System.arraycopy(buffer, startIdx, result, 0, toEnd);
329       System.arraycopy(buffer, 0, result, toEnd, endIdx);
330     }
331     return result;
332   }

334   @Override
335   public String toString() {
336     synchronized (sync) {
337       if (buffer == null) {
338         return "PipedOutputStream{readers=" + nrReadStreams + ", startIdx=" + startIdx
339                 + ", endIdx=" + endIdx
340                 + ", readerPerceivedEndIdx=" + readerPerceivedEndIdx
341                 + ", closed=" + writerClosed + '}';
342       } else {
343         return "PipedOutputStream{readers=" + nrReadStreams + ", bufferLength="
344                 + buffer.length + ", startIdx=" + startIdx
345                 + ", endIdx=" + endIdx
346                 + ", readerPerceivedEndIdx=" + readerPerceivedEndIdx
347                 + ((writerClosed)
348                         ? ", closed=" + writerClosed
349                         : ", unread=" + BaseEncoding.base64().encode(getUnreadBytesFromBuffer()))
350                 + '}';
351       }
352     }

355   public final class PipedInputStream extends InputStream {
356 
357     private boolean readerClosed = false;
358 
359     private PipedInputStream() {
360     }
361 
362     @SuppressFBWarnings("EI_EXPOSE_REP")
363     public PipedOutputStream getOutputStream() {
364       return PipedOutputStream.this;
365     }
366 
367     @Override
368     public int read() throws IOException {
369       long deadline = getNanoDeadline();
370       return readUntil(deadline);
371     }
372 
373     public int readUntil(final long deadlineNanos) throws IOException {
374       synchronized (sync) {
375         int availableToRead = 0;
376         while (!readerClosed && (availableToRead = availableToRead()) < 1 && !writerClosed) {
377           long timeToWaitNanos = deadlineNanos - TimeSource.nanoTime();
378           if (timeToWaitNanos <= 0) {
379             throw new IOTimeoutException(deadlineNanos, -timeToWaitNanos);
380           }
381           try {
382             TimeUnit.NANOSECONDS.timedWait(sync, timeToWaitNanos);
383           } catch (InterruptedException ex) {
384             throw new IOException("Interrupted while reading from "
385                     + PipedOutputStream.this, ex);
386           }
387         }
388         if (readerClosed) {
389           throw new IOException("Reader is closed for " + PipedOutputStream.this);
390         }
391         if (availableToRead == 0) {
392           if (!writerClosed) {
393             throw new IllegalStateException("Stream must be closed " + PipedOutputStream.this);
394           }
395           return -1;
396         }
397         int result = buffer[startIdx];
398         startIdx++;
399         if (startIdx >= buffer.length) {
400           startIdx = 0;
401         }
402         sync.notifyAll();
403         return result;
404       }
405     }
406 
407     @Override
408     public int read(final byte[] b, final int off, final int len) throws IOException {
409       long deadline = getNanoDeadline();
410       return readUntil(len, b, off, deadline);
411     }
412 
413     public int readUntil(final int len, final byte[] b, final int off, final long deadline)
414             throws IOException {
415       int bytesWritten = 0;
416       synchronized (sync) {
417         int availableToRead = 0;
418         while (!readerClosed && (availableToRead = availableToRead()) < 1 && !writerClosed) {
419           long timeToWaitNanos = deadline - TimeSource.nanoTime();
420           if (timeToWaitNanos <= 0) {
421             throw new IOTimeoutException(deadline, -timeToWaitNanos);
422           }
423           try {
424             TimeUnit.NANOSECONDS.timedWait(sync, timeToWaitNanos);
425           } catch (InterruptedException ex) {
426             throw new IOException("Interrupted while reading from " + PipedOutputStream.this, ex);
427           }
428         }
429         if (readerClosed) {
430           throw new IOException("Reader is closed for " + PipedOutputStream.this);
431         }
432         if (availableToRead == 0) {
433           if (!writerClosed) {
434             throw new IllegalStateException("Stream should be closed, " + PipedOutputStream.this);
435           }
436           return -1;
437         }
438         availableToRead = Math.min(availableToRead, len);
439         int readToEnd = Math.min(availableToRead, buffer.length - startIdx);
440         System.arraycopy(buffer, startIdx, b, off, readToEnd);
441         bytesWritten += readToEnd;
442         startIdx += readToEnd;
443         int remaining = availableToRead - readToEnd;
444         if (remaining > 0) {
445           System.arraycopy(buffer, 0, b, off + readToEnd, remaining);
446           bytesWritten += remaining;
447           startIdx = remaining;
448         } else if (startIdx >= buffer.length) {
449           startIdx = 0;
450         }
451         sync.notifyAll();
452         return bytesWritten;
453       }
454     }
455 
456     @Override
457     public int available() {
458       synchronized (sync) {
459         if (readerClosed) {
460           throw new UncheckedIOException("Reader is closed for " + PipedOutputStream.this, null);
461         }
462         return availableToRead();
463       }
464     }
465 
466     @Override
467     public void close() {
468       synchronized (sync) {
469         nrReadStreams--;
470         readerClosed = true;
471         if (writerClosed && nrReadStreams == 0 && availableToRead() == 0) {
472           bufferProvider.recycle(buffer);
473           buffer = null;
474         }
475         sync.notifyAll();
476       }
477     }
478   }
