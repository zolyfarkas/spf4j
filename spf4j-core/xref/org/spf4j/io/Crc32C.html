595   /**
596    * Return a masked representation of crc.
597    * <p/>
598    * Motivation: it is problematic to compute the CRC of a string that contains embedded CRCs.
599    * Therefore we recommend
600    * that CRCs stored somewhere (e.g., in files) should be masked before being stored.
601    */
602   public static int mask(final int crc) {
603     // Rotate right by 15 bits and add a constant.
604     return ((crc >>> 15) | (crc << 17)) + MASK_DELTA;
605   }
606 
607   /**
608    * Return the crc whose masked representation is masked_crc.
609    */
610   public static int unmask(final int maskedCrc) {
611     int rot = maskedCrc - MASK_DELTA;
612     return ((rot >>> 17) | (rot << 15));
613   }
