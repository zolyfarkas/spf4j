36  /**
37   * Why another buffered input stream?
38   * Main use case if for troubleshooting.
39   * Allows you to get more detail on where your stream processing failed.
40   *
41   * Implementation is a circular byte buffer, where you have 2 sizes to control the behavior:
42   *
43   * buffer size - the total sie of the buffer.
44   * read size - the maximum number of read bytes kept in the buffer.
45   *
46   *
47   * @author zoly
48   */
49  @ParametersAreNonnullByDefault
50  @CleanupObligation
51  public final class MemorizingBufferedInputStream extends FilterInputStream {
52  
53      private byte[] memory;
54  
55      private final  SizedRecyclingSupplier<byte[]> bufferProvider;
56  
57      private final int readSize;
58  
59      private final Charset charset;
60  
61      private int memIdx;
62  
63      private int startIdx;
64  
65      private int endIdx;
66  
67      private boolean isEof;
68  
69      private boolean isClosed;
70  
71      private long readBytes;
72  
73      public MemorizingBufferedInputStream(final InputStream in) {
74          this(in, 16384, 8192, ArraySuppliers.Bytes.GL_SUPPLIER, Charsets.UTF_8);
75      }
76  
77      public MemorizingBufferedInputStream(final InputStream in, final int size) {
78          this(in, size, size / 2, ArraySuppliers.Bytes.GL_SUPPLIER, Charsets.UTF_8);
79      }
80  
81      public MemorizingBufferedInputStream(final InputStream in, final Charset charset) {
82          this(in, 16384, 8192, ArraySuppliers.Bytes.GL_SUPPLIER, charset);
83      }
84  
85      public MemorizingBufferedInputStream(final InputStream in,
86              final int size, final int readSize,
87              final SizedRecyclingSupplier<byte[]> bufferProvider,
88              @Nullable final Charset charset) {
89          super(in);
90          if (readSize > size) {
91              throw new IllegalArgumentException("Read size " + readSize + " cannot be greater than " + size);
92          }
93          if (size < 2) {
94              throw new IllegalArgumentException("Buffer size " + size + " cannot be smaler than 2");
95          }
96          memory = bufferProvider.get(size);
97          this.bufferProvider = bufferProvider;
98          this.charset = charset;
99          this.startIdx = 0;
100         this.endIdx = 0;
101         this.readSize = readSize;
102         this.isEof = false;
103         this.isClosed = false;
104         this.readBytes = 0;
105     }
106 
107     @Override
108     @DischargesObligation
109     public synchronized void close() throws IOException {
110         if (!isClosed) {
111             isClosed = true;
112             bufferProvider.recycle(memory);
113             memory = null;
114             super.close();
115         }
116     }
117 
118     private int availableToWrite() {
119         if (memIdx <= endIdx) {
120             return memIdx + memory.length - endIdx - 1;
121         } else {
122             return memIdx - endIdx - 1;
123         }
124     }
125 
126     private int availableToRead() {
127         if (startIdx <= endIdx) {
128             return endIdx - startIdx;
129         } else {
130             return memory.length - startIdx + endIdx;
131         }
132     }
133 
134     private int availableInMemory() {
135         if (memIdx <= startIdx) {
136             return startIdx - memIdx;
137         } else {
138             return memory.length - memIdx + startIdx;
139         }
140     }
141 
142 
143     public synchronized byte[] getReadBytesFromBuffer() {
144         final int availableInMemory = availableInMemory();
145         if (availableInMemory == 0) {
146             return Arrays.EMPTY_BYTE_ARRAY;
147         }
148         byte[] result = new byte[availableInMemory];
149         if (memIdx < startIdx) {
150             System.arraycopy(memory, memIdx, result, 0, result.length);
151         } else {
152             final int toEnd = memory.length - memIdx;
153             System.arraycopy(memory, memIdx, result, 0, toEnd);
154             System.arraycopy(memory, 0, result, toEnd, startIdx);
155         }
156         return result;
157     }
158 
159     public synchronized byte[] getUnreadBytesFromBuffer() {
160         final int availableToRead = availableToRead();
161         if (availableToRead == 0) {
162             return Arrays.EMPTY_BYTE_ARRAY;
163         }
164         byte[] result = new byte[availableToRead];
165         if (startIdx < endIdx) {
166             System.arraycopy(memory, startIdx, result, 0, result.length);
167         } else {
168             final int toEnd = memory.length - startIdx;
169             System.arraycopy(memory, startIdx, result, 0, toEnd);
170             System.arraycopy(memory, 0, result, toEnd, endIdx);
171         }
172         return result;
173     }
174 
175 
176     private int tryCleanup(final int size) {
177         int canWrite = availableToWrite();
178         if (canWrite < size) {
179             int toFree = size - canWrite;
180             int availableToFree = availableInMemory();
181             if (toFree > availableToFree) {
182                 toFree = availableToFree;
183             }
184             memIdx = memIdx + toFree;
185             if (memIdx >= memory.length) {
186                 memIdx = memIdx - memory.length;
187             }
188             return toFree + canWrite;
189         } else {
190             return size;
191         }
192     }
193 
194     private void fill() throws IOException {
195         int size = tryCleanup(readSize);
196         if (size < readSize) {
197             throw new IllegalStateException("Illegal state " + this);
198         }
199         int canWriteInBulk = Math.min(size, memory.length - endIdx);
200         int read = super.read(memory, endIdx, canWriteInBulk);
201         if (read < 0) {
202             isEof = true;
203             return;
204         }
205         endIdx += read;
206         if (read < canWriteInBulk) {
207             return;
208         }
209         int wrapArround = size - canWriteInBulk;
210         if (wrapArround > 0) {
211             read = super.read(memory, 0, wrapArround);
212             if (read < 0) {
213                 isEof = true;
214                 return;
215             }
216             endIdx = read;
217         } else if (endIdx >= memory.length) {
218             endIdx = 0;
219         }
220     }
221 
222     @Override
223     public synchronized int read(final byte[] b, final int off, final int len) throws IOException {
224         if (isClosed) {
225             throw new IOException("Stream is closed " + this);
226         }
227         int availableToRead = availableToRead();
228         if (availableToRead <= 0) {
229             fill();
230             availableToRead = availableToRead();
231         }
232         if (availableToRead == 0) {
233             if (isEof) {
234                 return -1;
235             } else {
236                 throw new IllegalStateException("State=" + this);
237             }
238         }
239         int toRead = Math.min(availableToRead, len);
240         int readToEnd = Math.min(toRead, memory.length - startIdx);
241         System.arraycopy(memory, startIdx, b, off, readToEnd);
242         startIdx += readToEnd;
243         int wrapArround = toRead - readToEnd;
244         if (wrapArround > 0) {
245             System.arraycopy(memory, 0, b, off + readToEnd, wrapArround);
246             startIdx = wrapArround;
247         } else if (startIdx >= memory.length) {
248             startIdx = 0;
249         }
250         this.readBytes += toRead;
251         return toRead;
252     }
253 
254 
255     @Override
256     public synchronized int read() throws IOException {
257         if (isClosed) {
258             throw new IOException("Stream is closed " + this);
259         }
260         int availableToRead = availableToRead();
261         if (availableToRead <= 0) {
262             fill();
263             availableToRead = availableToRead();
264         }
265         if (availableToRead == 0) {
266             if (isEof) {
267                 return -1;
268             } else {
269                 throw new IllegalStateException("State=" + this);
270             }
271         }
272         int result = memory[startIdx++] & 0xff;
273         if (startIdx >= memory.length) {
274             startIdx = 0;
275         }
276         this.readBytes++;
277         return result;
278     }
279 
280     @Override
281     public synchronized int available() throws IOException {
282         if (isClosed) {
283             throw new IOException("Stream is closed " + this);
284         }
285         return availableToRead();
286     }
287 
288     @Override
289     public synchronized String toString() {
290         StringBuilder result;
291         if (isClosed) {
292            result = new StringBuilder(64);
293         } else {
294            result = new StringBuilder((availableToRead() + availableInMemory()) * 2 + 128);
295         }
296         result.append("MemorizingBufferedInputStream{\n");
297         if (isClosed) {
298             result.append("closed=true\n");
299         } else if (charset == null) {
300             final BaseEncoding base64 = BaseEncoding.base64();
301             result.append("readBytes=\"").append(base64.encode(getReadBytesFromBuffer())).append("\",\n");
302             result.append("unreadBytes=\"").append(base64.encode(getUnreadBytesFromBuffer())).append("\"\n");
303         } else {
304             result.append("readStr=\"").append(
305                     Strings.fromUtf8(getReadBytesFromBuffer())).append("\",\n");
306             result.append("unreadStr=\"").append(
307                     Strings.fromUtf8(getUnreadBytesFromBuffer())).append("\"\n");
308         }
309         result.append("memIdx=").append(this.memIdx).append("\"\n");
310         result.append("startIdx=").append(this.startIdx).append("\"\n");
311         result.append("endIdx=").append(this.endIdx).append("\"\n");
312         result.append('}');
313         return result.toString();
314     }
315 
316     public synchronized long getReadBytes() {
317         return readBytes;
318     }
319 
320     @Override
321     public synchronized long skip(final long n) throws IOException {
322         long nrSkipped = 0;
323         for (int i = 0; i < n; i++) {
324             int read = read();
325             if (read < 0) {
326                 break;
327             } else {
328                 nrSkipped++;
329             }
330         }
331         return nrSkipped;
332     }
