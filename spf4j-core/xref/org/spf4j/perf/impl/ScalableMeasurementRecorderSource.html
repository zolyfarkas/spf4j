18  package org.spf4j.perf.impl;
19  
20  import java.io.Closeable;
21  import java.io.IOException;
22  import java.util.HashMap;
23  import java.util.Map;
24  import java.util.concurrent.ScheduledFuture;
25  import javax.annotation.concurrent.ThreadSafe;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  import org.spf4j.base.AbstractRunnable;
29  import org.spf4j.concurrent.DefaultScheduler;
30  import org.spf4j.base.Pair;
31  import org.spf4j.perf.EntityMeasurements;
32  import org.spf4j.perf.EntityMeasurementsSource;
33  import org.spf4j.perf.MeasurementStore;
34  import org.spf4j.perf.MeasurementProcessor;
35  import org.spf4j.perf.MeasurementRecorder;
36  import org.spf4j.perf.MeasurementRecorderSource;
37  
38  @ThreadSafe
39  // a recorder instance is tipically alive for the entire life of the process
40  @edu.umd.cs.findbugs.annotations.SuppressWarnings("PMB_INSTANCE_BASED_THREAD_LOCAL")
41  public final class ScalableMeasurementRecorderSource implements
42          MeasurementRecorderSource, EntityMeasurementsSource, Closeable {
43  
44      private static final Logger LOG = LoggerFactory.getLogger(ScalableMeasurementRecorderSource.class);
45  
46      
47      private final Map<Thread, Map<Object, MeasurementProcessor>> measurementProcessorMap;
48      
49      private final ThreadLocal<Map<Object, MeasurementProcessor>> threadLocalMeasurementProcessorMap;
50      
51  
52      private final ScheduledFuture<?> samplingFuture;
53      private final MeasurementProcessor processorTemplate;
54      
55      ScalableMeasurementRecorderSource(final MeasurementProcessor processor,
56              final int sampleTimeMillis, final MeasurementStore database) {
57          this.processorTemplate = processor;
58          measurementProcessorMap = new HashMap<Thread, Map<Object, MeasurementProcessor>>();
59          threadLocalMeasurementProcessorMap = new ThreadLocal<Map<Object, MeasurementProcessor>>() {
60  
61              @Override
62              protected Map<Object, MeasurementProcessor> initialValue() {
63                  Map<Object, MeasurementProcessor> result = new HashMap<Object, MeasurementProcessor>();
64                  synchronized (measurementProcessorMap) {
65                      measurementProcessorMap.put(Thread.currentThread(), result);
66                  }
67                  return result;
68              }
69              
70          };
71          final AbstractRunnable persister = new AbstractRunnable(true) {
72              private volatile long lastRun = 0;
73  
74              @Override
75              public void doRun() throws IOException {
76                  long currentTime = System.currentTimeMillis();
77                  if (currentTime > lastRun) {
78                      lastRun = currentTime;
79                      for (EntityMeasurements m
80                              : ScalableMeasurementRecorderSource.this.getEntitiesMeasurementsAndReset().values()) {
81                          database.saveMeasurements(
82                                  m.getInfo(), currentTime, sampleTimeMillis, m.getMeasurementsAndReset());
83                      }
84                  } else {
85                      LOG.warn("Last measurement recording was at {} current run is {}, something is wrong",
86                              lastRun, currentTime);
87                  }
88              }
89          };
90          samplingFuture = DefaultScheduler.scheduleAllignedAtFixedRateMillis(persister, sampleTimeMillis);
91                  org.spf4j.base.Runtime.addHookAtBeginning(new AbstractRunnable(true) {
92              
93              @Override
94              public void doRun() throws Exception {
95                  persister.doRun();
96                  close();
97              }
98          });
99      }
100     
101     @Override
102     public MeasurementRecorder getRecorder(final Object forWhat) {
103         Map<Object, MeasurementProcessor> recorders = threadLocalMeasurementProcessorMap.get();
104         synchronized (recorders) {
105             MeasurementProcessor result = recorders.get(forWhat);
106             if (result == null)  {
107                 result = (MeasurementProcessor) processorTemplate.createLike(
108                         Pair.of(processorTemplate.getInfo().getMeasuredEntity(), forWhat));
109                 recorders.put(forWhat, result);
110             }
111             return result;
112         }
113     }
114 
115     @Override
116     public Map<Object, EntityMeasurements> getEntitiesMeasurements() {
117         Map<Object, EntityMeasurements> result = new HashMap<Object, EntityMeasurements>();
118         
119         synchronized (measurementProcessorMap) {
120             for (Map.Entry<Thread, Map<Object, MeasurementProcessor>> entry : measurementProcessorMap.entrySet()) {
121                 
122                 Map<Object, MeasurementProcessor> measurements = entry.getValue();
123                 synchronized (measurements) {
124                     for (Map.Entry<Object, MeasurementProcessor> lentry : measurements.entrySet()) {
125 
126                         Object what = lentry.getKey();
127                         EntityMeasurements existingMeasurement = result.get(what);
128                         if (existingMeasurement == null) {
129                             existingMeasurement = lentry.getValue().createClone();
130                         } else {
131                             existingMeasurement = existingMeasurement.aggregate(lentry.getValue().createClone());
132                         }
133                         result.put(what, existingMeasurement);
134                     }
135                 }
136             }
137         }
138         return result;
139     }
140 
141     @Override
142     public Map<Object, EntityMeasurements> getEntitiesMeasurementsAndReset() {
143         Map<Object, EntityMeasurements> result = new HashMap<Object, EntityMeasurements>();
144         
145         synchronized (measurementProcessorMap) {
146             for (Map.Entry<Thread, Map<Object, MeasurementProcessor>> entry : measurementProcessorMap.entrySet()) {
147                 
148                 Map<Object, MeasurementProcessor> measurements = entry.getValue();
149                 synchronized (measurements) {
150                     for (Map.Entry<Object, MeasurementProcessor> lentry : measurements.entrySet()) {
151 
152                         Object what = lentry.getKey();
153                         EntityMeasurements existingMeasurement = result.get(what);
154                         if (existingMeasurement == null) {
155                             existingMeasurement = lentry.getValue().reset();
156                         } else {
157                             existingMeasurement = existingMeasurement.aggregate(lentry.getValue().reset());
158                         }
159                         result.put(what, existingMeasurement);
160                     }
161                 }
162             }
163         }
164         return result;
165     }
166     
167     
168     
169     @Override
170     public void close() {
171         samplingFuture.cancel(false);
172     }
173     
174 }
