69    public static final MeasurementStore MEASUREMENT_STORE;
70  
71    static {
72      MeasurementStore mStore;
73      try {
74        mStore = buildStoreFromConfig(System.getProperty("spf4j.perf.ms.config", null));
75      } catch (IOException | ObjectCreationException ex) {
76        LOG.error("Cannot initialize measurement store, installing NOP store", ex);
77        mStore = new NopMeasurementStore();
78      }
79      if (!(mStore instanceof NopMeasurementStore)) {
80        Flusher.flushEvery(Integer.getInteger("spf4j.perf.ms.flushIntervalMillis", 60000), mStore);
81      }
82      MEASUREMENT_STORE = mStore;
83    }

86    }

89      return MEASUREMENT_STORE;

92    /**
93     * Configuration is a coma separated list of stores:
94     * TSDB@/path/to/file.tsdb,TSDB_TXT@/path/to/file.tsdbtxt,GRAPHITE_UDP@1.1.1.1:8080,GRAPHITE_TCP@1.1.1.1:8080
95     *
96     * @param configuration
97     * @return a measurement store.
98     */
99    @Nonnull
100   @SuppressFBWarnings("PATH_TRAVERSAL_IN") // the config is not supplied by a user.
101   private static MeasurementStore buildStoreFromConfig(@Nullable final String configuration)
102           throws IOException, ObjectCreationException {
103 
104     if (configuration == null || configuration.trim().isEmpty()) {
105       return new TSDBMeasurementStore(new File(
106               System.getProperty("spf4j.perf.ms.defaultTsdbFolderPath",
107                       System.getProperty("java.io.tmpdir"))
108               + File.separator
109               + CharSequences.validatedFileName(System.getProperty("spf4j.perf.ms.defaultTsdbFileNamePrefix",
110                       ManagementFactory.getRuntimeMXBean().getName() + ".tsdb2"))));
111     }
112 
113     List<String> stores;
114     try {
115       stores = Csv.readRow(new StringReader(configuration));
116     } catch (CsvParseException ex) {
117       throw new IllegalArgumentException("Invalid configuration " + configuration, ex);
118     }
119     final int size = stores.size();
120     if (size == 1) {
121       return fromString(stores.get(0));
122     } else {
123       MeasurementStore[] mstores = new MeasurementStore[size];
124       int i = 0;
125       for (String config : stores) {
126         mstores[i] = fromString(config);
127         i++;
128       }
129       return new MultiStore(mstores);
130     }
131   }
132 
133   public static MeasurementStore fromString(final String string) throws IOException, ObjectCreationException {
134     int atIdx = string.indexOf('@');
135     final int length = string.length();
136     if (atIdx < 0) {
137       atIdx = length;
138     }
139     StoreType type = StoreType.valueOf(string.substring(0, atIdx));
140     if (atIdx >= length) {
141       return type.create("");
142     } else {
143       return type.create(string.substring(atIdx + 1));
144     }
145   }

147   /**
148    * Create a Quantized Measurement recorder. (see concept at http://dtrace.org/blogs/bmc/2011/02/08/llquantize/ )
149    *
150    * A quantized measurement recorder is appropriate for low overhead measurement recording with good detail
151    * using quantized data resolution.
152    * For lower overhead recorder see createScalableMinMaxAvgRecorder.
153    *
154    * example : createScalableQuantizedRecorder("response time", "ms", 60000, 10, -3, 3, 5)
155    * will aggregate and persist measurements every minute.
156    * will have the following measurement buckets:
157    *
158    * QNI_-1000, Q-1000_-800, Q-800_-600, Q-600_-400, Q-400_-100, Q-100_-80, Q-80_-60, Q-60_-40, Q-40_-10, Q-10_-8,
159    * Q-8_-6, Q-6_-4, Q-4_-2, Q-2_0, Q0_2, Q2_4, Q4_6,
160    * Q6_8, Q8_10, Q10_40, Q40_60, Q60_80, Q80_100, Q100_400, Q400_600, Q600_800, Q800_1000, Q1000_PI
161    *
162    * where A_B equivalent Math notation is [A,B)
163    * where NI = negative infinity
164    * where PI is positive infinity
165    *
166    * @param forWhat an object identifying what is being measured, ex: "response time"
167    * @param unitOfMeasurement the unit of measurement of the measurements, ex "milliseconds"
168    * @param sampleTimeMillis the sampling (accumulating interval) ex: 60000 for minute level detail.
169    * @param factor the log factor of the magnitudes, ex: 10 for 0-1,1-10,10-100,100 - 1000 magnitudes.
170    * @param lowerMagnitude the lowest magnitude. ex: 10 for 10 * factor min bucket limit value.
171    * @param higherMagnitude th highest magnitude. ex 10 for 10 * factor max bucket limit value.
172    * @param quantasPerMagnitude number of equally divided measurement buckets per magnitude. ex: 10
173    * @return a measurement recorder that
174    */
175   public static MeasurementRecorder createScalableQuantizedRecorder(
176           final Object forWhat, final String unitOfMeasurement, final int sampleTimeMillis,
177           final int factor, final int lowerMagnitude,
178           final int higherMagnitude, final int quantasPerMagnitude) {
179     ScalableMeasurementRecorder mr = new ScalableMeasurementRecorder(new QuantizedAccumulator(forWhat, "",
180             unitOfMeasurement, factor, lowerMagnitude, higherMagnitude,
181             quantasPerMagnitude), sampleTimeMillis, MEASUREMENT_STORE, true);
182     mr.registerJmx();
183     return mr;
184   }
