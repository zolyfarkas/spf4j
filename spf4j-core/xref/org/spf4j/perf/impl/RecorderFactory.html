77    public static final MeasurementStore MEASUREMENT_STORE;
78  
79    static {
80      MeasurementStore mStore;
81      try {
82        mStore = buildStoreFromConfig(System.getProperty("spf4j.perf.ms.config", null));
83      } catch (IOException | ObjectCreationException ex) {
84        LOG.error("Cannot initialize measurement store, installing NOP store", ex);
85        mStore = new NopMeasurementStore();
86      }
87      if (!(mStore instanceof NopMeasurementStore) && Boolean.getBoolean("spf4j.perf.ms.periodicFlush")) {
88        Flusher.flushEvery(Integer.getInteger("spf4j.perf.ms.flushIntervalMillis", 60000), mStore);
89      }
90      MEASUREMENT_STORE = mStore;
91    }
92

94    }

97      return MEASUREMENT_STORE;

100   /**
101    * Configuration is a coma separated list of stores:
102    * TSDB@/path/to/file.tsdb,TSDB_TXT@/path/to/file.tsdbtxt,GRAPHITE_UDP@1.1.1.1:8080,GRAPHITE_TCP@1.1.1.1:8080
103    *
104    * @param configuration
105    * @return a measurement store.
106    */
107   @Nonnull
108   @SuppressFBWarnings("PATH_TRAVERSAL_IN") // the config is not supplied by a user.
109   private static MeasurementStore buildStoreFromConfig(@Nullable final String configuration)
110           throws IOException, ObjectCreationException {
111 
112     if (configuration == null || configuration.trim().isEmpty()) {
113       return new AvroMeasurementStore(Paths.get(System.getProperty("spf4j.perf.ms.defaultTsdbFolderPath",
114                       System.getProperty("java.io.tmpdir"))),
115               CharSequences.validatedFileName(System.getProperty("spf4j.perf.ms.defaultTsdbFileNamePrefix",
116                       ManagementFactory.getRuntimeMXBean().getName())));
117     }
118 
119     List<String> stores;
120     try {
121       stores = Csv.readRow(new StringReader(configuration));
122     } catch (CsvParseException ex) {
123       throw new IllegalArgumentException("Invalid configuration " + configuration, ex);
124     }
125     final int size = stores.size();
126     if (size == 1) {
127       return StoreType.fromString(stores.get(0));
128     } else {
129       MeasurementStore[] mstores = new MeasurementStore[size];
130       int i = 0;
131       for (String config : stores) {
132         mstores[i] = StoreType.fromString(config);
133         i++;
134       }
135       return new MultiStore(mstores);
136     }
137   }

139   /**
140    * Create a Quantized Measurement recorder. (see concept at http://dtrace.org/blogs/bmc/2011/02/08/llquantize/ )
141    *
142    * A quantized measurement recorder is appropriate for low overhead measurement recording with good detail
143    * using quantized data resolution.
144    * For lower overhead recorder see createScalableMinMaxAvgRecorder.
145    *
146    * example : createScalableQuantizedRecorder("response time", "ms", 60000, 10, -3, 3, 5)
147    * will aggregate and persist measurements every minute.
148    * will have the following measurement buckets:
149    *
150    * QNI_-1000, Q-1000_-800, Q-800_-600, Q-600_-400, Q-400_-100, Q-100_-80, Q-80_-60, Q-60_-40, Q-40_-10, Q-10_-8,
151    * Q-8_-6, Q-6_-4, Q-4_-2, Q-2_0, Q0_2, Q2_4, Q4_6,
152    * Q6_8, Q8_10, Q10_40, Q40_60, Q60_80, Q80_100, Q100_400, Q400_600, Q600_800, Q800_1000, Q1000_PI
153    *
154    * where A_B equivalent Math notation is [A,B)
155    * where NI = negative infinity
156    * where PI is positive infinity
157    *
158    * @param forWhat an object identifying what is being measured, ex: "response time"
159    * @param unitOfMeasurement the unit of measurement of the measurements, ex "milliseconds"
160    * @param sampleTimeMillis the sampling (accumulating interval) ex: 60000 for minute level detail.
161    * @param factor the log factor of the magnitudes, ex: 10 for 0-1,1-10,10-100,100 - 1000 magnitudes.
162    * @param lowerMagnitude the lowest magnitude. ex: 10 for 10 * factor min bucket limit value.
163    * @param higherMagnitude th highest magnitude. ex 10 for 10 * factor max bucket limit value.
164    * @param quantasPerMagnitude number of equally divided measurement buckets per magnitude. ex: 10
165    * @return a measurement recorder that
166    */
167   public static MeasurementRecorder createScalableQuantizedRecorder(
168           final Object forWhat, final String unitOfMeasurement, final int sampleTimeMillis,
169           final int factor, final int lowerMagnitude,
170           final int higherMagnitude, final int quantasPerMagnitude) {
171     ScalableMeasurementRecorder mr = new ScalableMeasurementRecorder(new QuantizedAccumulator(forWhat, "",
172             unitOfMeasurement, factor, lowerMagnitude, higherMagnitude,
173             quantasPerMagnitude), sampleTimeMillis, MEASUREMENT_STORE, true);
174     mr.registerJmx();
175     return mr;
176   }

264   public static MultiMeasurementRecorder createDirectRecorder(final Object measuredEntity, final String description,
265           final String[] measurementNames, final String[] measurementUnits) {
266     Aggregation[] aggs = new Aggregation[measurementNames.length];
267     Arrays.fill(aggs, Aggregation.UNKNOWN);
268     DirectStoreMultiAccumulator mr = new DirectStoreMultiAccumulator(
269             new MeasurementsInfoImpl(measuredEntity, description,
270                     measurementNames, measurementUnits, aggs, MeasurementType.UNTYPED), MEASUREMENT_STORE);
271     mr.registerJmx();
272     return mr;
273   }
