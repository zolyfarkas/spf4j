18  package org.spf4j.ds;
19  
20  import java.util.HashMap;
21  import java.util.Map;
22  import javax.annotation.Nonnull;
23  import javax.annotation.ParametersAreNonnullByDefault;
24  import org.spf4j.base.Pair;
25  
26  /**
27   *
28   * @author zoly
29   */
30  @ParametersAreNonnullByDefault
31  public final class VertexEdges<V, E> extends Pair<Map<E, V>, Map<E, V>> {
32  
33      VertexEdges(final Map<E, V> first, final Map<E, V> second) {
34          super(first, second);
35      }
36      
37      public VertexEdges() {
38          super(new HashMap<E, V>(2), new HashMap<E, V>(2));
39      }
40    
41      @Nonnull
42      public Map<E, V> getIncomming() {
43          return first;
44      }
45      
46  
47      @Nonnull
48      public Map<E, V> getOutgoing() {
49          return second;
50      }
51      
52      public VertexEdges copy() {
53          return new VertexEdges(new HashMap<E, V>(first), new HashMap<E, V>(second));
54      }
55  
56      
57  }
