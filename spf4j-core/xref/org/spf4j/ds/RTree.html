45      /**
46       * Creates a new RTree.
47       *
48       * @param maxEntries maximum number of entries per node
49       * @param minEntries minimum number of entries per node (except for the root
50       * node)
51       * @param numDims the number of dimensions of the RTree.
52       */
53      public RTree(final int maxEntries, final int minEntries, final int numDims,
54              final SeedPicker seedPicker) {
55          assert (minEntries <= (maxEntries / 2));
56          this.numDims = numDims;
57          this.maxEntries = maxEntries;
58          this.minEntries = minEntries;
59          this.seedPicker = seedPicker;
60          pointDims = new float[numDims];
61          root = buildRoot(true);
62      }

65          this(maxEntries, minEntries, numDims, SeedPicker.LINEAR);

68      private static final float DIM_FACTOR = -2.0f;
69      private static final float FUDGE_FACTOR = 1.001f;
70      private static final float INIT_COORD = (float) Math.sqrt(Float.MAX_VALUE);

72      private Node buildRoot(final boolean asLeaf) {
73          float[] initCoords = new float[numDims];
74          float[] initDimensions = new float[numDims];
75          for (int i = 0; i < this.numDims; i++) {
76              initCoords[i] = INIT_COORD;
77              initDimensions[i] = DIM_FACTOR * INIT_COORD;
78          }
79          return new Node(initCoords, initDimensions, asLeaf);
80      }

124     /**
125      * Searches the RTree for objects overlapping with the given rectangle.
126      *
127      * @param coords the corner of the rectangle that is the lower bound of
128      * every dimension (eg. the top-left corner)
129      * @param dimensions the dimensions of the rectangle.
130      * @return a list of objects whose rectangles overlap with the given
131      * rectangle.
132      */
133     @Nonnull
134     public List<T> search(final float[] coords, final float[] dimensions) {
135         assert (coords.length == numDims);
136         assert (dimensions.length == numDims);
137         LinkedList<T> results = new LinkedList<T>();
138         search(coords, dimensions, root, results);
139         return results;
140     }
141 
142     private void search(final float[] coords, final float[] dimensions, final Node n,
143             final LinkedList<T> results) {
144         if (n.leaf) {
145             for (Node e : n.children) {
146                 if (isOverlap(coords, dimensions, e.coords, e.dimensions)) {
147                     results.add(((Entry<T>) e).entry);
148                 }
149             }
150         } else {
151             for (Node c : n.children) {
152                 if (isOverlap(coords, dimensions, c.coords, c.dimensions)) {
153                     search(coords, dimensions, c, results);
154                 }
155             }
156         }
157     }

159     /**
160      * Deletes the entry associated with the given rectangle from the RTree
161      *
162      * @param coords the corner of the rectangle that is the lower bound in
163      * every dimension
164      * @param dimensions the dimensions of the rectangle
165      * @param entry the entry to delete
166      * @return true iff the entry was deleted from the RTree.
167      */
168     public boolean delete(final float[] coords, final float[] dimensions, final T entry) {
169         assert (coords.length == numDims);
170         assert (dimensions.length == numDims);
171         Node l = findLeaf(root, coords, dimensions, entry);
172         if (l == null) {
173             throw new RuntimeException("leaf not found for entry " + entry);
174         }
175         ListIterator<Node> li = l.children.listIterator();
176         T removed = null;
177         while (li.hasNext()) {
178             @SuppressWarnings("unchecked")
179             Entry<T> e = (Entry<T>) li.next();
180             if (e.entry.equals(entry)) {
181                 removed = e.entry;
182                 li.remove();
183                 break;
184             }
185         }
186         if (removed != null) {
187             condenseTree(l);
188             size--;
189         }
190         if (size == 0) {
191             root = buildRoot(true);
192         }
193         return (removed != null);
194     }
195 
196     public boolean delete(final float[] coords, final T entry) {
197         return delete(coords, pointDims, entry);
198     }

200     private Node findLeaf(final Node n, final float[] coords,
201             final float[] dimensions, final T entry) {
202         if (n.leaf) {
203             for (Node c : n.children) {
204                 if (((Entry) c).entry.equals(entry)) {
205                     return n;
206                 }
207             }
208             return null;
209         } else {
210             for (Node c : n.children) {
211                 if (isOverlap(c.coords, c.dimensions, coords, dimensions)) {
212                     Node result = findLeaf(c, coords, dimensions, entry);
213                     if (result != null) {
214                         return result;
215                     }
216                 }
217             }
218             return null;
219         }
220     }

222     private void condenseTree(final Node pn) {
223         Node n = pn;
224         Set<Node> q = new HashSet<Node>();
225         while (n != root) {
226             if (n.leaf && (n.children.size() < minEntries)) {
227                 q.addAll(n.children);
228                 n.parent.children.remove(n);
229             } else if (!n.leaf && (n.children.size() < minEntries)) {
230                 // probably a more efficient way to do this...
231                 LinkedList<Node> toVisit = new LinkedList<Node>(n.children);
232                 while (!toVisit.isEmpty()) {
233                     Node c = toVisit.pop();
234                     if (c.leaf) {
235                         q.addAll(c.children);
236                     } else {
237                         toVisit.addAll(c.children);
238                     }
239                 }
240                 n.parent.children.remove(n);
241             } else {
242                 tighten(n);
243             }
244             n = n.parent;
245         }
246         if (root.children.size() == 0) {
247             root = buildRoot(true);
248         } else if ((root.children.size() == 1) && (!root.leaf)) {
249             root = root.children.get(0);
250             root.parent = null;
251         } else {
252             tighten(root);
253         }
254         for (Node ne : q) {
255             @SuppressWarnings("unchecked")
256             Entry<T> e = (Entry<T>) ne;
257             insert(e.coords, e.dimensions, e.entry);
258         }
259         size -= q.size();
260     }

270     /**
271      * Inserts the given entry into the RTree, associated with the given
272      * rectangle.
273      *
274      * @param coords the corner of the rectangle that is the lower bound in
275      * every dimension
276      * @param dimensions the dimensions of the rectangle
277      * @param entry the entry to insert
278      */
279     public void insert(final float[] coords, final float[] dimensions, final T entry) {
280         assert (coords.length == numDims);
281         assert (dimensions.length == numDims);
282         Entry e = new Entry(coords, dimensions, entry);
283         Node l = chooseLeaf(root, e);
284         l.children.add(e);
285         size++;
286         e.parent = l;
287         if (l.children.size() > maxEntries) {
288             Node[] splits = splitNode(l);
289             adjustTree(splits[0], splits[1]);
290         } else {
291             adjustTree(l, null);
292         }
293     }

302         insert(coords, pointDims, entry);

305     private void adjustTree(final Node n, final Node nn) {
306         if (n == root) {
307             if (nn != null) {
308                 // build new root and add children.
309                 root = buildRoot(false);
310                 root.children.add(n);
311                 n.parent = root;
312                 root.children.add(nn);
313                 nn.parent = root;
314             }
315             tighten(root);
316             return;
317         }
318         tighten(n);
319         if (nn != null) {
320             tighten(nn);
321             if (n.parent.children.size() > maxEntries) {
322                 Node[] splits = splitNode(n.parent);
323                 adjustTree(splits[0], splits[1]);
324             }
325         }
326         if (n.parent != null) {
327             adjustTree(n.parent, null);
328         }
329     }

331     private Node[] splitNode(final Node n) {
332         // TODO: this class probably calls "tighten" a little too often.
333         // For instance the call at the end of the "while (!cc.isEmpty())" loop
334         // could be modified and inlined because it's only adjusting for the addition
335         // of a single node.  Left as-is for now for readability.
336         @SuppressWarnings("unchecked")
337         Node[] nn = new RTree.Node[]{n, new Node(n.coords, n.dimensions, n.leaf)};
338         final Node nn1 = nn[1];
339         nn1.parent = n.parent;
340         if (nn1.parent != null) {
341             nn1.parent.children.add(nn1);
342         }
343         LinkedList<Node> cc = new LinkedList<Node>(n.children);
344         n.children.clear();
345         Node[] ss = seedPicker == SeedPicker.LINEAR ? lPickSeeds(cc) : qPickSeeds(cc);
346         final Node nn0 = nn[0];
347         nn0.children.add(ss[0]);
348         nn1.children.add(ss[1]);
349         tighten(nn);
350         while (!cc.isEmpty()) {
351             final int size0 = nn0.children.size();
352             final int size1 = nn1.children.size();
353             if ((size0 >= minEntries)
354                     && (size1 + cc.size() == minEntries)) {
355                 nn1.children.addAll(cc);
356                 cc.clear();
357                 tighten(nn); // Not sure this is required.
358                 return nn;
359             } else if ((size1 >= minEntries)
360                     && (size0 + cc.size() == minEntries)) {
361                 nn0.children.addAll(cc);
362                 cc.clear();
363                 tighten(nn); // Not sure this is required.
364                 return nn;
365             }
366             Node c = seedPicker == SeedPicker.LINEAR ? lPickNext(cc) : qPickNext(cc, nn);
367             Node preferred;
368             float e0 = getRequiredExpansion(nn0.coords, nn0.dimensions, c);
369             float e1 = getRequiredExpansion(nn1.coords, nn1.dimensions, c);
370             if (e0 < e1) {
371                 preferred = nn0;
372             } else if (e0 > e1) {
373                 preferred = nn1;
374             } else {
375                 float a0 = getArea(nn0.dimensions);
376                 float a1 = getArea(nn1.dimensions);
377                 if (a0 < a1) {
378                     preferred = nn0;
379                 } else if (e0 > a1) {
380                     preferred = nn1;
381                 } else {
382                     if (size0 < size1) {
383                         preferred = nn0;
384                     } else if (size0 > size1) {
385                         preferred = nn1;
386                     } else {
387                         preferred = nn[(int) Math.round(Math.random())];
388                     }
389                 }
390             }
391             preferred.children.add(c);
392             tighten(preferred);
393         }
394         return nn;
395     }

397     // Implementation of Quadratic PickSeeds
398     @SuppressFBWarnings("MRC_METHOD_RETURNS_CONSTANT")
399     private Node[] qPickSeeds(final List<Node> nn) {
400         @SuppressWarnings("unchecked")
401         Node[] bestPair = new Node[2];
402         float maxWaste = -1.0f * Float.MAX_VALUE;
403         for (Node n1 : nn) {
404             for (Node n2 : nn) {
405                 if (n1 == n2) {
406                     continue;
407                 }
408                 float n1a = getArea(n1.dimensions);
409                 float n2a = getArea(n2.dimensions);
410                 float ja = 1.0f;
411                 for (int i = 0; i < numDims; i++) {
412                     final float n1coordsi = n1.coords[i];
413                     final float n2coordsi = n2.coords[i];
414                     float jc0 = Math.min(n1coordsi, n2coordsi);
415                     float jc1 = Math.max(n1coordsi + n1.dimensions[i], n2coordsi + n2.dimensions[i]);
416                     ja *= (jc1 - jc0);
417                 }
418                 float waste = ja - n1a - n2a;
419                 if (waste > maxWaste) {
420                     maxWaste = waste;
421                     bestPair[0] = n1;
422                     bestPair[1] = n2;
423                 }
424             }
425         }
426         nn.remove(bestPair[0]);
427         nn.remove(bestPair[1]);
428         return bestPair;
429     }

431     /**
432      * Implementation of QuadraticPickNext
433      *
434      * @param cc the children to be divided between the new nodes, one item will
435      * be removed from this list.
436      * @param nn the candidate nodes for the children to be added to.
437      */
438     private Node qPickNext(final LinkedList<Node> cc, final Node[] nn) {
439         float maxDiff = -1.0f * Float.MAX_VALUE;
440         Node nextC = null;
441         for (Node c : cc) {
442             float n0Exp = getRequiredExpansion(nn[0].coords, nn[0].dimensions, c);
443             float n1Exp = getRequiredExpansion(nn[1].coords, nn[1].dimensions, c);
444             float diff = Math.abs(n1Exp - n0Exp);
445             if (diff > maxDiff) {
446                 maxDiff = diff;
447                 nextC = c;
448             }
449         }
450         assert (nextC != null) : "No node selected from qPickNext";
451         cc.remove(nextC);
452         return nextC;
453     }

455     // Implementation of LinearPickSeeds
456     private Node[] lPickSeeds(final LinkedList<Node> nn) {
457         @SuppressWarnings("unchecked")
458         Node[] bestPair = new RTree.Node[2];
459         boolean foundBestPair = false;
460         float bestSep = 0.0f;
461         for (int i = 0; i < numDims; i++) {
462             float dimLb = Float.MAX_VALUE, dimMinUb = Float.MAX_VALUE;
463             float dimUb = -1.0f * Float.MAX_VALUE, dimMaxLb = -1.0f * Float.MAX_VALUE;
464             Node nMaxLb = null, nMinUb = null;
465             for (Node n : nn) {
466                 if (n.coords[i] < dimLb) {
467                     dimLb = n.coords[i];
468                 }
469                 if (n.dimensions[i] + n.coords[i] > dimUb) {
470                     dimUb = n.dimensions[i] + n.coords[i];
471                 }
472                 if (n.coords[i] > dimMaxLb) {
473                     dimMaxLb = n.coords[i];
474                     nMaxLb = n;
475                 }
476                 if (n.dimensions[i] + n.coords[i] < dimMinUb) {
477                     dimMinUb = n.dimensions[i] + n.coords[i];
478                     nMinUb = n;
479                 }
480             }
481             float sep = (nMaxLb == nMinUb) ? -1.0f
482                     : Math.abs((dimMinUb - dimMaxLb) / (dimUb - dimLb));
483             if (sep >= bestSep) {
484                 bestPair[0] = nMaxLb;
485                 bestPair[1] = nMinUb;
486                 bestSep = sep;
487                 foundBestPair = true;
488             }
489         }
490         // In the degenerate case where all points are the same, the above
491         // algorithm does not find a best pair.  Just pick the first 2
492         // children.
493         if (!foundBestPair) {
494             bestPair = new RTree.Node[]{nn.get(0), nn.get(1)};
495         }
496         nn.remove(bestPair[0]);
497         nn.remove(bestPair[1]);
498         return bestPair;
499     }

511     private void tighten(final Node... nodes) {
512         assert (nodes.length >= 1) : "Pass some nodes to tighten!";
513         for (Node n : nodes) {
514             assert (n.children.size() > 0) : "tighten() called on empty node!";
515             float[] minCoords = new float[numDims];
516             float[] maxCoords = new float[numDims];
517             for (int i = 0; i < numDims; i++) {
518                 minCoords[i] = Float.MAX_VALUE;
519                 maxCoords[i] = Float.MIN_VALUE;
520 
521                 for (Node c : n.children) {
522                     // we may have bulk-added a bunch of children to a node (eg. in
523                     // splitNode)
524                     // so here we just enforce the child->parent relationship.
525                     c.parent = n;
526                     if (c.coords[i] < minCoords[i]) {
527                         minCoords[i] = c.coords[i];
528                     }
529                     if ((c.coords[i] + c.dimensions[i]) > maxCoords[i]) {
530                         maxCoords[i] = (c.coords[i] + c.dimensions[i]);
531                     }
532                 }
533             }
534             for (int i = 0; i < numDims; i++) {
535                 // Convert max coords to dimensions
536                 maxCoords[i] -= minCoords[i];
537             }
538             System.arraycopy(minCoords, 0, n.coords, 0, numDims);
539             System.arraycopy(maxCoords, 0, n.dimensions, 0, numDims);
540         }
541     }

543     private Node chooseLeaf(final Node n, final Entry<T> e) {
544         if (n.leaf) {
545             return n;
546         }
547         float minInc = Float.MAX_VALUE;
548         Node next = null;
549         for (Node c : n.children) {
550             float inc = getRequiredExpansion(c.coords, c.dimensions, e);
551             if (inc < minInc) {
552                 minInc = inc;
553                 next = c;
554             } else if (inc == minInc) {
555                 float curArea = 1.0f;
556                 float thisArea = 1.0f;
557                 for (int i = 0; i < c.dimensions.length; i++) {
558                     if (next == null) {
559                         throw new IllegalStateException("Illegal state at " + c);
560                     }
561                     curArea *= next.dimensions[i];
562                     thisArea *= c.dimensions[i];
563                 }
564                 if (thisArea < curArea) {
565                     next = c;
566                 }
567             }
568         }
569         return chooseLeaf(next, e);
570     }

572     /**
573      * Returns the increase in area necessary for the given rectangle to cover
574      * the given entry.
575      */
576     private float getRequiredExpansion(final float[] coords, final float[] dimensions, final Node e) {
577         float area = getArea(dimensions);
578         float[] deltas = new float[dimensions.length];
579         for (int i = 0; i < deltas.length; i++) {
580             if (coords[i] + dimensions[i] < e.coords[i] + e.dimensions[i]) {
581                 deltas[i] = e.coords[i] + e.dimensions[i] - coords[i] - dimensions[i];
582             } else if (coords[i] + dimensions[i] > e.coords[i] + e.dimensions[i]) {
583                 deltas[i] = coords[i] - e.coords[i];
584             }
585         }
586         float expanded = 1.0f;
587         for (int i = 0; i < dimensions.length; i++) {
588             area *= dimensions[i] + deltas[i];
589         }
590         return (expanded - area);
591     }
592 
593     private float getArea(final float[] dimensions) {
594         float area = 1.0f;
595         for (int i = 0; i < dimensions.length; i++) {
596             area *= dimensions[i];
597         }
598         return area;
599     }

601     private boolean isOverlap(final float[] scoords, final float[] sdimensions,
602             final float[] coords, final float[] dimensions) {
603 
604         for (int i = 0; i < scoords.length; i++) {
605             boolean overlapInThisDimension = false;
606             if (scoords[i] == coords[i]) {
607                 overlapInThisDimension = true;
608             } else if (scoords[i] < coords[i]) {
609                 if (scoords[i] + FUDGE_FACTOR * sdimensions[i] >= coords[i]) {
610                     overlapInThisDimension = true;
611                 }
612             } else if (scoords[i] > coords[i]) {
613                 if (coords[i] + FUDGE_FACTOR * dimensions[i] >= scoords[i]) {
614                     overlapInThisDimension = true;
615                 }
616             }
617             if (!overlapInThisDimension) {
618                 return false;
619             }
620         }
621         return true;
622     }

672         pw.println("<html><head></head><body>");
673         visualize(root, pw, 0, 0, ubWidth, ubDepth);
674         pw.println("</body>");
675         pw.flush();
676         return sw.toString();
677     }

679     private void visualize(final Node n, final java.io.PrintWriter pw, final int x0,
680             final int y0, final int w, final int h) {
681         pw.printf("<div style=\"position:absolute; left: %d; top: %d; width: %d; height: %d; border: 1px dashed\">%n",
682                 x0, y0, w, h);
683         pw.println("<pre>");
684         pw.println("Node: " + n + " (root==" + (n == root) + ") \n");
685         pw.println("Coords: " + Arrays.toString(n.coords) + '\n');
686         pw.println("Dimensions: " + Arrays.toString(n.dimensions) + '\n');
687         pw.println("# Children: " + ((n.children == null) ? 0 : n.children.size()) + '\n');
688         pw.println("isLeaf: " + n.leaf + '\n');
689         pw.println("</pre>");
690         int numChildren = (n.children == null) ? 0 : n.children.size();
691         for (int i = 0; i < numChildren; i++) {
692             visualize(n.children.get(i), pw, (int) (x0 + (i * w / (float) numChildren)),
693                     y0 + ELEM_HEIGHT, (int) (w / (float) numChildren), h - ELEM_HEIGHT);
694         }
695         pw.println("</div>");
696     }
