53  /**
54   * Token bucket algorithm base call rate limiter. see https://en.wikipedia.org/wiki/Token_bucket for more detail.
55   * Unlike the Guava implementation, the token replenishment is done in a separate thread.
56   * As such permit acquisition methods have lower overhead
57   * (no System.nanotime invocation is made when permits available),
58   * This lower overhead comes at the cost of increasing the cost of RateLimiter object instance. (a scheduled runnable)
59   * This implementation also uses CAS to update the permit bucket which should yield better concurrency characteristics.
60   * there is a backoff to handle high contention better as well
61   * (see https://dzone.com/articles/wanna-get-faster-wait-bit)
62   *
63   * The different in performance can be observed in a benchmark where we try to limit 10000000 ops/s:
64   *
65   * Benchmark                                                                 Mode  Cnt        Score        Error  Units
66   * com.google.common.util.concurrent.GuavaRateLimiterBenchmark.acquire      thrpt   10  8197739.576 ± 163437.214  ops/s
67   * org.spf4j.failsafe.Spf4jRateLimiterBenchmark.acquire                     thrpt   10  9791489.540 ±  89480.010  ops/s
68   *
69   * The Guava implementation cannot really get as close to the desired rate as the spf4j implementation.
70   *
71   * The spf4j implementation tops of at about:
72   *
73   * Benchmark                               Mode  Cnt         Score        Error  Units
74   * Spf4jRateLimiterBenchmark.acquire      thrpt   10  15270385.653 ± 507217.444  ops/s
75   *
76   * these benchmarks are artificial and really measure the overhead of permit acquisition alone without doing anything
77   * else, real use cases will accompany a permit acquision with some work which impacts the CAS/lock acquisition perf...
78   *
79   * Rate Limiter also implements the PermitSupplier abstraction along with GuavaRateLimiter allowing interchangeability
80   * between the 2 implementations based on what trade-of work better for you.
81   * PermitSupplier allows interchangeability and combination with Semaphore (extends PermitSupplier) implementations.
82   *
83   * @author Zoltan Farkas
84   */
85  @Beta

179   /**
180    * create the rate limiter.
181    *
182    * @param maxAvailablePermits the maximum number of permits that can accumulate.
183    * @param minReplenishInterval the minimum replenish interval, since a scheduler replenishes the tokens asynchronously
184    * there is a limit that is relative to the system clock resolution.
185    * @param tu the time unit of the minimum replenish interval.
186    * @param scheduler the scheduler to use to replenish the bucket.
187    * @param nanoTimeSupplier the time supplier.
188    * @param maxBackoffNanos expected concurrency level
189    */
190   public RateLimiter(final long permitsPerReplenishInterval,
191           final Duration replenishmentInterval,
192           final long initialNrOfPermits,
193           final long maxAvailablePermits,
194           final long minReplenishInterval,
195           final TimeUnit tu,
196           final ScheduledExecutorService scheduler,
197           final LongSupplier nanoTimeSupplier,
198           final int maxBackoffNanos) {
199     this.maxBackoffNanos = maxBackoffNanos;
200     this.sync = new Object();
201     this.nanoTimeSupplier = nanoTimeSupplier;
202     this.permitReplenishIntervalNanos = replenishmentInterval.toNanos();
203     this.permitsPerReplenishInterval = permitsPerReplenishInterval;
204     assert permitsPerReplenishInterval >= 1;
205     if (maxAvailablePermits < permitsPerReplenishInterval) {
206       throw new IllegalArgumentException("Invalid max burst size: " + maxAvailablePermits
207               + ",  increase maxBurstSize to something larger than " + permitsPerReplenishInterval
208               + " we assume a clock resolution of " + permitReplenishIntervalNanos
209               + " and that is the minimum replenish interval");
210     }
211     this.permits = new AtomicLong(initialNrOfPermits);
212     lastReplenishmentNanos = nanoTimeSupplier.getAsLong();
213     accumulate = (long left, long right) -> {
214       long result = left + right;
215       return (result > maxAvailablePermits) ? maxAvailablePermits : result;
216     };
217     this.replenisher = scheduler.scheduleAtFixedRate(() -> {
218       synchronized (sync) {
219         Atomics.accumulate(permits, permitsPerReplenishInterval, accumulate, maxBackoffNanos);
220         lastReplenishmentNanos = nanoTimeSupplier.getAsLong();
221       }
222     }, permitReplenishIntervalNanos, permitReplenishIntervalNanos, TimeUnit.NANOSECONDS);
223   }
224 
225   @Override
226   public boolean addPermits(final int nrPermits) {
227     return Atomics.maybeAccumulate(permits, nrPermits, accumulate, maxBackoffNanos);
228   }

239   public boolean tryAcquire(final int nrPermits) {
240     return Atomics.maybeAccumulate(permits, nrPermits, (long prev, long x) -> {
241       long dif = prev - x;
242       // right will be -1d.
243       return (dif < 0) ? prev : dif;
244     }, maxBackoffNanos);
245   }

247   private final class ReservationHandler implements LongBinaryOperator, Acquisition {
248 
249     private long currTimeNanos;
250 
251     private final long deadlineNanos;
252 
253     private long nsUntilResourcesAvailable;
254 
255     private int reTimeCount;
256 
257     private boolean isSuccess;
258 
259     ReservationHandler(final long currTimeNanos, final long deadlineNanos) {
260       this.currTimeNanos = currTimeNanos;
261       this.deadlineNanos = deadlineNanos;
262       this.nsUntilResourcesAvailable = 0;
263       this.reTimeCount = RE_READ_TIME_AFTER_RETRIES;
264       this.isSuccess = false;
265     }
266 
267     @Override
268     public long applyAsLong(final long prev, final long nrPermits) {
269       long remaimingPermits = prev - nrPermits;
270       if (remaimingPermits >= 0) {
271         isSuccess = true;
272         return remaimingPermits;
273       }
274       if (reTimeCount > 0) {
275         reTimeCount--;
276       } else {
277         // re-reading current time on every try is pointless since System.nanotime can take 30ns to exec...
278         reTimeCount = RE_READ_TIME_AFTER_RETRIES;
279         currTimeNanos = nanoTimeSupplier.getAsLong();
280       }
281       long timeoutNs = deadlineNanos - currTimeNanos;
282       long nsUntilNextReplenishment
283               = permitReplenishIntervalNanos - (currTimeNanos - lastReplenishmentNanos);
284       long permitsNeeded = -remaimingPermits;
285       if (permitsNeeded <= permitsPerReplenishInterval) {
286         nsUntilResourcesAvailable = nsUntilNextReplenishment;
287         return remaimingPermits;
288       } else {
289         int numberOfReplenishMentsNeed = permitsNeeded % permitsPerReplenishInterval == 0
290                 ? (int) (permitsNeeded / permitsPerReplenishInterval)
291                 : (int) (permitsNeeded / permitsPerReplenishInterval + 1);
292         long nsNeeded = nsUntilNextReplenishment
293                 + (numberOfReplenishMentsNeed - 1) * permitReplenishIntervalNanos;
294         nsUntilResourcesAvailable = nsNeeded;
295         if (nsNeeded <= timeoutNs) {
296           isSuccess = true;
297           return remaimingPermits; // make reservation.
298         } else {
299           return prev;
300         }
301       }
302     }
303 
304     public long getNsUntilResourcesAvailable() {
305       return nsUntilResourcesAvailable;
306     }
307 
308     @Override
309     public String toString() {
310       return "ReservationHandler{" + "deadlineNanos=" + deadlineNanos + ", permits=" + permits
311               + ", nsUntilResourcesAvailable=" + nsUntilResourcesAvailable + '}';
312     }
313 
314     @Override
315     public boolean isSuccess() {
316       return isSuccess;
317     }
318 
319     @Override
320     public long permitAvailableEstimateInNanos() {
321       return nsUntilResourcesAvailable;
322     }
323   }

329   @Override
330   @SuppressFBWarnings("MDM_THREAD_YIELD") //fb has a point here...
331   public boolean tryAcquire(final int nrPermits, final long deadlineNanos) throws InterruptedException {
332     Acquisition acq = tryAcquireGetDelayNanos(nrPermits, deadlineNanos);
333     if (acq.isSuccess()) {
334       long permitAvailableEstimateInNanos = acq.permitAvailableEstimateInNanos();
335       if (permitAvailableEstimateInNanos > 0) {
336         TimeUnit.NANOSECONDS.sleep(permitAvailableEstimateInNanos);
337       }
338       return true;
339     } else {
340       return false;
341     }
342   }

345   @CheckReturnValue
346   @Override
347   public boolean tryAcquire(@Nonnegative final int nrPermits, @Nonnegative final long timeout, final TimeUnit unit)
348           throws InterruptedException {
349     if (timeout < 0) {
350       throw new IllegalArgumentException("incalid timeout " + timeout + ' ' + unit);
351     }
352     boolean tryAcquire = tryAcquire(nrPermits);
353     if (tryAcquire) {
354       return true;
355     }
356     if (timeout == 0) {
357       return false;
358     }
359     Acquisition acq = forceReserve(ExecutionContexts.computeDeadline(timeout, unit), nrPermits);
360     if (acq.isSuccess()) {
361       long permitAvailableEstimateInNanos = acq.permitAvailableEstimateInNanos();
362       if (permitAvailableEstimateInNanos > 0) {
363         TimeUnit.NANOSECONDS.sleep(permitAvailableEstimateInNanos);
364       }
365       return true;
366     } else {
367       return false;
368     }
369   }

371   @Override
372   public Acquisition tryAcquireEx(final int nrPermits, final long timeout, final TimeUnit unit)
373           throws InterruptedException {
374     boolean tryAcquire = tryAcquire(nrPermits);
375     if (tryAcquire) {
376       return Acquisition.SUCCESS;
377     }
378     Acquisition acq =  forceReserve(ExecutionContexts.computeDeadline(timeout, unit), nrPermits);
379     if (acq.isSuccess()) {
380       long permitAvailableEstimateInNanos = acq.permitAvailableEstimateInNanos();
381       if (permitAvailableEstimateInNanos > 0) {
382         TimeUnit.NANOSECONDS.sleep(permitAvailableEstimateInNanos);
383       }
384     }
385     return acq;
386   }

393       long permitAvailableEstimateInNanos = acq.permitAvailableEstimateInNanos();
394       if (permitAvailableEstimateInNanos > 0) {

396       }
397     }

401   /**
402    * @param nrPermits nr of permits to acquire
403    * @param timeout the maximum time to wait to acquire the permits.
404    * @param unit the unit of time to wait.
405    * @return negative value if cannot acquire number of permits within time, or if positive it is the amount of ms
406    * we can wait to act of the acquired permissions.
407    * the user of this method needs to be trusted, since it can violate the contract.
408    * @throws InterruptedException
409    */
410   @Signed
411   Acquisition tryAcquireGetDelayNanos(final int nrPermits, final long deadlineNanos) {
412     boolean tryAcquire = tryAcquire(nrPermits);
413     if (tryAcquire) {
414       return Acquisition.SUCCESS;
415     } else {
416       return forceReserve(deadlineNanos, nrPermits);
417     }
418   }
419 
420   private ReservationHandler forceReserve(final long deadlineNanos, final int nrPermits) {
421     // no curent permits available, reserve the slots and get the wait time
422     if (replenisher.isCancelled()) {
423       throw new IllegalStateException("RateLimiter is closed: " + this);
424     }
425     ReservationHandler rh = new ReservationHandler(nanoTimeSupplier.getAsLong(), deadlineNanos);
426     synchronized (sync) {
427       Atomics.accumulate(permits, nrPermits, rh, maxBackoffNanos);
428     }
429     return rh;
430   }
