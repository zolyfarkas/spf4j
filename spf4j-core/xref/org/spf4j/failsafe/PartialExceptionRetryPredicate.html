16  package org.spf4j.failsafe;
17  
18  import java.util.concurrent.Callable;
19  import org.spf4j.base.Callables;
20  
21  /**
22   *
23   * @author Zoltan Farkas
24   */
25  @FunctionalInterface
26  public interface PartialExceptionRetryPredicate<T, C extends Callable<? extends T>>
27          extends PartialTypedExceptionRetryPredicate<T, C, Throwable> {
28  
29    /**
30     * @deprecated use this method to migrate from deprecated API to new APIs (failsafe)
31     */
32    @Deprecated
33    static PartialExceptionRetryPredicate<?, ? extends Callable<?>> from(
34            final Callables.AdvancedRetryPredicate<Exception> oldStyle) {
35      return new PartialExceptionRetryPredicate<Object, Callable<? extends Object>>() {
36        @Override
37        public org.spf4j.failsafe.RetryDecision getExceptionDecision(final Throwable value, final Callable what) {
38          RetryDecision abort = org.spf4j.failsafe.RetryDecision.abort();
39          if (!(value instanceof Exception)) {
40            return abort;
41          }
42          Callables.AdvancedAction aa = oldStyle.apply((Exception) value);
43          switch (aa) {
44            case ABORT:
45              return abort;
46            case RETRY:
47            case RETRY_DELAYED:
48              return org.spf4j.failsafe.RetryDecision.retryDefault(what);
49            case RETRY_IMMEDIATE:
50              return org.spf4j.failsafe.RetryDecision.retry(0, what);
51            default:
52              throw new IllegalStateException("Invalid enum value " + aa);
53          }
54        }
55      };
56    }
