92      /**
93       * initializes the Object set of this hash table.
94       *
95       * @param initialCapacity an <code>int</code> value
96       * @return an <code>int</code> value
97       */
98      public int setUp(int initialCapacity) {
99          int capacity;
100         capacity = super.setUp(initialCapacity);
101         if (capacity == 0) {
102           _set = org.spf4j.base.Arrays.EMPTY_OBJ_ARRAY;
103           return 0;
104         } else {
105           _set = new Object[capacity];
106         }
107         Arrays.fill(_set, FREE);
108         return capacity;
109     }

112     /**
113      * Executes <tt>procedure</tt> for each element in the set.
114      *
115      * @param procedure a <code>TObjectProcedure</code> value
116      * @return false if the loop over the set terminated because
117      *         the procedure returned false for some value.
118      */
119     @SuppressWarnings({"unchecked"})
120     public boolean forEach(TObjectProcedure<? super T> procedure) {
121         Object[] set = _set;
122         for (int i = set.length; i-- > 0;) {
123             if (set[i] != FREE
124                     && set[i] != REMOVED
125                     && !procedure.execute((T) set[i])) {
126                 return false;
127             }
128         }
129         return true;
130     }

145     /**
146      * Locates the index of <tt>obj</tt>.
147      *
148      * @param obj an <code>Object</code> value
149      * @return the index of <tt>obj</tt> or -1 if it isn't in the set.
150      */
151     protected int index(Object obj) {
152         if (obj == null)
153             return indexForNull();
154 
155         if (_set.length == 0) {
156           return -1;
157         }
158 
159         // From here on we know obj to be non-null
160         final int hash = hash(obj) & 0x7fffffff;
161         int index = hash % _set.length;
162         Object cur = _set[index];
163 
164 
165         if (cur == FREE) {
166             return -1;
167         }
168 
169         if (cur == obj || equals(obj, cur)) {
170             return index;
171         }
172 
173         return indexRehashed(obj, index, hash, cur);
174     }
175 
176     /**
177      * Locates the index of non-null <tt>obj</tt>.
178      *
179      * @param obj   target key, know to be non-null
180      * @param index we start from
181      * @param hash
182      * @param cur
183      * @return
184      */
185     private int indexRehashed(Object obj, int index, int hash, Object cur) {
186         final Object[] set = _set;
187         final int length = set.length;
188 
189         // NOTE: here it has to be REMOVED or FULL (some user-given value)
190         // see Knuth, p. 529
191         int probe = 1 + (hash % (length - 2));
192 
193         final int loopIndex = index;
194 
195         do {
196             index -= probe;
197             if (index < 0) {
198                 index += length;
199             }
200             cur = set[index];
201             //
202             if (cur == FREE)
203                 return -1;
204 
205             //
206             if ((cur == obj || equals(obj, cur)))
207                 return index;
208         } while (index != loopIndex);
209 
210         return -1;
211     }

213     /**
214      * Locates the index <tt>null</tt>.
215      * <p/>
216      * null specific loop exploiting several properties to simplify the iteration logic
217      * - the null value hashes to 0 we so we can iterate from the beginning.
218      * - the probe value is 1 for this case
219      * - object identity can be used to match this case
220      * <p/>
221      * --> this result a simpler loop
222      *
223      * @return
224      */
225     private int indexForNull() {
226         int index = 0;
227         for (Object o : _set) {
228             if (o == null)
229                 return index;
230 
231             if (o == FREE)
232                 return -1;
233 
234             index++;
235         }
236 
237         return -1;
238     }

253     /**
254      * Locates the index at which <tt>key</tt> can be inserted.  if
255      * there is already a value equal()ing <tt>key</tt> in the set,
256      * returns that value's index as <tt>-index - 1</tt>.
257      * <p/>
258      * If a slot is found the value is inserted. When a FREE slot is used the consumeFreeSlot field is
259      * set to true. This field should be used in the method invoking insertKey() to pass to postInsertHook()
260      *
261      * @param key an <code>Object</code> value
262      * @return the index of a FREE slot at which key can be inserted
263      *         or, if key is already stored in the hash, the negative value of
264      *         that index, minus 1: -index -1.
265      */
266     protected int insertKey(T key) {
267         if (key == null)
268             return insertKeyForNull();
269 
270         final int hash = hash(key) & 0x7fffffff;
271         int index = hash % _set.length;
272         Object cur = _set[index];
273 
274         if (cur == FREE) {
275             _set[index] = key;  // insert value
276             return index;       // empty, all done
277         }
278 
279         if (cur == key || equals(key, cur)) {
280             return -index - 1;   // already stored
281         }
282 
283         return insertKeyRehash(key, index, hash, cur);
284     }

286     /**
287      * Looks for a slot using double hashing for a non-null key values and inserts the value
288      * in the slot
289      *
290      * @param key   non-null key value
291      * @param index natural index
292      * @param hash
293      * @param cur   value of first matched slot
294      * @return
295      */
296     private int insertKeyRehash(T key, int index, int hash, Object cur) {
297         final Object[] set = _set;
298         final int length = set.length;
299         // already FULL or REMOVED, must probe
300         // compute the double hash
301         final int probe = 1 + (hash % (length - 2));
302 
303         final int loopIndex = index;
304         int firstRemoved = -1;
305 
306         /**
307          * Look until FREE slot or we start to loop
308          */
309         do {
310             // Identify first removed slot
311             if (cur == REMOVED && firstRemoved == -1)
312                 firstRemoved = index;
313 
314             index -= probe;
315             if (index < 0) {
316                 index += length;
317             }
318             cur = set[index];
319 
320             // A FREE slot stops the search
321             if (cur == FREE) {
322                 if (firstRemoved != -1) {
323                     _set[firstRemoved] = key;
324                     return firstRemoved;
325                 } else {
326                     _set[index] = key;  // insert value
327                     return index;
328                 }
329             }
330 
331             if (cur == key || equals(key, cur)) {
332                 return -index - 1;
333             }
334 
335             // Detect loop
336         } while (index != loopIndex);
337 
338         // We inspected all reachable slots and did not find a FREE one
339         // If we found a REMOVED slot we return the first one found
340         if (firstRemoved != -1) {
341             _set[firstRemoved] = key;
342             return firstRemoved;
343         }
344 
345         // Can a resizing strategy be found that resizes the set?
346         throw new IllegalStateException("No free or removed slots available. Key set full?!!");
347     }

349     /**
350      * Looks for a slot using double hashing for a null key value and inserts the value.
351      * <p/>
352      * null specific loop exploiting several properties to simplify the iteration logic
353      * - the null value hashes to 0 we so we can iterate from the beginning.
354      * - the probe value is 1 for this case
355      * - object identity can be used to match this case
356      *
357      * @return
358      */
359     private int insertKeyForNull() {
360         int index = 0;
361         int firstRemoved = -1;
362 
363         // Look for a slot containing the 'null' value as key
364         for (Object o : _set) {
365             // Locate first removed
366             if (o == REMOVED && firstRemoved == -1)
367                 firstRemoved = index;
368 
369             if (o == FREE) {
370                 if (firstRemoved != -1) {
371                     _set[firstRemoved] = null;
372                     return firstRemoved;
373                 } else {
374                     _set[index] = null;  // insert value
375                     return index;
376                 }
377             }
378 
379             if (o == null) {
380                 return -index - 1;
381             }
382 
383             index++;
384         }
385 
386         // We inspected all reachable slots and did not find a FREE one
387         // If we found a REMOVED slot we return the first one found
388         if (firstRemoved != -1) {
389             _set[firstRemoved] = null;
390             return firstRemoved;
391         }
392 
393         // We scanned the entire key set and found nothing, is set full?
394         // Can a resizing strategy be found that resizes the set?
395         throw new IllegalStateException("Could not find insertion index for null key. Key set full!?!!");
396     }

433

438     /**
439      * Convenience methods for subclasses to use in throwing exceptions about
440      * badly behaved user objects employed as keys.  We have to throw an
441      * IllegalArgumentException with a rather verbose message telling the
442      * user that they need to fix their object implementation to conform
443      * to the general contract for java.lang.Object.
444      *
445      *
446      * @param o1 the first of the equal elements with unequal hash codes.
447      * @param o2 the second of the equal elements with unequal hash codes.
448      * @throws IllegalArgumentException the whole point of this method.
449      */
450     protected final IllegalArgumentException buildObjectContractViolation(Object o1, Object o2, String extra ) {
451         return new IllegalArgumentException("Equal objects must have equal hashcodes. " +
452                 "During rehashing, Trove discovered that the following two objects claim " +
453                 "to be equal (as in java.lang.Object.equals()) but their hashCodes (or " +
454                 "those calculated by your TObjectHashingStrategy) are not equal." +
455                 "This violates the general contract of java.lang.Object.hashCode().  See " +
456                 "bullet point two in that method's documentation. object #1 =" + objectInfo(o1) +
457                 "; object #2 =" + objectInfo(o2) + "\n" + extra);
458     }

472     protected static String reportPotentialConcurrentMod(int newSize, int oldSize) {
473         // Note that we would not be able to detect concurrent paired of put()-remove()
474         // operations with this simple check
475         if (newSize != oldSize)
476             return "[Warning] apparent concurrent modification of the key set. " +
477                     "Size before and after rehash() do not match " + oldSize + " vs " + newSize;
478 
479         return "";
480     }

482     /**
483      *
484      * @param newVal the key being inserted
485      * @param oldVal the key already stored at that position
486      * @param currentSize size of the key set during rehashing
487      * @param oldSize size of the key set before rehashing
488      * @param oldKeys the old key set
489      */
490     protected String dumpExtraInfo(Object newVal, Object oldVal, int currentSize, int oldSize, Object[] oldKeys) {
491         StringBuilder b = new StringBuilder();
492         //
493         b.append(dumpKeyTypes(newVal, oldVal));
494 
495         b.append(reportPotentialConcurrentMod(currentSize, oldSize));
496         b.append(detectKeyLoss(oldKeys, oldSize));
497 
498         // Is de same object already present? Double insert?
499         if (newVal == oldVal) {
500             b.append("Inserting same object twice, rehashing bug. Object= ").append(oldVal);
501         }
502 
503         return b.toString();
504     }

506     /**
507      * Detect inconsistent hashCode() and/or equals() methods
508      *
509      * @param keys
510      * @param oldSize
511      * @return
512      */
513     private static String detectKeyLoss(Object[] keys, int oldSize) {
514         StringBuilder buf = new StringBuilder();
515         Set<Object> k = makeKeySet(keys);
516         if (k.size() != oldSize) {
517             buf.append("\nhashCode() and/or equals() have inconsistent implementation");
518             buf.append("\nKey set lost entries, now got ").append(k.size()).append(" instead of ").append(oldSize);
519             buf.append(". This can manifest itself as an apparent duplicate key.");
520         }
521 
522         return buf.toString();
523     }

525     private static Set<Object> makeKeySet(Object[] keys) {
526         Set<Object> types = new HashSet<Object>();
527         for (Object o : keys) {
528             if (o != FREE && o != REMOVED) {
529                     types.add(o);
530             }
531         }
532 
533         return types;
534     }

536     private static String equalsSymmetryInfo(Object a, Object b) {
537         StringBuilder buf = new StringBuilder();
538         if (a == b) {
539             return  "a == b";
540         }
541 
542         if (a.getClass() != b.getClass()) {
543             buf.append("Class of objects differ a=").append(a.getClass()).append(" vs b=").append(b.getClass());
544 
545             boolean aEb = a.equals(b);
546             boolean bEa = b.equals(a);
547             if (aEb != bEa) {
548                 buf.append("\nequals() of a or b object are asymmetric");
549                 buf.append("\na.equals(b) =").append(aEb);
550                 buf.append("\nb.equals(a) =").append(bEa);
551             }
552         }
553 
554         return buf.toString();
555     }

557     protected static String objectInfo(Object o) {
558         return (o == null ? "class null" : o.getClass()) + " id= " + System.identityHashCode(o)
559                 + " hashCode= " + (o == null ? 0 : o.hashCode()) + " toString= " + String.valueOf(o);
560     }

562     private String dumpKeyTypes(Object newVal, Object oldVal) {
563         StringBuilder buf = new StringBuilder();
564         Set<Class<?>> types = new HashSet<Class<?>>();
565         for (Object o : _set) {
566             if (o != FREE && o != REMOVED) {
567                 if (o != null)
568                     types.add(o.getClass());
569                 else
570                     types.add(null);
571             }
572         }
573 
574         if (types.size() > 1) {
575             buf.append("\nMore than one type used for keys. Watch out for asymmetric equals(). " +
576                     "Read about the 'Liskov substitution principle' and the implications for equals() in java.");
577 
578             buf.append("\nKey types: ").append(types);
579             buf.append(equalsSymmetryInfo(newVal, oldVal));
580         }
581 
582         return buf.toString();
583     }
