18  package org.spf4j.stackmonitor;
19  
20  import com.google.common.annotations.Beta;
21  import com.google.common.base.Predicate;
22  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
23  import java.util.Arrays;
24  import java.util.concurrent.ConcurrentHashMap;
25  import java.util.concurrent.ConcurrentMap;
26  import java.util.function.Consumer;
27  import javax.annotation.Nullable;
28  
29  @Beta
30  public final class ObservableStackCollector extends AbstractStackCollector {
31  
32    private final Predicate<Thread> threadFilter;
33  
34    private final ConcurrentMap<Thread, Consumer<StackTraceElement[]>> stConsumers;
35  
36    public ObservableStackCollector(final boolean collectForMain, final String... xtraIgnoredThreads) {
37      this(FastStackCollector.createNameBasedFilter(collectForMain, xtraIgnoredThreads));
38    }
39  
40    public ObservableStackCollector(final Predicate<Thread> threadFilter) {
41      this.threadFilter = threadFilter;
42      this.stConsumers = new ConcurrentHashMap<>();
43    }
44  
45    @Nullable
46    public Consumer<StackTraceElement[]> registerConsumer(final Thread thread,
47            final Consumer<StackTraceElement[]> consumer) {
48      return stConsumers.put(thread, consumer);
49    }
50  
51    public Consumer<StackTraceElement[]> unregisterConsumer(final Thread thread) {
52      return stConsumers.remove(thread);
53    }
54  
55    private Thread[] requestFor = new Thread[]{};
56  
57    @Override
58    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CHECKED")
59    public void sample(final Thread ignore) {
60      Thread[] threads = FastStackCollector.getThreads();
61      final int nrThreads = threads.length;
62      if (requestFor.length < nrThreads) {
63        requestFor = new Thread[nrThreads - 1];
64      }
65      int j = 0;
66      for (int i = 0; i < nrThreads; i++) {
67        Thread th = threads[i];
68        if (ignore != th && !threadFilter.apply(th)) { // not interested in these traces
69          requestFor[j++] = th;
70        }
71      }
72      Arrays.fill(requestFor, j, requestFor.length, null);
73      StackTraceElement[][] stackDump = FastStackCollector.getStackTraces(requestFor);
74      for (int i = 0; i < j; i++) {
75        Thread t = requestFor[i];
76        StackTraceElement[] stackTrace = stackDump[i];
77        Consumer<StackTraceElement[]> consumer = this.stConsumers.get(t);
78        if (consumer != null) {
79          consumer.accept(stackTrace);
80        }
81        if (stackTrace != null && stackTrace.length > 0) {
82          addSample(stackTrace);
83        } else {
84          addSample(new StackTraceElement[]{
85            new StackTraceElement("Thread", t.getName(), "", 0)
86          });
87        }
88      }
89    }
90  
91  }
