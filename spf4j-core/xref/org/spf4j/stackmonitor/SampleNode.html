45      public SampleNode(final StackTraceElement[] stackTrace, final int from) {
46          sampleCount = 1;
47          if (from >= 0) {
48              subNodes = new THashMap();
49              subNodes.put(Method.getMethod(stackTrace[from]), new SampleNode(stackTrace, from - 1));
50          }
51      }

53      public static SampleNode createSampleNode(final StackTraceElement... stackTrace) {
54          SampleNode result = new SampleNode(1, null);
55          SampleNode prevResult = result;
56          for (int i = stackTrace.length - 1; i >= 0; i--) {
57              StackTraceElement elem = stackTrace[i];
58              if (prevResult.subNodes == null) {
59                  prevResult.subNodes = new THashMap<>();
60              }
61              SampleNode node = new SampleNode(1, null);
62              prevResult.subNodes.put(Method.getMethod(elem), node);
63              prevResult = node;
64          }
65          return result;
66      }

69      public static void addToSampleNode(final SampleNode node, final StackTraceElement... stackTrace) {
70          SampleNode prevResult = node;
71          prevResult.sampleCount++;
72          for (int i = stackTrace.length - 1; i >= 0; i--) {
73              StackTraceElement elem = stackTrace[i];
74              final Method method = Method.getMethod(elem);
75              SampleNode nNode;
76              if (prevResult.subNodes == null) {
77                  prevResult.subNodes = new THashMap<>();
78                  nNode = new SampleNode(1, null);
79                  prevResult.subNodes.put(method, nNode);
80              } else {
81                  nNode = prevResult.subNodes.get(method);
82                  if (nNode != null) {
83                      nNode.sampleCount++;
84                  } else {
85                      nNode = new SampleNode(1, null);
86                      prevResult.subNodes.put(method, nNode);
87                  }
88              }
89              prevResult = nNode;
90          }
91      }

93      public static SampleNode clone(final SampleNode node) {
94          if (node.subNodes == null) {
95              return new SampleNode(node.sampleCount, null);
96          }
97          final THashMap<Method, SampleNode> newSubNodes = new THashMap<>(node.subNodes.size());
98          node.subNodes.forEachEntry(new TObjectObjectProcedure<Method, SampleNode>() {
99  
100             @Override
101             public boolean execute(final Method a, final SampleNode b) {
102                 newSubNodes.put(a, SampleNode.clone(b));
103                 return true;
104             }
105         });
106         return new SampleNode(node.sampleCount, newSubNodes);
107     }

109     public static SampleNode aggregate(final SampleNode node1, final SampleNode node2) {
110         int newSampleCount = node1.sampleCount + node2.sampleCount;
111         TMap<Method, SampleNode> newSubNodes;
112         if (node1.subNodes == null && node2.subNodes == null) {
113             newSubNodes = null;
114         } else if (node1.subNodes == null) {
115             newSubNodes = cloneSubNodes(node2);
116         } else if (node2.subNodes == null) {
117             newSubNodes = cloneSubNodes(node1);
118         } else {
119             final THashMap<Method, SampleNode> ns = new THashMap<>(node1.subNodes.size() + node2.subNodes.size());
120 
121             node1.subNodes.forEachEntry(new TObjectObjectProcedure<Method, SampleNode>() {
122 
123                 @Override
124                 public boolean execute(final Method m, final SampleNode b) {
125                     SampleNode other = node2.subNodes.get(m);
126                     if (other == null) {
127                         ns.put(m, SampleNode.clone(b));
128                     } else {
129                         ns.put(m, aggregate(b, other));
130                     }
131                     return true;
132                 }
133             });
134             node2.subNodes.forEachEntry(new TObjectObjectProcedure<Method, SampleNode>() {
135 
136                 @Override
137                 public boolean execute(final Method m, final SampleNode b) {
138                     if (!node1.subNodes.containsKey(m)) {
139                         ns.put(m, SampleNode.clone(b));
140                     }
141                     return true;
142                 }
143             });
144             newSubNodes = ns;
145 
146         }
147         return new SampleNode(newSampleCount, newSubNodes);
148     }

150     public static TMap<Method, SampleNode> cloneSubNodes(final SampleNode node) {
151         final TMap<Method, SampleNode> ns = new THashMap<>(node.subNodes.size());
152         putAllClones(node.subNodes, ns);
153         return ns;
154     }
155 
156     public static void putAllClones(final TMap<Method, SampleNode> source,
157             final TMap<Method, SampleNode> destination) {
158         source.forEachEntry(new TObjectObjectProcedure<Method, SampleNode>() {
159 
160             @Override
161             public boolean execute(final Method a, final SampleNode b) {
162                 destination.put(a, SampleNode.clone(b));
163                 return true;
164             }
165         });
166     }

168     public SampleNode(final int count, @Nullable final TMap<Method, SampleNode> subNodes) {
169         this.sampleCount = count;
170         this.subNodes = subNodes;
171     }

173     public void addSample(final StackTraceElement[] stackTrace, final int from) {
174         sampleCount++;
175         if (from >= 0) {
176             Method method = Method.getMethod(stackTrace[from]);
177             SampleNode subNode = null;
178             if (subNodes == null) {
179                 subNodes = new THashMap();
180             } else {
181                 subNode = subNodes.get(method);
182             }
183             if (subNode == null) {
184                 subNodes.put(method, new SampleNode(stackTrace, from - 1));
185             } else {
186                 subNode.addSample(stackTrace, from - 1);
187             }
188         }
189     }

206     public int height() {
207         if (subNodes == null) {
208             return 1;
209         } else {
210             int subHeight = 0;
211             for (SampleNode node : subNodes.values()) {
212                 int nHeight = node.height();
213                 if (nHeight > subHeight) {
214                     subHeight = nHeight;
215                 }
216             }
217             return subHeight + 1;
218         }
219 
220     }

223     public int getNrNodes() {
224         if (subNodes == null) {
225             return 1;
226         } else {
227             int nrNodes = 0;
228             for (SampleNode node : subNodes.values()) {
229                 nrNodes += node.getNrNodes();
230             }
231             return nrNodes + 1;
232         }
233     }

235     @Nullable
236     public SampleNode filteredBy(final Predicate<Method> predicate) {
237 
238         int newCount = this.sampleCount;
239 
240         THashMap<Method, SampleNode> sns = null;
241         if (this.subNodes != null) {
242             for (Map.Entry<Method, SampleNode> entry : this.subNodes.entrySet()) {
243                 Method method = entry.getKey();
244                 SampleNode sn = entry.getValue();
245                 if (predicate.apply(method)) {
246                     newCount -= sn.getSampleCount();
247                 } else {
248                     if (sns == null) {
249                         sns = new THashMap<>();
250                     }
251                     SampleNode sn2 = sn.filteredBy(predicate);
252                     if (sn2 == null) {
253                         newCount -= sn.getSampleCount();
254                     } else {
255                         newCount -= sn.getSampleCount() - sn2.getSampleCount();
256                         sns.put(method, sn2);
257                     }
258 
259                 }
260             }
261         }
262         if (newCount == 0) {
263             return null;
264         } else if (newCount < 0) {
265             throw new IllegalStateException("child sample counts must be <= parent sample count, detail: " + this);
266         } else {
267             return new SampleNode(newCount, sns);
268         }

271     public interface InvocationHandler {
272 
273         void handle(Method from, Method to, int count, final Map<Method, Integer> ancestors);
274     }
275 
276     public void forEach(final InvocationHandler handler, final Method from,
277             final Method to, final Map<Method, Integer> ancestors) {
278 
279         handler.handle(from, to, sampleCount, ancestors);
280 
281         if (subNodes != null) {
282             Integer val = ancestors.get(to);
283             if (val != null) {
284                 val = val + 1;
285             } else {
286                 val = 1;
287             }
288             ancestors.put(to, val);
289             for (Map.Entry<Method, SampleNode> subs : subNodes.entrySet()) {
290                 Method toKey = subs.getKey();
291                 subs.getValue().forEach(handler, to, toKey, ancestors);
292             }
293             val = ancestors.get(to);
294             if (val == 1) {
295                 ancestors.remove(to);
296             } else {
297                 val = val - 1;
298                 ancestors.put(to, val);
299             }
300         }
301     }

303     public static final class InvocationCount {
304 
305         private int value;
306 
307         public InvocationCount(final int value) {
308             this.value = value;
309         }
310 
311         public int getValue() {
312             return value;
313         }
314 
315         public void setValue(final int value) {
316             this.value = value;
317         }
318 
319         @Override
320         public String toString() {
321             return "InvocationCount{" + "value=" + value + '}';
322         }
323     }
324 
325     @Nonnull
326     public static Graph<Method, InvocationCount> toGraph(final SampleNode rootNode) {
327         final HashMapGraph<Method, InvocationCount> result = new HashMapGraph<>();
328 
329         rootNode.forEach(new InvocationHandler() {
330             @Override
331             public void handle(final Method pfrom, final Method pto, final int count,
332                     final Map<Method, Integer> ancestors) {
333 
334                 Method from = pfrom;
335                 Method to = pto;
336                 Integer val = ancestors.get(from);
337                 if (val != null) {
338                     from = from.withId(val - 1);
339                 }
340                 val = ancestors.get(to);
341                 if (val != null) {
342                     to = to.withId(val);
343                 }
344 
345                 InvocationCount ic = result.getEdge(from, to);
346 
347                 if (ic == null) {
348                     result.add(new InvocationCount(count), from, to);
349                 } else {
350                     ic.setValue(count + ic.getValue());
351                 }
352 
353 
354             }
355         }, Method.ROOT, Method.ROOT, new HashMap<Method, Integer>());
356 
357         return result;
358 
359     }

361     @Override
362     public int hashCode() {
363         int hash = 5;
364         hash = 89 * hash + this.sampleCount;
365         return 89 * hash + Objects.hashCode(this.subNodes);
366     }

368     @Override
369     public boolean equals(final Object obj) {
370         if (obj == null) {
371             return false;
372         }
373         if (getClass() != obj.getClass()) {
374             return false;
375         }
376         final SampleNode other = (SampleNode) obj;
377         if (this.sampleCount != other.sampleCount) {
378             return false;
379         }
380         if (this.subNodes == other.subNodes) {
381             return true;
382         }
383         if (this.subNodes != null && other.subNodes == null && this.subNodes.isEmpty()) {
384             return true;
385         }
386         if (this.subNodes == null && other.subNodes != null && other.subNodes.isEmpty()) {
387             return true;
388         }
389         return Objects.equals(this.subNodes, other.subNodes);
390     }
