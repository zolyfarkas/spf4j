11  public final class Closeables {
12  
13      private Closeables() { }
14  
15      public static <T extends Exception> void closeAll(@Nullable final T throwed, final AutoCloseable ... closeables)
16      throws T {
17          T ex = throwed;
18          for (AutoCloseable closeable : closeables) {
19              try {
20                  closeable.close();
21              } catch (Exception ex1) {
22                  if (ex == null) {
23                      ex = (T) ex1;
24                  } else {
25                      ex = (T) Throwables.suppress(ex1, ex);
26                  }
27              }
28          }
29          if (ex != null) {
30              throw ex;
31          }
32  
33      }
34  
35  
36  }
