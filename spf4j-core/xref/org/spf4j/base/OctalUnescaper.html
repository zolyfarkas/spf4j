20  import java.io.IOException;
21  import java.io.Writer;
22  
23  
24  public final class OctalUnescaper extends CharSequenceTranslator {
25  
26      /**
27       * {@inheritDoc}
28       */
29      @Override
30      public int translate(final CharSequence input, final int index, final Writer out) throws IOException {
31          int remaining = input.length() - index - 1; // how many characters left, ignoring the first \
32          StringBuilder builder = new StringBuilder();
33          if (input.charAt(index) == '\\' && remaining > 0 && isOctalDigit(input.charAt(index + 1))) {
34              int next = index + 1;
35              int next2 = index + 2;
36              int next3 = index + 3;
37  
38              // we know this is good as we checked it in the if block above
39              builder.append(input.charAt(next));
40  
41              if (remaining > 1 && isOctalDigit(input.charAt(next2))) {
42                  builder.append(input.charAt(next2));
43                  if (remaining > 2 && isZeroToThree(input.charAt(next)) && isOctalDigit(input.charAt(next3))) {
44                      builder.append(input.charAt(next3));
45                  }
46              }
47  
48              out.write(Integer.parseInt(builder.toString(), 8));
49              return 1 + builder.length();
50          }
51          return 0;
52      }
53  
54      /**
55       * Checks if the given char is an octal digit. Octal digits are the character representations of the digits 0 to 7.
56       * @param ch the char to check
57       * @return true if the given char is the character representation of one of the digits from 0 to 7
58       */
59      private boolean isOctalDigit(final char ch) {
60          return ch >= '0' && ch <= '7';
61      }
62  
63      /**
64       * Checks if the given char is the character representation of one of the digit from 0 to 3.
65       * @param ch the char to check
66       * @return true if the given char is the character representation of one of the digits from 0 to 3
67       */
68      private boolean isZeroToThree(final char ch) {
69          return ch >= '0' && ch <= '3';
70      }
71  }
