6    * "improved" implementation based on DataTypeConverterImpl
7    * performance should be same/slightly faster than the JDK equivalent
8    * But most importantly you can encode/decode parts of a String, which should reduce the need

18      private static final byte[] DECODE_MAP = initDecodeMap();
19  
20      private static final byte PADDING = 127;
21  
22      private static byte[] initDecodeMap() {
23          byte[] map = new byte[128];
24          int i;
25          for (i = 0; i < 128; i++) {
26              map[i] = -1;
27          }
28  
29          for (i = 'A'; i <= 'Z'; i++) {
30              map[i] = (byte) (i - 'A');
31          }
32          for (i = 'a'; i <= 'z'; i++) {
33              map[i] = (byte) (i - 'a' + 26);
34          }
35          for (i = '0'; i <= '9'; i++) {
36              map[i] = (byte) (i - '0' + 52);
37          }
38          map['+'] = 62;
39          map['/'] = 63;
40          map['='] = PADDING;
41  
42          return map;
43      }

45      /**
46       * computes the length of binary data speculatively.
47       *
48       * <p>
49       * Our requirement is to create byte[] of the exact length to store the binary data. If we do this in a
50       * straight-forward way, it takes two passes over the data. Experiments show that this is a non-trivial overhead
51       * (35% or so is spent on the first pass in calculating the length.)
52       *
53       * <p>
54       * So the approach here is that we compute the length speculatively, without looking at the whole contents. The
55       * obtained speculative value is never less than the actual length of the binary data, but it may be bigger. So if
56       * the speculation goes wrong, we'll pay the cost of reallocation and buffer copying.
57       *
58       * <p>
59       * If the base64 text is tightly packed with no indentation nor illegal char (like what most web services produce),
60       * then the speculation of this method will be correct, so we get the performance benefit.
61       */
62      private static int guessLength(final CharSequence text, final int from, final int len) {
63          final int to = from + len;
64  
65          // compute the tail '=' chars
66          int j = to - 1;
67          for (; j >= 0; j--) {
68              byte code = DECODE_MAP[text.charAt(j)];
69              if (code == PADDING) {
70                  continue;
71              }
72              if (code == -1) // most likely this base64 text is indented. go with the upper bound
73              {
74                  return len / 4 * 3;
75              }
76              break;
77          }
78  
79          j++;    // text.charAt(j) is now at some base64 char, so +1 to make it the size
80          int padSize = to - j;
81          if (padSize > 2) // something is wrong with base64. be safe and go with the upper bound
82          {
83              return len / 4 * 3;
84          }
85  
86          // so far this base64 looks like it's unindented tightly packed base64.
87          // take a chance and create an array with the expected size
88          return len / 4 * 3 - padSize;
89      }

93          final int to = from + len;
94  
95          // compute the tail '=' chars
96          int j = to - 1;
97          for (; j >= 0; j--) {
98              byte code = DECODE_MAP[text[j]];

141     /**
142      * @param text base64Binary data is likely to be long, and decoding requires each character to be accessed twice
143      * (once for counting length, another for decoding.)
144      *
145      */
146     public static byte[] decodeBase64(final CharSequence text, final int from, final int len) {
147         final int buflen = guessLength(text, from, len);
148         final byte[] out = new byte[buflen];
149         int o = 0;
150 
151         int i;
152 
153         final byte[] quadruplet = new byte[4];
154         int q = 0;
155 
156         // convert each quadruplet to three bytes.
157         int to = from + len;
158         for (i = from; i < to; i++) {
159             char ch = text.charAt(i);
160             byte v = DECODE_MAP[ch];
161 
162             if (v != -1) {
163                 quadruplet[q++] = v;
164             }
165 
166             if (q == 4) {
167                 // quadruplet is now filled.
168                 out[o++] = (byte) ((quadruplet[0] << 2) | (quadruplet[1] >> 4));
169                 if (quadruplet[2] != PADDING) {
170                     out[o++] = (byte) ((quadruplet[1] << 4) | (quadruplet[2] >> 2));
171                 }
172                 if (quadruplet[3] != PADDING) {
173                     out[o++] = (byte) ((quadruplet[2] << 6) | (quadruplet[3]));
174                 }
175                 q = 0;
176             }
177         }
178 
179         if (buflen == o) { // speculation worked out to be OK
180             return out;
181         }
182 
183         // we overestimated, so need to create a new buffer
184         byte[] nb = new byte[o];
185         System.arraycopy(out, 0, nb, 0, o);
186         return nb;
187     }

239     private static char[] initEncodeMap() {
240         char[] map = new char[64];
241         int i;
242         for (i = 0; i < 26; i++) {
243             map[i] = (char) ('A' + i);
244         }
245         for (i = 26; i < 52; i++) {
246             map[i] = (char) ('a' + (i - 26));
247         }
248         for (i = 52; i < 62; i++) {
249             map[i] = (char) ('0' + (i - 52));
250         }
251         map[62] = '+';
252         map[63] = '/';
253 
254         return map;
255     }

289     public static void encodeBase64(final byte[] input, final int offset, final int len, final Appendable result)
290             throws IOException {
291         for (int i = offset; i < len; i += 3) {
292             switch (len - i) {
293                 case 1:
294                     result.append(encode(input[i] >> 2));
295                     result.append(encode(((input[i]) & 0x3) << 4));
296                     result.append("==");
297                     break;
298                 case 2:
299                     result.append(encode(input[i] >> 2));
300                     result.append(encode(
301                             ((input[i] & 0x3) << 4)
302                             | ((input[i + 1] >> 4) & 0xF)));
303                     result.append(encode((input[i + 1] & 0xF) << 2));
304                     result.append('=');
305                     break;
306                 default:
307                     result.append(encode(input[i] >> 2));
308                     result.append(encode(
309                             ((input[i] & 0x3) << 4)
310                             | ((input[i + 1] >> 4) & 0xF)));
311                     result.append(encode(
312                             ((input[i + 1] & 0xF) << 2)
313                             | ((input[i + 2] >> 6) & 0x3)));
314                     result.append(encode(input[i + 2] & 0x3F));
315                     break;
316             }
317         }
318     }

320     /**
321      * Encodes a byte array into a char array by doing base64 encoding.
322      *
323      * The caller must supply a big enough buffer.
324      *
325      * @return the value of {@code ptr+((len+2)/3)*4}, which is the new offset in the output buffer where the further
326      * bytes should be placed.
327      */
328     public static int encodeBase64(final byte[] input, final int offset,
329             final int len, final char[] buf, final int cptr) {
330         int ptr = cptr;
331         for (int i = offset; i < len; i += 3) {
332             switch (len - i) {
333                 case 1:
334                     buf[ptr++] = encode(input[i] >> 2);
335                     buf[ptr++] = encode(((input[i]) & 0x3) << 4);
336                     buf[ptr++] = '=';
337                     buf[ptr++] = '=';
338                     break;
339                 case 2:
340                     buf[ptr++] = encode(input[i] >> 2);
341                     buf[ptr++] = encode(
342                             ((input[i] & 0x3) << 4)
343                             | ((input[i + 1] >> 4) & 0xF));
344                     buf[ptr++] = encode((input[i + 1] & 0xF) << 2);
345                     buf[ptr++] = '=';
346                     break;
347                 default:
348                     buf[ptr++] = encode(input[i] >> 2);
349                     buf[ptr++] = encode(
350                             ((input[i] & 0x3) << 4)
351                             | ((input[i + 1] >> 4) & 0xF));
352                     buf[ptr++] = encode(
353                             ((input[i + 1] & 0xF) << 2)
354                             | ((input[i + 2] >> 6) & 0x3));
355                     buf[ptr++] = encode(input[i + 2] & 0x3F);
356                     break;
357             }
358         }
359         return ptr;
360     }

362     /**
363      * Encodes a byte array into another byte array by first doing base64 encoding then encoding the result in ASCII.
364      *
365      * The caller must supply a big enough buffer.
366      *
367      * @return the value of {@code ptr+((len+2)/3)*4}, which is the new offset in the output buffer where the further
368      * bytes should be placed.
369      */
370     public static int encodeBase64(final byte[] input, final int offset, final int len,
371             final byte[] out, final int cptr) {
372         int ptr = cptr;
373         byte[] buf = out;
374         int max = len + offset;
375         for (int i = offset; i < max; i += 3) {
376             switch (max - i) {
377                 case 1:
378                     buf[ptr++] = encodeByte(input[i] >> 2);
379                     buf[ptr++] = encodeByte(((input[i]) & 0x3) << 4);
380                     buf[ptr++] = '=';
381                     buf[ptr++] = '=';
382                     break;
383                 case 2:
384                     buf[ptr++] = encodeByte(input[i] >> 2);
385                     buf[ptr++] = encodeByte(
386                             ((input[i] & 0x3) << 4)
387                             | ((input[i + 1] >> 4) & 0xF));
388                     buf[ptr++] = encodeByte((input[i + 1] & 0xF) << 2);
389                     buf[ptr++] = '=';
390                     break;
391                 default:
392                     buf[ptr++] = encodeByte(input[i] >> 2);
393                     buf[ptr++] = encodeByte(
394                             ((input[i] & 0x3) << 4)
395                             | ((input[i + 1] >> 4) & 0xF));
396                     buf[ptr++] = encodeByte(
397                             ((input[i + 1] & 0xF) << 2)
398                             | ((input[i + 2] >> 6) & 0x3));
399                     buf[ptr++] = encodeByte(input[i + 2] & 0x3F);

401             }
402         }
