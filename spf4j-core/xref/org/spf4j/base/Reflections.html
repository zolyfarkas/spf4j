36      public static Class<?> primitiveToWrapper(final Class<?> clasz) {
37          if (clasz.isPrimitive()) {
38              return PRIMITIVE_MAP.get(clasz);
39          } else {
40              return clasz;

45      public static Object getAnnotationAttribute(@Nonnull final Annotation annot, @Nonnull final String attributeName) {
46          for (Method method : annot.annotationType().getDeclaredMethods()) {
47              if (method.getName().equals(attributeName)) {
48                  try {
49                      return method.invoke(annot);
50                  } catch (IllegalAccessException ex) {
51                      throw new RuntimeException(ex);
52                  } catch (IllegalArgumentException ex) {
53                      throw new RuntimeException(ex);
54                  } catch (InvocationTargetException ex) {
55                      throw new RuntimeException(ex);
56                  }
57              }
58          }
59          throw new IllegalArgumentException(attributeName + " attribute is not present on annotation " + annot);
60      }

62      public static Method getCompatibleMethod(final Class<?> c,
63              final String methodName,
64              final Class<?>... paramTypes) {
65          Method[] methods = c.getMethods();
66          for (Method m : methods) {
67  
68              if (!m.getName().equals(methodName)) {
69                  continue;
70              }
71  
72              Class<?>[] actualTypes = m.getParameterTypes();
73              if (actualTypes.length > paramTypes.length) {
74                  continue;
75              }
76  
77              boolean found = true;
78              int last = actualTypes.length - 1;
79              for (int j = 0; j < actualTypes.length; j++) {
80                  final Class<?> actType = actualTypes[j];
81                  final Class<?> paramType = paramTypes[j];
82                  found = canAssign(actType, paramType);
83                  if (!found && j == last) {
84                      if (actType.isArray()) {
85                          boolean matchvararg = true;
86                          Class<?> varargType = actType.getComponentType();
87                          for (int k = j; k < paramTypes.length; k++) {
88                              if (!canAssign(varargType, paramTypes[k])) {
89                                  matchvararg = false;
90                                  break;
91                              }
92                          }
93                          found = matchvararg;
94                      }
95                  } else if (j == last && actualTypes.length < paramTypes.length) {
96                      found = false;
97                  }
98                  if (!found) {
99                      break;
100                 }
101             }
102 
103             if (found) {
104                 return m;
105             }
106         }
107         return null;
108     }

111     public static Object invoke(final Method m, final Object object, final Object [] parameters)
112             throws IllegalAccessException, InvocationTargetException {
113         int np = parameters.length;
114         if (np > 0) {
115             Class<?> [] actTypes = m.getParameterTypes();
116             Class<?> lastParamClass =  actTypes[actTypes.length - 1];
117             if (Reflections.canAssign(lastParamClass, parameters[np - 1].getClass())) {
118                 return m.invoke(object, parameters);
119             } else if (lastParamClass.isArray()) {
120                 int lidx = actTypes.length - 1;
121                 int l = np - lidx;
122                 Object array = Array.newInstance(lastParamClass.getComponentType(), l);
123                 for (int k = 0; k < l; k++) {
124                     Array.set(array, k, parameters[lidx + k]);
125                 }
126                 Object [] newParams  = new Object [actTypes.length];
127                 System.arraycopy(parameters, 0, newParams, 0, lidx);
128                 newParams[lidx] = array;
129                 return m.invoke(object, newParams);
130             } else {
131                 throw new IllegalStateException();
132             }
133         } else {
134             return m.invoke(object);
135         }
136     }

140     public static boolean canAssign(final Class<?> to, final Class<?> from) {
141         boolean found = true;
142         if (!to.isAssignableFrom(from)) {
143             if (to.isPrimitive()) {
144                 Class<?> boxed = PRIMITIVE_MAP.get(to);
145                 found = boxed.equals(from);
146             } else if (from.isPrimitive()) {
147                 found = PRIMITIVE_MAP.get(from).equals(to);
148             } else {
149                 found = false;
150             }
151         }
152         return found;
153     }

155     static final class MethodDesc {
156 
157         @Nonnull
158         private final Class<?> clasz;
159         @Nonnull
160         private final String name;
161         @Nonnull
162         private final Class<?>[] paramTypes;
163 
164         public MethodDesc(final Class<?> clasz, final String name, final Class<?>[] paramTypes) {
165             this.clasz = clasz;
166             this.name = name;
167             this.paramTypes = paramTypes;
168         }
169 
170         public Class<?> getClasz() {
171             return clasz;
172         }
173 
174         public String getName() {
175             return name;
176         }
177 
178         public Class<?>[] getParamTypes() {
179             return paramTypes;
180         }
181 
182         @Override
183         public int hashCode() {
184             return 47 * this.clasz.hashCode() + this.name.hashCode();
185         }
186 
187         @Override
188         public boolean equals(final Object obj) {
189             if (obj == null) {
190                 return false;
191             }
192             if (obj instanceof MethodDesc) {
193                 final MethodDesc other = (MethodDesc) obj;
194                 if (!this.clasz.equals(other.clasz)) {
195                     return false;
196                 }
197                 if (!this.name.equals(other.name)) {
198                     return false;
199                 }
200                 return (Arrays.equals(this.paramTypes, other.paramTypes));
201             } else {
202                 return false;
203             }
204         }
205 
206         @Override
207         public String toString() {
208             return "MethodDesc{" + "clasz=" + clasz + ","
209                     + " name=" + name + ", paramTypes=" + Arrays.toString(paramTypes) + '}';
210         }
211 
212     }

214     private static final LoadingCache<MethodDesc, Method> CACHE_FAST
215             = new UnboundedRacyLoadingCache<MethodDesc, Method>(64,
216                     new CacheLoader<MethodDesc, Method>() {
217                         @Override
218                         public Method load(final MethodDesc k) throws Exception {
219                             final Method m = getCompatibleMethod(k.getClasz(), k.getName(), k.getParamTypes());
220                             AccessController.doPrivileged(new PrivilegedAction() {
221                                 @Override
222                                 public Object run() {
223                                     m.setAccessible(true);
224                                     return null; // nothing to return
225                                 }
226                             });
227                             return m;
228                         }
229                     });

231     public static Method getCompatibleMethodCached(final Class<?> c,
232             final String methodName,
233             final Class<?>... paramTypes) {
234         return CACHE_FAST.getUnchecked(new MethodDesc(c, methodName, paramTypes));
235     }
