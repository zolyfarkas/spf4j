18  package org.spf4j.concurrent;
19  
20  import com.google.common.io.BaseEncoding;
21  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
22  import java.net.NetworkInterface;
23  import java.net.SocketException;
24  import java.util.Enumeration;
25  import java.util.function.Supplier;
26  import javax.annotation.Nullable;
27  import javax.annotation.ParametersAreNonnullByDefault;
28  import org.spf4j.base.Strings;
29  import static org.spf4j.base.Strings.appendUnsignedString;
30  
31  /**
32   * Unique ID Generator Based on the assumptions:
33   * 1. host MAC address is used. (each network interface has a Unique ID) (encoded with provided encoder)
34   * 2. process id is used + current epoch seconds. it is assumed the PID is not recycled within a second.
35   * 3. A process sequence is used. UIDs will cycle after Long.MaxValue is reached.
36   *
37   * @author zoly
38   */
39  @ParametersAreNonnullByDefault
40  public final class UIDGenerator implements Supplier<CharSequence> {
41  
42      private final Sequence sequence;
43  
44      private final StringBuilder base;
45  
46      private final int maxSize;
47  
48      public UIDGenerator(final Sequence sequence) {
49          this(sequence, 0);
50      }
51  
52      public UIDGenerator(final Sequence sequence, final String prefix) {
53          this(sequence, BaseEncoding.base64Url(), 0, '.', prefix);
54      }
55  
56  
57      public UIDGenerator(final Sequence sequence, final long customEpoch) {
58          this(sequence, BaseEncoding.base64Url(), customEpoch, '.', "");
59      }
60  
61      public UIDGenerator(final Sequence sequence, final String prefix, final long customEpoch) {
62          this(sequence, BaseEncoding.base64Url(), customEpoch, '.', prefix);
63      }
64      
65      /**
66       * Construct a UID Generator
67       * @param sequence
68       * @param baseEncoding - if null MAC address based ID will not be included.
69       */
70      @SuppressFBWarnings("STT_TOSTRING_STORED_IN_FIELD")
71      public UIDGenerator(final Sequence sequence, @Nullable final BaseEncoding baseEncoding,
72              final long customEpoch, final char separator, final String prefix) {
73          this.sequence = sequence;
74          StringBuilder sb = new StringBuilder(16 + prefix.length());
75          sb.append(prefix);
76          if (baseEncoding != null) {
77              byte[] intfMac;
78              try {
79                  Enumeration<NetworkInterface> networkInterfaces = NetworkInterface.getNetworkInterfaces();
80                  if (networkInterfaces != null && networkInterfaces.hasMoreElements()) {
81                      do {
82                          intfMac = networkInterfaces.nextElement().getHardwareAddress();
83                      } while ((intfMac == null || intfMac.length == 0) && networkInterfaces.hasMoreElements());
84                      if (intfMac == null) {
85                          intfMac = new byte[] {0};
86                      }
87                  } else {
88                      intfMac = new byte[] {0};
89                  }
90              } catch (SocketException ex) {
91                  throw new RuntimeException(ex);
92              }
93              sb.append(baseEncoding.encode(intfMac)).append(separator);
94          }
95          appendUnsignedString(sb, org.spf4j.base.Runtime.PID, 5);
96          sb.append(separator);
97          appendUnsignedString(sb, (System.currentTimeMillis() - customEpoch) / 1000, 5);
98          sb.append(separator);
99          base = sb;
100         maxSize = base.length() + 16;
101     }
102 
103     public int getMaxSize() {
104         return maxSize;
105     }
106 
107     public CharSequence next() {
108         StringBuilder result = new StringBuilder(maxSize);
109         result.append(base);
110         Strings.appendUnsignedString(result, sequence.next(), 5);
111         return result;
112     }
113 
114 
115 
116     @Override
117     public String toString() {
118         return "UIDGenerator{" + "sequence=" + sequence + ", base=" + base + ", maxSize=" + maxSize + '}';
119     }
120 
121     @Override
122     public CharSequence get() {
123       return next();
124     }
125 
126 
127 
128 }
