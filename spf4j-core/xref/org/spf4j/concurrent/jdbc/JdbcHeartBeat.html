33  /**
34   * A class that does "heartbeats" (at a arbitrary inteval) to a database table.
35   * This is to detect the death of a process.
36   * The process is considered dead when: currentTime - lastheartbeat > beatInterval * 2
37   * When this class mechanism detects that it cannot perform the heartbeats it throws a Error.
38   * The sensible this for the process is to go down (and restart if it is a daemon).
39   *
40   * @author zoly
41   */
42  @SuppressFBWarnings(value = {"SQL_PREPARED_STATEMENT_GENERATED_FROM_NONCONSTANT_STRING",
43    "PMB_POSSIBLE_MEMORY_BLOAT"}, justification = "The db object names are configurable,"
44          + "we for know allow heartbeats to multiple data sources, should be one mostly")
45  @ParametersAreNonnullByDefault

73    @Override
74    public void close() throws SQLException, InterruptedException {
75      synchronized (jdbc) {
76        if (!isClosed) {
77          unregisterJmx();
78          ScheduledFuture<?> running = scheduledHearbeat;
79          if (running != null) {
80            scheduledHearbeat.cancel(true);
81          }
82          removeHeartBeatRow(jdbcTimeoutSeconds);
83          for (LifecycleHook hook : lifecycleHooks) {
84            hook.onClose();
85          }
86          isClosed = true;
87        }
88      }
89    }

109   private JdbcHeartBeat(final DataSource dataSource, final HeartBeatTableDesc hbTableDesc, final long intervalMillis,
110           final int jdbcTimeoutSeconds) throws SQLException, InterruptedException {
111     if (intervalMillis < 1000) {
112       throw new IllegalArgumentException("The heartbeat interval should be at least 1s and not "
113               + intervalMillis + " ms");
114     }
115 
116     this.jdbc = new JdbcTemplate(dataSource);
117     this.jdbcTimeoutSeconds = jdbcTimeoutSeconds;
118     this.intervalMillis = intervalMillis;
119     this.hbTableDesc = hbTableDesc;
120     this.isClosed = false;
121     String hbTableName = hbTableDesc.getTableName();
122     String lastHeartbeatColumn = hbTableDesc.getLastHeartbeatColumn();
123     String currentTimeMillisFunc = hbTableDesc.getCurrentTimeMillisFunc();
124     String intervalColumn = hbTableDesc.getIntervalColumn();
125     String ownerColumn = hbTableDesc.getOwnerColumn();
126 
127     this.updateHeartbeatSql = "UPDATE " + hbTableName + " SET " + lastHeartbeatColumn + " = " + currentTimeMillisFunc
128             + " WHERE " + ownerColumn + " = ? AND " + lastHeartbeatColumn + " + " + intervalColumn
129             + " * 2 > " + currentTimeMillisFunc;
130     this.deleteHeartBeatSql = "DELETE FROM " + hbTableName
131             + " WHERE " + ownerColumn + " = ?";
132     this.deleteSql = "DELETE FROM " + hbTableName + " WHERE " + lastHeartbeatColumn + " + " + intervalColumn
133             + " * 2 < " + currentTimeMillisFunc;
134     this.lifecycleHooks = new CopyOnWriteArrayList<>();
135     long startTimeNanos =  System.nanoTime();
136     createHeartbeatRow();
137     long duration = System.nanoTime() - startTimeNanos;
138     this.beatDurationNanos = Math.max(duration, TimeUnit.MILLISECONDS.toNanos(10));
139   }

161   void createHeartbeatRow()
162           throws SQLException, InterruptedException {
163 
164     jdbc.transactOnConnection((final Connection conn, final long deadlineNanos) -> {
165 
166       try (final PreparedStatement insert = conn.prepareStatement("insert into " + hbTableDesc.getTableName()
167               + " (" + hbTableDesc.getOwnerColumn() + ',' + hbTableDesc.getIntervalColumn() + ','
168               + hbTableDesc.getLastHeartbeatColumn() + ") VALUES (?, ?, "
169               + hbTableDesc.getCurrentTimeMillisFunc() + ")")) {
170         insert.setNString(1, org.spf4j.base.Runtime.PROCESS_ID);
171         insert.setLong(2, this.intervalMillis);
172         insert.setQueryTimeout((int) TimeUnit.NANOSECONDS.toSeconds(deadlineNanos - System.nanoTime()));
173         insert.executeUpdate();
174       }
175       return null;
176     }, jdbcTimeoutSeconds, TimeUnit.SECONDS);
177   }

188   int removeDeadHeartBeatRows(final Connection conn, final long deadlineNanos) throws SQLException {
189     try (final PreparedStatement stmt = conn.prepareStatement(deleteSql)) {
190       stmt.setQueryTimeout((int) TimeUnit.NANOSECONDS.toSeconds(deadlineNanos - System.nanoTime()));
191       return stmt.executeUpdate();

195   void removeHeartBeatRow(final int timeoutSeconds)
196           throws SQLException, InterruptedException {
197     jdbc.transactOnConnection((final Connection conn, final long deadlineNanos) -> {
198       try (final PreparedStatement stmt = conn.prepareStatement(deleteHeartBeatSql)) {
199         stmt.setNString(1, org.spf4j.base.Runtime.PROCESS_ID);
200         stmt.setQueryTimeout((int) TimeUnit.NANOSECONDS.toSeconds(deadlineNanos - System.nanoTime()));
201         int nrDeleted = stmt.executeUpdate();
202         if (nrDeleted != 1) {
203           throw new IllegalStateException("Heartbeat rows deleted: " + nrDeleted
204                   + " for " + org.spf4j.base.Runtime.PROCESS_ID);
205         }
206       }
207       return null;
208     }, timeoutSeconds, TimeUnit.SECONDS);
209   }

230   public void scheduleHeartbeat() {
231     synchronized (jdbc) {
232       if (isClosed) {
233         throw new IllegalStateException("Heartbeater is closed " + this);
234       }
235       long lr = lastRun;
236       long delay;
237       if (lr == 0) {
238         delay = intervalMillis;
239       } else if (lr > 0) {
240         delay = intervalMillis - (System.currentTimeMillis() - lr);
241       } else {
242         throw new IllegalStateException("The end of times are upon us :-) " + lr);
243       }
244       scheduledHearbeat = DefaultScheduler.INSTANCE.schedule(getHeartBeatRunnable(), delay, TimeUnit.MILLISECONDS);
245     }
246   }

248   @JmxExport
249   public void beat() throws SQLException, InterruptedException {
250     jdbc.transactOnConnection(new HandlerNano<Connection, Void, SQLException>() {
251       @Override
252       @SuppressFBWarnings("NP_LOAD_OF_KNOWN_NULL_VALUE")
253       public Void handle(final Connection conn, final long deadlineNanos) throws SQLException {
254         beat(conn, deadlineNanos);
255         return null;
256       }
257     }, jdbcTimeoutSeconds, TimeUnit.SECONDS);
258     lastRun = System.currentTimeMillis();
259   }
260 
261   void beat(final Connection conn, final long deadlineNanos) throws SQLException {
262     synchronized (jdbc) {
263       if (isClosed) {
264         throw new IllegalStateException("Heartbeater is closed " + this);
265       }
266     }
267     try (PreparedStatement stmt = conn.prepareStatement(updateHeartbeatSql)) {
268       stmt.setQueryTimeout((int) TimeUnit.NANOSECONDS.toSeconds(deadlineNanos - System.nanoTime()));
269       stmt.setNString(1, org.spf4j.base.Runtime.PROCESS_ID);
270       int rowsUpdated = stmt.executeUpdate();
271       if (rowsUpdated != 1) {
272         throw new IllegalStateException("Broken Heartbeat for "
273                 + org.spf4j.base.Runtime.PROCESS_ID + "sql : " + updateHeartbeatSql + " rows : " + rowsUpdated);
274       }
275       LOG.debug("Heart Beat for {}", org.spf4j.base.Runtime.PROCESS_ID);
276     }
277   }

279   boolean tryBeat(final Connection conn, final long deadlineNanos) throws SQLException {
280     if (System.currentTimeMillis() - lastRun > intervalMillis / 2) {
281       beat(conn, deadlineNanos);
282       return true;
283     } else {
284       return false;
285     }
286   }

293   @JmxExport
294   public long getLastRunDB() throws SQLException, InterruptedException {
295     return jdbc.transactOnConnection(new HandlerNano<Connection, Long, SQLException>() {
296       @Override
297       @SuppressFBWarnings("NP_LOAD_OF_KNOWN_NULL_VALUE")
298       public Long handle(final Connection conn, final long deadlineNanos) throws SQLException {
299         try (PreparedStatement stmt = conn.prepareStatement("select " + hbTableDesc.getLastHeartbeatColumn()
300                 + " FROM " + hbTableDesc.getTableName() + " where " + hbTableDesc.getOwnerColumn() + " = ?")) {
301           stmt.setQueryTimeout(10);
302           stmt.setNString(1, org.spf4j.base.Runtime.PROCESS_ID);
303           try (ResultSet rs = stmt.executeQuery()) {
304             if (rs.next()) {
305               long result = rs.getLong(1);
306               if (rs.next()) {
307                 throw new IllegalStateException("Multible beats for same owner " + org.spf4j.base.Runtime.PROCESS_ID);
308               }
309               return result;
310             } else {
311               return 0L;
312             }
313           }
314         }
315       }
316     }, jdbcTimeoutSeconds, TimeUnit.SECONDS);
317   }

336   private static final int HEARTBEAT_INTERVAL_MILLIS = Integer.getInteger("spf4j.heartbeat.intervalMillis", 10000);

339   /**
340    * Get a reference to the hearbeat instance.
341    * @param dataSource - the datasource the hearbeat goes against.
342    * @param hook - a hook to notify when heartbeat fails.
343    * @return - the heartbeat instance.
344    * @throws SQLException
345    * @throws InterruptedException
346    */
347   public static synchronized JdbcHeartBeat getHeartBeatAndSubscribe(final DataSource dataSource,
348           final HeartBeatTableDesc hbTableDesc,
349           @Nullable final LifecycleHook hook)
350           throws SQLException, InterruptedException {
351     JdbcHeartBeat beat = HEARTBEATS.get(dataSource);
352     if (beat == null) {
353       beat = new JdbcHeartBeat(dataSource, hbTableDesc, HEARTBEAT_INTERVAL_MILLIS, 5);
354       beat.scheduleHeartbeat();
355       beat.registerJmx();
356       beat.addLyfecycleHook(new LifecycleHook() {
357         @Override
358         public void onError(final Error error) {
359         }
360 
361         @Override
362         public void onClose() {
363           HEARTBEATS.remove(dataSource);
364         }
365       });
366       HEARTBEATS.put(dataSource, beat);
367     }
368     if (hook != null) {
369       beat.addLyfecycleHook(hook);
370     }
371     return beat;
372   }

384   private class ScheduledHeartBeat implements Runnable {
385 
386     @Override
387     public void run() {
388       try {
389         long lr = lastRun;
390         long currentTimeMillis = System.currentTimeMillis();
391         if (lr != 0) {
392           // not first beat.
393           long millisSinceLastBeat = currentTimeMillis - lr;
394           if (millisSinceLastBeat < intervalMillis / 2) {
395             // we had a heartbeat not too long ago... weshould skip beating now and reschedule.
396             synchronized (jdbc) {
397               if (!isClosed) {
398                 scheduleHeartbeat();
399               }
400             }
401             return;
402           } else if (((intervalMillis * 2) < millisSinceLastBeat)) {
403             // Unable to beat at inteval!
404             Error err = new Error("System to busy to provide regular heartbeat, lastRun = " + lr
405                     + ", intervalMillis = " + intervalMillis + ", currentTimeMillis = " + currentTimeMillis);
406 
407             handleError(err);
408           }
409         }
410         beat();
411         synchronized (jdbc) {
412           if (!isClosed) {
413             scheduleHeartbeat();
414           }
415         }
416       } catch (RuntimeException | SQLException | InterruptedException ex) {
417         Error err = new Error("System failed heartbeat", ex);
418         handleError(err);
419       }
420     }
421 
422     public void handleError(final Error err) throws Error {
423       for (LifecycleHook hook : lifecycleHooks) {
424         hook.onError(err);
425       }
426       RuntimeException ex = Iterables.forAll(lifecycleHooks, (final LifecycleHook t) -> {
427         try {
428           t.onClose();
429         } catch (SQLException | InterruptedException e) {
430           throw new RuntimeException(e);
431         }
432       });
433       if (ex != null) {
434         err.addSuppressed(ex);
435       }
436       lifecycleHooks.clear();
437       throw err;
438     }
