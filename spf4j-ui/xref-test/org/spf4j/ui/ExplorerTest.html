45  /**
46   * @author Zoltan Farkas
47   */
48  public class ExplorerTest {
49  
50    private static NoExitSecurityManagerInstaller installNoExitSecurityManager;
51  
52    @BeforeClass
53    public static void setUpOnce() {
54       FailOnThreadViolationRepaintManager.install();
55      installNoExitSecurityManager = NoExitSecurityManagerInstaller.installNoExitSecurityManager();
56    }
57  
58    @AfterClass
59    public static void tearDown() {
60      installNoExitSecurityManager.uninstall();
61    }
62  
63    @Test
64    @SuppressFBWarnings("MDM_THREAD_YIELD") // need to since assertj does not seem to apply things otherwise...
65    public void testExplorer() throws InterruptedException {
66      Assume.assumeTrue(TestUtils.isExecutedInCI());
67      ObservationAssert expectation = TestLoggers.sys().expectUncaughtException(2, TimeUnit.SECONDS,
68              UncaughtExceptionDetail.hasThrowable((Matcher) Matchers.any(ExitException.class)));
69      JFrame tFrame = GuiActionRunner.execute(() -> new Explorer());
70      FrameFixture window = new FrameFixture(tFrame);
71      Settings settings = window.robot().settings();
72      settings.delayBetweenEvents(130);
73      settings.eventPostingDelay(100);
74      window.show(); // shows the frame to test
75      Assert.assertTrue(window.isEnabled());
76      JMenuItemFixture openFileMenuItem = window.menuItem(new GenericTypeMatcher<JMenuItem>(JMenuItem.class) {
77        @Override
78        protected boolean isMatching(final JMenuItem component) {
79          return "Open".equals(component.getText());
80        }
81      });
82      openFileMenuItem.click();
83      Thread.sleep(100); // without this somehow things are not found.
84      JFileChooserFixture fileChooser = window.fileChooser("openFileDialog");
85      fileChooser.setCurrentDirectory(new File("src/test/resources"));
86      fileChooser.selectFile(new File(
87              "src/test/resources/com.google.common.io.AppendableWriterBenchmark.spf4jAppendable-Throughput.ssdump2")
88              .getAbsoluteFile());
89      fileChooser.approve();
90      Assert.assertNotNull(window.internalFrame(
91              "com.google.common.io.AppendableWriterBenchmark.spf4jAppendable-Throughput.ssdump2"));
92  
93      openFileMenuItem.click();
94      Thread.sleep(100);
95      fileChooser = window.fileChooser("openFileDialog");
96      fileChooser.setCurrentDirectory(new File("src/test/resources"));
97      Thread.sleep(100);
98      fileChooser.selectFile(new File("src/test/resources/"
99                      + "19156@ZMacBookPro.local.tsdb").getAbsoluteFile());
100     fileChooser.approve();
101     Assert.assertNotNull(window.internalFrame(
102             "19156@ZMacBookPro.local.tsdb"));
103 
104     openFileMenuItem.click();
105     Thread.sleep(100);
106     fileChooser = window.fileChooser("openFileDialog");
107     fileChooser.setCurrentDirectory(new File("src/test/resources"));
108     Thread.sleep(100);
109     fileChooser.selectFile(new File("src/test/resources/"
110                     + "test8381720042200787335.tsdb2").getAbsoluteFile());
111     fileChooser.approve();
112     Assert.assertNotNull(window.internalFrame(
113             "test8381720042200787335.tsdb2"));
114     JMenuItemFixture openJsonText = window.menuItem(new GenericTypeMatcher<JMenuItem>(JMenuItem.class) {
115       @Override
116       protected boolean isMatching(final JMenuItem component) {
117         return "From Text/URL".equals(component.getText());
118       }
119     });
120     openJsonText.click();
121     Thread.sleep(100);
122     DialogFixture dialog = window.dialog("fromTextDialog");
123     dialog.textBox("textBox").setText("{\"ROOT@49924@ZMacBookPro-2.local\":10,"
124             + "\"c\":[{\"m3@C2\":2,\"c\":[{\"m2@C2\":2,\"c\":[{\"m1@C2\":2}]}]},{\"m1@C1\":2},"
125             + "{\"m3@C1\":4,\"c\":[{\"m2@C1\":4,\"c\":[{\"m1@C1\":4}]}]},"
126             + "{\"m4@C1\":2,\"c\":[{\"m2@C1\":2,\"c\":[{\"m1@C1\":2}]}]}]}");
127     dialog.button("display").click();
128     Assert.assertNotNull(window.internalFrame("SampleNode Tree"));
129     window.close();
130     window.cleanUp();
131     expectation.assertObservation();
132   }
