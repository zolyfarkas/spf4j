49      public static JFreeChart createHeatJFreeChart(final String[] dsNames, final double[][] values,
50              final long startTimeMillis, final long stepMillis, final String uom, final String chartName) {
51          final QuantizedXYZDatasetImpl dataSet = new QuantizedXYZDatasetImpl(dsNames, values,
52                  startTimeMillis, stepMillis);
53          NumberAxis xAxis = new NumberAxis("Time");
54          xAxis.setStandardTickUnits(dataSet.createXTickUnits());
55          xAxis.setLowerMargin(0);
56          xAxis.setUpperMargin(0);
57          xAxis.setVerticalTickLabels(true);
58          NumberAxis yAxis = new NumberAxis(uom);
59          yAxis.setStandardTickUnits(dataSet.createYTickUnits());
60          yAxis.setLowerMargin(0);
61          yAxis.setUpperMargin(0);
62          XYBlockRenderer renderer = new XYBlockRenderer();
63          PaintScale scale;
64          if (dataSet.getMinValue() >= dataSet.getMaxValue()) {
65              if (dataSet.getMinValue() == Double.POSITIVE_INFINITY) {
66                  scale = new InverseGrayScale(0, 1);
67              } else {
68                  scale = new InverseGrayScale(dataSet.getMinValue(), dataSet.getMaxValue() + 1);
69              }
70          } else {
71              scale = new InverseGrayScale(dataSet.getMinValue(), dataSet.getMaxValue());
72          }
73          renderer.setPaintScale(scale);
74          renderer.setBlockWidth(1);
75          renderer.setBlockHeight(1);
76          XYPlot plot = new XYPlot(dataSet, xAxis, yAxis, renderer);
77          plot.setBackgroundPaint(Color.white);
78          plot.setDomainGridlinesVisible(false);
79          plot.setRangeGridlinesVisible(false);
80          plot.setRangeMinorGridlinesVisible(false);
81          JFreeChart chart = new JFreeChart(chartName, plot);
82          PaintScaleLegend legend = new PaintScaleLegend(scale, new NumberAxis("Count"));
83          legend.setMargin(0, 5, 0, 5);
84          chart.addSubtitle(legend);
85          chart.removeLegend();
86          chart.setBackgroundPaint(Color.white);
87          return chart;
88      }

90      private static JFreeChart createJFreeChart(final String chartName, final String uom,
91              final XYDataset timeseriescollection) {
92          JFreeChart jfreechart = ChartFactory.createTimeSeriesChart(chartName,
93                  "Time", uom, timeseriescollection, true, true, false);
94          XYPlot xyplot = (XYPlot) jfreechart.getPlot();
95          DateAxis dateaxis = (DateAxis) xyplot.getDomainAxis();
96          dateaxis.setVerticalTickLabels(true);
97          XYLineAndShapeRenderer xylineandshaperenderer = (XYLineAndShapeRenderer) xyplot.getRenderer();
98          xylineandshaperenderer.setBaseShapesVisible(true);
99          xylineandshaperenderer.setUseFillPaint(true);
100         xylineandshaperenderer.setLegendItemToolTipGenerator(new StandardXYSeriesLabelGenerator("Tooltip {0}"));
101         return jfreechart;
102     }

104     public static BufferedImage createMinMaxAvgCountImg(final String chartName, final long[] timestamps,
105             final double[] min, final double[] max, final double[] total, final double[] count,
106             final String uom, final int width, final int height) {
107 
108         BufferedImage bi = Charts.createTimeSeriesJFreeChart(chartName, timestamps,
109                 new String[]{"min", "max", "avg"}, uom,
110                 new double[][]{min, max, Arrays.divide(total, count)}).createBufferedImage(width, height - height / 3);
111         BufferedImage bi2 = Charts.createTimeSeriesJFreeChart(null, timestamps,
112                 new String[]{"count"}, "count", new double[][]{count}).createBufferedImage(width, height / 3);
113         BufferedImage combined = new BufferedImage(width, height, BufferedImage.TYPE_INT_RGB);
114         final Graphics graphics = combined.getGraphics();
115         try {
116             graphics.drawImage(bi, 0, 0, null);
117             graphics.drawImage(bi2, 0, height - height / 3, null);
118         } finally {
119             graphics.dispose();
120         }
121         return combined;
122     }

124     public static BufferedImage generateCountTotalChart(final String groupName, final long[][] timestamps,
125             final String[] measurementNames, final String uom1, final double[][] measurements, final int width,
126             final int height, final String[] measurementNames2, final String uom2, final double[][] measurements2) {
127         BufferedImage count = Charts.createTimeSeriesJFreeChart("Measurements for "
128                 + groupName + " generated by spf4j",
129                 timestamps, measurementNames, uom1, measurements).createBufferedImage(width, height / 2);
130         BufferedImage total = Charts.createTimeSeriesJFreeChart(null,
131                 timestamps, measurementNames2, uom2, measurements2).createBufferedImage(width, height / 2);
132         BufferedImage combined = new BufferedImage(width, height, BufferedImage.TYPE_INT_RGB);
133         final Graphics graphics = combined.getGraphics();
134         try {
135             graphics.drawImage(count, 0, 0, null);
136             graphics.drawImage(total, 0, height / 2, null);
137         } finally {
138             graphics.dispose();
139         }
140         return combined;
141     }

145     private static TimeSeriesCollection createTimeSeriesCollection(final String[] measurementNames,
146             final long[] timestamps, final double[][] measurements) {
147         TimeSeriesCollection timeseriescollection = new TimeSeriesCollection();
148         for (int i = 0; i < measurementNames.length; i++) {
149             TimeSeries tseries = new TimeSeries(measurementNames[i]);
150             for (int j = 0; j < timestamps.length; j++) {
151                 FixedMillisecond ts = new FixedMillisecond(timestamps[j]);
152                 tseries.add(ts, measurements[i][j]);
153             }
154             timeseriescollection.addSeries(tseries);
155         }
156         return timeseriescollection;
157     }

196     public static Pair<long[], double[][]> fillGaps(final long[] timestamps,
197             final long[][] data, final int sampleTime, final int nrColumns) {
198         long startTime = timestamps[0];
199         int nrSamples = (int) ((timestamps[timestamps.length - 1] - startTime) / sampleTime) + 1;
200         long[] lts = new long[nrSamples];
201         double[][] dr = new double[nrSamples][];
202         long nextTime = startTime;
203         int j = 0;
204         int maxDeviation = sampleTime / 2;
205         double[] nodata = new double[nrColumns];
206         for (int i = 0; i < nrColumns; i++) {
207             nodata[i] = Double.NaN;
208         }
209         for (int i = 0; i < nrSamples; i++) {
210             lts[i] = nextTime;
211 
212             if (Math.abs(timestamps[j] - nextTime) < maxDeviation) {
213                 dr[i] = Arrays.toDoubleArray(data[j]);
214                 j++;
215             } else {
216                 dr[i] = nodata;
217             }
218             nextTime += sampleTime;
219         }
220         return Pair.of(lts, dr);
221     }
