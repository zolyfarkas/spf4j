234     private void openMenuItemActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_openMenuItemActionPerformed
235       JFileChooser chooser = new JFileChooser();
236       chooser.setName("openFileDialog");
237       chooser.setMultiSelectionEnabled(true);
238       chooser.setDialogType(JFileChooser.OPEN_DIALOG);
239       chooser.setFileSelectionMode(JFileChooser.FILES_ONLY);
240       chooser.addChoosableFileFilter(Spf4jFileFilter.D3_JSON);
241       chooser.addChoosableFileFilter(Spf4jFileFilter.SPF4J_JSON);
242       chooser.addChoosableFileFilter(Spf4jFileFilter.SSDUMP);
243       chooser.addChoosableFileFilter(Spf4jFileFilter.SSDUMP2);
244       chooser.addChoosableFileFilter(Spf4jFileFilter.SSDUMP3);
245       chooser.addChoosableFileFilter(Spf4jFileFilter.SSDUMP2_GZ);
246       chooser.addChoosableFileFilter(Spf4jFileFilter.SSDUMP3_GZ);
247       chooser.addChoosableFileFilter(Spf4jFileFilter.PROFILE_AVRO);
248       chooser.addChoosableFileFilter(Spf4jFileFilter.TSDB);
249       chooser.addChoosableFileFilter(Spf4jFileFilter.TSDB2);
250       chooser.addChoosableFileFilter(Spf4jFileFilter.AVRO_TABLEDEF);
251       if (folder != null) {
252         chooser.setCurrentDirectory(folder);
253       }
254 
255       int returnVal = chooser.showOpenDialog(this);
256 
257       if (returnVal == JFileChooser.APPROVE_OPTION) {
258         File [] files = chooser.getSelectedFiles();
259         for (File file : files) {
260           try {
261             openFile(file);
262           } catch (IOException ex) {
263             throw new UncheckedIOException(ex);
264           }
265         }
266       }
267     }//GEN-LAST:event_openMenuItemActionPerformed

298   private void fromTextMenuItemActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_fromTextMenuItemActionPerformed
299 
300     final JDialog frame = new JDialog(this, "Enter json stack samples", true);
301     frame.setName("fromTextDialog");
302     frame.setJMenuBar(createContextMenuBar());
303     TextEntryPanel panel = new TextEntryPanel((title, samples) -> {
304         Instant now = Instant.now();
305         JInternalFrame f;
306         try {
307           f = new StackDumpJInternalFrame(new OneStackSampleSupplier(now, now, samples), title, true);
308         } catch (IOException ex) {
309          throw new UncheckedIOException(ex);
310         }
311         frame.setVisible(false);
312         f.setVisible(true);
313         desktopPane.add(f, javax.swing.JLayeredPane.DEFAULT_LAYER);
314     }, (ex) -> {
315       JOptionPane.showMessageDialog(frame, "Invalid input " + ex);
316       frame.setVisible(false);
317       frame.dispose();
318     });
319     frame.getContentPane().add(panel);
320     frame.pack();
321     frame.setVisible(true);
322   }//GEN-LAST:event_fromTextMenuItemActionPerformed

353   private void openFile(final File file) throws IOException {
354     String fileName = file.getName();
355     JInternalFrame frame;
356     if (Spf4jFileFilter.TSDB.accept(file)) {
357       frame = new TSDBViewJInternalFrame(file);
358       frame.setVisible(true);
359       desktopPane.add(frame, javax.swing.JLayeredPane.DEFAULT_LAYER);
360     } else if (Spf4jFileFilter.TSDB2.accept(file)) {
361       frame = new TSDB2ViewJInternalFrame(file);
362       frame.setVisible(true);
363       desktopPane.add(frame, javax.swing.JLayeredPane.DEFAULT_LAYER);
364     } else if (Spf4jFileFilter.AVRO_TABLEDEF.accept(file)) {
365       frame = new MStoreViewJInternalFrame(file);
366       frame.setVisible(true);
367       desktopPane.add(frame, javax.swing.JLayeredPane.DEFAULT_LAYER);
368     } else if (Spf4jFileFilter.SSDUMP.accept(file)) {
369       SampleNode samples = loadLegacyFormat(file);
370       setFrames(samples, fileName);
371     } else if (Spf4jFileFilter.SSDUMP2.accept(file) || Spf4jFileFilter.SSDUMP2_GZ.accept(file)) {
372       SampleNode samples = org.spf4j.ssdump2.Converter.load(file);
373       setFrames(samples, fileName);
374     } else if (Spf4jFileFilter.SSDUMP3.accept(file) || Spf4jFileFilter.SSDUMP3_GZ.accept(file)) {
375       Map<String, SampleNode> loadLabeledDumps = org.spf4j.ssdump2.Converter.loadLabeledDumps(file);
376       setFrames(loadLabeledDumps, fileName);
377     } else if (Spf4jFileFilter.D3_JSON.accept(file)) {
378       try (BufferedReader br = Files.newBufferedReader(file.toPath(), StandardCharsets.UTF_8)) {
379         Pair<Method, SampleNode> parse = SampleNode.parseD3Json(br);
380         setFrames(parse.getSecond(), fileName);
381       }
382     } else if (Spf4jFileFilter.SPF4J_JSON.accept(file)) {
383       try (BufferedReader br = Files.newBufferedReader(file.toPath(), StandardCharsets.UTF_8)) {
384         Pair<Method, SampleNode> parse = SampleNode.parse(br);
385         setFrames(parse.getSecond(), fileName);
386       }
387     } else if (Spf4jFileFilter.PROFILE_AVRO.accept(file)) {
388       setFrames(new AvroStackSampleSupplier(file.toPath()), fileName);
389     }else {
390       throw new IOException("Unsupported file format " + fileName);
391     }
392   }

394   private static SampleNode loadLegacyFormat(final File file) throws IOException {
395     InputStream fis = Files.newInputStream(file.toPath());
396     return loadLegacyFormat(fis);
397   }
398 
399   @VisibleForTesting
400   static SampleNode loadLegacyFormat(InputStream fis) throws IOException {
401     try (BufferedInputStream bis = new BufferedInputStream(fis)) {
402       final CodedInputStream is = CodedInputStream.newInstance(bis);
403       is.setRecursionLimit(Short.MAX_VALUE);
404       return Converter.fromProtoToSampleNode(ProtoSampleNodes.SampleNode.parseFrom(is));
405     }
406   }
