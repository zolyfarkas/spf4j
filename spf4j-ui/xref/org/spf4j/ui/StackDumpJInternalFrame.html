49      public StackDumpJInternalFrame(final String sampleFile, final boolean isPro) throws IOException {
50          super(sampleFile);
51          initComponents();
52          if (sampleFile.endsWith("ssdump")) {
53              try (BufferedInputStream bis = new BufferedInputStream(new FileInputStream(sampleFile))) {
54                  final CodedInputStream is = CodedInputStream.newInstance(bis);
55                  is.setRecursionLimit(Short.MAX_VALUE);
56                  samples = Converter.fromProtoToSampleNode(ProtoSampleNodes.SampleNode.parseFrom(is));
57              }
58          } else {
59              samples = org.spf4j.ssdump2.Converter.load(new File(sampleFile));
60          }
61          if (isPro) {
62              ssScrollPanel.setViewportView(new ZStackPanel(samples));
63          } else {
64              ssScrollPanel.setViewportView(new FlameStackPanel(samples));
65          }
66          ssScrollPanel.setVisible(true);
67          pack();
68  
69      }
