120   private static final class TextTransferHandler extends TransferHandler {
121     // Start and end position in the source text.
122     // We need this information when performing a MOVE
123     // in order to remove the dragged text from the source.
124 
125     Position p0 = null, p1 = null;
126 
127     /**
128      * Perform the actual import. This method supports both drag and drop and cut/copy/paste.
129      */
130     public boolean importData(TransferHandler.TransferSupport support) {
131       // If we can't handle the import, bail now.
132       if (!canImport(support)) {
133         return false;
134       }
135 
136       // Fetch the data -- bail if this fails
137       String data;
138       try {
139         data = (String) support.getTransferable().getTransferData(
140                 DataFlavor.stringFlavor);
141       } catch (UnsupportedFlavorException | IOException e) {
142         Logger.getLogger(TextEntryPanel.class.getName()).log(Level.WARNING, "Exception encountered", e);
143         return false;
144       }
145 
146       javax.swing.JTextPane tc = (javax.swing.JTextPane) support.getComponent();
147       tc.replaceSelection(data);
148       return true;
149     }
150 
151     /**
152      * Bundle up the data for export.
153      */
154     @Nullable
155     protected Transferable createTransferable(JComponent c) {
156       javax.swing.JTextPane source = (javax.swing.JTextPane) c;
157       int start = source.getSelectionStart();
158       int end = source.getSelectionEnd();
159       Document doc = source.getDocument();
160       if (start == end) {
161         return null;
162       }
163       try {
164         p0 = doc.createPosition(start);
165         p1 = doc.createPosition(end);
166       } catch (BadLocationException e) {
167         throw new RuntimeException(e);
168       }
169       String data = source.getSelectedText();
170       return new StringSelection(data);
171     }
172 
173     /**
174      * These text fields handle both copy and move actions.
175      */
176     public int getSourceActions(JComponent c) {
177       return COPY_OR_MOVE;
178     }
179 
180     /**
181      * When the export is complete, remove the old text if the action was a move.
182      */
183     protected void exportDone(JComponent c, Transferable data, int action) {
184       if (action != MOVE) {
185         return;
186       }
187 
188       if ((p0 != null) && (p1 != null) && (p0.getOffset() != p1.getOffset())) {
189         try {
190           JTextComponent tc = (JTextComponent) c;
191           tc.getDocument()
192                   .remove(p0.getOffset(), p1.getOffset() - p0.getOffset());
193         } catch (BadLocationException e) {
194           throw new RuntimeException(e);
195         }
196       }
197     }
198 
199     /**
200      * We only support importing strings.
201      */
202     public boolean canImport(TransferHandler.TransferSupport support) {
203       // we only import Strings
204       if (!support.isDataFlavorSupported(DataFlavor.stringFlavor)) {
205         return false;
206       }
207       return true;
208     }
209   }
