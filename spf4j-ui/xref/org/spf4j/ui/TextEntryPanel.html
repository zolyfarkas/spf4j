124   public static Object readAvroBin(final InputStream input, final Schema writerSchema)
125           throws IOException {
126     DatumReader reader = new SpecificDatumReader(writerSchema);
127     DecoderFactory decoderFactory = DecoderFactory.get();
128     Decoder decoder = decoderFactory.binaryDecoder(input, null);
129     return reader.read(null, decoder);
130   }

133   @SuppressFBWarnings({"UP_UNUSED_PARAMETER", "URLCONNECTION_SSRF_FD"})
134   private void displayActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_displayActionPerformed
135     try {
136       String text = jTextPane1.getText().trim();
137       if (text.startsWith("http")) {
138         URL url = new URL(text);
139         URLConnection conn = url.openConnection();
140         conn.setRequestProperty("Accept", "application/avro");
141         conn.connect();
142         String contentType = conn.getContentType();
143         if (!"application/avro".equals(contentType)) {
144           throw new IOException("Unsupported content type " + contentType);
145         }
146         try (InputStream is = new BufferedInputStream(conn.getInputStream())) {
147           List<LogRecord> recs =
148                   (List<LogRecord>) readAvroBin(is, Schema.createArray(LogRecord.SCHEMA$));
149           for (LogRecord rec : recs) {
150             List<StackSampleElement> stackSamples = rec.getStackSamples();
151             if (!stackSamples.isEmpty()) {
152               nodeConsumer.accept(rec.getMsg() + "; with trId=" + rec.getTrId(),
153                       Converter.convert(stackSamples.iterator()));
154             }
155           }
156         }
157       } else if (text.startsWith("[")) {
158         Schema schema = Schema.createArray(StackSampleElement.getClassSchema());
159         DatumReader reader = new SpecificDatumReader(schema);
160         List<StackSampleElement> samples;
161         try {
162           Decoder decoder = DecoderFactory.get().jsonDecoder(schema, text);
163           samples = (List<StackSampleElement>) reader.read(null, decoder);
164         } catch (IOException ex) {
165           throw new UncheckedIOException(ex);
166         }
167         nodeConsumer.accept("SampleNode Array", Converter.convert(samples.iterator()));
168       } else {
169         nodeConsumer.accept("SampleNode Tree", SampleNode.parse(new StringReader(text)).getSecond());
170       }
171     } catch (IOException | RuntimeException ex) {
172       errorConsumer.accept(ex);
173     }
174   }//GEN-LAST:event_displayActionPerformed

182   private static final class TextTransferHandler extends TransferHandler {
183     // Start and end position in the source text.
184     // We need this information when performing a MOVE
185     // in order to remove the dragged text from the source.
186 
187     Position p0 = null, p1 = null;
188 
189     /**
190      * Perform the actual import. This method supports both drag and drop and cut/copy/paste.
191      */
192     public boolean importData(TransferHandler.TransferSupport support) {
193       // If we can't handle the import, bail now.
194       if (!canImport(support)) {
195         return false;
196       }
197 
198       // Fetch the data -- bail if this fails
199       String data;
200       try {
201         data = (String) support.getTransferable().getTransferData(
202                 DataFlavor.stringFlavor);
203       } catch (UnsupportedFlavorException | IOException e) {
204         Logger.getLogger(TextEntryPanel.class.getName()).log(Level.WARNING, "Exception encountered", e);
205         return false;
206       }
207 
208       javax.swing.JTextPane tc = (javax.swing.JTextPane) support.getComponent();
209       tc.replaceSelection(data);
210       return true;
211     }
212 
213     /**
214      * Bundle up the data for export.
215      */
216     @Nullable
217     protected Transferable createTransferable(JComponent c) {
218       javax.swing.JTextPane source = (javax.swing.JTextPane) c;
219       int start = source.getSelectionStart();
220       int end = source.getSelectionEnd();
221       Document doc = source.getDocument();
222       if (start == end) {
223         return null;
224       }
225       try {
226         p0 = doc.createPosition(start);
227         p1 = doc.createPosition(end);
228       } catch (BadLocationException e) {
229         throw new RuntimeException(e);
230       }
231       String data = source.getSelectedText();
232       return new StringSelection(data);
233     }
234 
235     /**
236      * These text fields handle both copy and move actions.
237      */
238     public int getSourceActions(JComponent c) {
239       return COPY_OR_MOVE;
240     }
241 
242     /**
243      * When the export is complete, remove the old text if the action was a move.
244      */
245     protected void exportDone(JComponent c, Transferable data, int action) {
246       if (action != MOVE) {
247         return;
248       }
249 
250       if ((p0 != null) && (p1 != null) && (p0.getOffset() != p1.getOffset())) {
251         try {
252           JTextComponent tc = (JTextComponent) c;
253           tc.getDocument()
254                   .remove(p0.getOffset(), p1.getOffset() - p0.getOffset());
255         } catch (BadLocationException e) {
256           throw new RuntimeException(e);
257         }
258       }
259     }
260 
261     /**
262      * We only support importing strings.
263      */
264     public boolean canImport(TransferHandler.TransferSupport support) {
265       // we only import Strings
266       if (!support.isDataFlavorSupported(DataFlavor.stringFlavor)) {
267         return false;
268       }
269       return true;
270     }
271   }
