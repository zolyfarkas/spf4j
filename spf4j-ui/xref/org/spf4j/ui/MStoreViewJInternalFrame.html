91    @SuppressForbiden
92    public MStoreViewJInternalFrame(final File tsDb) throws IOException {
93      super(tsDb.getPath());
94      String fileName = tsDb.getName();
95      setName(fileName);
96      initComponents();
97      if (fileName.endsWith(".tabledef.avro")) {
98        reader = new AvroMeasurementStoreReader(tsDb.toPath());
99      } else {
100       throw new IllegalArgumentException("Not a tabledef.avrro file " + tsDb);
101     }
102     Collection<Schema> measurements = reader.getMeasurements((x) -> true);
103     long startDateMillis = System.currentTimeMillis();
104     try (AvroCloseableIterable<Observation> scan = reader.getObservations()) {
105       for (Observation obs : scan) {
106         long ts = obs.getRelTimeStamp();
107         if (ts < startDateMillis) {
108           startDateMillis = ts;
109         }
110       }
111     }
112 
113     Map<String, DefaultMutableTreeNode> gNodes = new HashMap<>();
114     DefaultMutableTreeNode root = new DefaultMutableTreeNode(fileName);
115     for (Schema info : measurements) {
116       String groupName = info.getProp(TimeSeriesRecord.RAW_NAME);
117       Pair<String, String> pair = Pair.from(groupName);
118       if (pair == null) {
119         DefaultMutableTreeNode child = new DefaultMutableTreeNode(info);
120         root.add(child);
121       } else {
122         groupName = pair.getFirst();
123         DefaultMutableTreeNode gNode = gNodes.get(groupName);
124         if (gNode == null) {
125           gNode = new DefaultMutableTreeNode(groupName);
126           gNodes.put(groupName, gNode);
127           root.add(gNode);
128         }
129         DefaultMutableTreeNode child = new DefaultMutableTreeNode(info);
130         gNode.add(child);
131       }
132     }
133     measurementTree.setModel(new DefaultTreeModel(root));
134     measurementTree.setCellRenderer(new TreeCellRenderer());
135     measurementTree.setVisible(true);
136     this.startDate.setValue(new Date(startDateMillis));
137   }

146       Object value = ((DefaultMutableTreeNode) node).getUserObject();

285     private void plotButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_plotButtonActionPerformed
286       TreePath[] selectionPaths = measurementTree.getSelectionPaths();
287       JPanel content = new JPanel();
288       content.setLayout(new BoxLayout(content, BoxLayout.Y_AXIS));
289       chartPannel.setViewportView(content);
290       try {
291         Set<Schema> selectedTables = getSelectedTables(selectionPaths);
292         for (Schema tableName : selectedTables) {
293           addChartToPanel(tableName, content);
294         }
295       } catch (IOException ex) {

297       }

301   @SuppressFBWarnings("UP_UNUSED_PARAMETER")
302     private void exportButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_exportButtonActionPerformed
303       TreePath[] selectionPaths = measurementTree.getSelectionPaths();
304       Set<Schema> selectedTables = getSelectedTables(selectionPaths);
305       if (!selectedTables.isEmpty()) {
306         JFileChooser chooser = new JFileChooser();
307         chooser.setDialogType(JFileChooser.SAVE_DIALOG);
308         int returnVal = chooser.showOpenDialog(this);
309         if (returnVal == JFileChooser.APPROVE_OPTION) {
310           File file = chooser.getSelectedFile();
311           for (Schema metric : selectedTables) {
312             try (AvroCloseableIterable<TimeSeriesRecord> obs = reader.getMeasurementData(metric,
313                     Instant.EPOCH, Instant.now())) {
314               CsvWriter writer = Csv.CSV.writer(Files.newBufferedWriter(file.toPath(), StandardCharsets.UTF_8));
315               CsvEncoder encoder = new CsvEncoder(writer, Schema.createArray(metric));
316               encoder.writeHeader();
317               GenericDatumWriter dw = new GenericDatumWriter(metric);
318               for (TimeSeriesRecord o : obs) {
319                 dw.write(o, encoder);
320               }
321               encoder.flush();
322             } catch (IOException ex) {
323               throw new UncheckedIOException(ex);
324             }
325           }
326         }
327       }
328     }//GEN-LAST:event_exportButtonActionPerformed

350   @SuppressFBWarnings("CLI_CONSTANT_LIST_INDEX")
351   private static Set<Schema> getSelectedTables(@Nullable final TreePath[] selectionPaths) {
352     if (selectionPaths == null) {
353       return Collections.EMPTY_SET;
354     }
355     Set<Schema> result = new THashSet<>();
356     for (TreePath path : selectionPaths) {
357       Object[] pathArr = path.getPath();
358       if (pathArr.length < 2) {
359         continue;
360       }
361       DefaultMutableTreeNode colNode = (DefaultMutableTreeNode) pathArr[pathArr.length - 1];
362       int depth = colNode.getDepth();
363       if (depth == 0) {
364         result.add((Schema) colNode.getUserObject());
365       } else {
366         Enumeration childEnum = colNode.children();
367         while (childEnum.hasMoreElements()) {
368           DefaultMutableTreeNode child = (DefaultMutableTreeNode) childEnum.nextElement();
369           result.add((Schema) child.getUserObject());
370         }
371       }
372     }
373     return result;
374   }

376   private void addChartToPanel(final Schema table, final JPanel content) throws IOException {
377     long startTime = ((Date) startDate.getValue()).getTime();
378     long endTime = ((Date) endDate.getValue()).getTime();
379     Integer aggMillis = Ints.tryParse(aggregationMillis.getText());
380     if (aggMillis == null) {
381       aggMillis = 0;
382     }
383     if (Charts2.canGenerateHeatChart(table)) {
384       JFreeChart chart = Charts2.createHeatJFreeChart(reader, table,
385               startTime, endTime, aggMillis);
386       ChartPanel pannel = new ChartPanel(chart);
387       pannel.setPreferredSize(new Dimension(600, 800));
388       pannel.setDomainZoomable(false);
389       pannel.setMouseZoomable(false);
390       pannel.setRangeZoomable(false);
391       pannel.setZoomAroundAnchor(false);
392       pannel.setZoomInFactor(1);
393       pannel.setZoomOutFactor(1);
394       content.add(pannel);
395     }
396     if (Charts2.canGenerateMinMaxAvgCount(table)) {
397       JFreeChart chart = Charts2.createMinMaxAvgJFreeChart(reader, table,
398               startTime, endTime, aggMillis);
399       ChartPanel pannel = new ChartPanel(chart);
400       pannel.setPreferredSize(new Dimension(600, 600));
401       content.add(pannel);
402 
403     }
404     if (Charts2.canGenerateCount(table)) {
405       JFreeChart chart = Charts2.createCountJFreeChart(reader, table,
406               startTime, endTime, aggMillis);
407       ChartPanel pannel = new ChartPanel(chart);
408       pannel.setPreferredSize(new Dimension(600, 600));
409       content.add(pannel);
410     } else {
411       List<JFreeChart> createJFreeCharts = Charts2.createJFreeCharts(reader, table, startTime, endTime, aggMillis);
412       for (JFreeChart chart : createJFreeCharts) {
413         ChartPanel pannel = new ChartPanel(chart);
414         pannel.setPreferredSize(new Dimension(600, 600));
415         content.add(pannel);
416       }
417     }
