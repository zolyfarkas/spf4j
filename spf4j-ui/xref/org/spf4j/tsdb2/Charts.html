26     public static boolean canGenerateMinMaxAvgCount(final TableDef info) {
27         int found = 0;
28         for (ColumnDef colDef : info.getColumns()) {
29             switch (colDef.getName()) {
30                 case "min":
31                 case "max":
32                 case "total":
33                 case "count":
34                     found++;
35                     break;
36                 default:
37             }
38         }
39         return found >= 4;
40      }

56      @SuppressFBWarnings("STT_STRING_PARSING_A_FIELD")
57      public static boolean canGenerateHeatChart(final TableDef info) {
58          for (ColumnDef colDef : info.getColumns()) {
59              if (colDef.name.startsWith("Q") && colDef.name.contains("_")) {
60                  return true;
61              }
62          }
63          return false;
64      }

73     public static JFreeChart createHeatJFreeChart(final TimeSeries data, final TableDef info) {
74          Pair<long[], double[][]> mData = fillGaps(data.getTimeStamps(), data.getValues(),
75                  info.getSampleTime(), info.getColumns().size());
76          ColumnDef columnDef = TSDBQuery.getColumnDef(info, "total");
77          return org.spf4j.perf.impl.chart.Charts.createHeatJFreeChart(TSDBQuery.getColumnNames(info),
78                  mData.getSecond(), data.getTimeStamps()[0], info.getSampleTime(),
79                  columnDef.getUnitOfMeasurement(), "Measurements distribution for "
80                  + info.getName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j");
81      }

83      public static JFreeChart createMinMaxAvgJFreeChart(final TimeSeries data, final TableDef info) {
84          long[][] vals = data.getValues();
85          double[] min = Arrays.getColumnAsDoubles(vals, TSDBQuery.getColumnIndex(info, "min"));
86          double[] max = Arrays.getColumnAsDoubles(vals, TSDBQuery.getColumnIndex(info, "max"));
87          final int totalColumnIndex = TSDBQuery.getColumnIndex(info, "total");
88          double[] total = Arrays.getColumnAsDoubles(vals, totalColumnIndex);
89          double[] count = Arrays.getColumnAsDoubles(vals, TSDBQuery.getColumnIndex(info, "count"));
90          for (int i = 0; i < count.length; i++) {
91              if (count[i] == 0) {
92                  min[i] = 0;
93                  max[i] = 0;
94              }
95          }
96          long[] timestamps = data.getTimeStamps();
97          return org.spf4j.perf.impl.chart.Charts.createTimeSeriesJFreeChart("Min,Max,Avg chart for "
98                  + info.getName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j", timestamps,
99                  new String[]{"min", "max", "avg"},
100                 TSDBQuery.getColumnDef(info, "total").getUnitOfMeasurement(),
101                 new double[][]{min, max, Arrays.divide(total, count)});
102     }

127     public static List<JFreeChart> createJFreeCharts(final TimeSeries data, final TableDef info) {
128         long[][] vals = data.getValues();
129         List<JFreeChart> result = new ArrayList<>();
130         Map<String, Pair<List<String>, List<double[]>>> measurementsByUom = new HashMap<>();
131         //String[] columnMetaData = TSDBQuery.getColumnUnitsOfMeasurement(info);
132         int i = 0;
133         for (ColumnDef colDef : info.getColumns()) {
134             String uom = colDef.getUnitOfMeasurement();
135             Pair<List<String>, List<double[]>> meas = measurementsByUom.get(uom);
136             if (meas == null) {
137                 meas = Pair.of((List<String>) new ArrayList<String>(), (List<double[]>) new ArrayList<double[]>());
138                 measurementsByUom.put(uom, meas);
139             }
140             meas.getFirst().add(colDef.getName());
141             meas.getSecond().add(Arrays.getColumnAsDoubles(vals, i));
142             i++;
143         }
144         long[] timestamps = data.getTimeStamps();
145         for (Map.Entry<String, Pair<List<String>, List<double[]>>> entry : measurementsByUom.entrySet()) {
146             Pair<List<String>, List<double[]>> p = entry.getValue();
147             final List<String> measurementNames = p.getFirst();
148             final List<double[]> measurements = p.getSecond();
149             result.add(org.spf4j.perf.impl.chart.Charts.createTimeSeriesJFreeChart("chart for "
150                     + info.getName() + ", sampleTime " + info.getSampleTime()
151                     + " ms, generated by spf4j", timestamps,
152                     measurementNames.toArray(new String[measurementNames.size()]), entry.getKey(),
153                     measurements.toArray(new double[measurements.size()][])));
154         }
155         return result;
156     }
