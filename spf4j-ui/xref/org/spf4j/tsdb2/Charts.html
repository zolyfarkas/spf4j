57    public static boolean canGenerateMinMaxAvgCount(final TableDef info) {
58      int found = 0;
59      for (ColumnDef colDef : info.getColumns()) {
60        switch (colDef.getName()) {
61          case "min":
62          case "max":
63          case "total":
64          case "count":
65            found++;
66            break;
67          default:
68        }
69      }
70      return found >= 4;
71    }

73    public static boolean canGenerateCount(final TableDef info) {
74      int found = 0;
75      for (ColumnDef colDef : info.getColumns()) {
76        switch (colDef.getName()) {
77          case "count":
78            found++;
79            break;
80          default:
81        }
82      }
83      return found >= 1;
84    }

86    @SuppressFBWarnings("STT_STRING_PARSING_A_FIELD")
87    public static boolean canGenerateHeatChart(final TableDef info) {
88      for (ColumnDef colDef : info.getColumns()) {
89        String colName = colDef.getName();
90        if (colName.startsWith("Q") && colName.contains("_")) {
91          return true;
92        }
93      }
94      return false;
95    }

104   public static JFreeChart createHeatJFreeChart(final TimeSeries data, final TableDef info) {
105     Pair<long[], double[][]> mData = fillGaps(data.getTimeStamps(), data.getValues(),
106             info.getSampleTime(), info.getColumns().size());
107     ColumnDef columnDef = TSDBQuery.getColumnDef(info, "total");
108     return org.spf4j.perf.impl.chart.Charts.createHeatJFreeChart(TSDBQuery.getColumnNames(info),
109             mData.getSecond(), data.getTimeStamps()[0], info.getSampleTime(),
110             columnDef.getUnitOfMeasurement(), "Measurements distribution for "
111             + info.getName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j");
112   }

114   public static JFreeChart createMinMaxAvgJFreeChart(final TimeSeries data, final TableDef info) {
115     long[][] vals = data.getValues();
116     double[] min = Arrays.getColumnAsDoubles(vals, TSDBQuery.getColumnIndex(info, "min"));
117     double[] max = Arrays.getColumnAsDoubles(vals, TSDBQuery.getColumnIndex(info, "max"));
118     final int totalColumnIndex = TSDBQuery.getColumnIndex(info, "total");
119     double[] total = Arrays.getColumnAsDoubles(vals, totalColumnIndex);
120     double[] count = Arrays.getColumnAsDoubles(vals, TSDBQuery.getColumnIndex(info, "count"));
121     for (int i = 0; i < count.length; i++) {
122       if (count[i] == 0) {
123         min[i] = 0;
124         max[i] = 0;
125       }
126     }
127     long[] timestamps = data.getTimeStamps();
128     return org.spf4j.perf.impl.chart.Charts.createTimeSeriesJFreeChart("Min,Max,Avg chart for "
129             + info.getName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j", timestamps,
130             new String[]{"min", "max", "avg"},
131             TSDBQuery.getColumnDef(info, "total").getUnitOfMeasurement(),
132             new double[][]{min, max, Arrays.divide(total, count)});
133   }

156   public static List<JFreeChart> createJFreeCharts(final TimeSeries data, final TableDef info) {
157     long[][] vals = data.getValues();
158     Map<String, Pair<List<String>, List<double[]>>> measurementsByUom = new HashMap<>();
159     //String[] columnMetaData = TSDBQuery.getColumnUnitsOfMeasurement(info);
160     int i = 0;
161     for (ColumnDef colDef : info.getColumns()) {
162       String uom = colDef.getUnitOfMeasurement();
163       Pair<List<String>, List<double[]>> meas = measurementsByUom.get(uom);
164       if (meas == null) {
165         meas = Pair.of((List<String>) new ArrayList<String>(), (List<double[]>) new ArrayList<double[]>());
166         measurementsByUom.put(uom, meas);
167       }
168       meas.getFirst().add(colDef.getName());
169       meas.getSecond().add(Arrays.getColumnAsDoubles(vals, i));
170       i++;
171     }
172     long[] timestamps = data.getTimeStamps();
173     List<JFreeChart> result = new ArrayList<>(measurementsByUom.size());
174     for (Map.Entry<String, Pair<List<String>, List<double[]>>> entry : measurementsByUom.entrySet()) {
175       Pair<List<String>, List<double[]>> p = entry.getValue();
176       final List<String> measurementNames = p.getFirst();
177       final List<double[]> measurements = p.getSecond();
178       result.add(org.spf4j.perf.impl.chart.Charts.createTimeSeriesJFreeChart("chart for "
179               + info.getName() + ", sampleTime " + info.getSampleTime()
180               + " ms, generated by spf4j", timestamps,
181               measurementNames.toArray(new String[measurementNames.size()]), entry.getKey(),
182               measurements.toArray(new double[measurements.size()][])));
183     }
184     return result;
185   }

189     TimeSeries data = TSDBQuery.getTimeSeries(database, TSDBQuery.getIds(td), startTime, endTime);
